840 // Lightweight std::vector<> like class to avoid dragging dependencies (also: windows implementation of STL with debug enabled is absurdly slow, so let's bypass it so our code runs fast in debug).
841 // Our implementation does NOT call c++ constructors because we don't use them in ImGui. Don't use this class as a straight std::vector replacement in your code!
842 template<typename T>
843 class ImVector
844 {
845 public:
846  int Size;
847  int Capacity;
848  T* Data;
849 
850  typedef T value_type;
851  typedef value_type* iterator;
852  typedef const value_type* const_iterator;
853 
854  ImVector() { Size = Capacity = 0; Data = NULL; }
855  ~ImVector() { if (Data) ImGui::MemFree(Data); }
856 
857  inline bool empty() const { return Size == 0; }
858  inline int size() const { return Size; }
859  inline int capacity() const { return Capacity; }
860 
861  inline value_type& operator[](int i) { IM_ASSERT(i < Size); return Data[i]; }
862  inline const value_type& operator[](int i) const { IM_ASSERT(i < Size); return Data[i]; }
863 
864  inline void clear() { if (Data) { Size = Capacity = 0; ImGui::MemFree(Data); Data = NULL; } }
865  inline iterator begin() { return Data; }
866  inline const_iterator begin() const { return Data; }
867  inline iterator end() { return Data + Size; }
868  inline const_iterator end() const { return Data + Size; }
869  inline value_type& front() { IM_ASSERT(Size > 0); return Data[0]; }
870  inline const value_type& front() const { IM_ASSERT(Size > 0); return Data[0]; }
871  inline value_type& back() { IM_ASSERT(Size > 0); return Data[Size-1]; }
872  inline const value_type& back() const { IM_ASSERT(Size > 0); return Data[Size-1]; }
873  inline void swap(ImVector<T>& rhs) { int rhs_size = rhs.Size; rhs.Size = Size; Size = rhs_size; int rhs_cap = rhs.Capacity; rhs.Capacity = Capacity; Capacity = rhs_cap; value_type* rhs_data = rhs.Data; rhs.Data = Data; Data = rhs_data; }
874 
875  inline int _grow_capacity(int new_size) { int new_capacity = Capacity ? (Capacity + Capacity/2) : 8; return new_capacity > new_size ? new_capacity : new_size; }
876 
877  inline void resize(int new_size) { if (new_size > Capacity) reserve(_grow_capacity(new_size)); Size = new_size; }
878  inline void reserve(int new_capacity)
879  {
880  if (new_capacity <= Capacity) return;
881  T* new_data = (value_type*)ImGui::MemAlloc((size_t)new_capacity * sizeof(value_type));
882  if (Data)
883  memcpy(new_data, Data, (size_t)Size * sizeof(value_type));
884  ImGui::MemFree(Data);
885  Data = new_data;
886  Capacity = new_capacity;
887  }
888 
889  inline void push_back(const value_type& v) { if (Size == Capacity) reserve(_grow_capacity(Size+1)); Data[Size++] = v; }
890  inline void pop_back() { IM_ASSERT(Size > 0); Size--; }
891 
892  inline iterator erase(const_iterator it) { IM_ASSERT(it >= Data && it < Data+Size); const ptrdiff_t off = it - Data; memmove(Data + off, Data + off + 1, ((size_t)Size - (size_t)off - 1) * sizeof(value_type)); Size--; return Data + off; }
893  inline iterator insert(const_iterator it, const value_type& v) { IM_ASSERT(it >= Data && it <= Data+Size); const ptrdiff_t off = it - Data; if (Size == Capacity) reserve(Capacity ? Capacity * 2 : 4); if (off < (int)Size) memmove(Data + off + 1, Data + off, ((size_t)Size - (size_t)off) * sizeof(value_type)); Data[off] = v; Size++; return Data + off; }
894 };

1085 // Helper: Manually clip large list of items.
1086 // If you are submitting lots of evenly spaced items and you have a random access to the list, you can perform coarse clipping based on visibility to save yourself from processing those items at all.
1087 // The clipper calculates the range of visible items and advance the cursor to compensate for the non-visible items we have skipped.
