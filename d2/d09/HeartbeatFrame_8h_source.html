48 struct HeartbeatHeader
49 {
50  union {
51  // the complete 64 bit header word, initialize with blockType 1 and size 1
52  uint64_t headerWord = 0x1100000000000000;
53  struct {
54  // bit 0 to 31: orbit number
55  uint32_t orbit;
56  // bit 32 to 43: bunch crossing id
57  uint16_t bcid:12;
58  // bit 44 to 47: reserved
59  uint16_t reserved:4;
60  // bit 48 to 51: trigger type
61  uint8_t triggerType:4;
62  // bit 52 to 55: reserved
63  uint8_t reservedTriggerType:4;
64  // bit 56 to 59: header length
65  uint8_t headerLength:4;
66  // bit 60 to 63: block type (=1 HBF/trigger Header)
67  uint8_t blockType:4;
68  };
69  }; // end union
70 
71  operator bool() const {return headerWord != 0 && blockType == 0x1;}
72 
73  bool operator<(const HeartbeatHeader& other) const {
74  return this->orbit < other.orbit;
75  }
76 
77  operator uint64_t() const {return headerWord;}
78 };
79 static_assert(sizeof(HeartbeatHeader) == 8, "Heartbeat header must be 64bit");

81 struct HeartbeatTrailer
82 {
83  union {
84  // the complete 64 bit trailer word, initialize with blockType 5 and size 1
85  uint64_t trailerWord = 0x5100000000000000;
86  struct {
87  // bit 0 to 31: data length in words
88  uint32_t dataLength;
89  // bit 32 to 52: detector specific status words
90  uint32_t status:21;
91  // bit 53: =1 in case a new physics trigger arrived within read-out period
92  uint16_t hbfTruncated:1;
93  // bit 54: =0 HBF correctly transmitted
94  uint16_t hbfStatus:1;
95  // bit 55: =1 HBa/0 HBr received
96  uint16_t hbAccept:1;
97  // bit 56 to 59: trailer length
98  uint16_t trailerLength:4;
99  // bit 60 to 63: block type (=5 HBF Trailer)
100  uint8_t blockType:4;
101  };
102  }; // end union
103 
104  operator bool() const {return trailerWord != 0 && blockType == 0x5;}
105 
106  operator uint64_t() const {return trailerWord;}
107 };
108 static_assert(sizeof(HeartbeatTrailer) == 8, "Heartbeat trailer must be 64bit");

152 template<typename HeaderT, typename TrailerT>
153 class ReverseParser {
154 public:
155  using HeaderType = HeaderT;
156  using TrailerType = TrailerT;
157 
158  struct FrameEntry {
159  const HeaderType* header = nullptr;
160  const TrailerType* trailer = nullptr;
161  const byte* payload = nullptr;
162  size_t length = 0;
163  };
164  static const size_t envelopeLength = sizeof(HeaderType) + sizeof(TrailerType);
165 
166  using CheckHeaderFct = std::function<bool(const HeaderType*)>;
167  using CheckTrailerFct = std::function<bool(const TrailerType*)>;
168  using GetFrameSizeFct = std::function<size_t(const TrailerType& )>;
169  using InsertFct = std::function<bool(FrameEntry )>;
170 
171  int parse(const byte* buffer, size_t bufferSize,
172  CheckHeaderFct checkHeader,
173  CheckTrailerFct checkTrailer,
174  GetFrameSizeFct getFrameSize,
175  InsertFct insert) {
176  if (buffer == nullptr || bufferSize == 0) return 0;
177  unsigned nFrames = 0;
178  auto position = bufferSize;
179  do {
180  FrameEntry entry;
181  if (sizeof(TrailerType) > position) break;
182  entry.trailer = reinterpret_cast<const TrailerType*>(buffer + position - sizeof(TrailerType));
183  if (!checkTrailer(entry.trailer)) break;
184  auto frameSize = getFrameSize(*entry.trailer);
185  if (frameSize > position) break;
186  entry.header = reinterpret_cast<const HeaderType*>(buffer + position - frameSize);
187  if (!checkHeader(entry.header)) break;
188  entry.payload = reinterpret_cast<const byte*>(entry.header + 1);
189  entry.length = frameSize - sizeof(HeaderType) - sizeof(TrailerType);
190  if (!insert(entry)) break;
191  ++nFrames;
192  position -= frameSize;
193  } while (position > 0);
194 
195  if (position == 0) {
196  // frames found and format consistent
197  return nFrames;
198  } else if (nFrames == 0) {
199  // no frames found at all, th buffer does not contain any
200  return 0;
201  }
202 
203  // format error detected
204  // TODO: decide about error policy
205  return -1;
206  }
207 };

234  bool operator<(const FrameIndex& other) const {
235  // std::map.find uses the logic !(a < b) && !(b < a)
236  // have to combine the two fields in one variable for
237  // comparison
238  uint64_t us = columnIndex;
239  us &= 0x00000fffffffffff;
240  us <<= 20;
241  us |= slot & 0xfffff;
242  uint64_t them = other.columnIndex;
243  them &= 0x00000fffffffffff;
244  them <<= 20;
245  them |= other.slot & 0xfffff;
246  return us < them;
247  }

264  size_t addSlot(SlotDataType slotData, byte* seqData, size_t seqSize) {
265  unsigned nFrames = mFrames.size();
266  unsigned currentSlot = mSlotData.size();
267  mSlotData.emplace_back(slotData);
268  using ParserT = o2::header::ReverseParser<HeartbeatHeader, HeartbeatTrailer>;
269  ParserT p;
270  p.parse(seqData, seqSize,
271  [](const typename ParserT::HeaderType* h) {return (*h);},
272  [](const typename ParserT::TrailerType* t) {return (*t);},
273  [](const typename ParserT::TrailerType& t) {
274  return t.dataLength + ParserT::envelopeLength;
275  },
276  [this, currentSlot](typename ParserT::FrameEntry entry) {
277  // insert the heartbeat header as column index in ascending
278  // order
279  auto position = mColumns.begin();
280  while (position != mColumns.end() && *position < *entry.header) {
281  position++;
282  }
283  if (position == mColumns.end() || *entry.header < *position) {
284  mColumns.emplace(position, *entry.header);
285  }
286 
287  // insert frame descriptor under key composed from header and slot
288  auto result = mFrames.emplace(FrameIndex{*entry.header, currentSlot},
289  FrameData{entry.payload, entry.length});
290  return result.second;
291  }
292  );
293  return mFrames.size() - nFrames;
294  }
