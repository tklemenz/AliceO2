25 GeneratorFromFile::GeneratorFromFile(const char* name)
26 {
27  mEventFile = TFile::Open(name);
28  if (mEventFile == nullptr) {
29  LOG(FATAL) << "EventFile " << name << " not found \n";
30  return;
31  }
32  // the kinematics will be stored inside a Tree "TreeK" with branch "Particles"
33  // different events are stored inside TDirectories
34 
35  // we need to probe for the number of events
36  TObject* object = nullptr;
37  do {
38  std::stringstream eventstringstr;
39  eventstringstr << "Event" << mEventsAvailable;
40  // std::cout << "probing for " << eventstring << "\n";
41  object = mEventFile->Get(eventstringstr.str().c_str());
42  // std::cout << "got " << object << "\n";
43  if (object != nullptr)
44  mEventsAvailable++;
45  } while (object != nullptr);
46  std::cout << "Found " << mEventsAvailable << " events in this file \n";
47 }

58 Bool_t GeneratorFromFile::ReadEvent(FairPrimaryGenerator* primGen)
59 {
60  if (mEventCounter < mEventsAvailable) {
61  // get the tree and the branch
62  std::stringstream treestringstr;
63  treestringstr << "Event" << mEventCounter << "/TreeK";
64  TTree* tree = (TTree*)mEventFile->Get(treestringstr.str().c_str());
65  if (tree == nullptr)
66  return kFALSE;
67 
68  auto branch = tree->GetBranch("Particles");
69  TParticle* primary = new TParticle();
70  branch->SetAddress(&primary);
71  for (int i = 0; i < branch->GetEntries(); ++i) {
72  branch->GetEntry(i); // fill primary
73  auto pdgid = primary->GetPdgCode();
74  auto px = primary->Px();
75  auto py = primary->Py();
76  auto pz = primary->Pz();
77  auto vx = primary->Vx();
78  auto vy = primary->Vy();
79  auto vz = primary->Vz();
80 
81  auto parent = -1;
82  bool wanttracking = true;
83  auto e = primary->Energy();
84  auto tof = primary->T();
85  auto weight = primary->GetWeight();
86  primGen->AddTrack(pdgid, px, py, pz, vx, vy, vz, parent, wanttracking, e, tof, weight);
87  }
88  mEventCounter++;
89  return kTRUE;
90  } else {
91  LOG(ERROR) << "GeneratorFromFile: Ran out of events\n";
92  }
93  return kFALSE;
94 }
