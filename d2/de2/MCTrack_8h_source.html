142  // set bit indicating that this track
143  // left a hit in detector with id iDet
144  void setHit(Int_t iDet) {
145  assert(0<=iDet && iDet < o2::Base::DetID::nDetectors);
146  auto prop = ((PropEncoding)mProp);
147  prop.hitmask |= 1 << iDet;
148  mProp = prop.i;
149  }

151  // did detector iDet see this track?
152  bool leftTrace(Int_t iDet) const {
153  return (((PropEncoding)mProp).hitmask & ( 1 << iDet )) > 0;
154  }

156  // determine how many detectors "saw" this track
157  int getNumDet() const
158  {
159  int count = 0;
160  for (auto i = o2::Base::DetID::First; i < o2::Base::DetID::nDetectors; ++i) {
161  if (leftTrace(i))
162  count++;
163  }
164  return count;
165  }

306 template <typename T>
307 inline Double_t MCTrackT<T>::GetRapidity() const
308 {
309  const auto e = GetEnergy();
310  Double_t y =
311  0.5 * std::log((e + static_cast<double>(mStartVertexMomentumZ)) / (e - static_cast<double>(mStartVertexMomentumZ)));
312  return y;
313 }
