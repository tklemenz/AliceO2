84 Bool_t GeometryManager::getOriginalMatrixFromPath(const char* path, TGeoHMatrix& m)
85 {
86  m.Clear();
87 
88  if (!sGeometry || !sGeometry->IsClosed()) {
89  LOG(ERROR) << "Can't get the original global matrix! gGeoManager doesn't exist or it is still opened!"
90  << FairLogger::endl;
91  return kFALSE;
92  }
93 
94  if (!sGeometry->CheckPath(path)) {
95  LOG(ERROR) << "Volume path " << path << " not valid!" << FairLogger::endl;
96  return kFALSE;
97  }
98 
99  TIter next(sGeometry->GetListOfPhysicalNodes());
100  sGeometry->cd(path);
101 
102  while (sGeometry->GetLevel()) {
103  TGeoPhysicalNode* physNode = nullptr;
104  next.Reset();
105  TGeoNode* node = sGeometry->GetCurrentNode();
106 
107  while ((physNode = (TGeoPhysicalNode*)next())) {
108  if (physNode->GetNode() == node) {
109  break;
110  }
111  }
112 
113  TGeoMatrix* lm = nullptr;
114  if (physNode) {
115  lm = physNode->GetOriginalMatrix();
116  if (!lm) {
117  lm = node->GetMatrix();
118  }
119  } else {
120  lm = node->GetMatrix();
121  }
122 
123  m.MultiplyLeft(lm);
124 
125  sGeometry->CdUp();
126  }
127  return kTRUE;
128 }

219 void GeometryManager::MatBudget::accountMaterial(const TGeoMaterial* material)
220 {
221  meanRho = material->GetDensity();
222  meanX2X0 = material->GetRadLen();
223  meanA = material->GetA();
224  meanZ = material->GetZ();
225  if (material->IsMixture()) {
226  TGeoMixture* mixture = (TGeoMixture*)material;
227  Double_t norm = 0.;
228  meanZ2A = 0.;
229  for (Int_t iel = 0; iel < mixture->GetNelements(); iel++) {
230  norm += mixture->GetWmixt()[iel];
231  meanZ2A += mixture->GetZmixt()[iel] * mixture->GetWmixt()[iel] / mixture->GetAmixt()[iel];
232  }
233  meanZ2A /= norm;
234  } else {
235  meanZ2A = meanZ / meanA;
236  }
237 }

240 GeometryManager::MatBudget GeometryManager::MeanMaterialBudget(float x0, float y0, float z0,
241  float x1, float y1, float z1)
242 {
243  //
244  // Calculate mean material budget and material properties between
245  // the points "0" and "1".
246  //
247  // "mparam" - parameters used for the energy and multiple scattering
248  // corrections:
249  //
250  // see MatBudget data members for provided information
251  //
252  // Origin: Marian Ivanov, Marian.Ivanov@cern.ch
253  //
254  // Corrections and improvements by
255  // Andrea Dainese, Andrea.Dainese@lnl.infn.it,
256  // Andrei Gheata, Andrei.Gheata@cern.ch
257  //
258  // Ported to O2: ruben.shahoyan@cern.ch
259  //
260 
261  double length, startD[3] = { x0, y0, z0 };
262  double dir[3] = { x1 - x0, y1 - y0, z1 - z0 };
263  if ((length = dir[0] * dir[0] + dir[1] * dir[1] + dir[2] * dir[2]) <
264  TGeoShape::Tolerance() * TGeoShape::Tolerance()) {
265  return MatBudget(); // return empty struct
266  }
267  length = TMath::Sqrt(length);
268  double invlen = 1. / length;
269  for (int i = 3; i--;) {
270  dir[i] *= invlen;
271  }
272 
273  // Initialize start point and direction
274  TGeoNode* currentnode = gGeoManager->InitTrack(startD, dir);
275  if (!currentnode) {
276  LOG(ERROR) << "start point out of geometry: " << x0 << ':' << y0 << ':' << z0 << FairLogger::endl;
277  return MatBudget(); // return empty struct
278  }
279 
280  MatBudget budTotal, budStep;
281  budStep.accountMaterial(currentnode->GetVolume()->GetMedium()->GetMaterial());
282  budStep.length = length;
283 
284  // Locate next boundary within length without computing safety.
285  // Propagate either with length (if no boundary found) or just cross boundary
286  gGeoManager->FindNextBoundaryAndStep(length, kFALSE);
287  Double_t stepTot = 0.0; // Step made
288  Double_t step = gGeoManager->GetStep();
289  // If no boundary within proposed length, return current step data
290  if (!gGeoManager->IsOnBoundary()) {
291  budStep.meanX2X0 = budStep.length / budStep.meanX2X0;
292  return MatBudget(budStep);
293  }
294  // Try to cross the boundary and see what is next
295  Int_t nzero = 0;
296  while (length > TGeoShape::Tolerance()) {
297  if (step < 2. * TGeoShape::Tolerance()) {
298  nzero++;
299  }
300  else {
301  nzero = 0;
302  }
303  if (nzero > 3) {
304  // This means navigation has problems on one boundary
305  // Try to cross by making a small step
306  const double* curPos = gGeoManager->GetCurrentPoint();
307  LOG(ERROR) << "Cannot cross boundary at (" << curPos[0] << ',' << curPos[1] << ',' << curPos[2] << ')'
308  << FairLogger::endl;
309  budTotal.normalize(stepTot);
310  budTotal.nCross = -1; // flag failed navigation
311  return MatBudget(budTotal);
312  }
313  stepTot += step;
314 
315  budTotal.meanRho += step * budStep.meanRho;
316  budTotal.meanX2X0 += step / budStep.meanX2X0;
317  budTotal.meanA += step * budStep.meanA;
318  budTotal.meanZ += step * budStep.meanZ;
319  budTotal.meanZ2A += step * budStep.meanZ2A;
320  budTotal.nCross++;
321 
322  if (step >= length) {
323  break;
324  }
325  currentnode = gGeoManager->GetCurrentNode();
326  if (!currentnode) {
327  break;
328  }
329  length -= step;
330  budStep.accountMaterial(currentnode->GetVolume()->GetMedium()->GetMaterial());
331  gGeoManager->FindNextBoundaryAndStep(length, kFALSE);
332  step = gGeoManager->GetStep();
333  }
334  budTotal.normalize(stepTot);
335  return MatBudget(budTotal);
336 }
