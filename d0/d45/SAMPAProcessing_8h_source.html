124 template<typename T>
125 inline
126 T SAMPAProcessing::getGamma4(T time, T startTime, T ADC)
127 {
128  const static ParameterElectronics &eleParam = ParameterElectronics::defaultInstance();
129  Vc::float_v tmp0 = (time-startTime)/eleParam.getPeakingTime();
130  Vc::float_m cond = (tmp0 > 0);
131  Vc::float_v tmp;
132  tmp(cond) = tmp0;
133  Vc::float_v tmp2=tmp*tmp;
134  return 55.f*ADC*Vc::exp(-4.f*tmp)*tmp2*tmp2;
135 }
136 }
