53 bool DataSourceDevice::ConditionalRun() {
54  static const auto reftime = std::chrono::system_clock::now();
55  if (mRate > 0.001) {
56  auto timeSinceRef = std::chrono::duration_cast<TimeScale>(std::chrono::system_clock::now() - reftime);
57  auto timespan = timeSinceRef.count() - mLastTime;
58  TimeScale::rep period = (float)TimeScale::period::den / mRate;
59  if (timespan < period) {
60  TimeScale sleepfor(period - timespan);
61  std::this_thread::sleep_for(sleepfor);
62  }
63  mLastTime = std::chrono::duration_cast<TimeScale>(std::chrono::system_clock::now() - reftime).count();
64  }
65  LOG(DEBUG) << "DataSourceDevice::Processing::START";
66  LOG(DEBUG) << "ConditionalRun thread" << pthread_self();
67  // This is dummy because a source does not really have inputs.
68  // However, in order to be orthogonal between sources and
69  // processing code, we still specify it.
70  InputRecord dummyInputs{{}, {}};
71  try {
72  mContext.prepareForTimeslice(mCurrentTimeslice);
73  mRootContext.prepareForTimeslice(mCurrentTimeslice);
74  mCurrentTimeslice += 1;
75 
76  // Avoid runaway process in case we have nothing to do.
77  if ((!mStatefulProcess) && (!mStatelessProcess)) {
78  LOG(ERROR) << "No callback defined";
79  sleep(1);
80  }
81 
82  ProcessingContext processingContext{dummyInputs, mServiceRegistry, mAllocator};
83  if (mStatelessProcess) {
84  LOG(DEBUG) << "Has stateless process callback";
85  mStatelessProcess(processingContext);
86  }
87  if (mStatefulProcess) {
88  LOG(DEBUG) << "Has stateful process callback";
89  mStatefulProcess(processingContext);
90  }
91  size_t nMsg = mContext.size() + mRootContext.size();
92  LOG(DEBUG) << "Process produced " << nMsg << " messages";
93  DataProcessor::doSend(*this, mContext);
94  DataProcessor::doSend(*this, mRootContext);
95  } catch(std::exception &e) {
96  if (mError) {
97  ErrorContext errorContext{dummyInputs, mServiceRegistry, e};
98  mError(errorContext);
99  } else {
100  LOG(DEBUG) << "Uncaught exception: " << e.what();
101  }
102  } catch(...) {
103  LOG(DEBUG) << "Unknown exception type.";
104  LOG(DEBUG) << "DataSourceDevice::Processing::END";
105  return false;
106  }
107  LOG(DEBUG) << "DataSourceDevice::Processing::END";
108  return true;
109 }
