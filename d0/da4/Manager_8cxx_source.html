104  if (singleKeys) {
105  f->WriteObject(&mConditionCache, "CDBentriesMap");

151 Bool_t Manager::initFromSnapshot(const char *snapshotFileName, Bool_t overwrite)
152 {
153  // initialize manager from a CDB snapshot, that is add the entries
154  // to the entries map and the ids to the ids list taking them from
155  // the map and the list found in the input file
156 
157  // if the manager is locked it cannot initialize from a snapshot
158  if (mLock) {
159  LOG(ERROR) << "Being locked I cannot initialize from the snapshot!" << FairLogger::endl;
160  return kFALSE;
161  }
162 
163  // open the file
164  TString snapshotFile(snapshotFileName);
165  if (snapshotFile.BeginsWith("alien://")) {
166  if (!gGrid) {
167  TGrid::Connect("alien://", "");
168  if (!gGrid) {
169  LOG(ERROR) << "Connection to alien failed!" << FairLogger::endl;
170  return kFALSE;
171  }
172  }
173  }
174 
175  TFile *f = TFile::Open(snapshotFileName);
176  if (!f || f->IsZombie()) {
177  LOG(ERROR) << "Cannot open file " << snapshotFileName << FairLogger::endl;
178  return kFALSE;
179  }
180 
181  // retrieve entries' map from snapshot file
182  TMap *entriesMap = nullptr;
183  TIter next(f->GetListOfKeys());
184  TKey *key;
185  while ((key = (TKey *) next())) {
186  if (strcmp(key->GetClassName(), "TMap") != 0) {
187  continue;
188  }
189  entriesMap = (TMap *) key->ReadObj();
190  break;
191  }
192  if (!entriesMap || entriesMap->GetEntries() == 0) {
193  LOG(ERROR) << "Cannot get valid map of CDB entries from snapshot file" << FairLogger::endl;
194  return kFALSE;
195  }
196 
197  // retrieve ids' list from snapshot file
198  TList *idsList = nullptr;
199  TIter nextKey(f->GetListOfKeys());
200  TKey *keyN;
201  while ((keyN = (TKey *) nextKey())) {
202  if (strcmp(keyN->GetClassName(), "TList") != 0) {
203  continue;
204  }
205  idsList = (TList *) keyN->ReadObj();
206  break;
207  }
208  if (!idsList || idsList->GetEntries() == 0) {
209  LOG(ERROR) << "Cannot get valid list of CDB entries from snapshot file" << FairLogger::endl;
210  return kFALSE;
211  }
212 
213  // Add each (entry,id) from the snapshot to the memory: entry to the cache, id to the list of ids.
214  // If "overwrite" is false: add the entry to the cache and its id to the list of ids
215  // only if neither of them is already there.
216  // If "overwrite" is true: write the snapshot entry,id in any case. If something
217  // was already there for that calibration type, remove it and issue a warning
218  TIter iterObj(entriesMap->GetTable());
219  TPair *pair = nullptr;
220  Int_t nAdded = 0;
221  while ((pair = dynamic_cast<TPair *>(iterObj.Next()))) {
222  TObjString *os = (TObjString *) pair->Key();
223  TString path = os->GetString();
224  TIter iterId(idsList);
225  ConditionId *id = nullptr;
226  ConditionId *correspondingId = nullptr;
227  while ((id = dynamic_cast<ConditionId *>(iterId.Next()))) {
228  TString idpath(id->getPathString());
229  if (idpath == path) {
230  correspondingId = id;
231  break;
232  }
233  }
234  if (!correspondingId) {
235  LOG(ERROR) << R"(id for ")" << path.Data()
236  << R"(" not found in the snapshot (while entry was). This entry is skipped!)" << FairLogger::endl;
237  break;
238  }
239  Bool_t cached = mConditionCache.Contains(path.Data());
240  Bool_t registeredId = kFALSE;
241  TIter iter(mIds);
242  ConditionId *idT = nullptr;
243  while ((idT = dynamic_cast<ConditionId *>(iter.Next()))) {
244  if (idT->getPathString() == path) {
245  registeredId = kTRUE;
246  break;
247  }
248  }
249 
250  if (overwrite) {
251  if (cached || registeredId) {
252  LOG(WARNING) << R"(An entry was already cached for ")" << path.Data()
253  << R"(". Removing it before caching from snapshot)" << FairLogger::endl;
254  unloadFromCache(path.Data());
255  }
256  mConditionCache.Add(pair->Key(), pair->Value());
257  mIds->Add(id);
258  nAdded++;
259  } else {
260  if (cached || registeredId) {
261  LOG(WARNING) << R"(An entry was already cached for ")" << path.Data()
262  << R"(". Not adding this object from snapshot)" << FairLogger::endl;
263  } else {
264  mConditionCache.Add(pair->Key(), pair->Value());
265  mIds->Add(id);
266  nAdded++;
267  }
268  }
269  }
270 
271  // mCondition is the new owner of the cache
272  mConditionCache.SetOwnerKeyValue(kTRUE, kTRUE);
273  entriesMap->SetOwnerKeyValue(kFALSE, kFALSE);
274  mIds->SetOwner(kTRUE);
275  idsList->SetOwner(kFALSE);
276  LOG(INFO) << nAdded << " new (entry,id) cached. Total number " << mConditionCache.GetEntries() << FairLogger::endl;
277 
278  f->Close();
279  delete f;
280 
281  return kTRUE;
282 }

409 void Manager::alienToCvmfsUri(TString &uriString) const
410 {
411  // convert alien storage uri to local:///cvmfs storage uri (called when OCDB_PATH is set)
412 
413  TObjArray *arr = uriString.Tokenize('?');
414  TIter iter(arr);
415  TObjString *str = nullptr;
416  TString entryKey = "";
417  TString entryValue = "";
418  TString newUriString = "";
419  while ((str = (TObjString *) iter.Next())) {
420  TString entry(str->String());
421  Int_t indeq = entry.Index('=');
422  entryKey = entry(0, indeq + 1);
423  entryValue = entry(indeq + 1, entry.Length() - indeq);
424 
425  if (entryKey.Contains("folder", TString::kIgnoreCase)) {
426 
427  TRegexp re_RawFolder("^/alice/data/20[0-9]+/OCDB");
428  TRegexp re_MCFolder("^/alice/simulation/2008/v4-15-Release");
429  TString rawFolder = entryValue(re_RawFolder);
430  TString mcFolder = entryValue(re_MCFolder);
431  if (!rawFolder.IsNull()) {
432  entryValue.Replace(0, 6, "/cvmfs/alice-ocdb.cern.ch/calibration");
433  // entryValue.Replace(entryValue.Length()-4, entryValue.Length(), "");
434  } else if (!mcFolder.IsNull()) {
435  entryValue.Replace(0, 36, "/cvmfs/alice-ocdb.cern.ch/calibration/MC");
436  } else {
437  LOG(FATAL) << "Environment variable for cvmfs OCDB folder set for an invalid OCDB storage:\n "
438  << entryValue.Data() << FairLogger::endl;
439  }
440  } else {
441  newUriString += entryKey;
442  }
443  newUriString += entryValue;
444  newUriString += '?';
445  }
446  newUriString.Prepend("local://");
447  newUriString.Remove(TString::kTrailing, '?');
448  uriString = newUriString;
449 }

483

685 void Manager::validateCvmfsCase() const
686 {
687  // The OCDB_PATH variable contains the path to the directory in /cvmfs/ which is
688  // an AliRoot tag based snapshot of the AliEn file catalogue (e.g.
689  // /cvmfs/alice.cern.ch/x86_64-2.6-gnu-4.1.2/Packages/OCDB/v5-05-76-AN).
690  // The directory has to contain:
691  // 1) <data|MC>/20??.list.gz gzipped text files listing the OCDB files (seen by that AliRoot tag)
692  // 2) bin/getOCDBFilesPerRun.sh (shell+awk) script extracting from 1) the list
693  // of valid files for the given run.
