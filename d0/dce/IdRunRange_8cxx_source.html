35 Bool_t IdRunRange::isOverlappingWith(const IdRunRange &other) const
36 {
37  // check if this runRange overlaps other runRange
38 
39  if (!(isValid() && other.isValid())) {
40  LOG(ERROR) << "Comparing invalid run ranges!" << FairLogger::endl;
41  return kFALSE;
42  }
43 
44  if (isAnyRange() || other.isAnyRange()) {
45  LOG(ERROR) << "Comparing unspecified ranges!" << FairLogger::endl;
46  return kFALSE;
47  }
48 
49  return ((mFirstRun < other.mFirstRun && other.mFirstRun <= mLastRun) ||
50  (other.mFirstRun <= mFirstRun && mFirstRun <= other.mLastRun));
51 }
