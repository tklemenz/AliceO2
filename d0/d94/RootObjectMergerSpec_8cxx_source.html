47 DataProcessorSpec getRootObjectMergerSpec() {
48  // set up the processing function
49  // creating the shared pointer of worker instance directly in variable capture
50  // this does not take any options into account but sets up a fixed object
51  // the shared pointer makes sure to clean up the instance when the processing
52  // function gets out of scope
53  auto processingFct = [merger = std::make_shared<Merger>(10)] (ProcessingContext &pc) {
54  using DataHeader = o2::header::DataHeader;
55  for (auto & input : pc.inputs()) {
56  auto dh = o2::header::get<const DataHeader>(input.header);
57  std::cout << dh->dataOrigin.str
58  << " " << dh->dataDescription.str
59  << " " << dh->payloadSize
60  << std::endl;
61  auto obj = framework::DataRefUtils::as<TObject>(input);
62  // FIXME: mergeObject should probably use either a shared_ptr
63  // or a unique_ptr to indicate ownership.
64  auto merged = merger->mergeObject(obj.release());
65  if (!merged) continue;
66  std::cout << "Merger: got merged object " << merged->GetTitle() << std::endl;
67  merged->Print();
68  }
69  };
70 
71  return {
72  "qc_merger",
73  {InputSpec{"qc_producer", "QC", "ROOTOBJECT", 0, InputSpec::QA}},
74  Outputs{},
75  AlgorithmSpec(processingFct)
76  };
