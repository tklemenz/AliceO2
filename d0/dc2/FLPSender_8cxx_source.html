66 void FLPSender::Run()
67 {
68  // base buffer, to be copied from for every timeframe body (zero-copy)
69  FairMQMessagePtr baseMsg(NewMessage(mEventSize));
70 
71  // store the channel reference to avoid traversing the map on every loop iteration
72  FairMQChannel& dataInChannel = fChannels.at(mInChannelName).at(0);
73 
74  while (CheckCurrentState(RUNNING)) {
75  // initialize f2e header
76  auto* header = new f2eHeader;
77  if (mTestMode > 0) {
78  // test-mode: receive and store id part in the buffer.
79  FairMQMessagePtr id(NewMessage());
80  if (dataInChannel.Receive(id) > 0) {
81  header->timeFrameId = *(static_cast<uint16_t*>(id->GetData()));
82  header->flpIndex = mIndex;
83  } else {
84  // if nothing was received, try again
85  delete header;
86  continue;
87  }
88  } else {
89  // regular mode: use the id generated locally
90  header->timeFrameId = mTimeFrameId;
91  header->flpIndex = mIndex;
92 
93  if (++mTimeFrameId == UINT16_MAX - 1) {
94  mTimeFrameId = 0;
95  }
96  }
97 
98  FairMQParts parts;
99 
100  parts.AddPart(NewMessage(header, sizeof(f2eHeader), [](void* data, void* hint){ delete static_cast<f2eHeader*>(hint); }, header));
101  parts.AddPart(NewMessage());
102 
103  // save the arrival time of the message.
104  mArrivalTime.push(steady_clock::now());
105 
106  if (mTestMode > 0) {
107  // test-mode: initialize and store data part in the buffer.
108  parts.At(1)->Copy(*baseMsg);
109  mSTFBuffer.push(move(parts));
110  } else {
111  // regular mode: receive data part from input
112  if (dataInChannel.Receive(parts.At(1)) >= 0) {
113  mSTFBuffer.push(move(parts));
114  } else {
115  // if nothing was received, try again
116  continue;
117  }
118  }
119 
120  // if offset is 0 - send data out without staggering.
121  if (mSendOffset == 0 && mSTFBuffer.size() > 0) {
122  sendFrontData();
123  } else if (mSTFBuffer.size() > 0) {
124  if (duration_cast<milliseconds>(steady_clock::now() - mArrivalTime.front()).count() >= (mSendDelay * mSendOffset)) {
125  sendFrontData();
126  } else {
127  // LOG(INFO) << "buffering...";
128  }
129  }
130  }
131 }

133 inline void FLPSender::sendFrontData()
134 {
135  f2eHeader header = *(static_cast<f2eHeader*>(mSTFBuffer.front().At(0)->GetData()));
136  uint16_t currentTimeframeId = header.timeFrameId;
137 
138  // for which EPN is the message?
139  int direction = currentTimeframeId % mNumEPNs;
140  // LOG(INFO) << "Sending event " << currentTimeframeId << " to EPN#" << direction << "...";
141 
142  if (Send(mSTFBuffer.front(), mOutChannelName, direction, 0) < 0) {
143  LOG(ERROR) << "Failed to queue sub-timeframe #" << currentTimeframeId << " to EPN[" << direction << "]";
144  }
145  mSTFBuffer.pop();
146  mArrivalTime.pop();
147 }
