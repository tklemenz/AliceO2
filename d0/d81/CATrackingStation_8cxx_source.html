108 void TrackingStation::Init(TClonesArray* points, o2::ITS::GeometryTGeo* geo) {
109  if (mNZBins < 1) mNZBins = 2;
110  if (mNPhiBins < 1) mNPhiBins = 1;
111  mDZInv = mNZBins / (mZMax - mZMin);
112  mDPhiInv = mNPhiBins / k2PI;
113  //
114  mBins = new ClBinInfo_t[mNZBins * mNPhiBins];
115  mOccBins = new int[mNZBins * mNPhiBins];
116  mNClusters = points->GetEntriesFast();
117  if(mNClusters == 0) return;
118  mSortedClInfo.reserve(mNClusters);
119  mVIDOffset = ((Hit*)points->UncheckedAt(0))->GetDetectorID();
120  // prepare detectors info
121  int detID = -1;
122  mIndex.resize(geo->getNumberOfChipsPerLayer(mID),-1);
123  mDetectors.reserve(geo->getNumberOfChipsPerLayer(mID));
124  // prepare cluster info
125  ClearSortedInfo();
126  mSortedClInfo.reserve(mNClusters);
127  ClsInfo_t cl;
128  for (int iCl = 0; iCl < points->GetEntriesFast(); ++iCl) { //Fill this layer with detectors
129  Hit* c = (Hit*)points->UncheckedAt(iCl);
130  if (detID == c->GetDetectorID()) {
131  continue;
132  }
133  detID = c->GetDetectorID();
134  ITSDetInfo_t det;
135  det.index = iCl;
136  mIndex[detID - mVIDOffset] = mDetectors.size();
137 
138  //
139  /*
140  RS: using origL2G * T2L matrix will add bias, sine origL2G is for chip, while T2L is for misaligned sensor
141  TGeoHMatrix m;
142  geo->getOriginalMatrix(detID,m);
143  //
144  const TGeoHMatrix *tm = &geo->getMatrixT2L(detID);
145  m.Multiply(tm);
146  double txyz[3] = {0.,0.,0.}, xyz[3] = {0.,0.,0.};
147  m.LocalToMaster(txyz,xyz);
148  det.xTF = sqrt(xyz[0] * xyz[0] + xyz[1] * xyz[1]);
149  */
150  Point3D<float> loc(0.f,0.f,0.f);
151  auto glo = geo->getMatrixSensor(detID)(loc);
152 
153  det.xTF = det.xTF = glo.Rho();
154  det.phiTF = glo.Phi();
155  det.sinTF = sinf(det.phiTF);
156  det.cosTF = cosf(det.phiTF);
157  //
158  // RS: did not understand this routine - just 1st HIT of the chip is selected?
159  // to discuss with Max
160  /*
161  // compute the real radius (with misalignment)
162  TGeoHMatrix mmisal(*(geo->getMatrix(detID)));
163  mmisal.Multiply(tm);
164  xyz[0] = 0.;
165  xyz[1] = 0.;
166  xyz[2] = 0.;
167  mmisal.LocalToMaster(txyz,xyz);
168  det.xTFmisal = sqrt(xyz[0] * xyz[0] + xyz[1] * xyz[1]);
169  mDetectors.push_back(det);
170  */
171  //
172  c->GetStartPosition(cl.x,cl.y,cl.z);
173  cl.r = sqrt(cl.x*cl.x + cl.y*cl.y);
174  cl.phi = atan2(cl.y,cl.x);
175  BringTo02Pi(cl.phi);
176  cl.zphibin = GetBinIndex(GetZBin(cl.z),GetPhiBin(cl.phi));
177  cl.detid = detID - mVIDOffset;
178  //
179  mSortedClInfo.push_back(cl);
180  //
181  } // end loop on detectors
182 }

184 void TrackingStation::SortClusters(const float vtx[3]) {
185  // sort clusters and build fast lookup table
186  //
187  //
188  if (vtx) {
189  for (int icl = mNClusters;icl--;) {
190  mSortedClInfo[icl].x -= vtx[0];
191  mSortedClInfo[icl].y -= vtx[1];
192  }
193  }
194  sort(mSortedClInfo.begin(), mSortedClInfo.end()); // sort in phi, z
195  //
196  // fill cells in phi,z
197  int currBin = -1;
198  for (int icl = 0;icl < mNClusters; ++icl) {
199  ClsInfo_t &t = mSortedClInfo[icl];
200  if (t.zphibin > currBin) { // register new occupied bin
201  currBin = t.zphibin;
202  mBins[currBin].first = icl;
203  mBins[currBin].index = mNOccBins;
204  mOccBins[mNOccBins++] = currBin;
205  }
206  mBins[currBin].ncl++;
207  }
208 }

256 int TrackingStation::SelectClusters(float zmin,float zmax,float phimin,float phimax) {
257  // prepare occupied bins in the requested region
258  if (!mNOccBins) return 0;
259  if (zmax < mZMin || zmin > mZMax || zmin > zmax) return 0;
260  mFoundBins.clear();
261  mQueryZBmin = GetZBin(zmin);
262  if (mQueryZBmin < 0) mQueryZBmin = 0;
263  mQueryZBmax = GetZBin(zmax);
264  if (mQueryZBmax >= mNZBins) mQueryZBmax = mNZBins - 1;
265  BringTo02Pi(phimin);
266  BringTo02Pi(phimax);
267  mQueryPhiBmin = GetPhiBin(phimin);
268  mQueryPhiBmax = GetPhiBin(phimax);
269  int dbz = 0;
270  mNFoundClusters = 0;
271  int nbcheck = mQueryPhiBmax - mQueryPhiBmin + 1; //TODO:(MP) check if a circular buffer is feasible
272  if (nbcheck > 0) { // no wrapping around 0-2pi, fast case
273  for (int ip = mQueryPhiBmin;ip <= mQueryPhiBmax;ip++) {
274  int binID = GetBinIndex(mQueryZBmin,ip);
275  if ( !(dbz = (mQueryZBmax-mQueryZBmin)) ) { // just one Z bin in the query range
276  ClBinInfo_t& binInfo = mBins[binID];
277  if (!binInfo.ncl) continue;
278  mNFoundClusters += binInfo.ncl;
279  mFoundBins.push_back(binID);
280  continue;
281  }
282  int binMax = binID + dbz;
283  for ( ; binID <= binMax; binID++) {
284  ClBinInfo_t& binInfo = mBins[binID];
285  if (!binInfo.ncl) continue;
286  mNFoundClusters += binInfo.ncl;
287  mFoundBins.push_back(binID);
288  }
289  }
290  } else { // wrapping
291  nbcheck += mNPhiBins;
292  for (int ip0 = 0;ip0 <= nbcheck;ip0++) {
293  int ip = mQueryPhiBmin + ip0;
294  if (ip >= mNPhiBins) ip -= mNPhiBins;
295  int binID = GetBinIndex(mQueryZBmin,ip);
296  if ( !(dbz = (mQueryZBmax - mQueryZBmin)) ) { // just one Z bin in the query range
297  ClBinInfo_t& binInfo = mBins[binID];
298  if (!binInfo.ncl) continue;
299  mNFoundClusters += binInfo.ncl;
300  mFoundBins.push_back(binID);
301  continue;
302  }
303  int binMax = binID + dbz;
304  for (;binID <= binMax;binID++) {
305  ClBinInfo_t& binInfo = mBins[binID];
306  if (!binInfo.ncl) continue;
307  mNFoundClusters += binInfo.ncl;
308  mFoundBins.push_back(binID);
309  }
310  }
311  }
312  mFoundClusterIterator = mFoundBinIterator = 0;
313  return mNFoundClusters;
314 }

316 int TrackingStation::GetNextClusterInfoID() {
317  if (mFoundBinIterator < 0) return 0;
318  int currBin = mFoundBins[mFoundBinIterator];
319  if (mFoundClusterIterator < mBins[currBin].ncl) { // same bin
320  return mBins[currBin].first + mFoundClusterIterator++;
321  }
322  if (++mFoundBinIterator < int(mFoundBins.size())) { // need to change bin
323  currBin = mFoundBins[mFoundBinIterator];
324  mFoundClusterIterator = 1;
325  return mBins[currBin].first;
326  }
327  mFoundBinIterator = -1;
328  return -1;
329 }
