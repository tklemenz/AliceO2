103 
104  LOG(INFO) << "FileSink::Init >>>>> SHOULD CREATE THE FILE AND TREE";
105 
106  mFileOption = "RECREATE";

108

113  mFolder = new TFolder("cbmout", "Main Output Folder");

115  TFolder* foldMFT = mFolder->AddFolder("MFT","MFT");
116 
117  auto* BranchNameList = new TList();
118 
119  for ( mNObjects = 0 ; mNObjects < mBranchNames.size() ; mNObjects++ ) {
120

122 
123  if (mClassNames[mNObjects].find("TClonesArray(") == 0) {

127  LOG(INFO) << "FileSink::Init >>>>> Create a TClonesArray of this class: " << mClassNames[mNObjects].c_str() << "";

132 
133  mTree->Branch(mBranchNames[mNObjects].c_str(),"TClonesArray", &mOutputObjects[mNObjects]);

139  LOG(INFO) << "FileSink::Init >>>>> Create the branch EventHeader" << "";
140 
141  mOutputObjects [mNObjects] = new EventHeader();
142  mTree->Branch(mBranchNames[mNObjects].c_str(),"o2::MFT::EventHeader", &mOutputObjects[mNObjects]);
143  foldEventHeader->Add(mOutputObjects[mNObjects]);
144  BranchNameList->AddLast(new TObjString(mBranchNames[mNObjects].c_str()));
145 
146  } else {
147 
148  LOG(ERROR) << R"(!!! Unknown output object ")" << mClassNames[mNObjects] << R"(" !!!)";
149 
150  }
151  }
152 
153  mFolder->Write();
154  BranchNameList->Write("BranchList", TObject::kSingleKey);
155  BranchNameList->Delete();

158  OnData(mInputChannelName, &FileSink::storeData);
