56 Bool_t GeoCave::read(fstream& fin,FairGeoMedia* media)
57 {
58  // Reads the geometry from file
59  if (!media) { return kFALSE; }
60  const Int_t maxbuf=256;
61  char buf[maxbuf];
62  FairGeoNode* volu=nullptr;
63  FairGeoMedium* medium;
64  Bool_t rc=kTRUE;
65  do {
66  fin.getline(buf,maxbuf);
67  if (buf[0]!='\0' && buf[0]!='/' && !fin.eof()) {
68  if (strcmp(buf,name)==0) {
69  volu=new FairGeoNode;
70  volu->SetName(buf);
71  volu->setVolumeType(kFairGeoTopNode);
72  volu->setActive();
73  fin.getline(buf,maxbuf);
74  TString shape(buf);
75  FairGeoBasicShape* sh=pShapes->selectShape(shape);
76  if (sh) { volu->setShape(sh); }
77  else { rc=kFALSE; }
78  fin.getline(buf,maxbuf);
79  medium=media->getMedium(buf);
80  if (!medium) {
81  medium=new FairGeoMedium();
82  media->addMedium(medium);
83  }
84  volu->setMedium(medium);
85  Int_t n=0;
86  if (sh) { n=sh->readPoints(&fin,volu); }
87  if (n<=0) { rc=kFALSE; }
88  } else { rc=kFALSE; }
89  }
90  } while (rc && !volu && !fin.eof());
91  if (volu && rc) {
92  volumes->Add(volu);
93  masterNodes->Add(new FairGeoNode(*volu));
94  } else {
95  delete volu;
96  volu=nullptr;
97  rc=kFALSE;
98  }
99  return rc;
100 }

109 void GeoCave::write(fstream& fout)
110 {

112  fout.setf(ios::fixed,ios::floatfield);
113  FairGeoNode* volu=getVolume(name);
114  if (volu) {
