133 void Detector::initFieldTrackingParams(int& integration, float& maxfield)
134 {
135  auto vmc = TVirtualMC::GetMC();
136  auto field = vmc->GetMagField();
137  // set reasonable default values
138  integration = 2;
139  maxfield = 10;
140  // see if we can query the o2 field
141  if (auto o2field = dynamic_cast<o2::field::MagneticField*>(field)) {
142  integration = o2field->Integral(); // default integration method?
143  maxfield = o2field->Max();
144  }
145  else {
146  LOG(INFO) << "No magnetic field found; using default tracking values " << integration << " " << maxfield << " to initialize media\n";
147  }
148 }
