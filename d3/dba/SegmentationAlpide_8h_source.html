95 inline void SegmentationAlpide::localToDetectorUnchecked(float xRow, float zCol, int& iRow, int& iCol)
96 {
97  // convert to row/col w/o over/underflow check
98  xRow += 0.5 * (ActiveMatrixSizeRows+PassiveEdgeTop-PassiveEdgeReadOut); // coordinate wrt left edge of Active matrix
99  zCol += 0.5 * ActiveMatrixSizeCols; // coordinate wrt bottom edge of Active matrix
100  iRow = int(xRow / PitchRow);
101  iCol = int(zCol / PitchCol);
102  if (xRow<0) iRow -= 1;
103  if (zCol<0) iCol -= 1;
104 }

107 inline bool SegmentationAlpide::localToDetector(float xRow, float zCol, int& iRow, int& iCol)
108 {
109  // convert to row/col
110  xRow += 0.5 * (ActiveMatrixSizeRows+PassiveEdgeTop-PassiveEdgeReadOut); // coordinate wrt left edge of Active matrix
111  zCol += 0.5 * ActiveMatrixSizeCols; // coordinate wrt bottom edge of Active matrix
112  if (xRow<0 || xRow>=ActiveMatrixSizeRows || zCol<0 || zCol>=ActiveMatrixSizeCols) {
113  iRow = iCol = -1;
114  return false;
115  }
116  iRow = int(xRow / PitchRow);
117  iCol = int(zCol / PitchCol);
118  return true;
119 }
