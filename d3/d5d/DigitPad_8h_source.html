103 void DigitPad::setDigit(size_t trackID, float charge)
104 {
105  static FairRootManager *mgr = FairRootManager::Instance();
106  bool isKnown = false;
107  MCCompLabel tempLabel(trackID, mgr->GetEntryNr());
108  for(auto &mcLabel : mMClabel) {
109  if(compareMClabels(tempLabel, mcLabel.first)) {
110  ++mcLabel.second;
111  isKnown=true;
112  }
113  }
114  if(!isKnown) mMClabel.emplace_back(tempLabel, 1);
115  mChargePad += charge;
116 }
