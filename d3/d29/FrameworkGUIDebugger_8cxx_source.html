97 void displayHistory(const DeviceInfo &info, DeviceControl &control) {
98  if (info.history.empty()) {
99  return;
100  }
101  int startPos = info.historyPos;
102  const int historySize = info.history.size();
103 
104  int triggerStartPos = startPos + 1 % historySize;
105  int triggerStopPos = startPos % historySize;
106 
107  int j = startPos;
108  // We look for a stop trigger, so that we know where to stop the search for
109  // out start search. If no stop trigger is found, we search until the end
110  if (control.logStopTrigger[0]) {
111  while((j % historySize) != ((startPos + 1) % historySize)) {
112  assert(j >= 0);
113  assert(j < historySize);
114  auto &line = info.history[j];
115  if (strstr(line.c_str(), control.logStopTrigger)) {
116  triggerStopPos = (j + 1) % historySize;
117  break;
118  }
119  // Wrap in case we end up below 0
120  j = (j == 0) ? historySize - 1 : j - 1;
121  }
122  }
123 
124  // Look for the last instance of the start trigger before the
125  // last stop trigger.
126  j = startPos + 1;
127  if (control.logStartTrigger[0]) {
128  while((j % historySize) != triggerStopPos) {
129  assert(historySize > j);
130  assert(historySize == 1000);
131  auto &line = info.history[j];
132  if (strstr(line.c_str(), control.logStartTrigger)) {
133  triggerStartPos = j;
134  }
135  j = (j + 1) % historySize;
136  }
137  }
138 
139  // We start from the last trigger found. Eventually this is the first
140  // line in the ring buffer, if no trigger is specified.
141  size_t ji = triggerStartPos % historySize;
142  size_t je = triggerStopPos % historySize;
143  size_t iterations = 0;
144  while (historySize
145  && ((ji % historySize) != je)) {
146  assert(iterations < historySize);
147  iterations++;
148  assert(historySize == 1000);
149  assert(ji < historySize);
150  auto &line = info.history[ji];
151  auto logLevel = info.historyLevel[ji];
152 
153  // Skip empty lines
154  if (line.empty()) {
155  ji = (ji + 1) % historySize;
156  continue;
157  }
158  // Print matching lines
159  if (strstr(line.c_str(), control.logFilter) != 0) {
160  auto color = colorForLogLevel(logLevel);
161  // We filter twice, once on input, to reduce the
162  // stream, a second time at display time, to avoid
163  // showing unrelevant messages from past.
164  if (logLevel >= control.logLevel) {
165  ImGui::TextColored(color, line.c_str(), line.c_str() + line.size());
166  }
167  }
168  ji = (ji + 1) % historySize;
169  }
170 }

181 void
182 historyBar(DeviceGUIState &state,
183  const DeviceSpec &spec,
184  const DeviceMetricsInfo &metricsInfo) {
185  bool open = ImGui::TreeNode(state.label.c_str());
186  if (open) {
187  ImGui::Text("# channels: %lu", spec.outputChannels.size() + spec.inputChannels.size());
188  ImGui::TreePop();
189  }
190  ImGui::NextColumn();
191 
192  if (gState.selectedMetric == -1) {
193  ImGui::NextColumn();
194  return;
195  }
196 
197  auto currentMetricName = gState.availableMetrics[gState.selectedMetric];
198 
199  size_t i = metricIdxByName(currentMetricName, metricsInfo);
200  // We did not find any plot, skipping this.
201  if (i == metricsInfo.metricLabelsIdx.size()) {
202  ImGui::NextColumn();
203  return;
204  }
205  auto &metric = metricsInfo.metrics[i];
206 
207  switch(metric.type) {
208  case MetricType::Int:
209  {
210  HistoData<int> data;
211  data.mod = metricsInfo.timestamps[i].size();
212  data.first = metric.pos - data.mod;
213  data.size = metricsInfo.intMetrics[metric.storeIdx].size();
214  data.points = metricsInfo.intMetrics[metric.storeIdx].data();
215 
216  auto getter = [](void *hData, int idx) -> float {
217  auto histoData = reinterpret_cast<HistoData<int> *>(hData);
218  size_t pos = (histoData->first + static_cast<size_t>(idx)) % histoData->mod;
219  assert(pos >= 0 && pos < 1024);
220  return histoData->points[pos];
221  };
222  ImGui::PlotLines(currentMetricName.c_str(),
223  getter,
224  &data,
225  data.size);
226  ImGui::NextColumn();
227  }
228  break;
229  case MetricType::Float:
230  {
231  HistoData<float> data;
232  data.mod = metricsInfo.timestamps[i].size();
233  data.first = metric.pos - data.mod;
234  data.size = metricsInfo.floatMetrics[metric.storeIdx].size();
235  data.points = metricsInfo.floatMetrics[metric.storeIdx].data();
236 
237  auto getter = [](void *hData, int idx) -> float {
238  auto histoData = reinterpret_cast<HistoData<float> *>(hData);
239  size_t pos = (histoData->first + static_cast<size_t>(idx)) % histoData->mod;
240  assert(pos >= 0 && pos < 1024);
241  return histoData->points[pos];
242  };
243  ImGui::PlotLines(currentMetricName.c_str(),
244  getter,
245  &data,
246  data.size);
247  ImGui::NextColumn();
248  }
249  break;
250  default:
251  ImGui::NextColumn();
252  return;
253  break;
254  }
255 }

257 void
258 displayDeviceHistograms(const std::vector<DeviceInfo> &infos,
259  const std::vector<DeviceSpec> &devices,
260  std::vector<DeviceControl> &controls,
261  const std::vector<DeviceMetricsInfo> &metricsInfos) {
262  bool graphNodes = true;
263  showTopologyNodeGraph(&graphNodes, infos, devices);
264  ImGui::SetNextWindowPos(ImVec2(0, ImGui::GetIO().DisplaySize.y - 300), 0);
265  ImGui::SetNextWindowSize(ImVec2(ImGui::GetIO().DisplaySize.x, 300), 0);
266 
267  // Calculate the unique set of metrics, as available in the metrics service
268  std::set<std::string> allMetricsNames;
269  for (const auto &metricsInfo : metricsInfos) {
270  for (const auto &labelsPairs : metricsInfo.metricLabelsIdx) {
271  allMetricsNames.insert(labelsPairs.first);
272  }
273  }
274  using NamesIndex = std::vector<std::string>;
275  gState.availableMetrics.clear();
276  std::copy(allMetricsNames.begin(),
277  allMetricsNames.end(),
278  std::back_inserter(gState.availableMetrics));
279 
280  ImGui::Begin("Devices");
281  ImGui::Combo("Select metric",
282  &gState.selectedMetric,
283  [](void *data, int idx, const char **outText) -> bool {
284  NamesIndex *v = reinterpret_cast<NamesIndex*>(data);
285  if (idx >= v->size()) {
286  return false;
287  }
288  *outText = v->at(idx).c_str();
289  return true;
290  },
291  &gState.availableMetrics,
292  gState.availableMetrics.size()
293  );
294 
295  // Calculate the full timestamp range for the selected metric
296  if (gState.selectedMetric >= 0) {
297  auto currentMetricName = gState.availableMetrics[gState.selectedMetric];
298  size_t minTime = -1;
299  size_t maxTime = 0;
300  for (auto &metricInfo : metricsInfos) {
301  size_t mi = metricIdxByName(currentMetricName, metricInfo);
302  if (mi == metricInfo.metricLabelsIdx.size()) {
303  continue;
304  }
305  auto &metric = metricInfo.metrics[mi];
306  size_t minRangePos = metric.pos % metricInfo.timestamps.size();
307  size_t maxRangePos = (size_t)(metric.pos) - 1 % metricInfo.timestamps.size();
308  auto &timestamps = metricInfo.timestamps[mi];
309  size_t curMinTime = timestamps[minRangePos];
310  size_t curMaxTime = timestamps[maxRangePos];
311  minTime = minTime < curMinTime ? minTime : curMinTime;
312  maxTime = maxTime > curMaxTime ? maxTime : curMaxTime;
313  }
314  ImGui::Text("min timestamp: %zu, max timestamp: %zu",
315  minTime, maxTime);
316  }
317  ImGui::BeginChild("ScrollingRegion", ImVec2(0,-ImGui::GetItemsLineHeightWithSpacing()), false, ImGuiWindowFlags_HorizontalScrollbar);
318  ImGui::Columns(2);
319  ImGui::SetColumnOffset(1, 300);
320  for (size_t i = 0; i < gState.devices.size(); ++i) {
321  DeviceGUIState &guiState = gState.devices[i];
322  const DeviceSpec &spec = devices[i];
323  const DeviceMetricsInfo &metricsInfo = metricsInfos[i];
324 
325  historyBar(guiState, spec, metricsInfo);
326  }
327  ImGui::Columns(1);
328  ImGui::EndChild();
329  ImGui::End();
330 }
