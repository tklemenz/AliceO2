96  // get rotation data
97  Double_t ac[Geometry::NCellsC], bc[Geometry::NCellsC], gc[Geometry::NCellsC];
98  for (Int_t i = 0; i < Geometry::NCellsC; i++) {

111  for (Int_t i = 0; i < Geometry::NCellsC; i++) {

194 void Detector::SetOneMCP(TGeoVolume* ins)
195 {
196  Double_t x, y, z;
197  Float_t pinstart[3] = { 2.95, 2.95, 4.34 };
198  Float_t pmcp[3] = { 2.949, 2.949, 2.8 }; // MCP
199  Float_t ptop[3] = { 1.324, 1.324, 1. }; // cherenkov radiator
200  Float_t ptopref[3] = { 1.3241, 1.3241, 1. }; // cherenkov radiator wrapped with reflection
201  Float_t preg[3] = { 1.324, 1.324, 0.005 }; // photcathode
202  Double_t prfv[3] = { 0.0002, 1.323,
203  1. }; // vertical refracting layer bettwen radiators and bettwen radiator and not optical Air
204  Double_t prfh[3] = { 1.323, 0.0002, 1. }; // horizontal refracting layer bettwen radiators a
205  Double_t pal[3] = { 2.648, 2.648, 0.25 }; // 5mm Al top on th eeach radiator
206  // Entry window (glass)
207  TVirtualMC::GetMC()->Gsvolu("0TOP", "BOX", getMediumID(kOpGlass), ptop, 3); // glass radiator
208  TGeoVolume* top = gGeoManager->GetVolume("0TOP");
209  TVirtualMC::GetMC()->Gsvolu("0TRE", "BOX", getMediumID(kAir), ptopref, 3); // air: wrapped radiator
210  TGeoVolume* topref = gGeoManager->GetVolume("0TRE");
211  TVirtualMC::GetMC()->Gsvolu("0REG", "BOX", getMediumID(kOpGlassCathode), preg, 3);
212  TGeoVolume* cat = gGeoManager->GetVolume("0REG");
213  TVirtualMC::GetMC()->Gsvolu("0MCP", "BOX", getMediumID(kGlass), pmcp, 3); // glass
214  TGeoVolume* mcp = gGeoManager->GetVolume("0MCP");
215  TVirtualMC::GetMC()->Gsvolu("0RFV", "BOX", getMediumID(kOpAir), prfv, 3); // optical Air vertical
216  TGeoVolume* rfv = gGeoManager->GetVolume("0RFV");
217  TVirtualMC::GetMC()->Gsvolu("0RFH", "BOX", getMediumID(kOpAir), prfh, 3); // optical Air horizontal
218  TGeoVolume* rfh = gGeoManager->GetVolume("0RFH");
219 
220  TVirtualMC::GetMC()->Gsvolu("0PAL", "BOX", getMediumID(kAl), pal, 3); // 5mmAl top on the radiator
221  TGeoVolume* altop = gGeoManager->GetVolume("0PAL");
222 
223  // wrapped radiator + reflectiong layers
224  Int_t ntops = 0, nrfvs = 0, nrfhs = 0;
225  Float_t xin = 0, yin = 0, xinv = 0, yinv = 0, xinh = 0, yinh = 0;
226  x = y = z = 0;
227  topref->AddNode(top, 1, new TGeoTranslation(0, 0, 0));
228  xinv = -ptop[0] - prfv[0];
229  topref->AddNode(rfv, 1, new TGeoTranslation(xinv, 0, 0));
230  printf(" GEOGEO refv %f , 0,0 \n", xinv);
231  xinv = ptop[0] + prfv[0];
232  topref->AddNode(rfv, 2, new TGeoTranslation(xinv, 0, 0));
233  printf(" GEOGEO refv %f , 0,0 \n", xinv);
234  yinv = -ptop[1] - prfh[1];
235  topref->AddNode(rfh, 1, new TGeoTranslation(0, yinv, 0));
236  printf(" GEOGEO refh , 0, %f, 0 \n", yinv);
237  yinv = ptop[1] + prfh[1];
238  topref->AddNode(rfh, 2, new TGeoTranslation(0, yinv, 0));
239 
240  // container for radiator, cathod
241  for (Int_t ix = 0; ix < 2; ix++) {
242  xin = -pinstart[0] + 0.3 + (ix + 0.5) * 2 * ptopref[0];
243  for (Int_t iy = 0; iy < 2; iy++) {
244  z = -pinstart[2] + 2 * pal[2] + ptopref[2];
245  yin = -pinstart[1] + 0.3 + (iy + 0.5) * 2 * ptopref[1];
246  ntops++;
247  ins->AddNode(topref, ntops, new TGeoTranslation(xin, yin, z));
248  printf(" 0TOP full %i x %f y %f z %f \n", ntops, xin, yin, z);
249  z = -pinstart[2] + 2 * pal[2] + 2 * ptopref[2] + preg[2];
250  ins->AddNode(cat, ntops, new TGeoTranslation(xin, yin, z));
251  // cat->Print();
252  printf(" GEOGEO CATHOD x=%f , y= %f z= %f num %i\n", xin, yin, z, ntops);
253  }
254  }
255  // Al top
256  z = -pinstart[2] + pal[2];
257  ins->AddNode(altop, 1, new TGeoTranslation(0, 0, z));
258 
259  // MCP
260  z = -pinstart[2] + 2 * pal[2] + 2 * ptopref[2] + 2 * preg[2] + pmcp[2];
261  // z=-pinstart[2] + 2*ptopref[2] + preg[2];
262  ins->AddNode(mcp, 1, new TGeoTranslation(0, 0, z));
263 }

266 {
267 
268  TLorentzVector position;
269  static auto* refMC = TVirtualMC::GetMC();
270  if (refMC->IsTrackEntering()) {
271  refMC->TrackPosition(position);
272  float x = position.X();
273  float y = position.Y();
274  float z = position.Z();
275 
276  float time = refMC->TrackTime() * 1.0e12;
277  int trackID = refMC->GetStack()->GetCurrentTrackNumber();
278  int detID = v->getMCid();
279  float etot = refMC->Etot();
280  int iPart = refMC->TrackPid();
281  float enDep = refMC->Edep();
282  if (iPart == 50000050) // If particles is photon then ...
283  {
284  if (RegisterPhotoE(etot)) {
285  AddHit(x, y, z, time, enDep, trackID, detID);
286  }
287  }
288  return kTRUE;
289  }
290  return kFALSE;
291 }

419 Int_t Detector::ReadOptProperties(const std::string filePath)
420 {
421  std::ifstream infile;
422  infile.open(filePath.c_str());
423 
424  // Check if file is opened correctly
425  if (infile.fail() == true) {
426  // AliFatal(Form("Error opening ascii file: %s", filePath.c_str()));
427  return -1;
428  }
429 
430  std::string comment; // dummy, used just to read 4 first lines and move the cursor to the 5th, otherwise unused
431  if (!getline(infile, comment)) { // first comment line
432  // AliFatal(Form("Error opening ascii file (it is probably a folder!): %s", filePath.c_str()));
433  return -2;
434  }
435  getline(infile, comment); // 2nd comment line
436 
437  // Get number of elements required for the array
438  Int_t nLines;
439  infile >> nLines;
440  if (nLines < 0 || nLines > 1e4) {
441  // AliFatal(Form("Input arraySize out of range 0..1e4: %i. Check input file: %s", kNbins, filePath.c_str()));
442  return -4;
443  }
444 
445  getline(infile, comment); // finish 3rd line after the nEntries are read
446  getline(infile, comment); // 4th comment line
447 
448  // read the main body of the file (table of values: energy, absorption length and refractive index)
449  Int_t iLine = 0;
450  std::string sLine;
451  getline(infile, sLine);
452  while (!infile.eof()) {
453  if (iLine >= nLines) {
454  // AliFatal(Form("Line number: %i reaches range of declared arraySize: %i. Check input file: %s", iLine,
455  // kNbins, filePath.c_str()));
456  return -5;
457  }
458  std::stringstream ssLine(sLine);
459  // First column:
460  Double_t energy;
461  ssLine >> energy;
462  energy *= 1e-9; // Convert eV -> GeV immediately
463  mPhotonEnergyD.push_back(energy);
464  // Second column:
