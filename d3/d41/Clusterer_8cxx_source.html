78 void Clusterer::updateChip(int ip)
79 {
80  PixelReader::PixelData* pix = &mChipData.pixels[ip];
81  if (mCol != pix->col) { // switch the buffers
82  Int_t *tmp = mCurr;
83  mCurr = mPrev;
84  mPrev = tmp;
85  if (pix->col > mCol+1) std::fill(mPrev, mPrev+kMaxRow, -1);
86  std::fill(mCurr, mCurr+kMaxRow, -1);
87  mCol = pix->col;
88  }
89 
90  Bool_t attached=false;
91  UShort_t row = pix->row;
92  Int_t neighbours[]{mCurr[row-1], mPrev[row], mPrev[row+1], mPrev[row-1]};
93  for (auto pci : neighbours) {
94  if (pci<0) continue;
95  auto &ci = mPreClusterIndices[pci];
96  if (attached) {
97  auto &newci = mPreClusterIndices[mCurr[row]];
98  if (ci < newci) newci = ci;
99  else ci = newci;
100  } else {
101  auto &firstIndex = mPreClusterHeads[ci];
102  mPixels.emplace_back(firstIndex, pix);
103  firstIndex = mPixels.size() - 1;
104  mCurr[row] = pci;
105  attached = true;
106  }
107  }
108 
109  if (attached) return;
110 
111  //start new precluster
112  mPreClusterHeads.push_back(mPixels.size());
113  mPixels.emplace_back(-1,pix);
114  Int_t lastIndex = mPreClusterIndices.size();
115  mPreClusterIndices.push_back(lastIndex);
116  mCurr[row] = lastIndex;
117 
118 }

121 void Clusterer::finishChip(std::vector<Cluster> &clusters)
122 {
123  constexpr Float_t SigmaX2 = Segmentation::PitchRow*Segmentation::PitchRow / 12.; //FIXME
124  constexpr Float_t SigmaY2 = Segmentation::PitchCol*Segmentation::PitchCol / 12.; //FIXME
125 
126  std::array<Label,Cluster::maxLabels> labels;
127  std::array<const PixelData*,Cluster::kMaxPatternBits*2> pixArr;
128 
129  Int_t noc = clusters.size();
130  for (Int_t i1=0; i1<mPreClusterHeads.size(); ++i1) {
131  const auto ci = mPreClusterIndices[i1];
132  if (ci<0) continue;
133  UShort_t rowMax=0, rowMin=65535;
134  UShort_t colMax=0, colMin=65535;
135  Float_t x=0., z=0.;
136  int nlab = 0, npix = 0;
137  Int_t next = mPreClusterHeads[i1];
138  while (next >= 0) {
139  const auto &dig = mPixels[next];
140  const auto pix = dig.second; // PixelReader.PixelData*
141  x += pix->row;
142  z += pix->col;
143  if (pix->row < rowMin) rowMin = pix->row;
144  if (pix->row > rowMax) rowMax = pix->row;
145  if (pix->col < colMin) colMin = pix->col;
146  if (pix->col > colMax) colMax = pix->col;
147  if (npix<pixArr.size()) pixArr[npix] = pix; // needed for cluster topology
148  fetchMCLabels(pix, labels, nlab);
149  npix++;
150  next = dig.first;
151  }
152  mPreClusterIndices[i1] = -1;
153  for (Int_t i2=i1+1; i2<mPreClusterHeads.size(); ++i2) {
154  if (mPreClusterIndices[i2] != ci) continue;
155  next = mPreClusterHeads[i2];
156  while (next >= 0) {
157  const auto &dig = mPixels[next];
158  const auto pix = dig.second; // PixelReader.PixelData*
159  x += pix->row;
160  z += pix->col;
161  if (pix->row < rowMin) rowMin = pix->row;
162  if (pix->row > rowMax) rowMax = pix->row;
163  if (pix->col < colMin) colMin = pix->col;
164  if (pix->col > colMax) colMax = pix->col;
165  if (npix<pixArr.size()) pixArr[npix] = pix; // needed for cluster topology
166  // add labels
167  fetchMCLabels(pix, labels, nlab);
168  npix++;
169  next = dig.first;
170  }
171  mPreClusterIndices[i2] = -1;
172  }
173 
174  Point3D<float> xyzLoc( Segmentation::getFirstRowCoordinate() + x*Segmentation::PitchRow/npix, 0.f,
175  Segmentation::getFirstColCoordinate() + z*Segmentation::PitchCol/npix );
176  auto xyzTra = mGeometry->getMatrixT2L(mChipData.chipID)^(xyzLoc); // inverse transform from Local to Tracking frame
177 
178  clusters.emplace_back();
179  Cluster &c = clusters[noc];
180  c.SetUniqueID(noc); // Let the cluster remember its position within the cluster array
181  c.setROFrame(mChipData.roFrame);
182  c.setSensorID(mChipData.chipID);
183  c.setPos(xyzTra);
184  c.setErrors(SigmaX2, SigmaY2, 0.f);
185  c.setNxNzN(rowMax-rowMin+1,colMax-colMin+1,npix);
186  if (mClsLabels) {
187  for (int i=nlab;i--;) mClsLabels->addElement(noc,labels[i]);
188  }
189  noc++;
190 
191 #ifdef _ClusterTopology_
192  unsigned short colSpan=(colMax+1-colMin), rowSpan=(rowMax+1-rowMin), colSpanW=colSpan, rowSpanW=rowSpan;
193  if (colSpan*rowSpan>Cluster::kMaxPatternBits) { // need to store partial info
194  // will curtail largest dimension
195  if (colSpan>rowSpan) {
196  if ( (colSpanW=Cluster::kMaxPatternBits/rowSpan)==0 ) {
197  colSpanW = 1;
198  rowSpanW = Cluster::kMaxPatternBits;
199  }
200  }
201  else {
202  if ( (rowSpanW=Cluster::kMaxPatternBits/colSpan)==0 ) {
203  rowSpanW = 1;
204  colSpanW = Cluster::kMaxPatternBits;
205  }
206  }
207  }
208  c.setPatternRowSpan(rowSpanW,rowSpanW<rowSpan);
209  c.setPatternColSpan(colSpanW,colSpanW<colSpan);
210  c.setPatternRowMin(rowMin);
211  c.setPatternColMin(colMin);
212  if (npix>pixArr.size()) npix = pixArr.size();
213  for (int i=0;i<npix;i++) {
214  const auto pix = pixArr[i];
215  unsigned short ir = pix->row - rowMin, ic = pix->col - colMin;
216  if (ir<rowSpanW && ic<colSpanW) c.setPixel(ir,ic);
217  }
218 #endif //_ClusterTopology_
219 
220  }
221 }

224 void Clusterer::fetchMCLabels(const PixelReader::PixelData* pix,
225  std::array<Label,Cluster::maxLabels> &labels,
226  int &nfilled) const
227 {
228  // transfer MC labels to cluster

234  for (;ic--;) { // check if the label is already present
235  if (labels[ic]==lbl) break;
236  }
237  if (ic<0) { // label not found
238  labels[nfilled++] = lbl;
239  if (nfilled>=Cluster::maxLabels) break;
240  }
241  }
