102 Int_t Segmentation::getDetElemLocalID(Int_t half, Int_t disk, Int_t ladder, Int_t sensor) const {
103 
104  Int_t localId = 0;
105 
106  if (half == 1) localId += getHalf(0)->getHalfDisk(disk)->getNChips();
107 
108  for (Int_t iLad = 0; iLad < getHalf(half)->getHalfDisk(disk)->getNLadders(); iLad++) {
109  if (iLad < ladder) {
110  localId += getHalf(half)->getHalfDisk(disk)->getLadder(iLad)->getNSensors();
111  } else {
112  for (Int_t iSens = 0; iSens < getHalf(half)->getHalfDisk(disk)->getLadder(iLad)->getNSensors(); iSens++) {
113  if(iSens == sensor) return localId;
114  localId++;
115  }
116  }
117  }
118 
119  return -1;
