89  value_type operator()() {
90  value_type v;
91  int trials = 0;
92  while ((v = mModel(mGenerator)) < min || v >= max) {
93  if (trials++ > 1000) {
94  // this is a protection, just picked a reasonable threshold for number of trials

190 template <class RealType = double
191  , class _BASE = std::normal_distribution<RealType>

193 class normal_distribution : public _BASE {
194 public:
195  using result_type = typename _BASE::result_type;
196 
197  normal_distribution(result_type _mean,
198  result_type _stddev
199  ) : _BASE(_mean, _stddev), mean(_mean), stddev(_stddev) {}
200 
201  const double sqrt2pi = 2.5066283;
202  const result_type mean;
203  const result_type stddev;
204 
206  // if value_type is an integral type we want to have the probability
207  // that the result value is in the range [v, v+1) whereas the step
208  // can be something else than 1
209  // also the values outside the specified range should be excluded
210  // and the probability for intervals in the range has to be scaled
211  template<typename value_type>
212  double getProbability(value_type v) const {
213  return (exp(-(v-mean)*(v-mean)/(2*stddev*stddev)))/(stddev * sqrt2pi);
214  }
215 };
