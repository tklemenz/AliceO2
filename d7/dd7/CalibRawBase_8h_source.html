154 inline CalibRawBase::ProcessStatus CalibRawBase::processEvent(int eventNumber)
155 {
156  if (mGBTFrameContainers.size()) {

159  else if (mRawReaders.size()) {
160  return processEventRawReader(eventNumber);
161  }
162  else {

168 inline CalibRawBase::ProcessStatus CalibRawBase::processEventGBT()
169 {
170  if (!mGBTFrameContainers.size()) return ProcessStatus::NoReaders;
171  resetEvent();
172 
173  // loop over raw readers, fill digits for 500 time bins and process
174  // digits
175 
176  const int nRowIROC = mMapper.getNumberOfRowsROC(0);
177 
178  ProcessStatus status = ProcessStatus::Ok;
179 
180  std::vector<Digit> digits(80);
181  for (auto& reader_ptr : mGBTFrameContainers) {
182  auto reader = reader_ptr.get();
183  int readTimeBins = 0;
184  for (int i=0; i<mTimeBinsPerCall; ++i) {
185  if (reader->getData(digits)) {
186  for (auto& digi : digits) {
187  CRU cru(digi.getCRU());
188  const int roc = cru.roc();
189  // TODO: OROC case needs subtraction of number of pad rows in IROC
190  const PadRegionInfo& regionInfo = mMapper.getPadRegionInfo(cru.region());
191  const PartitionInfo& partInfo = mMapper.getPartitionInfo(cru.partition());
192 
193  // row is local in region (CRU)
194  const int row = digi.getRow();
195  const int pad = digi.getPad();
196  if (row==255 || pad==255) continue;
197 
198  int rowOffset = 0;
199  switch (mPadSubset) {
200  case PadSubset::ROC: {
201  rowOffset = regionInfo.getGlobalRowOffset();
202  rowOffset -= (cru.rocType()==RocType::OROC)*nRowIROC;
203  break;
204  }
205  case PadSubset::Region: {
206  break;
207  }
208  case PadSubset::Partition: {
209  rowOffset = regionInfo.getGlobalRowOffset();
210  rowOffset -= partInfo.getGlobalRowOffset();
211  break;
212  }
213  }
214 
215  // modify row depending on the calibration type used
216  const int timeBin= i; //digi.getTimeStamp();
217  const float signal = digi.getChargeFloat();
218 
219  updateCRU(cru, row, pad, timeBin, signal );
220  updateROC(roc, row+rowOffset, pad, timeBin, signal );
221  }
222  ++readTimeBins;
223  }
224 
225  digits.clear();
226  }
227 
228  // set status, don't overwrite decision
229  if (status == ProcessStatus::Ok) {
230  if (readTimeBins == 0 ) {
231  return ProcessStatus::NoMoreData;
232  }
233  else if (readTimeBins < mTimeBinsPerCall) {
234  status = ProcessStatus::Truncated;
235  }
236  }
237 
238  // notify that one raw reader processing finalized for this event
239  endReader();
240  }
241 
242  endEvent();
243  ++mNevents;
244  return status;
245 }

248 inline CalibRawBase::ProcessStatus CalibRawBase::processEventRawReader(int eventNumber)
249 {
250  if (!mRawReaders.size()) return ProcessStatus::NoReaders;
251  resetEvent();
252 
253  // loop over raw readers, fill digits for 500 time bins and process
254  // digits
255 
256  const int nRowIROC = mMapper.getNumberOfRowsROC(0);
257 
258  ProcessStatus status = ProcessStatus::Ok;
259 
260  mProcessedTimeBins = 0;
261  int processedReaders = 0;
262  bool hasData = false;
263 
264  uint64_t lastEvent = 0;
265  for (auto& reader_ptr : mRawReaders) {
266  auto reader = reader_ptr.get();
267 
268  lastEvent = std::max(lastEvent, reader->getLastEvent());
269 
270  if (eventNumber>=0) {
271  reader->loadEvent(eventNumber);
272  mPresentEventNumber = eventNumber;
273  }
274  else if (eventNumber==-1) {
275  mPresentEventNumber = reader->loadNextEvent();
276  }
277  else if (eventNumber==-2) {
278  mPresentEventNumber = reader->loadPreviousEvent();
279  }
280 
281  o2::TPC::PadPos padPos;
282  while (std::shared_ptr<std::vector<uint16_t>> data = reader->getNextData(padPos)) {
283  if (!data) continue;
284 
285  mProcessedTimeBins = std::max(mProcessedTimeBins, data->size());
286 
287  CRU cru(reader->getRegion());
288  const int roc = cru.roc();
289  // TODO: OROC case needs subtraction of number of pad rows in IROC
290  const PadRegionInfo& regionInfo = mMapper.getPadRegionInfo(cru.region());
291  const PartitionInfo& partInfo = mMapper.getPartitionInfo(cru.partition());
292 
293  // row is local in region (CRU)
294  const int row = padPos.getRow();
295  const int pad = padPos.getPad();
296  if (row==255 || pad==255) continue;
297 
298  int timeBin=0;
299  for (const auto& signalI : *data) {
300 
301  int rowOffset = 0;
302  switch (mPadSubset) {
303  case PadSubset::ROC: {
304  rowOffset = regionInfo.getGlobalRowOffset();
305  rowOffset -= (cru.rocType()==RocType::OROC)*nRowIROC;
306  break;
307  }
308  case PadSubset::Region: {
309  break;
310  }
311  case PadSubset::Partition: {
312  rowOffset = regionInfo.getGlobalRowOffset();
313  rowOffset -= partInfo.getGlobalRowOffset();
314  break;
315  }
316  }
317 
318  // modify row depending on the calibration type used
319  const float signal = float(signalI);
320  //const FECInfo& fecInfo = mTPCmapper.getFECInfo(PadSecPos(roc, row, pad));
321  //printf("Call update: %d, %d, %d, %d (%d), %.3f -- reg: %02d -- FEC: %02d, Chip: %02d, Chn: %02d\n", roc, row, pad, timeBin, i, signal, cru.region(), fecInfo.getIndex(), fecInfo.getSampaChip(), fecInfo.getSampaChannel());
322  updateCRU(cru, row, pad, timeBin, signal );
323  updateROC(roc, row+rowOffset, pad, timeBin, signal );
324  ++timeBin;
325  hasData=true;
326  }
327  }
328 
329  // notify that one raw reader processing finalized for this event
330  endReader();
331  ++processedReaders;
332  }
333 
334  // set status, don't overwrite decision
335  if (!hasData) {
336  return ProcessStatus::NoMoreData;
337  }
338  else if (processedReaders < mRawReaders.size()) {
339  status = ProcessStatus::Truncated;
340  }
341  else if (mPresentEventNumber == lastEvent) {
342  status = ProcessStatus::LastEvent;
343  }
344 
345  endEvent();
346  ++mNevents;
347  return status;
348 }
