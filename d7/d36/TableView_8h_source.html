89  size_t addRow(RowDescType rowData, byte* seqData, size_t seqSize) {
90  unsigned nFrames = mFrames.size();
91  unsigned currentRow = mRowData.size();
92  ParserType p;
93  p.parse(seqData, seqSize,
94  [](const typename ParserT::HeaderType& h) {return (h);},
95  [](const typename ParserT::TrailerType& t) {return (t);},
96  [](const typename ParserT::TrailerType& t) {
97  return t.dataLength + ParserT::totalOffset;
98  },
99  [this, currentRow](typename ParserT::FrameInfo entry) {
100  // insert the header as column index in ascending order
101  auto position = mColumns.begin();
102  while (position != mColumns.end() && *position < *entry.header) {
103  position++;
104  }
105  if (position == mColumns.end() || *entry.header < *position) {
106  mColumns.emplace(position, *entry.header);
107  }
108 
109  // insert frame descriptor under key composed from header and row
110  auto result = mFrames.emplace(FrameIndex{*entry.header, currentRow},
111  FrameData{entry.payload, entry.length});
112  return result.second;
113  }
114  );
115  auto insertedFrames = mFrames.size() - nFrames;
116  if (insertedFrames > 0) {
117  mRowData.emplace_back(rowData);
118  }
119  return insertedFrames;
120  }
