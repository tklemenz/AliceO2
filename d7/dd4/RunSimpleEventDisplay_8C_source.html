233 void FillMaxHists(Int_t type=0)
234 {
235  //
236  // type: 0 fill side and sector, 1: fill sector only
237  //
238  Float_t kEpsilon = 0.000000000001;
239  CalPad *pad=mEvDisp.getCalPadMax();
240  TH2F *hSide=0x0;
241  TH2F *hROC=0x0;
242  ResetHists(type);
243  const int runNumber = TString(gSystem->Getenv("RUN_NUMBER")).Atoi();
244  //const int eventNumber = mEvDisp.getNumberOfProcessedEvents() - 1;
245  const int eventNumber = mEvDisp.getPresentEventNumber();
246  for (Int_t iROC=0; iROC<72; iROC++){
247  // TODO: remove again at some point
248  if (iROC >0) break;
249  hROC=mHMaxOROC;
250  hSide=mHMaxC;
251  if (iROC<36) hROC = mHMaxIROC;
252  if (iROC%36<18) hSide = mHMaxA;
253  if (iROC%36==mSelectedSector%36) {
254 // TString title=Form("Max Values %cROC %c%02d (%02d)",(iROC<36)?'I':'O',(iROC%36<18)?'A':'C',iROC%18,iROC);
255  TString title=Form("Max Values Run %d Event %d", runNumber, eventNumber);
256  if (hROC) hROC->SetTitle(title.Data());
257  }
258  auto& calRoc = pad->getCalArray(iROC);
259  const int nRows = mMapper.getNumberOfRowsROC(iROC);
260  for (Int_t irow=0; irow<nRows; irow++){
261  const int nPads = mMapper.getNumberOfPadsInRowROC(iROC, irow);
262  for (Int_t ipad=0; ipad<nPads; ipad++){
263  Float_t value = calRoc.getValue(irow,ipad);
264  //printf("iROC: %02d, sel: %02d, row %02d, pad: %02d, value: %.5f\n", iROC, mSelectedSector, irow, ipad, value);
265  if (TMath::Abs(value)>kEpsilon){
266  if (!type&&hSide){
267  const GlobalPosition2D global2D = mMapper.getPadCentre(PadSecPos(Sector(iROC%36), PadPos(irow, ipad+(iROC>=36)*mMapper.getNumberOfRowsROC(iROC))));
268  Int_t binx = 1+TMath::Nint((global2D.X()+250.)*hSide->GetNbinsX()/500.);
269  Int_t biny = 1+TMath::Nint((global2D.Y()+250.)*hSide->GetNbinsY()/500.);
270  hSide->SetBinContent(binx,biny,value);
271  }
272  const int nPads = mMapper.getNumberOfPadsInRowROC(iROC,irow);
273  const Int_t cpad=ipad-nPads/2;
274  if ((iROC%36==mSelectedSector%36)&&hROC) {
275  //printf(" ->>> Fill: iROC: %02d, sel: %02d, row %02d, pad: %02d, value: %.5f\n", iROC, mSelectedSector, irow, ipad, value);
276  hROC->Fill(irow,cpad,value);
277  }
278  }
279  if ((iROC%36==mSelectedSector%36)&&hROC) {
280  hROC->SetUniqueID(iROC);
281  }
282  }
283  }
284  }
285  if (!type) Update("MaxValsA;MaxValsC");
286  Update("MaxValsI;MaxValsO");
287 }

308 Int_t FindROCFromXY(const Float_t x, const Float_t y, const Int_t side)
309 {
310  //
311  //
312  //
313 
314  Float_t r=TMath::Sqrt(x*x+y*y);
315  static const float innerWall = mMapper.getPadCentre(PadPos(0,0)).X()-5.;
316  static const float outerWall = mMapper.getPadCentre(PadPos(151,0)).X()+5.;
317  static const float outerIROC = mMapper.getPadCentre(PadPos(62,0)).X();
318  static const float innerOROC = mMapper.getPadCentre(PadPos(63,0)).X();
319  static const float betweenROC = (outerIROC+innerOROC)/2.;
320  //check radial boundary
321  if (r<innerWall || r>outerWall) return -1;
322  //check for IROC or OROC
323  Int_t type=0;
324  if (r>betweenROC) type=1;
325  Int_t alpha=TMath::Nint(TMath::ATan2(y,x)/TMath::Pi()*180);
326 // printf("%6.3f %6.3f %03d\n",x, y, alpha);
327  if (alpha<0) alpha+=360;
328  Int_t roc=alpha/20+side*18+type*36;
329  return roc;
330 }
