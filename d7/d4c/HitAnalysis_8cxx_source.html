74 InitStatus HitAnalysis::Init()
75 {
76  // Get the FairRootManager
77  FairRootManager *mgr = FairRootManager::Instance();
78  if (!mgr) {
79  LOG(ERROR) << "Could not instantiate FairRootManager. Exiting ..." << FairLogger::endl;
80  return kERROR;
81  }
82 
83  mHits = mgr->InitObjectAs<const std::vector<o2::ITSMFT::Hit>*>("ITSHit");
84  if (!mHits) {
85  LOG(ERROR) << "ITS points not registered in the FairRootManager. Exiting ..." << FairLogger::endl;
86  return kERROR;
87  }
88 
89  // Create geometry, initialize chip array
90  GeometryTGeo* geom = GeometryTGeo::Instance();
91  if ( !geom->isBuilt() ) geom->Build(true);
92  geom->fillMatrixCache( bit2Mask(TransformType::L2G) ); // make sure T2L matrices are loaded
93 
94  mGeometry = geom;
95 
96  if (mProcessChips) {
97  for (int chipid = 0; chipid < mGeometry->getNumberOfChips(); chipid++) {
98  mChips[chipid] = new Chip(nullptr, chipid, &mGeometry->getMatrixSensor(chipid));
99  }
100  LOG(DEBUG) << "Created " << mChips.size() << " chips." << FairLogger::endl;
101 
102  // Test whether indices match:
103  LOG(DEBUG) << "Testing for integrity of chip indices" << FairLogger::endl;
104  for (int i = 0; i < mGeometry->getNumberOfChips(); i++) {
105  if (mChips[i]->GetChipIndex() != i) {
106  LOG(ERROR) << "Chip index mismatch for entry " << i << ", value " << mChips[i]->GetChipIndex() <<
107  FairLogger::endl;
108  }
109  }
110  LOG(DEBUG) << "Test for chip index integrity finished" << FairLogger::endl;
111  }
112 
113  // Create histograms
114  // Ranges to be adjusted
115  Double_t maxLengthX(Segmentation::SensorSizeRows),
116  maxLengthY(Segmentation::SensorThickness),
117  maxLengthZ(Segmentation::SensorSizeCols);
118 
119  mLineSegment = new TH1D("lineSegment", "Length of the line segment within the chip", 500, 0.0, 0.01);
120  mLocalX0 = new TH1D("localX0", "X position in local (chip) coordinates at the start of a hit", 5000, -2 * maxLengthX,
121  2 * maxLengthX);
122  mLocalX1 = new TH1D("localX1", "X position in local (chip) coordinates at the end of a hit", 500, -0.005, 0.005);
123  mLocalY0 = new TH1D("localY0", "Y position in local (chip) coordinates at the start of a hit", 5000, -2 * maxLengthY,
124  2. * maxLengthY);
125  mLocalY1 = new TH1D("localY1", "Y position in local (chip) coordinates at the end of a hit", 500, -0.005, 0.005);
126  mLocalZ0 = new TH1D("localZ0", "Z position in local (chip) coordinates at the start of a hit", 5000, 2 * maxLengthZ,
127  -2 * maxLengthZ);
128  mLocalZ1 = new TH1D("localZ1", "Z position in local (chip) coordinates at the end of a hit", 500, -0.005, 0.005);
129  mHitCounter = new TH1F("hitcounter", "Simple hit counter", 1, 0.5, 1.5);
130 
131  mIsInitialized = kTRUE;
132  return kSUCCESS;
133 }

155 void HitAnalysis::ProcessChips()
156 {
157  // Add test: All chips must be empty
158  Int_t nchipsNotEmpty(0);
159  std::vector<int> nonEmptyChips;
160  for (auto chipiter: mChips) {
161  if (chipiter.second->GetNumberOfHits() > 0) {
162  nonEmptyChips.push_back(chipiter.second->GetChipIndex());
163  nchipsNotEmpty++;
164  }
165  }
166  if (nchipsNotEmpty) {
167  LOG(ERROR) << "Number of non-empty chips larger 0: " << nchipsNotEmpty << FairLogger::endl;
168  for (auto inditer : nonEmptyChips) {
169  LOG(ERROR) << "Chip index " << inditer << FairLogger::endl;
170  }
171  }
172 
173  // Assign hits to chips
174  for(auto& hit : *mHits) {
175  try {
176  mChips[hit.GetDetectorID()]->InsertHit(&hit);
177  } catch (Chip::IndexException &e) {
178  LOG(ERROR) << e.what() << FairLogger::endl;
179  }
180  }
181 
182  // Add test: Total number of hits assigned to chips must be the same as the size of the points array
183  Int_t nHitsAssigned(0);
184  for (auto chipiter : mChips) {
185  nHitsAssigned += chipiter.second->GetNumberOfHits();
186  }
187  if (nHitsAssigned != mHits->size()) {
188  LOG(ERROR) << "Number of points mismatch: Read(" << mHits->size() << "), Assigned(" << nHitsAssigned <<
189  ")" << FairLogger::endl;
190  }
191 
192  // Values for line segment calculation
193  double x0, x1, y0, y1, z0, z1, tof, edep, steplength;
194 
195  // loop over chips, get the line segment
196  for (auto chipiter: mChips) {
197  Chip &mychip = *(chipiter.second);
198  if (!mychip.GetNumberOfHits()) {
199  continue;
200  }
201  //LOG(DEBUG) << "Processing chip with index " << mychip.GetChipIndex() << FairLogger::endl;
202  for (int ihit = 0; ihit < mychip.GetNumberOfHits(); ihit++) {
203  if (mychip.GetHitAt(ihit)->IsEntering()) { continue; }
204  mychip.LineSegmentLocal(ihit, x0, x1, y0, y1, z0, z1, tof, edep);
205  steplength = TMath::Sqrt(x1 * x1 + y1 * y1 + z1 * z1);
206  mLineSegment->Fill(steplength);
207  mLocalX0->Fill(x0);
208  mLocalX1->Fill(x1);
209  mLocalY0->Fill(y0);
210  mLocalY1->Fill(y1);
211  mLocalZ0->Fill(z0);
212  mLocalZ1->Fill(z1);
213  }
214  }
215  // Clear all chips
216  for (std::map<int, Chip *>::iterator chipiter = mChips.begin(); chipiter != mChips.end(); ++chipiter) {
217  chipiter->second->Clear();
218  }
219 }

221 void HitAnalysis::ProcessHits()
222 {
223  for (auto& hit : *mHits) {
224  auto loc = mGeometry->getMatrixL2G(hit.GetDetectorID())^(hit.GetPos()); // global->local end position
225  auto locS = mGeometry->getMatrixL2G(hit.GetDetectorID())^(hit.GetPosStart()); // global->local start position
226  auto glo = mGeometry->getMatrixL2G(hit.GetDetectorID())(loc);
227  auto gloS = mGeometry->getMatrixL2G(hit.GetDetectorID())(locS);
228  mLocalX0->Fill(locS.X());
229  mLocalY0->Fill(locS.Y());
230  mLocalZ0->Fill(locS.Z());
231  loc.SetXYZ(loc.X()-locS.X(),loc.Y()-locS.Y(),loc.Z()-locS.Z());
232  mLocalX1->Fill(loc.X());
233  mLocalY1->Fill(loc.Y());
234  mLocalZ1->Fill(loc.Z());
235  }
236 }
