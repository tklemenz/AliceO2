31 void fakePayload(std::vector<byte> &buffer, std::function<void(T&,int)> filler, int numOfElements) {
32  auto payloadSize = sizeof(T)*numOfElements;
33  LOG(INFO) << "Payload size " << payloadSize << "\n";
34  buffer.resize(buffer.size() + payloadSize);
35 
36  T *payload = reinterpret_cast<T*>(buffer.data() + sizeof(HeartbeatHeader));
37  for (int i = 0; i < numOfElements; ++i) {
38  new (payload + i) T();
39  // put some random toy time stamp to each cluster
40  filler(payload[i], i);
41  }
42 }

62 void DataPublisherDevice::InitTask()
63 {
64  mInputChannelName = GetConfig()->GetValue<std::string>(OptionKeyInputChannelName);
65  mOutputChannelName = GetConfig()->GetValue<std::string>(OptionKeyOutputChannelName);
66  // TODO: this turned out to be a missing feature in the description fields of
67  // the different headers. The idea is to use a field of n bytes as an integer
68  // but create the integers from a char sequence. This was thought to be done
69  // at compile time, but here we want to have it configurable at runtime.
70  //
71  // GetConfig()->GetValue<std::string>(OptionKeyDataDescription).c_str()
72  // GetConfig()->GetValue<std::string>(OptionKeyDataOrigin).c_str()
73  //
74  // Still needed:
75  // * a registry for descriptors (probably for different descriptor types of
76  // even the same size) to ensure uniqueness
77  // * create the unsigned integer value once from the configurable string and
78  // check in the registry
79  // * constructors and assignment operators taking the integer type as argument
80  if (GetConfig()->GetValue<std::string>(OptionKeyDataDescription) == "TPCCLUSTER")
81  {
82  mDataDescription = DataDescription("CLUSTERS");
83  mDataOrigin = DataOrigin("TPC");
84  }
85  else if (GetConfig()->GetValue<std::string>(OptionKeyDataDescription) == "ITSRAW")
86  {
87  mDataDescription = DataDescription("CLUSTERS");
88  mDataOrigin = DataOrigin("ITS");
89  }
90  mSubSpecification = GetConfig()->GetValue<SubSpecificationT>(OptionKeySubspecification);
91  mFileName = GetConfig()->GetValue<std::string>(OptionKeyFileName);
92 
93  OnData(mInputChannelName.c_str(), &DataPublisherDevice::HandleData);
94 
95  // reserve space for the HBH at the beginning
96  mFileBuffer.resize(sizeof(o2::header::HeartbeatHeader));
97 
98  if (!mFileName.empty()) {
99  AppendFile(mFileName.c_str(), mFileBuffer);
100  }
101  else if (mDataDescription == DataDescription("CLUSTERS")
102  && mDataOrigin == DataOrigin("TPC"))
103  {
104  auto f = [](TPCTestCluster &cluster, int idx) {cluster.timeStamp = idx;};
105  fakePayload<TPCTestCluster>(mFileBuffer, f, 1000);
106  LOG(INFO) << "Payload size (after) " << mFileBuffer.size() << "\n";
107  // For the moment, add the data as another part to this message
108  }
109  else if (mDataDescription == DataDescription("CLUSTERS")
110  && mDataOrigin == DataOrigin("ITS"))
111  {
112  auto f = [](ITSRawData &cluster, int idx) {cluster.timeStamp = idx;};
113  fakePayload<ITSRawData>(mFileBuffer, f, 500);
114  }
115 
116  mFileBuffer.resize(mFileBuffer.size() + sizeof(o2::header::HeartbeatTrailer));
117  auto* hbhOut = reinterpret_cast<o2::header::HeartbeatHeader*>(&mFileBuffer[0]);
118  auto* hbtOut = reinterpret_cast<o2::header::HeartbeatTrailer*>(&mFileBuffer[mFileBuffer.size() - sizeof(o2::header::HeartbeatTrailer)]);
119  *hbhOut = o2::header::HeartbeatHeader();
120  *hbtOut = o2::header::HeartbeatTrailer();
121 }

130 bool DataPublisherDevice::HandleO2LogicalBlock(const byte* headerBuffer,
131  size_t headerBufferSize,
132  const byte* dataBuffer,
133  size_t dataBufferSize)
134 {
135  // AliceO2::header::hexDump("data buffer", dataBuffer, dataBufferSize);
136  const auto* dataHeader = o2::header::get<o2::header::DataHeader>(headerBuffer);
137  const auto* hbfEnvelope = o2::header::get<o2::header::HeartbeatFrameEnvelope>(headerBuffer);
138 
139  // TODO: not sure what the return value is supposed to indicate, it's
140  // not handled in O2Device::ForEach at the moment
141  // indicate that the block has not been processed by a 'false'
142  if (!dataHeader ||
143  (dataHeader->dataDescription) != o2::header::gDataDescriptionHeartbeatFrame) return false;
144 
145  if (!hbfEnvelope) {
146  LOG(ERROR) << "no heartbeat frame envelope header found";
147  return false;
148  }
149 
150  // TODO: consistency checks
151  // hbfEnvelope->header;
152  // hbfEnvelope->trailer;
153  // - block type in both HBH and HBT
154  // - HBH size + payload size (specified in HBT) + HBT size == dataBufferSize
155  // - dynamically adjust start of the trailer (if this contains more than one
156  // 64 bit word
157 
158  // TODO: make tool for reading and manipulation of the HeartbeatFrame/Envelop
159 
160 
161  // assume everything valid
162  // write the HBH and HBT as envelop to the buffer of the file data
163  auto* hbhOut = reinterpret_cast<o2::header::HeartbeatHeader*>(&mFileBuffer[0]);
164  auto* hbtOut = reinterpret_cast<o2::header::HeartbeatTrailer*>(&mFileBuffer[mFileBuffer.size() - sizeof(o2::header::HeartbeatTrailer)]);
165 
166  // copy HBH and HBT, but set the length explicitely to 1
167  // TODO: handle all kinds of corner cases, or add an assert
168  *hbhOut = hbfEnvelope->header;
169  hbhOut->headerLength = 1;
170  *hbtOut = hbfEnvelope->trailer;
171  hbtOut->dataLength = mFileBuffer.size() - sizeof(o2::header::HeartbeatFrameEnvelope);
172 
173  // top level subframe header, the DataHeader is going to be used with
174  // configured description, origin and sub specification
175  o2::header::DataHeader dh;
176  dh.dataDescription = mDataDescription;
177  dh.dataOrigin = mDataOrigin;
178  dh.subSpecification = mSubSpecification;
179  dh.payloadSize = mFileBuffer.size();
180 
181  O2Message outgoing;
182 
183  LOG(DEBUG) << "Sending buffer of size " << mFileBuffer.size() << "\n";
184  LOG(DEBUG) << "Orbit number " << hbhOut->orbit << "\n";
185  // build multipart message from header and payload
186  // TODO: obviously there is a lot to do here, avoid copying etc, this
187  // is just a proof of principle
188  // NewSimpleMessage(mFileBuffer) does not work with the vector
189 
190 
191  // TODO: fix payload size in dh
192  auto *buffer = new char[mFileBuffer.size()];
193  memcpy(buffer, mFileBuffer.data(), mFileBuffer.size());
194  AddMessage(outgoing, dh, NewMessage(buffer, mFileBuffer.size(),
195  [](void* data, void* hint) { delete[] reinterpret_cast<char *>(data); }, nullptr));
196 
197  // send message
198  Send(outgoing, mOutputChannelName.c_str());
199  outgoing.fParts.clear();
200 
201  return true;
202 }

204 bool DataPublisherDevice::AppendFile(const char* name, std::vector<byte>& buffer)
205 {
206  bool result = true;
207  std::ifstream ifile(name, std::ifstream::binary);
208  if (ifile.bad()) return false;
209 
210  // get length of file:
211  ifile.seekg (0, ifile.end);
212  int length = ifile.tellg();
213  ifile.seekg (0, ifile.beg);
214 
215  // allocate memory:
216  int position = buffer.size();
217  buffer.resize(buffer.size() + length);
218 
219  // read data as a block:
220  ifile.read(reinterpret_cast<char*>(&buffer[position]),length);
221  if (!(result = ifile.good())) {
222  LOG(ERROR) << "failed to read " << length << " byte(s) from file " << name << std::endl;
223  }
224 
225  ifile.close();
226 
227  return result;
228 }
