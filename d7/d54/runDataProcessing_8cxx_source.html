104 // This is the handler for the parent inner loop.
105 // So far the only responsibility for it are:
106 //
107 // - Echo children output in a sensible manner
108 //
109 //
110 // - TODO: allow single child view?

113 void doParent(fd_set *in_fdset,
114  int maxFd,
115  std::vector<DeviceInfo> infos,
116  std::vector<DeviceSpec> specs,
117  std::vector<DeviceControl> controls,
118  std::vector<DeviceMetricsInfo> metricsInfos,
119  std::map<int,size_t> &socket2Info,
120  bool batch) {
121  void *window = nullptr;
122  decltype(getGUIDebugger(infos, specs, metricsInfos, controls)) debugGUICallback;
123 
124  if (batch == false) {
125  window = initGUI("O2 Framework debug GUI");
126  debugGUICallback = getGUIDebugger(infos, specs, metricsInfos, controls);
127  }
128  if (batch == false && window == nullptr) {
129  LOG(WARN) << "Could not create GUI. Switching to batch mode. Do you have GLFW on your system?";
130  batch = true;
131  }
132  // FIXME: I should really have some way of exiting the
133  // parent..
134  while (batch || pollGUI(window, debugGUICallback)) {
135  // Exit this loop if all the children say they want to quit.
136  bool allReadyToQuit = true;
137  for (auto &info : infos) {
138  allReadyToQuit &= info.readyToQuit;
139  }
140  if (allReadyToQuit) {
141  break;
142  }
143 
144  // Wait for children to say something. When they do
145  // print it.
146  fd_set *fdset = (fd_set *)malloc(sizeof(fd_set));
147  timeval timeout;
148  timeout.tv_sec = 0;
149  timeout.tv_usec = 16666; // This should be enough to allow 60 HZ redrawing.
150  memcpy(fdset, in_fdset, sizeof(fd_set));
151  int numFd = select(maxFd, fdset, nullptr, nullptr, &timeout);
152  if (numFd == 0) {
153  continue;
154  }
155  for (int si = 0; si < maxFd; ++si) {
156  if (FD_ISSET(si, fdset)) {
157  assert(socket2Info.find(si) != socket2Info.end());
158  auto &info = infos[socket2Info[si]];
159 
160  bool fdActive = getChildData(si, info);
161  // If the pipe was closed due to the process exiting, we
162  // can avoid the select.
163  if (!fdActive) {
164  info.active = false;
165  close(si);
166  FD_CLR(si, in_fdset);
167  }
168  --numFd;
169  }
170  // FIXME: no need to check after numFd gets to 0.
171  }
172  // Display part. All you need to display should actually be in
173  // `infos`.
174  // TODO: split at \n
175  // TODO: update this only once per 1/60 of a second or
176  // things like this.
177  // TODO: have multiple display modes
178  // TODO: graphical view of the processing?
179  assert(infos.size() == controls.size());
180  std::smatch match;
181  std::string token;
182  const std::string delimiter("\n");
183  for (size_t di = 0, de = infos.size(); di < de; ++di) {
184  DeviceInfo &info = infos[di];
185  DeviceControl &control = controls[di];
186  DeviceMetricsInfo &metrics = metricsInfos[di];
187 
188  if (info.unprinted.empty()) {
189  continue;
190  }
191 
192  auto s = info.unprinted;
193  size_t pos = 0;
194  info.history.resize(info.historySize);
195  info.historyLevel.resize(info.historySize);
196 
197  while ((pos = s.find(delimiter)) != std::string::npos) {
198  token = s.substr(0, pos);
199  auto logLevel = LogParsingHelpers::parseTokenLevel(token);
200 
201  // Check if the token is a metric from SimpleMetricsService
202  // if yes, we do not print it out and simply store it to be displayed
203  // in the GUI.
204  // Then we check if it is part of our Poor man control system
205  // if yes, we execute the associated command.
206  if (parseMetric(token, match)) {
207  LOG(INFO) << "Found metric with key " << match[2]
208  << " and value " << match[4];
209  processMetric(match, metrics);
210  } else if (parseControl(token, match)) {
211  auto command = match[1];
212  auto validFor = match[2];
213  LOG(INFO) << "Found control command " << command
214  << " from pid " << info.pid
215  << " valid for " << validFor;
216  if (command == "QUIT") {
217  if (validFor == "ALL") {
218  for (auto &deviceInfo : infos) {
219  deviceInfo.readyToQuit = true;
220  }
221  }
222  }
223  } else if (!control.quiet
224  && (strstr(token.c_str(), control.logFilter) != nullptr)
225  && logLevel >= control.logLevel ) {
226  assert(info.historyPos >= 0);
227  assert(info.historyPos < info.history.size());
228  info.history[info.historyPos] = token;
229  info.historyLevel[info.historyPos] = logLevel;
230  info.historyPos = (info.historyPos + 1) % info.history.size();
231  std::cout << "[" << info.pid << "]: " << token << std::endl;
232  }
233  // We keep track of the maximum log error a
234  // device has seen.
235  if (logLevel > info.maxLogLevel
236  && logLevel > LogParsingHelpers::LogLevel::Info) {
237  info.maxLogLevel = logLevel;
238  }
239  s.erase(0, pos + delimiter.length());
240  }
241  info.unprinted = s;
242  }
243  // FIXME: for the gui to work correctly I would actually need to
244  // run the loop more often and update whenever enough time has
245  // passed.
246  }

261  populateBoostProgramOptions(optsDesc, spec.options, gHiddenDeviceOptions);
262  r.fConfig.AddToCmdLineOptions(optsDesc, true);

303 int createPipes(int maxFd, int *pipes) {
304  auto p = pipe(pipes);
305  maxFd = maxFd > pipes[0] ? maxFd : pipes[0];
306  maxFd = maxFd > pipes[1] ? maxFd : pipes[1];
307 
308  if (p == -1) {
309  std::cerr << "Unable to create PIPE: ";
310  switch(errno) {
311  case EFAULT:
312  assert(false && "EFAULT while reading from pipe");
313  break;
314  case EMFILE:
315  std::cerr << "Too many active descriptors";
316  break;
317  case ENFILE:
318  std::cerr << "System file table is full";
319  break;
320  default:
321  std::cerr << "Unknown PIPE" << std::endl;
322  };
323  // Kill immediately both the parent and all the children
324  kill(-1*getpid(), SIGKILL);
325  }
326  return maxFd;
327 }

329 // Kill all the active children. Exit code
330 // is != 0 if any of the children had an error.
331 int killChildren(std::vector<DeviceInfo> &infos) {
332  int exitCode = 0;
333  for (auto &info : infos) {
334  if (exitCode == 0
335  && info.maxLogLevel >= LogParsingHelpers::LogLevel::Error){
336  LOG(ERROR) << "Child " << info.pid << " had at least one "
337  << "message above severity ERROR";
338  exitCode = 1;
339  }
340  if (!info.active) {
341  continue;
342  }
343  kill(info.pid, SIGKILL);
344  int status;
345  waitpid(info.pid, &status, 0);
346  }
347  return exitCode;
348 }
