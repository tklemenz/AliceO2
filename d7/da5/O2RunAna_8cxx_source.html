186  nextfile=dynamic_cast<TFile*>(fileList->At(k));
187  if (nextfile) {
188  nextfile->Get("FAIRGeom");
189  }
190  if (gGeoManager) {
191  break;
192  }
193  }
194  gFile=currentfile;
195  }
196  } else { // if(fInputFile )
197  // NO input file but there is a geometry file
198  if (mLoadGeo) {
199  if (mInputGeoFile!=nullptr) { //First check if the user has a separate Geo file!
200  TIter next(mInputGeoFile->GetListOfKeys());
201  TKey* key;
202  while ((key = dynamic_cast<TKey*>(next()))) {

206  gGeoManager = dynamic_cast<TGeoManager*>(key->ReadObj());
207  break;
208  }
209  }
210  }
211  }
212  gROOT->GetListOfBrowsables()->Add(fTask);
213 
214  // Init the RTDB containers
215  // auto par=dynamic_cast<FairBaseParSet*>(fRtdb->getContainer("FairBaseParSet"));
216  if (/*par &&*/ mInFileIsOpen) {
217 
218  LOG(INFO) << "Parameter and input file are available, Assure that basic info is there for the run!" << FairLogger::endl;
219  FairSystemInfo sysInfo;
220  LOG(DEBUG) << "MEMORY BEFORE READEVENT(0) " << sysInfo.GetMaxMemory() << FairLogger::endl;

223 
224  // This is *CREATING* an instance of an event header (pointed to by fEvtHeader)
225  // (or returning an existing instance)
226  // happening in the base class
227  GetEventHeader();
228 
229  // we fill the event header
230  fRootManager->FillEventHeader(fEvtHeader);
231 
232  fRunId = fEvtHeader->GetRunId();
233 
234  // Copy the Event Header Info to Output
235  fEvtHeader->Register(mStoreEventHeader);
236 
237  // Init the containers in Tasks
238  fRtdb->initContainers(fRunId);
239 
240  // not sure what this is doing
241  fTask->SetParTask();
242 
243  }
244  else
245  {
246  LOG(INFO) << "Initializing without input file or Mixed input"

248  FairEventHeader* evt = GetEventHeader();

261 
262  fRtdb->initContainers(fRunId);

269  mField = fieldfact->createFairField();

279  // LOG(INFO) << "The following branches were requestes as input by task\n";

287  fTrajFilter->Init();
288  }
289 
290  // create the output tree after tasks initialisation
291  fOutFile->cd();

302 void O2RunAna::Run(Int_t Ev_start, Int_t Ev_end)
303 {
304  gFRAIsInterrupted = kFALSE;
305 
306  if (false /*fTimeStamps*/) {
307  RunTSBuffers();
308  }
309  else
310  {
311  UInt_t tmpId =0;
312  // if (fInputFile==0) {
313  if (!mInFileIsOpen) {
314  DummyRun(Ev_start,Ev_end);
315  return;
316  }
317 
318  Int_t MaxAllowed=fRootManager->CheckMaxEventNo(Ev_end);
319  if ( MaxAllowed != -1 ) {
320  if (Ev_end==0) {
321  if (Ev_start==0) {
322  Ev_end=MaxAllowed;
323  } else {
324  Ev_end = Ev_start;
325  if ( Ev_end > MaxAllowed ) {
326  Ev_end = MaxAllowed;
327  }
328  Ev_start=0;
329  }
330  } else {
331  if (Ev_end > MaxAllowed) {
332  cout << "-------------------Warning---------------------------" << endl;
333  cout << " -W O2RunAna : File has less events than requested!!" << endl;
334  cout << " File contains : " << MaxAllowed << " Events" << endl;
335  cout << " Requested number of events = " << Ev_end << " Events"<< endl;
336  cout << " The number of events is set to " << MaxAllowed << " Events"<< endl;
337  cout << "-----------------------------------------------------" << endl;
338  Ev_end = MaxAllowed;
339  }
340  }
341  LOG(INFO) << "O2RunAna::Run() After checking, the run will run from event " << Ev_start << " to " << Ev_end << "." << FairLogger::endl;
342  }
343  else {
344  LOG(INFO) << "O2RunAna::Run() continue running without stop" << FairLogger::endl;
345  }
346 
347  if (fGenerateRunInfo) {
348  mRunInfo.Reset();
349  }
350 
351  Int_t readEventReturn = 0;
352 
353  for (int i=Ev_start; i< Ev_end || MaxAllowed==-1 ; i++) {
354 
355  gSystem->IgnoreInterrupt();
356  // gFRAIsInterrupted = kFALSE;
357  signal(SIGINT, FRA_handler_ctrlc);
358 
359  if ( gFRAIsInterrupted ) {
360  LOG(WARNING) << "O2RunAna::Run() Event loop was interrupted by the user!" << FairLogger::endl;
361  break;
362  }
363 
364 #ifdef LAZY
365  // FIXME: fix event timings (should be the same as in nonlazy case)
366  for(auto& name : mRequestedBranches) {
367  std::cerr << "READING " << name << "\n";
368  fRootManager->ReadBranchEvent(name.c_str(), i);
369  }
370 #else
371  readEventReturn = fRootManager->ReadEvent(i);
372  if ( readEventReturn != 0 ) {
373  LOG(WARNING) << "O2RunAna::Run() fRootManager->ReadEvent(" << i << ") returned " << readEventReturn << ". Breaking the event loop" << FairLogger::endl;
374  break;
375  }
376 #endif
377  fRootManager->FillEventHeader(fEvtHeader);
378 
379  // fill the event info structure
380  mEventInfo->eventNumber = i;
381  mEventInfo->eventTime = fRootManager->GetEventTime();
382 
383 
384  tmpId = fEvtHeader->GetRunId();
385  if ( tmpId != fRunId ) {
386  fRunId = tmpId;
387  if ( !mStatic ) {
388  Reinit( fRunId );
389  fTask->ReInitTask();
390  }
391  }
392  //std::cout << "WriteoutBufferData with time: " << fRootManager->GetEventTime();
393  fRootManager->StoreWriteoutBufferData(fRootManager->GetEventTime());
394  fTask->ExecuteTask("");
395  Fill();
396  fRootManager->DeleteOldWriteoutBufferData();
397  fTask->FinishEvent();
398 
399  if (fGenerateRunInfo) {
400  mRunInfo.StoreInfo();
401  }
402  if (nullptr != FairTrajFilter::Instance()) {
403  FairTrajFilter::Instance()->Reset();
404  }
405 
406  }
407 
408  fRootManager->StoreAllWriteoutBufferData();
409  fTask->FinishTask();
410  if (fGenerateRunInfo) {
411  mRunInfo.WriteInfo();
412  }
413  fRootManager->LastFill();
414  fRootManager->Write();
415  }
416 }
