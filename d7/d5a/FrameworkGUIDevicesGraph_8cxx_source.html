33 NodeColor
34 decideColorForNode(const DeviceInfo &info) {
35  NodeColor result;
36  if (info.active == false) {
37  result.normal = PaletteHelpers::RED;
38  result.hovered = PaletteHelpers::SHADED_RED;
39  return result;
40  }
41  switch(info.maxLogLevel) {
42  case LogParsingHelpers::LogLevel::Error:
43  result.normal = PaletteHelpers::SHADED_RED;
44  result.hovered = PaletteHelpers::RED;
45  break;
46  case LogLevel::Warning:
47  result.normal = PaletteHelpers::SHADED_YELLOW;
48  result.hovered = PaletteHelpers::YELLOW;
49  break;
50  case LogLevel::Info:
51  result.normal = PaletteHelpers::SHADED_GREEN;
52  result.hovered = PaletteHelpers::GREEN;
53  break;
54  default:
55  result.normal = PaletteHelpers::GRAY;
56  result.hovered = PaletteHelpers::LIGHT_GRAY;
57  break;
58  }
59  return result;
60 }
