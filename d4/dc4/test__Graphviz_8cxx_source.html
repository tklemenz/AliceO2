26 void lineByLineComparision(const std::string &as, const std::string &bs) {
27  std::istringstream a(as);
28  std::istringstream b(bs);
29 
30  char bufferA[1024];
31  char bufferB[1024];
32 
33  while (a.good() && b.good()) {
34  a.getline(bufferA, 1024);
35  b.getline(bufferB, 1024);
36  BOOST_CHECK_EQUAL(std::string(bufferA), std::string(bufferB));
37  }
38  BOOST_CHECK(a.eof());
39  BOOST_CHECK(b.eof());
40 }
