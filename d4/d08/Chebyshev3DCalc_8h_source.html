197 inline Float_t Chebyshev3DCalc::chebyshevEvaluation1D(Float_t x, const Float_t *array, int ncf)
198 {
199  if (ncf <= 0) {
200  return 0;
201  }
202 
203  Float_t b0, b1, b2, x2 = x + x;
204  b0 = array[--ncf];
205  b1 = b2 = 0;
206 
207  for (int i = ncf; i--;) {
208  b2 = b1;
209  b1 = b0;
210  b0 = array[i] + x2 * b1 - b2;
211  }
212  return b0 - x * b1;
213 }

217 inline Float_t Chebyshev3DCalc::Eval(const Float_t *par) const
218 {
219  if (!mNumberOfRows) {
220  return 0.;
221  }
222  int ncfRC;
223  for (int id0 = mNumberOfRows; id0--;) {
224  int nCLoc = mNumberOfColumnsAtRow[id0]; // number of significant coefs on this row
225  int col0 = mColumnAtRowBeginning[id0]; // beginning of local column in the 2D boundary matrix
226  for (int id1 = nCLoc; id1--;) {
227  int id = id1 + col0;
228  mTemporaryCoefficients2D[id1] = (ncfRC = mCoefficientBound2D0[id])
229  ? chebyshevEvaluation1D(par[2], mCoefficients + mCoefficientBound2D1[id], ncfRC)
230  : 0.0;
231  }
232  mTemporaryCoefficients1D[id0] = nCLoc > 0 ? chebyshevEvaluation1D(par[1], mTemporaryCoefficients2D, nCLoc) : 0.0;
233  }
234  return chebyshevEvaluation1D(par[0], mTemporaryCoefficients1D, mNumberOfRows);
235 }
