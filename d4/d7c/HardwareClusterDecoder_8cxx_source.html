36 int HardwareClusterDecoder::decodeClusters(std::vector<std::pair<const ClusterHardwareContainer*, std::size_t>>& inputClusters, std::vector<ClusterNativeContainer>& outputClusters, const std::vector<MCLabelContainer>* inMCLabels, std::vector<MCLabelContainer>* outMCLabels)
37 {
38  if (!inMCLabels) outMCLabels = nullptr;
39  int nRowClusters[Constants::MAXSECTOR][Constants::MAXGLOBALPADROW] = {0};
40  int containerRowCluster[Constants::MAXSECTOR][Constants::MAXGLOBALPADROW] = {0};
41  Mapper& mapper = Mapper::instance();
42  int numberOfOutputContainers = 0;
43  for (int loop = 0;loop < 2;loop++)
44  {
45  for (int i = 0;i < inputClusters.size();i++)
46  {
47  if (outMCLabels && inputClusters[i].second > 1)
48  {
49  LOG(ERROR) << "Decoding of ClusterHardware to ClusterNative with MC labels is yet only support for single 8kb pages of ClusterHardwareContainer\n";
50  return(1);
51  }
52  for (int j = 0;j < inputClusters[i].second;j++)
53  {
54  const char* tmpPtr = reinterpret_cast<const char*> (inputClusters[i].first);
55  tmpPtr += j * 8192; //TODO: FIXME: Compute correct offset based on the size of the actual packet in the RDH
56  const ClusterHardwareContainer& cont = *(reinterpret_cast<const ClusterHardwareContainer*> (tmpPtr));
57  const CRU cru(cont.CRU);
58  const Sector sector = cru.sector();
59  const PadRegionInfo& region = mapper.getPadRegionInfo(cru.region());
60  const int rowOffset = region.getGlobalRowOffset();
61 
62  for (int k = 0;k < cont.numberOfClusters;k++)
63  {
64  const int padRowGlobal = rowOffset + cont.clusters[k].row;
65  int& nCls = nRowClusters[sector][padRowGlobal];
66  if (loop == 1)
67  {
68  //Fill cluster in the respective output buffer
69  const ClusterHardware& cIn = cont.clusters[k];
70  ClusterNative& cOut = outputClusters[containerRowCluster[sector][padRowGlobal]].clusters[nCls];
71  MCLabelContainer& mcOut = (*outMCLabels)[containerRowCluster[sector][padRowGlobal]];
72  float pad = cIn.getPad();
73  cOut.setPad(pad);
74  cOut.setTimeFlags(cIn.getTimeLocal() + cont.timeBinOffset, cIn.flags);
75  cOut.setSigmaPad(std::sqrt(cIn.getSigmaPad2()));
76  cOut.setSigmaTime(std::sqrt(cIn.getSigmaTime2()));
77  cOut.qMax = cIn.qMax;
78  cOut.qTot = cIn.qTot;
79  mIntegrator.integrateCluster(sector, padRowGlobal, pad, cIn.qTot);
80  if (outMCLabels)
81  {
82  for (const auto& element : (*inMCLabels)[i].getLabels(k)) {
83  mcOut.addElement(nCls, element);
84  }
85  }
86  }
87  else
88  {
89  //Count how many output buffers we need (and how large they are below)
90  if (nCls == 0) numberOfOutputContainers++;
91  }
92  nCls++;
93  }
94  }
95  }
96  if (loop == 1)
97  {
98  //We are done with filling the buffers, sort all output buffers
99  for (int i = 0;i < outputClusters.size();i++)
100  {
101  if (outMCLabels) {
102  sortClustersAndMC(outputClusters[i].clusters, (*outMCLabels)[i]);
103  } else {
104  auto& cl = outputClusters[i].clusters;
105  std::sort(cl.data(), cl.data() + cl.size(), ClusterNativeContainer::sortComparison);
106  }
107  }
108  }
109  else
110  {
111  //Now we know the size of all output buffers, allocate them
112  outputClusters.resize(numberOfOutputContainers);
113  if (outMCLabels) outMCLabels->resize(numberOfOutputContainers);
114  numberOfOutputContainers = 0;
115  for (int i = 0;i < Constants::MAXSECTOR;i++)
116  {
117  for (int j = 0;j < Constants::MAXGLOBALPADROW;j++)
118  {
119  if (nRowClusters[i][j] == 0) continue;
120  outputClusters[numberOfOutputContainers].clusters.resize(nRowClusters[i][j]);
121  outputClusters[numberOfOutputContainers].sector = i;
122  outputClusters[numberOfOutputContainers].globalPadRow = j;
123  containerRowCluster[i][j] = numberOfOutputContainers++;
124  mIntegrator.initRow(i, j);
125  }
126  }
127  memset(nRowClusters, 0, sizeof(nRowClusters));
128  }
129  }
130  return(0);
131 }

133 void HardwareClusterDecoder::sortClustersAndMC(std::vector<ClusterNative> clusters, MCLabelContainer mcTruth)
134 {
135  std::vector<unsigned int> indizes(clusters.size());
136  for (int i = 0;i < indizes.size();i++) indizes[i] = i;
137  std::sort(indizes.data(), indizes.data() + indizes.size(), [&clusters](const auto a, const auto b) {
138  return ClusterNativeContainer::sortComparison(clusters[a], clusters[b]);
139  });
140  std::vector<ClusterNative> tmpCl = std::move(clusters);
141  MCLabelContainer tmpMC = std::move(mcTruth);
142  for (int i = 0;i < clusters.size();i++)
143  {
144  clusters[i] = tmpCl[indizes[i]];
145  gsl::span<const MCCompLabel> mcArray = tmpMC.getLabels(indizes[i]);
146  for (int k = 0;k < mcArray.size();k++)
147  {
148  mcTruth.addElement(i, mcArray[k]);
149  }
150  }
151 }
