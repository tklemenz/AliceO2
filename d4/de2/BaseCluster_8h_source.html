26 // Basic cluster class with X,Y,Z position detector ID information + user fields
27 // The position is ALWAYS stored in tracking frame and is misaligned (in opposite
28 // to AliRoot). The errors are defined in *ideal* tracking frame
29 // DetectorID should correspond to continuous (no jumps between detector layers
30 // planes etc.) internal sensor ID within detector
31 // Detector specific clusters should be composed by including it as data member
32 template <typename T>
33 class BaseCluster : public TObject // temprarily derive from TObject
34 {
35  private:
36 
37  Point3D<T> mPos; // cartesian position
38  T mSigmaY2; // error in Y direction (usually rphi)
39  T mSigmaZ2; // error in Z direction (usually Z)
40  T mSigmaYZ; // non-diagonal term of error matrix
41  std::uint16_t mSensorID=0; // the sensor id
42  std::int8_t mCount = 0; // user field reserved for counting
43  std::uint8_t mBits = 0; // user field reserved for bit flags
44  enum masks_t : std::int32_t { kUserBitsMask = 0xff };
