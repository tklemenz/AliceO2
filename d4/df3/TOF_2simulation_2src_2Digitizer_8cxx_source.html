35 void Digitizer::processHit(HitType* hit)
36 {
37  mNumDigit = 0;
38 
39  Float_t pos[3] = { hit->GetX(), hit->GetY(), hit->GetZ() };
40  Float_t deltapos[3];
41  Int_t detInd[5];
42  Int_t detIndOtherPad[5];
43 
44  Geo::getPadDxDyDz(pos, detInd, deltapos); // Get DetId and residuals
45 
46  detIndOtherPad[0] = detInd[0], detIndOtherPad[1] = detInd[1],
47  detIndOtherPad[2] = detInd[2]; // same sector, plate, strip
48 
49  Int_t otherraw = 0;
50  if (detInd[3] == 0)
51  otherraw = 1;
52 
53  Int_t iZshift = otherraw ? 1 : -1;
54 
55  Int_t channel = Geo::getIndex(detInd);
56 
57  Float_t charge = getCharge(hit->GetEnergyLoss());
58  Float_t time = getShowerTimeSmeared(hit->GetTime() * 1E3, charge); // in ps from now!
59 
60  Float_t xLocal = deltapos[0];
61  Float_t zLocal = deltapos[2];
62 
63  // PadId - Pad Identifier
64  // E | F --> PadId = 5 | 6
65  // A | B --> PadId = 1 | 2
66  // C | D --> PadId = 3 | 4
67 
68  // check the fired PAD 1 (A)
69  if (isFired(xLocal, zLocal, charge))
70  addDigit(channel, time, xLocal, zLocal, charge, 0, 0, detInd[3]);
71 
72  // check PAD 2
73  detIndOtherPad[3] = otherraw;
74  detIndOtherPad[4] = detInd[4];
75  channel = Geo::getIndex(detIndOtherPad);
76  xLocal = deltapos[0]; // recompute local coordinates
77  if (otherraw)
78  zLocal = deltapos[2] - Geo::ZPAD; // recompute local coordinates
79  else
80  zLocal = deltapos[2] + Geo::ZPAD;
81  if (isFired(xLocal, zLocal, charge)) {
82  addDigit(channel, time, xLocal, zLocal, charge, 0, iZshift, detInd[3]);
83  }
84 
85  // check PAD 3
86  detIndOtherPad[3] = detInd[3];
87  detIndOtherPad[4] = detInd[4] - 1;
88  if (detIndOtherPad[4] >= 0) {
89  channel = Geo::getIndex(detIndOtherPad);
90  xLocal = deltapos[0] + Geo::XPAD; // recompute local coordinates
91  zLocal = deltapos[2]; // recompute local coordinates
92  if (isFired(xLocal, zLocal, charge)) {
93  addDigit(channel, time, xLocal, zLocal, charge, -1, 0, detInd[3]);
94  }
95  }
96 
97  // check PAD 5
98  detIndOtherPad[3] = detInd[3];
99  detIndOtherPad[4] = detInd[4] + 1;
100  if (detIndOtherPad[4] < Geo::NPADX) {
101  channel = Geo::getIndex(detIndOtherPad);
102  xLocal = deltapos[0] - Geo::XPAD; // recompute local coordinates
103  zLocal = deltapos[2]; // recompute local coordinates
104  if (isFired(xLocal, zLocal, charge)) {
105  addDigit(channel, time, xLocal, zLocal, charge, 1, 0, detInd[3]);
106  }
107  }
108 
109  // check PAD 4
110  detIndOtherPad[3] = otherraw;
111  detIndOtherPad[4] = detInd[4] - 1;
112  if (detIndOtherPad[4] >= 0) {
113  channel = Geo::getIndex(detIndOtherPad);
114  xLocal = deltapos[0] + Geo::XPAD; // recompute local coordinates
115  if (otherraw)
116  zLocal = deltapos[2] - Geo::ZPAD; // recompute local coordinates
117  else
118  zLocal = deltapos[2] + Geo::ZPAD;
119  if (isFired(xLocal, zLocal, charge)) {
120  addDigit(channel, time, xLocal, zLocal, charge, -1, iZshift, detInd[3]);
121  }
122  }
123 
124  // check PAD 6
125  detIndOtherPad[3] = otherraw;
126  detIndOtherPad[4] = detInd[4] + 1;
127  if (detIndOtherPad[4] < Geo::NPADX) {
128  channel = Geo::getIndex(detIndOtherPad);
129  xLocal = deltapos[0] - Geo::XPAD; // recompute local coordinates
130  if (otherraw)
131  zLocal = deltapos[2] - Geo::ZPAD; // recompute local coordinates
132  else
133  zLocal = deltapos[2] + Geo::ZPAD;
134  if (isFired(xLocal, zLocal, charge)) {
135  addDigit(channel, time, xLocal, zLocal, charge, 1, iZshift, detInd[3]);
136  }
137  }
138 }

140 void Digitizer::addDigit(Int_t channel, Float_t time, Float_t x, Float_t z, Float_t charge, Int_t iX, Int_t iZ,
141  Int_t padZfired)
142 {
143  // TOF digit requires: channel, time and time-over-threshold
144 
145  time = getDigitTimeSmeared(time, x, z, charge); // add time smearing
146 
147  charge *= getFractionOfCharge(x, z);
148  Float_t tot = 12; // time-over-threshold
149 
150  Float_t xborder = Geo::XPAD * 0.5 - TMath::Abs(x);
151  Float_t zborder = Geo::ZPAD * 0.5 - TMath::Abs(z);
152  Float_t border = TMath::Min(xborder, zborder);
153 
154  Float_t timewalkX = x * mTimeWalkeSlope;
155  Float_t timewalkZ = (z - (padZfired - 0.5) * Geo::ZPAD) * mTimeWalkeSlope;
156 
157  if (border < 0) { // keep the effect onlu if hit out of pad
158  border *= -1;
159  Float_t extraTimeSmear = border * mTimeSlope;
160  time += gRandom->Gaus(mTimeDelay, extraTimeSmear);
161  } else {
162  border = 1 - border;
163  // if(border > 0) printf("deltat =%f\n",mTimeDelay*border*border*border);
164  // else printf("deltat=0\n");
165  // getchar();
166  if (border > 0)
167  time += mTimeDelay * border * border * border;
168  }
169  time += TMath::Sqrt(timewalkX * timewalkX + timewalkZ * timewalkZ) - mTimeDelayCorr - mTimeWalkeSlope * 2;
170 
171  mTime[mNumDigit] = time;
172  mTot[mNumDigit] = tot;
173  mXshift[mNumDigit] = iX;
174  mZshift[mNumDigit] = iZ;
175  mNumDigit++;
176 }

178 Float_t Digitizer::getShowerTimeSmeared(Float_t time, Float_t charge)
179 {
180  // add the smearing common to all the digits belongin to the same shower
181  return time + gRandom->Gaus(0, mShowerResolution);

184 Float_t Digitizer::getDigitTimeSmeared(Float_t time, Float_t x, Float_t z, Float_t charge)
185 {
186  // add the smearing component which is indepedent for any digit even if belonging to the same shower (in case of
187  // multiple hits)
188  return time + gRandom->Gaus(0, mDigitResolution); // sqrt(33**2 + 50**2) ps = 60 ps

200 Bool_t Digitizer::isFired(Float_t x, Float_t z, Float_t charge)
201 {
202  if (TMath::Abs(x) > Geo::XPAD * 0.5 + 0.3)
203  return kFALSE;
204  if (TMath::Abs(z) > Geo::ZPAD * 0.5 + 0.3)
205  return kFALSE;
206 
207  Float_t effX = getEffX(x);
208  Float_t effZ = getEffZ(z);
209 
210  Float_t efficiency = TMath::Min(effX, effZ);
211 
212  if (gRandom->Rndm() > efficiency)
213  return kFALSE;
214 
215  return kTRUE;
216 }

218 Float_t Digitizer::getEffX(Float_t x)
219 {
220  Float_t xborder = Geo::XPAD * 0.5 - TMath::Abs(x);
221 
222  if (xborder > 0) {
223  if (xborder > mBound1)
224  return mEffCenter;
225  else if (xborder > mBound2)
226  return mEffBoundary1 + (mEffCenter - mEffBoundary1) * (xborder - mBound2) / (mBound1 - mBound2);
227  else
228  return mEffBoundary2 + (mEffBoundary1 - mEffBoundary2) * xborder / mBound2;
229  } else {
230  xborder *= -1;
231  if (xborder > mBound4)
232  return 0;
233  else if (xborder > mBound3)
234  return mEffBoundary3 - mEffBoundary3 * (xborder - mBound3) / (mBound4 - mBound3);
235  else
236  return mEffBoundary2 + (mEffBoundary3 - mEffBoundary2) * xborder / mBound3;
237  }
238 
239  return 0;
240 }

242 Float_t Digitizer::getEffZ(Float_t z)
243 {
244  Float_t zborder = Geo::ZPAD * 0.5 - TMath::Abs(z);
245 
246  if (zborder > 0) {
247  if (zborder > mBound1)
248  return mEffCenter;
249  else if (zborder > mBound2)
250  return mEffBoundary1 + (mEffCenter - mEffBoundary1) * (zborder - mBound2) / (mBound1 - mBound2);
251  else
252  return mEffBoundary2 + (mEffBoundary1 - mEffBoundary2) * zborder / mBound2;
253  } else {
254  zborder *= -1;
255  if (zborder > mBound4)
256  return 0;
257  else if (zborder > mBound3)
258  return mEffBoundary3 - mEffBoundary3 * (zborder - mBound3) / (mBound4 - mBound3);
259  else
260  return mEffBoundary2 + (mEffBoundary3 - mEffBoundary2) * zborder / mBound3;
261  }
262 
263  return 0;

268 void Digitizer::initParameters()
269 {
270  // boundary references for interpolation of efficiency and resolution
271  mBound1 = 0.4; // distance from border when efficiency starts to decrese
272  mBound2 = 0.15; // second step in the fired pad
273  mBound3 = 0.55; // distance from border (not fired pad)
274  mBound4 = 0.9; // distance from border (not fired pad) when efficiency vanishes
275 
276  // resolution parameters
277  mTOFresolution = 60; // TOF global resolution in ps
278  mShowerResolution = 50; // smearing correlated for all digits of the same hit
279  if (mTOFresolution > mShowerResolution)
280  mDigitResolution = TMath::Sqrt(mTOFresolution * mTOFresolution -
281  mShowerResolution * mShowerResolution); // independent smearing for each digit
282  else {
283  mShowerResolution = mTOFresolution;
284  mDigitResolution = 0;
285  }
286 
287  mTimeSlope = 100; // ps/cm extra smearing if hit out of pad propto the distance from the border
288  mTimeDelay = 70; // time delay if hit out of pad
289  mTimeWalkeSlope = 40; // ps/cm
290 
291  if (mMode == 0) {
292  mTimeSlope = 0;
293  mTimeDelay = 0;
294  mTimeWalkeSlope = 0;
295  }
296 
297  mTimeDelayCorr = mTimeDelay / 3.5;
298  if (mShowerResolution > mTimeDelayCorr)
299  mShowerResolution = TMath::Sqrt(mShowerResolution * mShowerResolution - mTimeDelayCorr * mTimeDelayCorr);
300  else
301  mShowerResolution = 0;
302 
303  if (mShowerResolution > mTimeWalkeSlope * 0.8)
304  mShowerResolution = TMath::Sqrt(mShowerResolution * mShowerResolution - mTimeWalkeSlope * mTimeWalkeSlope * 0.64);
305  else
306  mShowerResolution = 0;
307 
308  // efficiency parameters
309  mEffCenter = 0.995; // efficiency in the center of the fired pad
310  mEffBoundary1 = 0.94; // efficiency in mBound2
311  mEffBoundary2 = 0.833; // efficiency in the pad border
312  mEffBoundary3 = 0.1; // efficiency in mBound3
313 }
