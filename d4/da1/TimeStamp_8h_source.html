31  // number of bunches and the 40 MHz clock with 25 ns bunch spacing
32  // gives revolution time of 89.1 us and 11.223345 kHz
33  // this depends on the assumption that the particles are moving effectively
34  // at speed of light. There are also documents specifying the orbit time
35  // to 89.4 us
36  // Note: avoid to define the revolution frequency and use the integral numbers
37  // for bunch places and bunch spacing in nano seconds
38  // TODO: this eventually needs to be configurable
39  const int gNumberOfBunches = 3564;
40  const int gBunchSpacingNanoSec = 25;
41  const int gOrbitTimeNanoSec = std::ratio<gNumberOfBunches*gBunchSpacingNanoSec>::num;

46  // the type of the clock tick depends on whether to use also the bunches
47  // as substructure of the orbit.
48  // a trait class to extrat the properties of the clock, namely the type
49  // of the tick and the period
50  template <typename T>
51  struct Property {
52  // the default does not specify anything and is never going to be used
53  };
54  template <>
55  struct Property<OrbitPrecision> {
56  typedef uint32_t rep;
57  // avoid rounding errors by using the integral numbers in the std::ratio
58  // template to define the period
59  typedef std::ratio_multiply<std::ratio<gOrbitTimeNanoSec>, std::nano> period;
60  };
61  template <>
62  struct Property<BunchPrecision> {
63  typedef uint64_t rep;
64  // this is effectively the LHC clock and the ratio is the
65  // bunch spacing
66  typedef std::ratio_multiply<std::ratio<gBunchSpacingNanoSec>, std::nano> period;
67  };
68 };

70 // a chrono clock implementation
71 // - always relative to run start
72 // - need run start to calculate the epoch

94  time_point now() noexcept {
95  // tp1 - tp2 results in a duration, we use to create a time_point with characteristics
