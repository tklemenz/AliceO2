118 void HwClusterer::processDigits(
119  const std::vector<std::vector<std::vector<std::tuple<Digit const*, int, int>>>>& digits,
120  const std::vector<std::vector<std::vector<std::shared_ptr<HwClusterFinder>>>>& clusterFinder,
121  std::vector<std::vector<Cluster>>& cluster,
122  std::vector<std::vector<std::vector<std::pair<int,int>>>>& label,
123  const CfConfig config)
124 {
125 // std::thread::id this_id = std::this_thread::get_id();
126 // g_display_mutex.lock();
127 // std::cout << "thread " << this_id << " started.\n";
128 // g_display_mutex.unlock();
129 
130  int timeDiff = (config.iMaxTimeBin+1) - config.iMinTimeBin;
131  if (timeDiff < 0) return;
132  const Mapper& mapper = Mapper::instance();
133  std::vector<std::vector<HwClusterFinder::MiniDigit>> iAllBins(timeDiff,std::vector<HwClusterFinder::MiniDigit>(config.iMaxPads));
134 
135  for (unsigned iCRU = config.iCRUMin; iCRU <= config.iCRUMax; ++iCRU) {
136  if (iCRU % config.iThreadMax != config.iThreadID) continue;
137 
138  for (int iRow = 0; iRow < mapper.getNumberOfRowsPartition(iCRU); iRow++){
139 
140  /*
141  * prepare local storage
142  */
143  short t,p;
144  if (config.iEnableNoiseSim && config.iNoiseObject != nullptr) {
145  for (t=timeDiff; t--;) {
146  for (p=config.iMaxPads; p--;) {
147  iAllBins[t][p].charge = config.iNoiseObject->getValue(CRU(iCRU),iRow,p);
148  iAllBins[t][p].index = -1;
149  iAllBins[t][p].event = -1;
150  }
151  }
152  } else {
153  for (auto &bins : iAllBins) std::fill(bins.begin(),bins.end(),HwClusterFinder::MiniDigit());
154 // std::fill(&iAllBins[0][0], &iAllBins[0][0]+timeDiff*config.iMaxPads, HwClusterFinder::MiniDigit());
155  }
156 
157  /*
158  * fill in digits
159  */
160  for (auto& digit : digits[iCRU][iRow]){
161  const Int_t iTime = std::get<0>(digit)->getTimeStamp();
162  const Int_t iPad = std::get<0>(digit)->getPad() + 2; // offset to have 2 empty pads on the "left side"
163  const Float_t charge = std::get<0>(digit)->getChargeFloat();
164 
165  // std::cout << iCRU << " " << iRow << " " << iPad << " " << iTime << " (" << iTime-minTime << "," << timeDiff << ") " << charge << std::endl;
166  iAllBins[iTime-config.iMinTimeBin][iPad].charge += charge;
167  iAllBins[iTime-config.iMinTimeBin][iPad].index = std::get<1>(digit);
168  iAllBins[iTime-config.iMinTimeBin][iPad].event = std::get<2>(digit);
169  if (config.iEnablePedestalSubtraction && config.iPedestalObject != nullptr) {
170  const float pedestal = config.iPedestalObject->getValue(CRU(iCRU),iRow,iPad-2);
171  //printf("digit: %.2f, pedestal: %.2f\n", iAllBins[iTime-config.iMinTimeBin][iPad], pedestal);
172  iAllBins[iTime-config.iMinTimeBin][iPad].charge -= pedestal;
173  }
174  }
175 
176  /*
177  * copy data to cluster finders
178  */
179  const unsigned iPadsPerCF = static_cast<const unsigned>(clusterFinder[iCRU][iRow][0]->getNpads());
180  const unsigned iTimebinsPerCF = static_cast<const unsigned>(clusterFinder[iCRU][iRow][0]->getNtimebins());
181  std::vector<std::vector<std::shared_ptr<HwClusterFinder>>::const_reverse_iterator> cfWithCluster;
182  int time;
183  unsigned pad;
184  for (time = 0; time < timeDiff; ++time){ // ordering important!!
185  for (pad = 0; pad < config.iMaxPads; pad = pad + (iPadsPerCF -2 -2 )) {
186  const Short_t cf = pad / (iPadsPerCF-2-2);
187  clusterFinder[iCRU][iRow][cf]->addTimebin(iAllBins[time].begin()+pad,time+config.iMinTimeBin,(config.iMaxPads-pad)>=iPadsPerCF?iPadsPerCF:(config.iMaxPads-pad));
188  }
189 
190  /*
191  * search for clusters and store reference to CF if one was found
192  */
193  if (clusterFinder[iCRU][iRow][0]->getTimebinsAfterLastProcessing() == iTimebinsPerCF-2 -2) {
194  /*
195  * ordering is important: from right to left, so that the CFs could inform each other if cluster was found
196  */
197  for (auto rit = clusterFinder[iCRU][iRow].crbegin(); rit != clusterFinder[iCRU][iRow].crend(); ++rit) {
198  if ((*rit)->findCluster()) {
199  cfWithCluster.push_back(rit);
200  }
201  }
202  }
203  }
204 
205  /*
206  * add empty timebins to find last clusters
207  */
208  if (!config.iIsContinuousReadout) {
209  // +2 so that for sure all data is processed
210  for (time = 0; time < clusterFinder[iCRU][iRow][0]->getNtimebins()+2; ++time){
211  for (auto rit = clusterFinder[iCRU][iRow].crbegin(); rit != clusterFinder[iCRU][iRow].crend(); ++rit) {
212  (*rit)->addZeroTimebin(time+timeDiff+config.iMinTimeBin,iPadsPerCF);
213  }
214 
215  /*
216  * search for clusters and store reference to CF if one was found
217  */
218  if (clusterFinder[iCRU][iRow][0]->getTimebinsAfterLastProcessing() == iTimebinsPerCF-2 -2) {
219  /*
220  * ordering is important: from right to left, so that the CFs could inform each other if cluster was found
221  */
222  for (auto rit = clusterFinder[iCRU][iRow].crbegin(); rit != clusterFinder[iCRU][iRow].crend(); ++rit) {
223  if ((*rit)->findCluster()) {
224  cfWithCluster.push_back(rit);
225  }
226  }
227  }
228  }
229  for (auto rit = clusterFinder[iCRU][iRow].crbegin(); rit != clusterFinder[iCRU][iRow].crend(); ++rit) {
230  (*rit)->setTimebinsAfterLastProcessing(0);
231  }
232  }
233 
234  /*
235  * collect found cluster
236  */
237  for (auto &cf_rit : cfWithCluster) {
238  auto cc = (*cf_rit)->getClusterContainer();
239  for (auto& c : *cc) cluster[iCRU].push_back(c);
240 
241  auto ll = (*cf_rit)->getClusterDigitIndices();
242  for (auto& l : *ll) {
243  label[iCRU].push_back(l);
244  }
245 
246  (*cf_rit)->clearClusterContainer();
247  }
248 
249  }
250  }
251

253 // std::cout << "thread " << this_id << " finished.\n";
254 // g_display_mutex.unlock();

258 void HwClusterer::Process(std::vector<o2::TPC::Digit> const &digits, MCLabelContainer const* mcDigitTruth, int eventCount)
259 {
260  mClusterArray->clear();
261  if(mClusterMcLabelArray) mClusterMcLabelArray->clear();
262 
263 
264  /*
265  * clear old storages
266  */
267  for (auto& cs : mClusterStorage) cs.clear();
268  for (auto& cdis : mClusterDigitIndexStorage) cdis.clear();
269  for (auto& dc : mDigitContainer ) {
270  for (auto& dcc : dc) dcc.clear();
271  }
272 
273  int iTimeBin;
274  int iTimeBinMin = (mIsContinuousReadout)?mLastTimebin + 1 : 0;
275  //int iTimeBinMin = mLastTimebin + 1;
276  int iTimeBinMax = mLastTimebin;
277 
278  /*
279  * Loop over digits
280  */
281  int digitIndex = 0;
282  for (const auto& digit : digits) {
283 
284  /*
285  * add current digit to storage
286  */
287 
288  iTimeBin = digit.getTimeStamp();
289  if (digit.getCRU() < static_cast<int>(mCRUMin) || digit.getCRU() > static_cast<int>(mCRUMax)) {
290  LOG(DEBUG) << "Digit [" << digitIndex << "] is out of CRU range (" << digit.getCRU() << " < " << mCRUMin << " or > " << mCRUMax << ")" << FairLogger::endl;
291  // Necessary because MCTruthContainer requires continuous indexing
292  ++digitIndex;
293  continue;
294  }
295  if (iTimeBin < iTimeBinMin) {
296  LOG(DEBUG) << "Digit [" << digitIndex << "] time stamp too small (" << iTimeBin << " < " << iTimeBinMin << ")" << FairLogger::endl;
297  // Necessary because MCTruthContainer requires continuous indexing
298  ++digitIndex;
299  continue;
300  }
301 
302  iTimeBinMax = std::max(iTimeBinMax,iTimeBin);
303  if (mcDigitTruth == nullptr)
304  mDigitContainer[digit.getCRU()][digit.getRow()].emplace_back(std::make_tuple(&digit,-1,eventCount));
305  else {
306  mDigitContainer[digit.getCRU()][digit.getRow()].emplace_back(std::make_tuple(&digit,digitIndex,eventCount));
307  }
308  ++digitIndex;
309  }
310 
311  if (mcDigitTruth != nullptr && mClusterMcLabelArray != nullptr )
312  mLastMcDigitTruth[eventCount] = std::make_unique<MCLabelContainer>(*mcDigitTruth);
313 
314  ProcessTimeBins(iTimeBinMin, iTimeBinMax);
315 
316  mLastMcDigitTruth.erase(eventCount-mTimebinsPerCF);
317 
318  LOG(DEBUG) << "Event ranged from time bin " << iTimeBinMin << " to " << iTimeBinMax << "." << FairLogger::endl;
319 }

383 void HwClusterer::ProcessTimeBins(int iTimeBinMin, int iTimeBinMax)
384 {
385 
386  /*
387  * vector to store threads for parallel processing
388  */
389  std::vector<std::thread> thread_vector;
390 
391  LOG(DEBUG) << "Starting " << mNumThreads << " threads, hardware supports " << std::thread::hardware_concurrency() << " parallel threads." << FairLogger::endl;
392 
393  for (unsigned threadId = 0; threadId < std::min(mNumThreads,mCRUMax); ++threadId) {
394  struct CfConfig cfConfig = {
395  threadId,
396  mNumThreads,
397  mCRUMin,
398  mCRUMax,
399  static_cast<unsigned>(mPadsMax)+2+2,
400  iTimeBinMin,
401  iTimeBinMax,
402  mEnableNoiseSim,
403  mEnablePedestalSubtraction,
404  mIsContinuousReadout,
405  mNoiseObject,
406  mPedestalObject
407  };
408  thread_vector.emplace_back(
409  processDigits, // function name
410  std::ref(mDigitContainer), // digit container for individual CRUs
411  std::ref(mClusterFinder), // cluster finder for individual CRUs
412  std::ref(mClusterStorage), // container to store found clusters
413  std::ref(mClusterDigitIndexStorage), // container to store found cluster MC Labels
414  cfConfig // configuration
415  );
416  }
417 
418 
419  /*
420  * wait for threads to join
421  */
422  for (std::thread& t: thread_vector) {
423  t.join();
424  }
425 
426  /*
427  * collect clusters from individual cluster finder
428  */
429 
430  // map to count unique MC labels
431  std::map<MCCompLabel,int> labelCount;
432 
433  // multiset to sort labels according to occurrence
434  auto mcComp = [](const std::pair<MCCompLabel, int>& a, const std::pair<MCCompLabel, int>& b) { return a.second > b.second;};
435  std::multiset<std::pair<MCCompLabel,int>,decltype(mcComp)> labelSort(mcComp);
436 
437  // for each CRU
438  for (unsigned cru = 0; cru < mClusterStorage.size(); ++cru) {
439  std::vector<Cluster>* clustersFromCRU = &mClusterStorage[cru];
440  std::vector<std::vector<std::pair<int,int>>>* labelsFromCRU = &mClusterDigitIndexStorage[cru];
441 
442  // for each found cluster
443  for(unsigned c = 0; c < clustersFromCRU->size(); ++c) {
444  const auto clusterPos = mClusterArray->size();
445  mClusterArray->emplace_back(clustersFromCRU->at(c));
446  if (mClusterMcLabelArray == nullptr) continue;
447  labelCount.clear();
448  labelSort.clear();
449 
450  // for each used digit
451  for (auto &digitIndex : labelsFromCRU->at(c)) {
452  if (digitIndex.first < 0) continue;
453  for (auto &l : mLastMcDigitTruth[digitIndex.second]->getLabels(digitIndex.first)) {
454  labelCount[l]++;
455  }
456  }
457  for (auto &l : labelCount) labelSort.insert(l);
458  for (auto &l : labelSort) mClusterMcLabelArray->addElement(clusterPos,l.first);
459  }
460  }
461 
462  mLastTimebin = iTimeBinMax;
463 }
