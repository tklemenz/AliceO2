53 void ConditionsMQClient::Run()
54 {
55  Backend* backend;
56 
57  while (CheckCurrentState(RUNNING)) {
58 
59  boost::this_thread::sleep(boost::posix_time::milliseconds(1000));
60  boost::posix_time::ptime startTime = boost::posix_time::microsec_clock::local_time();
61 
62  if (mDataSource == "OCDB") {
63  backend = new BackendOCDB();
64  } else if (mDataSource == "Riak") {
65  backend = new BackendRiak();
66  } else {
67  LOG(ERROR) << R"(")" << mDataSource << R"(" is not a valid Data Source)";
68  return;
69  }
70 
71  boost::filesystem::path dataPath(mObjectPath);
72  boost::filesystem::recursive_directory_iterator endIterator;
73 
74  // Traverse the filesystem and retrieve the name of each root found
75  if (boost::filesystem::exists(dataPath) && boost::filesystem::is_directory(dataPath)) {
76  for (static boost::filesystem::recursive_directory_iterator directoryIterator(dataPath);
77  directoryIterator != endIterator; ++directoryIterator) {
78  if (boost::filesystem::is_regular_file(directoryIterator->status())) {
79 
80  // Retrieve the key from the filename by erasing the directory structure and trimming the file extension
81  std::string str = directoryIterator->path().string();
82  str.erase(0, mObjectPath.length());
83  std::size_t pos = str.rfind(".");
84  std::string key = str.substr(0, pos);
85 
86  if (mOperationType == "GET") {
87  std::string* messageString = new string();
88  backend->Serialize(messageString, key, mOperationType, mDataSource);
89 
90  unique_ptr<FairMQMessage> request(fTransportFactory->CreateMessage(
91  const_cast<char*>(messageString->c_str()), messageString->length(), CustomCleanup, messageString));
92  unique_ptr<FairMQMessage> reply(fTransportFactory->CreateMessage());
93 
94  if (fChannels.at("data-get").at(0).Send(request) > 0) {
95  if (fChannels.at("data-get").at(0).Receive(reply) > 0) {
96  LOG(DEBUG) << "Received a condition with a size of " << reply->GetSize();
97  backend->UnPack(std::move(reply));
98  }
99  }
100  } else if (mOperationType == "PUT") {
101  std::string* messageString = new string();
102  backend->Pack(directoryIterator->path().string(), key, messageString);
103 
104  unique_ptr<FairMQMessage> request(fTransportFactory->CreateMessage(
105  const_cast<char*>(messageString->c_str()), messageString->length(), CustomCleanup, messageString));
106 
107  if (fChannels.at("data-put").at(0).Send(request) > 0) {
108  LOG(DEBUG) << "Message sent" << endl;
109  }
110  }
111  }
112  }
113  } else {
114  LOG(ERROR) << "Path " << mObjectPath << " not existing or not a directory";
115  }
116 
117  boost::posix_time::ptime endTime = boost::posix_time::microsec_clock::local_time();
118  LOG(DEBUG) << " Time elapsed: " << (endTime - startTime).total_milliseconds() << "ms";
119  }
120 }
