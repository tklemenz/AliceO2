52 bool Track::GetPhiZat(float r, float bfield,float &phi, float &z) const {
53  float rp4=getCurvature(bfield);
54 
55  float xt=getY(), yt=getY();
56  float x = 0.f, y = 0.f;
57  float sn=sin(getAlpha()), cs=cos(getAlpha());
58  float a = x*cs + y*sn;
59  y = -x*sn + y*cs; x=a;
60  xt-=x; yt-=y;
61 
62  sn=rp4*xt - getSnp(); cs=rp4*yt + sqrt((1.- getSnp())*(1.+getSnp()));
63  a=2*(xt*getSnp() - yt*sqrt((1.-getSnp())*(1.+getSnp())))-rp4*(xt*xt + yt*yt);
64  float d = -a/(1 + sqrt(sn*sn + cs*cs));
65 
66  if (fabs(d) > r) {
67  if (r>1e-1) return false;
68  r = fabs(d);
69  }
70 
71  float rcurr=sqrt(getX()*getX() + getY()*getY());
72  float phicurr=getPhi();
73 
74  if (getX()>=0.) {
75  phi=phicurr+asin(d/r)-asin(d/rcurr);
76  } else {
77  phi=phicurr+asin(d/r)+asin(d/rcurr)-kPI;
78  }
79 
80  //return a phi in [0,2pi
81  if (phi<0.) phi+=2.*kPI;
82  else if (phi>=2.*kPI) phi-=2.*kPI;
83  z=getZ()+getTgl()*(sqrt((r-d)*(r+d))-sqrt((rcurr-d)*(rcurr+d)));
84 
85  return true;
86 }
