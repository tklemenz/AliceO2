141 template <class T>
142 void CalArray<T>::initData()
143 {
144  const auto& mapper = Mapper::instance();
145 
146  switch (mPadSubset) {
147  case PadSubset::ROC: {
148  mData.resize(ROC(mPadSubsetNumber).rocType() == RocType::IROC? mapper.getPadsInIROC() : mapper.getPadsInOROC());
149  if (mName.length()) setName(boost::str(format("ROC_%1$02d") % mPadSubsetNumber));
150  break;
151  }
152  case PadSubset::Partition: {
153  mData.resize(mapper.getPartitionInfo(mPadSubsetNumber % mapper.getNumberOfPartitions()).getNumberOfPads());
154  if (mName.length()) setName(boost::str(format("Partition_%1$03d") % mPadSubsetNumber));
155  break;
156  }
157  case PadSubset::Region: {
158  mData.resize(mapper.getPadRegionInfo(mPadSubsetNumber % mapper.getNumberOfPadRegions()).getNumberOfPads());
159  if (mName.length()) setName(boost::str(format("Region_%1$03d") % mPadSubsetNumber));
160  break;
161  }
162  }
163 }
