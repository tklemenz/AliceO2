255 void GBTFrameContainer::addGBTFrame(unsigned word3, unsigned word2, unsigned word1, unsigned word0) {
256  if (!mEnableStoreGBTFrames && (mGBTFrames.size() > 1)) {
257  mGBTFrames[0] = mGBTFrames[1];
258  mGBTFrames[1].setData(word3, word2, word1, word0);
259  } else {
260  mGBTFrames.emplace_back(word3, word2, word1, word0);
261  }
262  processFrame(mGBTFrames.end()-1);
263 };
