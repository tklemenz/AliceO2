275 //__________________________________________________________________________
276 TGeoHMatrix* GeometryTGeo::extractMatrixSensor(int index) const
277 {
278  // extract matrix transforming from the PHYSICAL sensor frame to global one
279  // Note, the if the effective sensitive layer thickness is smaller than the
280  // total physical sensor tickness, this matrix is biased and connot be used
281  // directly for transformation from sensor frame to global one.
282  //
283  // Therefore we need to add a shift
284 
285  int lay, stav, sstav, mod, chipInMod;
286  getChipId(index, lay, stav, sstav, mod, chipInMod);
287 
288  int wrID = mLayerToWrapper[lay];
289 
290  TString path = Form("/cave_1/%s_2/", GeometryTGeo::getITSVolPattern());
291 
292  if (wrID >= 0) {
293  path += Form("%s%d_1/", getITSWrapVolPattern(), wrID);
294  }
295 
296  path +=
297  Form("%s%d_1/%s%d_%d/", GeometryTGeo::getITSLayerPattern(), lay, GeometryTGeo::getITSStavePattern(), lay, stav);
298 
299  if (mNumberOfHalfStaves[lay] > 0) {
300  path += Form("%s%d_%d/", GeometryTGeo::getITSHalfStavePattern(), lay, sstav);
301  }
302  if (mNumberOfModules[lay] > 0) {
303  path += Form("%s%d_%d/", GeometryTGeo::getITSModulePattern(), lay, mod);
304  }
305  path +=
306  Form("%s%d_%d/%s%d_1", GeometryTGeo::getITSChipPattern(), lay, chipInMod, GeometryTGeo::getITSSensorPattern(), lay);
307 
308  static TGeoHMatrix matTmp;
309  gGeoManager->PushPath();
310 
311  if (!gGeoManager->cd(path.Data())) {
312  gGeoManager->PopPath();
313  LOG(ERROR) << "Error in cd-ing to " << path.Data() << FairLogger::endl;
314  return nullptr;
315  } // end if !gGeoManager
316 
317  matTmp = *gGeoManager->GetCurrentMatrix(); // matrix may change after cd
318  // RSS
319  // printf("%d/%d/%d %s\n",lay,stav,detInSta,path.Data());
320  // mat->Print();
321  // Restore the modeler state.
322  gGeoManager->PopPath();
323 
324  // account for the difference between sensitive layer and physical sensor ticknesses
325  static TGeoTranslation tra(0.,0.5*(Segmentation::SensorThickness-Segmentation::SensLayerThickness),0.);
326 
327  matTmp *= tra;
328 
329  return &matTmp;
330 }

332 //__________________________________________________________________________
333 void GeometryTGeo::Build(int loadTrans)
334 {
335  if ( isBuilt() ) {
336  LOG(WARNING) << "Already built" << FairLogger::endl;
337  return; // already initialized
338  }
339 
340  if (!gGeoManager) {
341  // RSTODO: in future there will be a method to load matrices from the CDB
342  LOG(FATAL) << "Geometry is not loaded" << FairLogger::endl;
343  }
344 
345  mNumberOfLayers = extractNumberOfLayers();
346  if (!mNumberOfLayers) {
347  return;
348  }
349 
350  mNumberOfStaves.resize(mNumberOfLayers);
351  mNumberOfHalfStaves.resize(mNumberOfLayers);
352  mNumberOfModules.resize(mNumberOfLayers);
353  mNumberOfChipsPerModule.resize(mNumberOfLayers);
354  mNumberOfChipRowsPerModule.resize(mNumberOfLayers);
355  mNumberOfChipsPerHalfStave.resize(mNumberOfLayers);
356  mNumberOfChipsPerStave.resize(mNumberOfLayers);
357  mNumberOfChipsPerLayer.resize(mNumberOfLayers);
358  mLastChipIndex.resize(mNumberOfLayers);
359  int numberOfChips = 0;
360 
361  for (int i = 0; i < mNumberOfLayers; i++) {
362  mNumberOfStaves[i] = extractNumberOfStaves(i);
363  mNumberOfHalfStaves[i] = extractNumberOfHalfStaves(i);
364  mNumberOfModules[i] = extractNumberOfModules(i);
365  mNumberOfChipsPerModule[i] = extractNumberOfChipsPerModule(i, mNumberOfChipRowsPerModule[i]);
366  mNumberOfChipsPerHalfStave[i] = mNumberOfChipsPerModule[i] * Max(1, mNumberOfModules[i]);
367  mNumberOfChipsPerStave[i] = mNumberOfChipsPerHalfStave[i] * Max(1, mNumberOfHalfStaves[i]);
368  mNumberOfChipsPerLayer[i] = mNumberOfChipsPerStave[i] * mNumberOfStaves[i];
369  numberOfChips += mNumberOfChipsPerLayer[i];
370  mLastChipIndex[i] = numberOfChips - 1;
371  }
372  setSize(numberOfChips);
373  fillTrackingFramesCache();
374  //
375  fillMatrixCache(loadTrans);
376 
377 }

379 //__________________________________________________________________________
380 void GeometryTGeo::fillMatrixCache(int mask)
381 {
382  // populate matrix cache for requested transformations
383  //
384  if (mSize<1) {
385  LOG(WARNING) << "The method Build was not called yet" << FairLogger::endl;
386  Build(mask);
387  return;
388  }
389 
390  // build matrices
391  if ( (mask & bit2Mask(TransformType::L2G)) && !getCacheL2G().isFilled() ) {
392  // Matrices for Local (Sensor!!! rather than the full chip) to Global frame transformation
393  LOG(INFO) << "Loading ITS L2G matrices from TGeo" << FairLogger::endl;
394  auto& cacheL2G = getCacheL2G();
395  cacheL2G.setSize(mSize);
396 
397  for (int i=0;i<mSize;i++) {
398  TGeoHMatrix *hm = extractMatrixSensor(i);
399  cacheL2G.setMatrix( Mat3D( *hm ) , i);
400  }
401  }
402 
403  if ( (mask & bit2Mask(TransformType::T2L)) && !getCacheT2L().isFilled() ) {
404  // matrices for Tracking to Local (Sensor!!! rather than the full chip) frame transformation
405  LOG(INFO) << "Loading ITS T2L matrices from TGeo" << FairLogger::endl;
406  auto& cacheT2L = getCacheT2L();
407  cacheT2L.setSize(mSize);
408  for (int i=0;i<mSize;i++) {
409  TGeoHMatrix& hm = createT2LMatrix(i);
410  cacheT2L.setMatrix( Mat3D(hm) , i);
411  }
412  }
413 
414  if ( (mask & bit2Mask(TransformType::T2G)) && !getCacheT2G().isFilled() ) {
415  LOG(WARNING) << "It is faster to use 2D rotation for T2G instead of full Transform3D matrices" << FairLogger::endl;
416  // matrices for Tracking to Global frame transformation
417  LOG(INFO) << "Loading ITS T2G matrices from TGeo" << FairLogger::endl;
418  auto& cacheT2G = getCacheT2G();
419  cacheT2G.setSize(mSize);
420 
421  for (int i=0;i<mSize;i++) {
422  TGeoHMatrix& mat = createT2LMatrix(i);
423  mat.MultiplyLeft( extractMatrixSensor(i) );
424  cacheT2G.setMatrix( Mat3D(mat), i);
425  }
426  }
427 
428  if ( (mask & bit2Mask(TransformType::T2GRot)) && !getCacheT2GRot().isFilled() ) {
429  // 2D rotation matrices for Tracking frame to Global rotations
430  LOG(INFO) << "Loading ITS T2G rotation 2D matrices" << FairLogger::endl;
431  auto& cacheT2Gr = getCacheT2GRot();
432  cacheT2Gr.setSize(mSize);
433  for (int i=0;i<mSize;i++) {
434  cacheT2Gr.setMatrix( Rot2D(getSensorRefAlpha(i)) , i);
435  }
436  }
437 
438 }

454 //__________________________________________________________________________
455 int GeometryTGeo::extractNumberOfLayers()
456 {
457  int numberOfLayers = 0;
458 
459  TGeoVolume* itsV = gGeoManager->GetVolume(getITSVolPattern());
460  if (!itsV) {
461  LOG(FATAL) << "ITS volume " << getITSVolPattern() << " is not in the geometry" << FairLogger::endl;
462  }
463 
464  // Loop on all ITSV nodes, count Layer volumes by checking names
465  // Build on the fly layer - wrapper correspondence
466  TObjArray* nodes = itsV->GetNodes();
467  int nNodes = nodes->GetEntriesFast();
468 
469  for (int j = 0; j < nNodes; j++) {
470  int lrID = -1;
471  TGeoNode* nd = (TGeoNode*)nodes->At(j);
472  const char* name = nd->GetName();
473 
474  if (strstr(name, getITSLayerPattern())) {
475  numberOfLayers++;
476  if ((lrID = extractVolumeCopy(name, GeometryTGeo::getITSLayerPattern())) < 0) {
477  LOG(FATAL) << "Failed to extract layer ID from the " << name << FairLogger::endl;
478  exit(1);
479  }
480 
481  mLayerToWrapper[lrID] = -1; // not wrapped
482  } else if (strstr(name, getITSWrapVolPattern())) { // this is a wrapper volume, may cointain layers
483  int wrID = -1;
484  if ((wrID = extractVolumeCopy(name, GeometryTGeo::getITSWrapVolPattern())) < 0) {
485  LOG(FATAL) << "Failed to extract wrapper ID from the " << name << FairLogger::endl;
486  exit(1);
487  }
488 
489  TObjArray* nodesW = nd->GetNodes();
490  int nNodesW = nodesW->GetEntriesFast();
491 
492  for (int jw = 0; jw < nNodesW; jw++) {
493  TGeoNode* ndW = (TGeoNode*)nodesW->At(jw);
494  if (strstr(ndW->GetName(), getITSLayerPattern())) {
495  if ((lrID = extractVolumeCopy(ndW->GetName(), GeometryTGeo::getITSLayerPattern())) < 0) {
496  LOG(FATAL) << "Failed to extract layer ID from the " << name << FairLogger::endl;
497  exit(1);
498  }
499  numberOfLayers++;
500  mLayerToWrapper[lrID] = wrID;
501  }
502  }
503  }
504  }
505  return numberOfLayers;
506 }

604  snprintf(stavnam, 30, "%s%d", getITSHalfStavePattern(), lay);
605  volLd = gGeoManager->GetVolume(stavnam);
606  }
607  }
608  if (!volLd) { // no substaves on this layer, check staves
609  snprintf(stavnam, 30, "%s%d", getITSStavePattern(), lay);
610  volLd = gGeoManager->GetVolume(stavnam);
611  }
612  if (!volLd) {
613  LOG(FATAL) << "can't find volume containing chips on layer " << lay << FairLogger::endl;
614  }
615 
616  // Loop on all stave nodes, count Chip volumes by checking names
617  int nNodes = volLd->GetNodes()->GetEntries();
618 
619  double xmin = 1e9, xmax = -1e9, zmin = 1e9, zmax = -1e9;
620  double lab[3], loc[3] = { 0, 0, 0 };
621  double dx = -1, dz = -1;
622 
623  for (int j = 0; j < nNodes; j++) {
624  // AliInfo(Form("L%d %d of %d %s %s ->
625  // %d",lay,j,nNodes,volLd->GetNodes()->At(j)->GetName(),GetITSChipPattern(),numberOfChips));
626  TGeoNodeMatrix* node = (TGeoNodeMatrix*)volLd->GetNodes()->At(j);
627  if (!strstr(node->GetName(), getITSChipPattern())) {
628  continue;
629  }
630  node->LocalToMaster(loc, lab);
631  if (lab[0] > xmax) {
632  xmax = lab[0];
633  }
634  if (lab[0] < xmin) {
635  xmin = lab[0];
636  }
637  if (lab[2] > zmax) {
638  zmax = lab[2];
639  }
640  if (lab[2] < zmin) {
641  zmin = lab[2];
642  }
643 
644  numberOfChips++;
645 
646  if (dx < 0) {
647  TGeoShape* chShape = node->GetVolume()->GetShape();
648  TGeoBBox* bbox = dynamic_cast<TGeoBBox*>(chShape);
649  if (!bbox) {
650  LOG(FATAL) << "Chip " << node->GetName() << " volume is of unprocessed shape " << chShape->IsA()->GetName()
651  << FairLogger::endl;
652  } else {
653  dx = 2 * bbox->GetDX();
654  dz = 2 * bbox->GetDZ();
655  }
656  }
657  }
658 
659  double spanX = xmax - xmin;
660  double spanZ = zmax - zmin;
661  nrow = TMath::Nint(spanX / dx + 1);
662  int ncol = TMath::Nint(spanZ / dz + 1);
663  if (nrow * ncol != numberOfChips) {
664  LOG(ERROR) << "Inconsistency between Nchips=" << numberOfChips << " and Nrow*Ncol=" << nrow << "*" << ncol << "->"
665  << nrow * ncol << FairLogger::endl
666  << "Extracted chip dimensions (x,z): " << dx << " " << dz << " Module Span: " << spanX << " " << spanZ
667  << FairLogger::endl;
668  }
669  return numberOfChips;
670 }

703 void GeometryTGeo::extractSensorXAlpha(int isn, float &x, float &alp)
704 {
705  // calculate r and phi of the impact of the normal on the sensor
706  // (i.e. phi of the tracking frame alpha and X of the sensor in this frame)
707  double locA[3] = {-100.,0.,0.}, locB[3] = {100.,0.,0.}, gloA[3], gloB[3];
708  const TGeoHMatrix* matL2G = extractMatrixSensor(isn);
709 
710  matL2G->LocalToMaster(locA, gloA);
711  matL2G->LocalToMaster(locB, gloB);
712  double dx = gloB[0] - gloA[0], dy = gloB[1] - gloA[1];
713  double t = (gloB[0] * dx + gloB[1] * dy) / (dx * dx + dy * dy);
714  double xp = gloB[0] - dx * t, yp = gloB[1] - dy * t;
715  x = Sqrt(xp*xp+yp*yp);
716  alp = ATan2(yp,xp);
717  BringTo02Pi(alp);
718 }
