38 Double_t SimulationAlpide::computeIncidenceAngle(TLorentzVector dir) const
39 {
40  Vector3D<float> glob(dir.Px()/dir.P(), dir.Py()/dir.P(), dir.Pz()/dir.P());
41  auto loc = (*mMat)^(glob);
42  Vector3D<float> normal(0.f,-1.f,0.f);
43  return TMath::Abs(loc.Dot(normal)/TMath::Sqrt(loc.Mag2()));

48 void SimulationAlpide::Hits2Digits(Double_t eventTime, UInt_t &minFr, UInt_t &maxFr)
49 {
50  Int_t nhits = GetNumberOfHits();
51 
52  // convert hits to digits, returning the min and max RO frames processed
53  //
54  // addNoise(mParam[Noise]);
55  // RS: attention: the noise will be added just before transferring the digits of the given frame to the output,
56  // because at this stage we don't know to which RO frame the noise should be added
57 
58  for (Int_t h = 0; h < nhits; ++h) {
59 
60  const Hit *hit = GetHitAt(h);
61  double hTime0 = hit->GetTime()*sec2ns + eventTime - mParams->getTimeOffset(); // time from the RO start, in ns
62  if (hTime0 > UINT_MAX) {
63  LOG(WARNING) << "Hit RO Frame undefined: time: " << hTime0 << " is in far future: hitTime: "
64  << hit->GetTime() << " EventTime: " << eventTime << " ChipOffset: "
65  << mParams->getTimeOffset() << FairLogger::endl;
66  return;
67  }
68 
69  // calculate RO Frame for this hit
70  if (hTime0<0) hTime0 = 0.;
71  UInt_t roframe = static_cast<UInt_t>(hTime0/mParams->getROFrameLenght());
72  if (roframe<minFr) minFr = roframe;
73  if (roframe>maxFr) maxFr = roframe;
74 
75  // check if the hit time is not in the dead time of the chip
76  // RS: info from L.Musa: no inefficiency on the frame boundary
77  //if ( mParams->getROFrameLenght() - roframe%mParams->getROFrameLenght() < mParams->getROFrameDeadTime()) continue;
78 
79  switch (mParams->getHit2DigitsMethod()) {
80  case DigiParams::p2dCShape :
81  Hit2DigitsCShape(hit, roframe, eventTime);
82  break;
83  case DigiParams::p2dSimple :
84  Hit2DigitsSimple(hit, roframe, eventTime);
85  break;
86  default:
87  LOG(ERROR) << "Unknown point to digit mode " << mParams->getHit2DigitsMethod() << FairLogger::endl;
88  break;
89  }
90  }
91 }

94 void SimulationAlpide::Hit2DigitsCShape(const Hit *hit, UInt_t roFrame, double eventTime)
95 {
96  // convert single hit to digits with CShape generation method
97 
98  float nSteps = static_cast<Float_t>(mParams->getNSimSteps());
99  Vector3D<float> xyzLocS( (*mMat)^(hit->GetPosStart()) ); // start position
100  Vector3D<float> xyzLocE( (*mMat)^(hit->GetPos()) ); // end position
101  Vector3D<float> step(xyzLocE);
102  step -= xyzLocS;
103  step /= nSteps; // position increment at each step
104  // the electrons will injected in the middle of each step
105  xyzLocS += step/2;
106  xyzLocE -= step/2;
107 
108  int rowS=-1,colS=-1,rowE=-1,colE=-1, nSkip=0;
109  // get entrance pixel row and col
110  while (!Segmentation::localToDetector(xyzLocS.X(), xyzLocS.Z(), rowS, colS)) { // guard-ring ?
111  if (++nSkip>=nSteps) return; // did not enter to sensitive matrix
112  xyzLocS += step;
113  }
114  // get exit pixel row and col
115  while (!Segmentation::localToDetector(xyzLocE.X(), xyzLocE.Z(), rowE, colE)) { // guard-ring ?
116  if (++nSkip>=nSteps) return; // did not enter to sensitive matrix
117  xyzLocE -= step;
118  }
119  // estimate the limiting min/max row and col where the non-0 response is possible
120  if (rowS>rowE) std::swap(rowS,rowE);
121  if (colS>colE) std::swap(colS,colE);
122  rowS -= AlpideRespSimMat::NPix/2;
123  rowE += AlpideRespSimMat::NPix/2;
124  if (rowS<0) rowS = 0;
125  if (rowE>=Segmentation::NRows) rowE = Segmentation::NRows-1;
126  colS -= AlpideRespSimMat::NPix/2;
127  colE += AlpideRespSimMat::NPix/2;
128  if (colS<0) colS = 0;
129  if (colE>=Segmentation::NCols) colE = Segmentation::NCols-1;
130  int rowSpan = rowE-rowS+1, colSpan = colE-colS+1; // size of plaquet response is expected
131 
132  float respMatrix[rowSpan][colSpan]; // response accumulated here
133  std::fill(&respMatrix[0][0],&respMatrix[0][0]+rowSpan*colSpan,0.f);
134 
135  float nElectrons = hit->GetEnergyLoss()*mParams->getEnergyToNElectrons(); // total number of deposited electrons
136  nElectrons /= nSteps; // N electrons injected per step
137  if (nSkip) nSteps -= nSkip;
138 
139  int rowPrev=-1, colPrev=-1, row, col;
140  float cRowPix=0.f, cColPix=0.f; // local coordinated of the current pixel center
141 
142  const o2::ITSMFT::AlpideSimResponse* resp = mParams->getAlpSimResponse();
143 
144  // take into account that the AlpideSimResponse has min/max thickness non-symmetric around 0
145  xyzLocS.SetY( xyzLocS.Y() + resp->getDepthShift());
146 
147  for (int iStep=nSteps;iStep--;) {
148  // Get the pixel ID
149  Segmentation::localToDetector(xyzLocS.X(), xyzLocS.Z(), row, col);
150  if (row!=rowPrev || col!=colPrev) { // update pixel and coordinates of its center
151  if (!Segmentation::detectorToLocal(row, col, cRowPix, cColPix)) continue; // should not happen
152  rowPrev = row;
153  colPrev = col;
154  }
155  bool flipCol, flipRow;
156  // note that response needs coordinates along column row (locX) (locZ) then depth (locY)
157  auto rspmat = resp->getResponse(xyzLocS.X()-cRowPix, xyzLocS.Z()-cColPix, xyzLocS.Y(), flipRow, flipCol);
158  //printf("#%d Lx:%e Lz:%e Ly:%e r:%d c:%d | dZ:%e dX:%e | %p\n",iStep,xyzLocS.X(), xyzLocS.Z(),xyzLocS.Y(),row,col,
159  // xyzLocS.Z()-cColPix, xyzLocS.X()-cRowPix, rspmat);
160 
161  xyzLocS += step;
162  if (!rspmat) continue;
163 
164  for (int irow=AlpideRespSimMat::NPix; irow--;) {
165  int rowDest = row+irow-AlpideRespSimMat::NPix/2 - rowS; // destination row in the respMatrix
166  if (rowDest<0 || rowDest>=rowSpan) continue;
167  for (int icol=AlpideRespSimMat::NPix; icol--;) {
168  int colDest = col+icol-AlpideRespSimMat::NPix/2 - colS; // destination column in the respMatrix
169  if (colDest<0 || colDest>=colSpan) continue;
170 
171  respMatrix[rowDest][colDest] += rspmat->getValue(irow,icol,flipRow,flipCol);
172  }
173  }
174  }
175 
176  double hTime = hit->GetTime()*sec2ns + eventTime; // time in ns
177 
178  // fire the pixels assuming Poisson(n_response_electrons)
179  for (int irow=rowSpan;irow--;) {
180  for (int icol=colSpan;icol--;) {
181  float nEleResp = respMatrix[irow][icol];
182  if (!nEleResp) continue;
183  int nEle = gRandom->Poisson(nElectrons*nEleResp);
184  if (nEle) addDigit(roFrame, irow+rowS, icol+colS, nEle, hit->getCombLabel(), hTime);
185  }
186  }
187 
188 }

191 void SimulationAlpide::Hit2DigitsSimple(const Hit *hit, UInt_t roFrame, double eventTime)
192 {
193  // convert single hit to digits with 1 to 1 mapping
194  Point3D<float> glo( 0.5*(hit->GetX() + hit->GetStartX()),
195  0.5*(hit->GetY() + hit->GetStartY()),
196  0.5*(hit->GetZ() + hit->GetStartZ()) );
197  auto loc = (*mMat)^( glo );
198 
199  int ix, iz;
200  if (!Segmentation::localToDetector(loc.X(), loc.Z(), ix, iz)) {
201  LOG(DEBUG) << "Out of the chip" << FairLogger::endl;
202  return;
203  }
204  addDigit(roFrame, ix, iz, hit->GetEnergyLoss()*mParams->getEnergyToNElectrons(), hit->getCombLabel(), hit->GetTime()*sec2ns + eventTime);
205 }

210 {
211  UInt_t row = 0;
212  UInt_t col = 0;
213  Int_t nhits = 0;
214  constexpr float ns2sec = 1e-9;

216  float mean = mParams->getNoisePerPixel()*Segmentation::NPixels;

220  nhits = gRandom->Poisson(mean);

222  for (Int_t i = 0; i < nhits; ++i) {
223  row = gRandom->Integer(Segmentation::NRows);

225  // RS TODO: why the noise was added with 0 charge? It should be above the threshold!
