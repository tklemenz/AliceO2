37 // A container to hold and manage MC truth information/labels.
38 // The actual MCtruth type is a generic template type and can be supplied by the user
39 // It is meant to manage associations from one "dataobject" identified by an index into an array
40 // to multiple TruthElements
41 
42 template <typename TruthElement>
43 class MCTruthContainer
44 {
45  private:
46  std::vector<MCTruthHeaderElement>
47  mHeaderArray; // the header structure array serves as an index into the actual storage
48  std::vector<TruthElement> mTruthArray; // the buffer containing the actual truth information
49 
50  size_t getSize(int dataindex) const
51  {
52  // calculate size / number of labels from a difference in pointed indices
53  const auto size = (dataindex < mHeaderArray.size() - 1)
54  ? mHeaderArray[dataindex + 1].index - mHeaderArray[dataindex].index
55  : mTruthArray.size() - mHeaderArray[dataindex].index;
56  return size;
57  }
58 
59  public:
60  // constructor
61  MCTruthContainer() = default;
62  // destructor
63  ~MCTruthContainer() = default;
64  // copy constructor
65  MCTruthContainer(const MCTruthContainer& other) = default;
66  // assignment operator
67  MCTruthContainer& operator=(const MCTruthContainer &other) = default;
68 
69  // access
70  MCTruthHeaderElement getMCTruthHeader(uint dataindex) const { return mHeaderArray[dataindex]; }
71  // access the element directly (can be encapsulated better away)... needs proper element index
72  // which can be obtained from the MCTruthHeader startposition and size
73  TruthElement const& getElement(uint elementindex) const { return mTruthArray[elementindex]; }
74  // return the number of original data indexed here
75  size_t getIndexedSize() const { return mHeaderArray.size(); }
76  // return the number of elements managed in this container
77  size_t getNElements() const { return mTruthArray.size(); }
78 
79  // get individual "view" container for a given data index
80  // the caller can do modifications on this view (such as sorting)
81  gsl::span<TruthElement> getLabels(int dataindex) {
82  if(dataindex >= getIndexedSize()) return gsl::span<TruthElement>();
83  return gsl::span<TruthElement>(&mTruthArray[mHeaderArray[dataindex].index], getSize(dataindex));
84  }
85 
86  // get individual const "view" container for a given data index
87  // the caller can't do modifications on this view
88  gsl::span<const TruthElement> getLabels(int dataindex) const {
89  if(dataindex >= getIndexedSize()) return gsl::span<const TruthElement>();
90  return gsl::span<const TruthElement>(&mTruthArray[mHeaderArray[dataindex].index], getSize(dataindex));
91  }

93  void clear()
94  {
95  mHeaderArray.clear();
96  mTruthArray.clear();
97  }
98 
99  // add element for a particular dataindex
100  // at the moment only strictly consecutive modes are supported
101  void addElement(uint dataindex, TruthElement const& element)
102  {
103  if (dataindex < mHeaderArray.size()) {
104  // look if we have something for this dataindex already
105  // must currently be the last one
106  if (dataindex != (mHeaderArray.size() - 1)) {
107  throw std::runtime_error("MCTruthContainer: unsupported code path");
108  }
109  } else {
110  assert(dataindex == mHeaderArray.size());
111  // add a new one
112  mHeaderArray.emplace_back(mTruthArray.size());
113  }
114  auto& header = mHeaderArray[dataindex];
115  mTruthArray.emplace_back(element);
116  }
