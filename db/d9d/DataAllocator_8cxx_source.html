35 std::string
36 DataAllocator::matchDataHeader(const OutputSpec &spec, size_t timeslice) {
37  // FIXME: we should take timeframeId into account as well.
38  for (auto &output : mAllowedOutputs) {
39  if (DataSpecUtils::match(output.matcher, spec.origin, spec.description, spec.subSpec)
40  && ((timeslice % output.maxTimeslices) == output.timeslice)) {
41  return output.channel;
42  }
43  }
44  std::ostringstream str;
45  str << "Worker is not authorised to create message with "
46  << "origin(" << spec.origin.str << ")"
47  << "description(" << spec.description.str << ")"
48  << "subSpec(" << spec.subSpec << ")";
49  throw std::runtime_error(str.str());
50 }

52 DataChunk
53 DataAllocator::newChunk(const OutputSpec &spec, size_t size) {
54  std::string channel = matchDataHeader(spec, mContext->timeslice());
55 
56  DataHeader dh;
57  dh.dataOrigin = spec.origin;
58  dh.dataDescription = spec.description;
59  dh.subSpecification = spec.subSpec;
60  dh.payloadSize = size;
61  dh.payloadSerializationMethod = o2::header::gSerializationMethodNone;
62 
63  DataProcessingHeader dph{mContext->timeslice(), 1};
64  //we have to move the incoming data
65  o2::header::Stack headerStack{dh, dph};
66  FairMQMessagePtr headerMessage = mDevice->NewMessageFor(channel, 0,
67  headerStack.buffer.get(),
68  headerStack.bufferSize,
69  &o2::header::Stack::freefn,
70  headerStack.buffer.get());
71  headerStack.buffer.release();
72  FairMQMessagePtr payloadMessage = mDevice->NewMessageFor(channel, 0, size);
73  auto dataPtr = payloadMessage->GetData();
74  auto dataSize = payloadMessage->GetSize();
75 
76  FairMQParts parts;
77  parts.AddPart(std::move(headerMessage));
78  parts.AddPart(std::move(payloadMessage));
79  assert(parts.Size() == 2);
80  mContext->addPart(std::move(parts), channel);
81  assert(parts.Size() == 0);
82  return DataChunk{reinterpret_cast<char*>(dataPtr), dataSize};
83 }

147 DataAllocator::addPartToContext(FairMQMessagePtr&& payloadMessage,
148  const OutputSpec &spec,
149  o2::header::SerializationMethod serializationMethod)
150 {
151  std::string channel = matchDataHeader(spec, mRootContext->timeslice());
152  auto headerMessage = headerMessageFromSpec(spec, channel, serializationMethod);
153 
154  FairMQParts parts;
155 
156  // FIXME: this is kind of ugly, we know that we can change the content of the
157  // header message because we have just created it, but the API declares it const
158  const DataHeader *cdh = o2::header::get<DataHeader>(headerMessage->GetData());
159  DataHeader *dh = const_cast<DataHeader *>(cdh);
160  dh->payloadSize = payloadMessage->GetSize();
161  parts.AddPart(std::move(headerMessage));
162  parts.AddPart(std::move(payloadMessage));
163  mContext->addPart(std::move(parts), channel);
164 }
