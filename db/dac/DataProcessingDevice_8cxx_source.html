81 bool
82 DataProcessingDevice::HandleData(FairMQParts &iParts, int /*index*/) {
83  // Initial part. Let's hide all the unnecessary and have
84  // simple lambdas for each of the steps I am planning to have.
85  assert(!mInputs.empty());
86 
87  // These duplicate references are created so that each function
88  // does not need to know about the whole class state, but I can
89  // fine grain control what is exposed at each state.
90  auto &metricsService = mServiceRegistry.get<MetricsService>();
91  auto &statefulProcess = mStatefulProcess;
92  auto &statelessProcess = mStatelessProcess;
93  auto &errorCallback = mError;
94  auto &serviceRegistry = mServiceRegistry;
95  auto &allocator = mAllocator;
96  auto &processingCount = mProcessingCount;
97  auto &relayer = mRelayer;
98  auto &device = *this;
99  auto &context = mContext;
100  auto &rootContext = mRootContext;
101  auto &forwards = mForwards;
102  auto &inputsSchema = mInputs;
103  auto &errorCount = mErrorCount;
104 
105  std::vector<std::unique_ptr<FairMQMessage>> currentSetOfInputs;
106 
107  // This is the actual hidden state for the outer loop. In case we decide we
108  // want to support multithreaded dispatching of operations, I can simply
109  // move these to some thread local store and the rest of the lambdas
110  // should work just fine.
111  FairMQParts &parts = iParts;
112  std::vector<int> completed;
113 
114 
115  // This is how we validate inputs. I.e. we try to enforce the O2 Data model
116  // and we do a few stats. We bind parts as a lambda captured variable, rather
117  // than an input, because we do not want the outer loop actually be exposed
118  // to the implementation details of the messaging layer.
119  auto isValidInput = [&metricsService, &parts]() -> bool {
120  metricsService.post("inputs/parts/total", (int)parts.Size());
121 
122  for (size_t i = 0; i < parts.Size() ; ++i) {
123  LOG(DEBUG) << " part " << i << " is " << parts.At(i)->GetSize() << " bytes";
124  }
125  if (parts.Size() % 2) {
126  return false;
127  }
128  for (size_t hi = 0; hi < parts.Size()/2; ++hi) {
129  auto pi = hi*2;
130  auto dh = o2::header::get<DataHeader>(parts.At(pi)->GetData());
131  if (!dh) {
132  LOG(ERROR) << "Header is not a DataHeader?";
133  return false;
134  }
135  if (dh->payloadSize != parts.At(pi+1)->GetSize()) {
136  LOG(ERROR) << "DataHeader payloadSize mismatch";
137  return false;
138  }
139  auto dph = o2::header::get<DataProcessingHeader>(parts.At(pi)->GetData());
140  if (!dph) {
141  LOG(ERROR) << "Header stack does not contain DataProcessingHeader";
142  return false;
143  }
144  LOG(DEBUG) << "Timeslice is " << dph->startTime;
145  LOG(DEBUG) << " DataOrigin is " << dh->dataOrigin.str;
146  LOG(DEBUG) << " DataDescription is " << dh->dataDescription.str;
147  }
148  return true;
149  };
150 
151  //
152  auto reportError = [&errorCount, &metricsService](const char *message) {
153  LOG(ERROR) << message;
154  errorCount++;
155  metricsService.post("dataprocessing/errors", errorCount);
156  };
157 
158  auto putIncomingMessageIntoCache = [&parts,&relayer,&reportError]() {
159  // We relay execution to make sure we have a complete set of parts
160  // available.
161  for (size_t pi = 0; pi < (parts.Size()/2); ++pi) {
162  auto headerIndex = 2*pi;
163  auto payloadIndex = 2*pi+1;
164  assert(payloadIndex < parts.Size());
165  auto relayed = relayer.relay(std::move(parts.At(headerIndex)),
166  std::move(parts.At(payloadIndex)));
167  if (relayed == DataRelayer::WillNotRelay) {
168  reportError("Unable to relay part.");
169  return;
170  }
171  LOG(DEBUG) << "Relaying part idx: " << headerIndex;
172  }
173  };
174 
175  // For the moment we have a simple "immediately dispatch" policy for stuff
176  // in the cache. This could be controlled from the outside e.g. by waiting
177  // for a few sets of inputs to arrive before we actually dispatch the
178  // computation, however this can be defined at a later stage.
179  auto canDispatchSomeComputation = [&completed, &relayer]() -> bool {
180  completed = relayer.getReadyToProcess();
181  return completed.empty() == false;
182  };
183 
184  // We use this to get a list with the actual indexes in the cache which
185  // indicate a complete set of inputs. Notice how I fill the completed
186  // vector and return it, so that I can have a nice for loop iteration later
187  // on.
188  auto getCompleteInputSets = [&relayer,&completed,&metricsService]() -> std::vector<int> {
189  LOG(DEBUG) << "Getting parts to process";
190  completed = relayer.getReadyToProcess();
191  int pendingInputs = (int)relayer.getParallelTimeslices() - completed.size();
192  metricsService.post("inputs/relayed/pending", pendingInputs);
193  if (completed.empty()) {
194  metricsService.post("inputs/relayed/incomplete", 1);
195  }
196  return completed;
197  };
198 
199  // This is needed to convert from a pair of pointers to an actual DataRef
200  // and to make sure the ownership is moved from the cache in the relayer to
201  // the execution.
202  auto fillInputs = [&relayer, &inputsSchema, &currentSetOfInputs](int timeslice) -> InputRecord {
203  currentSetOfInputs = std::move(relayer.getInputsForTimeslice(timeslice));
204  InputRecord registry{inputsSchema, currentSetOfInputs};
205  return registry;
206  };
207 
208  // This is the thing which does the actual computation. No particular reason
209  // why we do the stateful processing before the stateless one.
210  // PROCESSING:{START,END} is done so that we can trigger on begin / end of processing
211  // in the GUI.
212  auto dispatchProcessing = [&processingCount,
213  &allocator,
214  &statefulProcess,
215  &statelessProcess,
216  &metricsService,
217  &context,
218  &rootContext,
219  &serviceRegistry,
220  &device](int i, InputRecord &record) {
221  if (statefulProcess) {
222  LOG(DEBUG) << "PROCESSING:START:" << i;
223  metricsService.post("dataprocessing/stateful_process", processingCount++);
224  ProcessingContext processContext{record, serviceRegistry, allocator};
225  statefulProcess(processContext);
226  LOG(DEBUG) << "PROCESSING:END:" << i;
227  }
228  if (statelessProcess) {
229  LOG(DEBUG) << "PROCESSING:START:" << i;
230  metricsService.post("dataprocessing/stateless_process", processingCount++);
231  ProcessingContext processContext{record, serviceRegistry, allocator};
232  statelessProcess(processContext);
233  LOG(DEBUG) << "PROCESSING:END:" << i;
234  }
235  DataProcessor::doSend(device, context);
236  DataProcessor::doSend(device, rootContext);
237  };
238 
239  // Error handling means printing the error and updating the metric
240  auto errorHandling = [&errorCallback,
241  &metricsService,
242  &serviceRegistry](std::exception &e, InputRecord &record) {
243  LOG(ERROR) << "Exception caught: " << e.what() << std::endl;
244  if (errorCallback) {
245  metricsService.post("error", 1);
246  ErrorContext errorContext{record, serviceRegistry, e};
247  errorCallback(errorContext);
248  }
249  };

251  // I need a preparation step which gets the current timeslice id and
252  // propagates it to the various contextes (i.e. the actual entities which
253  // create messages) because the messages need to have the timeslice id into
254  // it.
255  auto prepareForCurrentTimeSlice = [&rootContext, &context, &relayer](int i) {
256  size_t timeslice = relayer.getTimesliceForCacheline(i);
257  LOG(DEBUG) << "Timeslice for cacheline is " << timeslice;
258  rootContext.prepareForTimeslice(timeslice);
259  context.prepareForTimeslice(timeslice);
260  };
261 
262  // This is how we do the forwarding, i.e. we push
263  // the inputs which are shared between this device and others
264  // to the next one in the daisy chain.
265  // FIXME: do it in a smarter way than O(N^2)
266  auto forwardInputs = [&reportError, &forwards, &device, &currentSetOfInputs]
267  (int timeslice, InputRecord &record) {
268  assert(record.size()*2 == currentSetOfInputs.size());
269  LOG(DEBUG) << "FORWARDING:START:" << timeslice;
270  for (size_t ii = 0, ie = record.size(); ii != ie; ++ii) {
271  DataRef input = record.getByPos(ii);
272  assert(input.header);
273  auto dh = o2::header::get<DataHeader>(input.header);
274  if (!dh) {
275  reportError("Header is not a DataHeader?");
276  continue;
277  }
278  auto dph = o2::header::get<DataProcessingHeader>(input.header);
279  if (!dph) {
280  reportError("Header stack does not contain DataProcessingHeader");
281  continue;
282  }
283 
284  auto &header = currentSetOfInputs[ii*2];
285  auto &payload = currentSetOfInputs[ii*2+1];
286 
287  for (auto forward : forwards) {
288  LOG(DEBUG) << "Input part content";
289  LOG(DEBUG) << dh->dataOrigin.str;
290  LOG(DEBUG) << dh->dataDescription.str;
291  LOG(DEBUG) << dh->subSpecification;
292  if (DataSpecUtils::match(forward.matcher, dh->dataOrigin,
293  dh->dataDescription,
294  dh->subSpecification)) {
295 
296  if (header.get() == nullptr) {
297  LOG(ERROR) << "Missing header!";
298  continue;
299  }
300  auto fdph = o2::header::get<DataProcessingHeader>(header.get()->GetData());
301  if (fdph == nullptr) {
302  LOG(ERROR) << "Forwarded data does not have a DataProcessingHeader";
303  continue;
304  }
305  auto fdh = o2::header::get<DataHeader>(header.get()->GetData());
306  if (fdh == nullptr) {
307  LOG(ERROR) << "Forwarded data does not have a DataHeader";
308  continue;
309  }
310  LOG(DEBUG) << "Forwarding data to " << forward.channel;
311  LOG(DEBUG) << "Forwarded timeslice is " << fdph->startTime;
312  LOG(DEBUG) << "Forwarded channel is " << forward.channel;
313  FairMQParts forwardedParts;
314  forwardedParts.AddPart(std::move(header));
315  forwardedParts.AddPart(std::move(payload));
316  assert(forwardedParts.Size() == 2);
317  assert(o2::header::get<DataProcessingHeader>(forwardedParts.At(0)->GetData()));
318  LOG(DEBUG) << o2::header::get<DataProcessingHeader>(forwardedParts.At(0)->GetData())->startTime;
319  LOG(DEBUG) << forwardedParts.At(0)->GetSize();
320  // FIXME: this should use a correct subchannel
321  device.Send(forwardedParts, forward.channel, 0);
322  }
323  }
324  }
325  LOG(DEBUG) << "FORWARDING:END";
326  };
327 
328  // Second part. This is the actual outer loop we want to obtain, with
329  // implementation details which can be read. Notice how most of the state
330  // is actually hidden. For example we do not expose what "input" is. This
331  // will allow us to keep the same toplevel logic even if the actual meaning
332  // of input is changed (for example we might move away from multipart
333  // messages).
334  if (isValidInput() == false) {
335  reportError("Parts should come in couples. Dropping it.");
336  return true;
337  }
338  putIncomingMessageIntoCache();
339  if (canDispatchSomeComputation() == false) {
340  return true;
341  }
342 
343  for (auto cacheline : getCompleteInputSets()) {
344  prepareForCurrentTimeSlice(cacheline);
345  InputRecord record = fillInputs(cacheline);
346  try {
347  dispatchProcessing(cacheline, record);
348  } catch(std::exception &e) {
349  errorHandling(e, record);
350  }
351  forwardInputs(cacheline, record);
352  }
353 
354  return true;
355 }
