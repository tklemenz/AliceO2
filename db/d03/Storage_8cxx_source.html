53 void Storage::getSelection(/*const*/ ConditionId *id)
54 {
55  // return required version and subversion from the list of selection criteria
56 
57  TIter iter(&mSelections);
58  ConditionId *aSelection;
59 
60  // loop on the list of selection criteria
61  while ((aSelection = (ConditionId *) iter.Next())) {
62  // check if selection element contains id's path and run (range)
63  if (aSelection->isSupersetOf(*id)) {
64  LOG(DEBUG) << "Using selection criterion: " << aSelection->ToString().Data() << " " << FairLogger::endl;
65  // return required version and subversion
66 
67  id->setVersion(aSelection->getVersion());
68  id->setSubVersion(aSelection->getSubVersion());
69  return;
70  }
71  }
72 
73  // no valid element is found in the list of selection criteria -> return
74  LOG(DEBUG) << "Looking for objects with most recent version" << FairLogger::endl;
75  return;
76 }

79 {

103  printSelectionList();

106 void Storage::addSelection(const ConditionId &selection)
107 {
108  // add a selection criterion
109 
110  IdPath path = selection.getPath();
111  if (!path.isValid()) {
112  return;
113  }
114 
115  TIter iter(&mSelections);
116  const ConditionId *anId;
117  while ((anId = (ConditionId *) iter.Next())) {
118  if (selection.isSupersetOf(*anId)) {
119  LOG(WARNING) << "This selection is more general than a previous one and will hide it!" << FairLogger::endl;
120  LOG(WARNING) << (anId->ToString()).Data() << FairLogger::endl;
121  mSelections.AddBefore(anId, new ConditionId(selection));
122  return;
123  }
124  }
125  mSelections.AddFirst(new ConditionId(selection));
126 }

198 Condition *Storage::getObject(const ConditionId &query)
199 {
200  // get an Condition object from the database
201 
202  // check if query's path and runRange are valid
203  // query is invalid also if version is not specified and subversion is!
204  if (!query.isValid()) {
205  LOG(ERROR) << "Invalid query: " << query.ToString().Data() << FairLogger::endl;
206  return nullptr;
207  }
208 
209  // query is not specified if path contains wildcard or runrange = [-1,-1]
210  if (!query.isSpecified()) {
211  LOG(ERROR) << "Unspecified query: " << query.ToString().Data() << FairLogger::endl;
212  return nullptr;
213  }
214 
215  // This is needed otherwise TH1 objects (histos, TTree's) are lost when file is closed!
216  Bool_t oldStatus = TH1::AddDirectoryStatus();
217  TH1::AddDirectory(kFALSE);
218 
219  Condition *entry = getCondition(query);
220 
221  if (oldStatus != kFALSE) {
222  TH1::AddDirectory(kTRUE);
223  }
224 
225  // if drain storage is set, drain entry into drain storage
226  if (entry && (Manager::Instance())->isdrainSet()) {
227  Manager::Instance()->drain(entry);
228  }
229 
230  return entry;
231 }

247 TList *Storage::getAllObjects(const ConditionId &query)
248 {
249  // get multiple Condition objects from the database
250

252  LOG(ERROR) << "Invalid query: " << query.ToString().Data() << FairLogger::endl;
253  return nullptr;
254  }
255 
256  if (query.isAnyRange()) {
257  LOG(ERROR) << "Unspecified run or runrange: " << query.ToString().Data() << FairLogger::endl;
258  return nullptr;
259  }
260 
261  // This is needed otherwise TH1 objects (histos, TTree's) are lost when file is closed!
262  Bool_t oldStatus = TH1::AddDirectoryStatus();
263  TH1::AddDirectory(kFALSE);
264 
265  TList *result = getAllEntries(query);
266 
267  if (oldStatus != kFALSE) {
268  TH1::AddDirectory(kTRUE);
269  }
270 
271  Int_t nEntries = result->GetEntries();
272 
273  LOG(INFO) << nEntries << " objects retrieved. Request was: " << query.ToString().Data() << FairLogger::endl;
274  for (int i = 0; i < nEntries; i++) {
275  Condition *entry = (Condition *) result->At(i);
276  LOG(INFO) << entry->getId().ToString().Data() << FairLogger::endl;
277  }
278 
279  // if drain storage is set, drain entries into drain storage
280  if ((Manager::Instance())->isdrainSet()) {
281  for (int i = 0; i < result->GetEntries(); i++) {
282  Condition *entry = (Condition *) result->At(i);
283  Manager::Instance()->drain(entry);
284  }

386 void Storage::queryStorages(Int_t run, const char *pathFilter, Int_t version, ConditionMetaData *md)
387 {
388  // query CDB for files valid for given run, and fill list mValidFileIds
389  // Actual query is done in virtual function queryValidFiles()
390  // If version is not specified, the query will fill mValidFileIds
391  // with highest versions
392 
393  mRun = run;
394 
395  mPathFilter = pathFilter;
396  if (!mPathFilter.isValid()) {
397  LOG(ERROR) << "Filter not valid: " << pathFilter << FairLogger::endl;
398  mPathFilter = "*";
399  return;
400  }
401 
402  mVersion = version;
403 
404  LOG(INFO) << "Querying files valid for run " << mRun << R"( and path ")" << pathFilter << R"(" into CDB storage ")"
405  << mType.Data() << "://" << mBaseFolder.Data() << R"(")" << FairLogger::endl;
406 
407  // In mValidFileIds, clear id for the same 3level path, if any
408  LOG(DEBUG) << R"(Clearing list of CDB ConditionId's previously loaded for path ")" << pathFilter << R"(")"
409  << FairLogger::endl;
410  IdPath filter(pathFilter);
411  for (Int_t i = mValidFileIds.GetEntries() - 1; i >= 0; --i) {
412  ConditionId *rmMe = dynamic_cast<ConditionId *>(mValidFileIds.At(i));
413  IdPath rmPath = rmMe->getPathString();
414  if (filter.isSupersetOf(rmPath)) {
415  LOG(DEBUG) << R"(Removing id ")" << rmPath.getPathString().Data() << R"(" matching: ")" << pathFilter << R"(")"
416  << FairLogger::endl;
417  delete mValidFileIds.RemoveAt(i);
418  }
419  }
420 
421  if (mConditionMetaDataFilter) {
422  delete mConditionMetaDataFilter;
423  mConditionMetaDataFilter = nullptr;
424  }
425  if (md) {
426  mConditionMetaDataFilter = dynamic_cast<ConditionMetaData *>(md->Clone());
427  }
428 
429  queryValidFiles();
430 
431  LOG(INFO) << mValidFileIds.GetEntries() << " valid files found!" << FairLogger::endl;
432 }
