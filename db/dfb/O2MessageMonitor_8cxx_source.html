69 void O2MessageMonitor::Run()
70 {
71  //check socket type of data channel
72  std::string type;
73  std::vector<FairMQChannel> subChannels = fChannels["data"];
74  if (subChannels.size()>0) {
75  type = subChannels[0].GetType();
76  }
77 
78  while (CheckCurrentState(RUNNING) && (--mIterations)!=0) {
79  this_thread::sleep_for(chrono::milliseconds(mDelay));
80 
81  O2Message message;
82  NameHeader48 nameHeader;
83 
84  //maybe send a request
85  nameHeader = mName;
86  AddMessage(message,{mDataHeader,nameHeader},NewSimpleMessage(mPayload));
87  Send(message, "data");
88  message.fParts.clear();
89 
90  //message in;
91  Receive(message, "data");
92  LOG(INFO) << "== New message=============================";
93  ForEach(message, &O2MessageMonitor::HandleO2frame);
94  message.fParts.clear();
95 
96  //maybe a reply message
97  if (type=="rep") {
98  nameHeader = "My name is reply";
99  AddMessage(message,{mDataHeader,nameHeader},NewSimpleMessage("I am a reply"));
100  Send(message, "data");
101  }
102  }
103 }
