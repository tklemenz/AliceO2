17 void InteractionSampler::init()
18 {
19  // (re-)initialize sample and check parameters consistency
20 
21  int nBCSet = mBCFilling.count();
22  if (!nBCSet) {
23  LOG(WARNING) << "No bunch filling provided, impose default one" << FairLogger::endl;
24  ;
25  setDefaultBunchFilling();
26  nBCSet = mBCFilling.count();
27  }
28 
29  if (mMuBC < 0. && mIntRate < 0.) {
30  LOG(WARNING) << "No IR or muBC is provided, setting default IR" << FairLogger::endl;
31  ;
32  mIntRate = DefIntRate;
33  }
34 
35  if (mMuBC > 0.) {
36  mIntRate = mMuBC * nBCSet * LHCRevFreq;
37  LOG(INFO) << "Deducing IR=" << mIntRate << "Hz from " << nBCSet << " BCs at mu=" << mMuBC << FairLogger::endl;
38  } else {
39  mMuBC = mIntRate / (nBCSet * LHCRevFreq);
40  LOG(INFO) << "Deducing mu=" << mMuBC << " per BC from IR=" << mIntRate << " with " << nBCSet << " BCs"
41  << FairLogger::endl;
42  }
43 
44  mBCMin = 0;
45  mBCMax = -1;
46  for (int i = 0; i < LHCBCSlots; i++) {
47  if (!mBCFilling[i])
48  continue;
49  if (mBCMin > i)
50  mBCMin = i;
51  if (mBCMax < i)
52  mBCMax = i;
53  }
54  double muexp = TMath::Exp(-mMuBC);
55  mProbNoInteraction = 1. - muexp;
56  mMuBCZTRed = mMuBC * muexp / mProbNoInteraction;
57  mBCCurrent = mBCMin + gRandom->Integer(mBCMax - mBCMin + 1);
58  mIntBCCache = 0;
59  mOrbit = mPeriod = 0;
60 }

132 o2::MCInteractionRecord InteractionSampler::generateCollisionTime()
133 {
134  // generate single interaction record
135  if (mIntRate < 0)
136  init();
137 
138  if (mIntBCCache < 1) { // do we still have interaction in current BC?
139  mIntBCCache = simulateInteractingBC(); // decide which BC interacts and N collisions
140  }
141  double timeInt = mTimeInBC.back();
142  timeInt += mBCCurrent * BCSpacingLHC + mOrbit * OrbitDuration;
143  if (mPeriod)
144  timeInt += mPeriod * PeriodDuration;
145  mTimeInBC.pop_back();
146  mIntBCCache--;
147  return o2::MCInteractionRecord(timeInt, mBCCurrent, mOrbit, mPeriod);
148 }

151 int InteractionSampler::simulateInteractingBC()
152 {
153  // in order not to test random numbers for too many BC's with low mu,
154  // we estimate from very beginning how many BC's happen w/o interaction
155  // Returns number of collisions assigned to selected BC
156  double prob = gRandom->Rndm();
157  while (prob > 0.) { // skip BCs w/o interaction
158  nextCollidingBC(); // pick next interacting bunch
159  prob -= mProbNoInteraction;
160  }
161  // once BC is decided, enforce at least one interaction
162  int ncoll = genPoissonZT();
163  // assign random time withing a bunch
164  for (int i = ncoll; i--;) {
165  mTimeInBC.push_back(gRandom->Gaus(mBCTimeRMS));
166  }
167  if (ncoll > 1) { // sort in DECREASING time order (we are reading vector from the end)
168  std::sort(mTimeInBC.begin(), mTimeInBC.end(), [](const double a, const double b) { return a > b; });
169  }
170  return ncoll;
171 }
