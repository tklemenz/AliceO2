68 TObject* ViewerDevice::receiveDataObjectFromMerger()
69 {
70  TObject* receivedObject;
71  unique_ptr<FairMQMessage> request(NewMessage());
72 
73  if (fChannels.at("data-in").at(0).ReceiveAsync(request) >= 0) {
74  TMessageWrapper tm(request->GetData(), request->GetSize());
75  receivedObject = static_cast<TObject*>(tm.ReadObject(tm.GetClass()));
76  } else {
77  receivedObject = nullptr;
78  }
79 
80  return receivedObject;
