103  static o2::ITSMFT::ClusterShape* PixelsToClusterShape(std::vector<Pixel> v) {
104  if (v.size() == 0) return new o2::ITSMFT::ClusterShape();
105 
106  // To remove the multiple hits problem
107  Pixel p0 = v[0];
108  for (auto p = v.begin()+1; p != v.end(); ++p) {
109  if (fabs(p0.GetRow() - p->GetRow()) > v.size() || fabs(p0.GetCol() - p->GetCol()) > v.size()) {
110  v.erase(p);
111  p--;
112  }
113  }
114 
115  vector<UInt_t> r;
116  vector<UInt_t> c;
117  UInt_t min_r = UINT_MAX, min_c = UINT_MAX;
118  for (auto & p : v) {
119  if (p.GetRow() < min_r) min_r = p.GetRow();
120  if (p.GetCol() < min_c) min_c = p.GetCol();
121  if (find(begin(r), end(r), p.GetRow()) == end(r)) r.push_back(p.GetRow());
122  if (find(begin(c), end(c), p.GetCol()) == end(c)) c.push_back(p.GetCol());
123  }
124 
125  UInt_t rows = r.size();
126  UInt_t cols = c.size();
127  std::vector<UInt_t> pindex;
128  for (auto & p : v) {
129  Int_t rd = p.GetRow() - min_r;
130  Int_t cd = p.GetCol() - min_c;
131  Int_t index = rd * cols + cd;
132  pindex.push_back(index);
133  }
134 
135  return new o2::ITSMFT::ClusterShape(rows, cols, pindex);
136  }
