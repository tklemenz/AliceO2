122 // Poor man variant class. Does not take ownership of anything passed to it.
123 // FIXME: we should really use C++17 std::variant when it
124 // comes about
125 class Variant {
126  using storage_t = std::aligned_union<8, int, int64_t, const char *, float, double, bool>::type;
127 public:
128  Variant(VariantType type = VariantType::Unknown) : mType{type} {}
129 
130  template <typename T> Variant(T value)
131  : mType{variant_trait<T>::type()} {
132  variant_helper<storage_t, decltype(value)>::set(&mStore, value);
133  }
134 
135  template <typename T> Variant(std::initializer_list<T>)
136  : mType{VariantType::Unknown} {
137  static_assert(sizeof(T) == 0,
138  "brace-enclosed initializer list forbidden for Variant"
139  "\n did you accidentally put braces around the default value?");
140  }

142  Variant(const Variant &other)
143  : mType(other.mType)
144  {
145  // In case this is a string we need to duplicate it to avoid
146  // double deletion.
147  if (mType == variant_trait<const char *>::type()) {
148  variant_helper<storage_t, const char *>::set(&mStore, other.get<const char *>());
149  } else {
150  mStore = other.mStore;
151  }
152  }
153 
154  Variant(Variant &&other)
155  : mType(other.mType)
156  {
157  // In case this is a string we need to duplicate it to avoid
158  // double deletion.
159  if (mType == variant_trait<const char *>::type()) {
160  mStore = other.mStore;
161  *reinterpret_cast<char **>(&(other.mStore)) = nullptr;
162  } else {
163  mStore = other.mStore;
164  }
165  }
166 
167  ~Variant() {
168  // In case we allocated a string out of bound, we
169  // should delete it.
170  if (mType == variant_trait<const char *>::type()
171  || mType == variant_trait<char *>::type()
172  ) {
173  free(*reinterpret_cast<void **>(&mStore));
174  }
175  }
176 
177  void operator= ( const Variant &other) {
178  if (mType == variant_trait<const char *>::type()) {
179  variant_helper<storage_t, const char *>::set(&mStore, other.get<const char *>());
180  } else {
181  mStore = other.mStore;
182  }
183  }

185  template<typename T>
186  T get() const {
187  if (mType != variant_trait<T>::type()) {
188  throw std::runtime_error("Mismatch between types");
189  }
190  return variant_helper<storage_t, T>::get(&mStore);
191  }

193  template<typename T>
