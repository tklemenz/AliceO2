79 void LadderSegmentation::createSensors() {

85 
86  Geometry * mftGeom = Geometry::instance();
87 
88  for (Int_t iSensor=0; iSensor<mNSensors; iSensor++) {

93  mftGeom->getLadderID(GetUniqueID()),
94  iSensor);
95

97 
98  new ((*mChips)[iSensor]) ChipSegmentation(*chip);
