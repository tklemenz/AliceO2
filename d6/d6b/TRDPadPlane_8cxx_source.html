170 int TRDPadPlane::GetPadRowNumber(double z) const
171 {
172  //
173  // Finds the pad row number for a given z-position in local supermodule system
174  //
175 
176  int row = 0;
177  int nabove = 0;
178  int nbelow = 0;
179  int middle = 0;
180 
181  if ((z > GetRow0()) || (z < GetRowEnd())) {
182  row = -1;
183 
184  } else {
185  nabove = mNrows + 1;
186  nbelow = 0;
187  while (nabove - nbelow > 1) {
188  middle = (nabove + nbelow) / 2;
189  if (z == (mPadRow[middle - 1] + mPadRowSMOffset)) {
190  row = middle;
191  }
192  if (z > (mPadRow[middle - 1] + mPadRowSMOffset)) {
193  nabove = middle;
194  } else {
195  nbelow = middle;
196  }
197  }
198  row = nbelow - 1;
199  }
200 
201  return row;
202 }
