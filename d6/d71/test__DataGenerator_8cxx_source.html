37 template<typename DistributionType, typename... Args>
38 bool testWithDistribution(Args&&... args)
39 {
40  using value_type = typename DistributionType::result_type;
41  o2::test::DataGenerator<value_type, DistributionType> dg(std::forward<Args>(args)...);
42 
43  std::vector<int> throws(dg.nbins);
44  const unsigned nRolls = 1000000;
45 
46  for (unsigned n = 0; n < nRolls; n++) {
47  value_type v = dg();
48  unsigned bin = v/dg.step - dg.min;
49  BOOST_REQUIRE(bin < dg.nbins);
50  throws[bin]++;
51  }
52 
53  int mostAbundantValueBin = 0;
54  int mostAbundantValueCount = 0;
55  auto highestProbability = dg.getProbability(dg.min);
56  highestProbability = 0;
57  for (auto i : dg) {
58  int bin = i/dg.step - dg.min;
59  BOOST_REQUIRE(bin >= 0);
60  if (mostAbundantValueCount < throws[bin]) {
61  mostAbundantValueBin = bin;
62  mostAbundantValueCount = throws[bin];
63  }
64  if (highestProbability < dg.getProbability(i)) {
65  highestProbability = dg.getProbability(i);
66  }
67  std::cout << std::setw(4) << std::right << i << ": "
68  << std::setw(11) << std::left << dg.getProbability(i)
69  << " -- "
70  << throws[bin]
71  << std::endl;
72  }
73  std::vector<int> mostProbableValueBins;
74  for (auto i : dg) {
75  int bin = i/dg.step - dg.min;
76  if (dg.getProbability(i) >= highestProbability) {
77  mostProbableValueBins.push_back(bin);
78  }
79  }
80  BOOST_CHECK(std::find(mostProbableValueBins.begin(), mostProbableValueBins.end(), mostAbundantValueBin) != mostProbableValueBins.end());
81 
82  return true;
83 }
