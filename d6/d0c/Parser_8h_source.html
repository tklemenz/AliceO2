94  template<typename InputType>
95  int parse(const InputType* buffer, size_t bufferSize,
96  CheckHeaderFct checkHeader,
97  CheckTrailerFct checkTrailer,
98  GetFrameSizeFct getFrameSize,
99  InsertFct insert) {
100  static_assert(sizeof(InputType) == 1,
101  "ForwardParser currently only supports byte type buffer"
102  );
103  if (buffer == nullptr || bufferSize == 0) return 0;
104  auto position = 0;
105  std::vector<FrameInfo> frames;
106  do {
107  FrameInfo entry;
108 
109  // check the header
110  if (sizeof(HeaderType) + position > bufferSize) break;
111  entry.header = reinterpret_cast<const HeaderType*>(buffer + position);
112  if (!checkHeader(*entry.header)) break;
113 
114  // extract frame size from header, this is expected to be the
115  // total frome size including header, payload and optional trailer
116  auto frameSize = getFrameSize(*entry.header);
117  if (frameSize + position > bufferSize) break;
118 
119  // payload starts right after the header
120  entry.payload = reinterpret_cast<typename FrameInfo::PtrT>(entry.header + 1);
121  entry.length = frameSize - totalOffset;
122 
123  // optionally extract and check trailer
124  if (tailOffset > 0) {
125  entry.trailer = nullptr;
126  } else {
127  auto trailerStart = buffer + position + frameSize - tailOffset;
128  entry.trailer = reinterpret_cast<const TrailerType*>(trailerStart);
129  if (!checkTrailer(*entry.trailer)) break;
130  }
131 
132  // store the extracted frame info and continue with remaining buffer
133  frames.emplace_back(entry);
134  position += frameSize;
135  } while (position < bufferSize);
136 
137  if (position == bufferSize) {
138  // frames found and format consistent, insert entries to target
139  // Note: the complete block must be consistent
140  for (auto entry : frames) {
141  if (!insert(entry)) break;
142  }
143  return frames.size();
144  } else if (frames.size() == 0) {
145  // no frames found at all, the buffer does not contain any
146  return 0;
147  }
148 
149  // format error detected
150  // TODO: decide about error policy
151  return -1;
152  }

224  FrameInfo entry;
225 
226  // start from end, extract and check trailer
227  if (sizeof(TrailerType) > position) break;
228  entry.trailer = reinterpret_cast<const TrailerType*>(buffer + position - sizeof(TrailerType));
229  if (!checkTrailer(*entry.trailer)) break;
230 
231  // get the total frame size
232  auto frameSize = getFrameSize(*entry.trailer);
233  if (frameSize > position) break;
234 
235  // extract and check header
236  auto headerStart = buffer + position - frameSize;
237  entry.header = reinterpret_cast<const HeaderType*>(headerStart);
238  if (!checkHeader(*entry.header)) break;
239 
240  // payload immediately after header
241  entry.payload = reinterpret_cast<typename FrameInfo::PtrT>(entry.header + 1);
242  entry.length = frameSize - sizeof(HeaderType) - sizeof(TrailerType);
243  frames.emplace_back(entry);
244  position -= frameSize;
245  } while (position > 0);
246 
247  if (position == 0) {
248  // frames found and format consistent, the complete block must be consistent
249  for (auto entry : frames) {
250  if (!insert(entry)) break;
251  }
252  return frames.size();
253  } else if (frames.size() == 0) {
254  // no frames found at all, the buffer does not contain any
255  return 0;
