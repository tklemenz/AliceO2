96 void MergerDevice::subscribeCustomCommands()
97 {
98  ddsCustomCmd->subscribe([&](const string& command, const string& condition, uint64_t senderId) {
99  stringstream ss;
100  ptree request;
101 
102  ss << command;
103  read_json(ss, request);
104 
105  if (request.get<string>("command") == "check-state") {
106  const ptree response = createCheckStateResponse(request);
107  sendControlResponse(response, to_string(senderId));
108  } else if (request.get<string>("command") == "get-metrics") {

110  sendControlResponse(response, to_string(senderId));
111  }

172 void MergerDevice::handleReceivedDataObject()

206 TObject* MergerDevice::receiveDataObjectFromProducer()
207 {
208  int respondeCode;
209  TObject* receivedDataObject;
210  unique_ptr<FairMQMessage> input(NewMessage());
211 
212  if ((respondeCode = fChannels.at("data-in").at(0).ReceiveAsync(input)) == -2) {
213  if ((respondeCode = fChannels.at("data-in").at(0).ReceiveAsync(input)) == -2) {
214  mLastReceiveBufferOverloadTime = clock();
215  mReceiveBufferOverloaded = true;
216  LOG(DEBUG) << "Buffer of data-in channel is full. Waiting for free buffer...";
217 
218  while ((respondeCode = fChannels.at("data-in").at(0).ReceiveAsync(input)) == -2) {
219  this_thread::sleep_for(chrono::milliseconds(10));
220  }
221 
222  mReceiveBufferOverloaded = false;
223  LOG(DEBUG) << "Buffer was released after " << double(clock() - mLastReceiveBufferOverloadTime) / CLOCKS_PER_SEC
224  << " seconds.";
225  }
226  }
227 
228  if (respondeCode >= 0) {
229  TMessage* message = new TMessageWrapper(input->GetData(), input->GetSize());
230  receivedDataObject = reinterpret_cast<TObject*>(message->ReadObject(message->GetClass()));
231  delete message;
232  } else {
233  LOG(ERROR) << "Received empty message from producer, nothing to merge";
234  receivedDataObject = nullptr;
235  }
236 
237  return receivedDataObject;
238 }

277 string MergerDevice::getVmRSSUsage()
278 {

280 
281  if (procSelfStatus.is_open()) {
282  string line;
283 
284  while (getline(procSelfStatus, line)) {
285  if (line.substr(0, 5) == "VmRSS") {
286  istringstream ss(line);
287  string dump;
288  ss >> dump >> memory;
289  procSelfStatus.seekg(0, ios::beg);
290  break;
291  }
292  }
293  } else {
294  LOG(ERROR) << "Error during opening file /proc/self/status";
295  }
296 
297  return to_string(memory);

300 string MergerDevice::calculateCpuUsage()
301 {
302  auto measureTime = high_resolution_clock::now();
303  auto elapsedTime = std::chrono::duration_cast<std::chrono::microseconds>(measureTime - lastCpuMeasuredTime).count();
304  auto currentCpuClockTime = clock();
305  auto elapsedCpuClockTics = currentCpuClockTime - lastCpuMeasuredValue;
306 
307  lastCpuMeasuredTime = measureTime;
308  lastCpuMeasuredValue = currentCpuClockTime;
309 
310  double normalizeFactor = 1.0 / (elapsedTime / 1000000.0);
311 
312  return to_string(static_cast<int>(round(elapsedCpuClockTics * normalizeFactor)));
313 }
