50  DataRef getByPos(int pos) const {
51  if (pos*2 >= mCache.size() || pos < 0) {
52  throw std::runtime_error("Unknown argument requested at position " + std::to_string(pos));

70  // If we ask for a char const *, we simply point to the payload. Notice this
71  // is meant for C-style strings. If you want to actually get hold of the buffer,
72  // use get<DataRef> (or simply get) as that will give you the size as well.
73  // FIXME: check that the string is null terminated.
74  template <typename T>

89  // DataRef is special. Since there is no point in storing one in a payload,
90  // what it actually does is to return the DataRef used to hold the
