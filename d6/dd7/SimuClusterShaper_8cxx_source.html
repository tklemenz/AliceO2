60 void SimuClusterShaper::FillClusterRandomly() {
61  Int_t matrixSize = mCShape->GetNRows()*mCShape->GetNCols();
62 
63  // generate UNIQUE random numbers
64  UInt_t i = 0, j = 0;
65  auto *bits = new TBits(mNpixOn);
66 
67  if (mFireCenter) {
68  bits->SetBitNumber(mCShape->GetCenterIndex());
69  i++;
70  }
71  while (i < mNpixOn) {
72  j = gRandom->Integer(matrixSize); // [0, matrixSize-1]
73  if (bits->TestBitNumber(j)) continue;
74  bits->SetBitNumber(j);
75  i++;
76  }
77 
78  Int_t bit = 0;
79  for (i = 0; i < mNpixOn; ++i) {
80  j = bits->FirstSetBit(bit);
81  mCShape->AddShapeValue(j);
82  bit = j+1;
83  }
84  delete bits;
85 }

89 void SimuClusterShaper::FillClusterSorted() {
90  UInt_t matrixSize = mCShape->GetNRows()*mCShape->GetNCols();
91  if (matrixSize == 1) {
92  mCShape->AddShapeValue(mCShape->GetCenterIndex());
93  return;
94  }
95 
96  ReComputeCenters();
97 
98  std::map<Double_t, UInt_t> sortedpix;
99  Float_t pX = 0.f, pZ = 0.f;
100 
101  for (UInt_t i = 0; i < matrixSize; ++i) {
102  UInt_t r = i / mCShape->GetNRows();
103  UInt_t c = i % mCShape->GetNRows();
104  UInt_t nx = mHitC - mCShape->GetCenterC() + c;
105  UInt_t nz = mHitR - mCShape->GetCenterR() + r;
106  Segmentation::detectorToLocal(nx, nz, pX, pZ);
107  Double_t d = sqrt(pow(mHitX-pX,2)+pow(mHitZ-pZ,2));
108 
109  // what to do when you reached the border?
110  if (d > 1.) continue;
111  sortedpix[d] = i;
112  }
113 
114  // border case
115  if (sortedpix.size() < mNpixOn) mNpixOn = sortedpix.size();
116  for (std::map<Double_t, UInt_t>::iterator it = sortedpix.begin(); it != std::next(sortedpix.begin(),mNpixOn); ++it) {
117  // std::cout << " " << it->second << std::endl;
118  mCShape->AddShapeValue(it->second);
119  }
120 }

135 void SimuClusterShaper::ReComputeCenters() {
136  UInt_t r = 0, c = 0;
137  Float_t pX = 0.f, pZ = 0.f;
138  Segmentation::detectorToLocal(mHitC, mHitR, pX, pZ);
139 
140  // c is even
141  if (mCShape->GetNCols() % 2 == 0) {
142  if (mHitX > pX) { // n/2 - 1
143  c = mCShape->GetNCols()/2 - 1;
144  } else { // n/2
145  c = mCShape->GetNCols()/2;
146  }
147  } else { // c is odd
148  c = (mCShape->GetNCols()-1)/2;
149  }
150 
151  // r is even
152  if (mCShape->GetNRows() % 2 == 0) {
153  if (mHitZ > pZ) { // n/2 - 1
154  r = mCShape->GetNRows()/2 - 1;
155  } else { // n/2
156  r = mCShape->GetNRows()/2;
157  }
158  } else { // r is odd
159  r = (mCShape->GetNRows()-1)/2;
160  }
161 
162  mCShape->SetCenter(r, c);
