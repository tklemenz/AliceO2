40 Int_t CalibPedestal::updateROC(const Int_t roc, const Int_t row, const Int_t pad,
41  const Int_t timeBin, const Float_t signal)
42 {
43  Int_t adcValue = Int_t(signal);
44  if (timeBin<mFirstTimeBin || timeBin>mLastTimeBin) return 0;
45  if (adcValue<mADCMin || adcValue>mADCMax) return 0;
46 
47  const GlobalPadNumber padInROC = mMapper.getPadNumberInROC(PadROCPos(roc, row, pad));
48  Int_t bin = padInROC * mNumberOfADCs + (adcValue-mADCMin);
49  vectorType& adcVec = *getVector(ROC(roc), kTRUE);
50  ++(adcVec[bin]);
51 
52  //printf("bin: %5d, val: %.2f\n", bin, adcVec[bin]);
53 
54  return 0;

74 void CalibPedestal::analyse()
75 {
76  ROC roc;
77 
78  std::vector<float> fitValues;
79 
80  for (auto& vecPtr : mADCdata) {
81  auto vec = vecPtr.get();
82  if (!vec) {
83  ++roc;
84  continue;
85  }
86 
87  CalROC& calROCPedestal = mPedestal.getCalArray(roc);
88  CalROC& calROCNoise = mNoise.getCalArray(roc);
89 
90  float *array = vec->data();
91 
92  const size_t numberOfPads = (roc.rocType() == RocType::IROC) ? mMapper.getPadsInIROC() : mMapper.getPadsInOROC();
93 
94  for (Int_t ichannel=0; ichannel<numberOfPads; ++ichannel) {
95  size_t offset = ichannel * mNumberOfADCs;
96  fitGaus(mNumberOfADCs, array+offset, float(mADCMin), float(mADCMax+1), fitValues);
97 
98  const float pedestal = fitValues[1];
99  const float noise = fitValues[2];
100  calROCPedestal.setValue(ichannel, pedestal);
101  calROCNoise.setValue(ichannel, noise);
102 
103  //printf("roc: %2d, channel: %4d, pedestal: %.2f, noise: %.2f\n", roc.getRoc(), ichannel, pedestal, noise);
104 
105  }
106 
107  ++roc;
108  }
109 }
