73 Producer* createProducer(const char* type, Args&&... args) {
74  if (std::string(type).compare("TH1F") == 0) {
75  using ProducerType = typename impl::conditional<TH1Producer, 3, Args...>::type;
76  return impl::createProducer<ProducerType>(type, std::forward<Args>(args)...);
77  } else if (std::string(type).compare("TH2F") == 0) {
78  using ProducerType = typename impl::conditional<TH2Producer, 3, Args...>::type;
79  return impl::createProducer<ProducerType>(type, std::forward<Args>(args)...);
80  } else if (std::string(type).compare("TH3F") == 0) {
81  using ProducerType = typename impl::conditional<TH3Producer, 3, Args...>::type;
82  return impl::createProducer<ProducerType>(type, std::forward<Args>(args)...);
83  } else if (std::string(type).compare("THnF") == 0) {
84  using ProducerType = typename impl::conditional<THnProducer, 3, Args...>::type;
85  return impl::createProducer<ProducerType>(type, std::forward<Args>(args)...);
86  } else if (std::string(type).compare("TTree") == 0) {
87  using ProducerType = typename impl::conditional<TreeProducer, 4, Args...>::type;
88  return impl::createProducer<ProducerType>(type, std::forward<Args>(args)...);
89  }
90  LOG(ERROR) << "Unknown type of producer: " << type
91  << ". Possible types are TH1F,TH2F,TH3F,THnF,TTree.";
92  return nullptr;
93 }

107 DataProcessorSpec getRootObjectProducerSpec() {
108  return {
109  "qc_producer",
110  Inputs{},
111  {
112  OutputSpec{"QC", "ROOTOBJECT", 0, OutputSpec::QA}
113  },
114  AlgorithmSpec{
115  [](InitContext &ic) {
116  // get the option from the init context
117  std::shared_ptr<Producer> producer;
118  auto type = ic.options().get<std::string>("objectType");
119  auto name = ic.options().get<std::string>("objectName");
120  auto title = ic.options().get<std::string>("objectTitle");
121  if (type != "TTree") {
122  auto nBins = ic.options().get<int>("nBins");
123  producer.reset(createProducer(type.c_str(), name.c_str(), title.c_str(), nBins));
124  } else {
125  auto nBranches = ic.options().get<int>("nBranches");
126  auto nTreeEntries = ic.options().get<int>("nTreeEntries");
127  producer.reset(createProducer(type.c_str(), name.c_str(), title.c_str(), nBranches, nTreeEntries));
128  }
129 
130  if (!producer) {
131  throw std::runtime_error("failed to create producer instance");
132  }
133 
134  // set up the processing function
135  // using by-copy capture of the worker instance shared pointer
136  // the shared pointer makes sure to clean up the instance when the processing
137  // function gets out of scope
138  auto processingFct = [producer] (ProcessingContext &pc) {
139  pc.allocator().adopt(OutputSpec{"QC", "ROOTOBJECT", 0, OutputSpec::QA},
140  producer->produceData());
141  };
142 
143  // return the actual processing function as a lambda function using variables
144  // of the init function
145  return processingFct;
146  }
147  },
148  Options{
149  {"objectType", VariantType::String, "", {"Type of the produced histogram"}},
150  {"objectName", VariantType::String, "", {"Name of the produced histogram"}},
151  {"objectTitle", VariantType::String, "", {"Title of the produced histogram"}},
152  {"nBins", VariantType::Int, -1, {"Number of bins in histogram"}},
153  {"nBranches", VariantType::Int, -1, {"Number of branches in tree"}},
154  {"nTreeEntries", VariantType::Int, -1, {"Number of entries in tree"}},
155  }
156  };
157 }
