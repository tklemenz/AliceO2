87 bool HwClusterFinder::findCluster()
88 {
89  mTimebinsAfterLastProcessing = 0;
90  int foundNclusters = 0;
91 
92  //
93  // Set region to look in for peaks, max. array size +-2 in both dimensions
94  // to have enough space for 5x5 clusters with peak in inner region.
95  //
96  // pMin pMax
97  // | |
98  // V V
99  // o o o o o o o o <- (mTimebins-1) for itterating through array, mGlobalTimeOfLast
100  // o o o o o o o o
101  // o o|o¯o¯o¯o|o o <- tMax
102  // o o|o o o o|o o
103  // o o|o_o_o_o|o o <- tMin
104  // o o o o o o o o
105  // o o o o o o o o <- 0
106  //
107  //
108 
109  //
110  // In time direction
111  //
112  short tMax = (mTimebins-1) - 2;
113  short tMin = 2;
114 
115  //
116  // In pad direction
117  //
118  int pMin = 2;
119  int pMax = (mPads-1)-2;
120 
121  float qMax;
122  float qTot;
123  float charge;
124  float meanP, meanT;
125  float sigmaP, sigmaT;
126  short minP, minT;
127  short maxP, maxT;
128  short deltaP, deltaT;
129  //
130  // peak finding
131  //
132  short t,p,tt,pp;
133  for (t=tMin; t<=tMax; ++t) {
134  for (p=pMin; p<=pMax; ++p) {
135  //printf("t:%d, p:%d\n",t,p);
136  //
137  // find peak in 3x3 matrix
138  //
139  // ---> pad direction
140  // o o o o o |
141  // o i i i o |
142  // o i C i o V Time direction
143  // o i i i o
144  // o o o o o
145  //
146  if (mData[t ]->at(p).charge < mChargeThreshold) continue;
147 
148  // Require at least one neighboring time bin with signal
149  if (mRequireNeighbouringTimebin && (mData[t-1]->at(p ).charge <= 0 && mData[t+1]->at(p ).charge <= 0)) continue;
150  // Require at least one neighboring pad with signal
151  if (mRequireNeighbouringPad && (mData[t ]->at(p-1).charge <= 0 && mData[t ]->at(p+1).charge <= 0)) continue;
152 
153  // check for local maximum
154  if (mData[t-1]->at(p ).charge >= mData[t]->at(p).charge) continue;
155  if (mData[t+1]->at(p ).charge > mData[t]->at(p).charge) continue;
156  if (mData[t ]->at(p-1).charge >= mData[t]->at(p).charge) continue;
157  if (mData[t ]->at(p+1).charge > mData[t]->at(p).charge) continue;
158  if (mData[t-1]->at(p-1).charge >= mData[t]->at(p).charge) continue;
159  if (mData[t+1]->at(p+1).charge > mData[t]->at(p).charge) continue;
160  if (mData[t+1]->at(p-1).charge > mData[t]->at(p).charge) continue;
161  if (mData[t-1]->at(p+1).charge >= mData[t]->at(p).charge) continue;
162 // printf("##\n");
163 // printf("## cluster found at t=%d, p=%d (in row %d of CRU %d)\n",t,p,mRow,mCRU);
164 // printf("##\n");
165 // printCluster(t,p);
166 
167  //
168  // cluster was found!!
169  //
170 
171  // prepare temp storage
172  for (tt=0; tt<mClusterSizeTime; ++tt) {
173  for (pp=0; pp<mClusterSizePads; ++pp){
174  mTmpCluster[tt][pp] = MiniDigit();
175  }
176  }
177 
178  //
179  // Cluster peak (C) and surrounding inner 3x3 matrix (i) is always
180  // used taken for the found cluster
181  //
182  for (tt=1; tt<4; ++tt) {
183  for (pp=1; pp<4; ++pp) {
184  if ( mRequirePositiveCharge && mData[t+(tt-2)]->at(p+(pp-2)).charge < 0) continue;
185  mTmpCluster[tt][pp] = mData[t+(tt-2)]->at(p+(pp-2));
186 // mData[t+(tt-2)][p+(pp-2)] = 0;
187  }
188  }
189 
190  //
191  // The outer cells of the 5x5 matrix (o) are taken only if the
192  // neighboring inner cell (i) has a signal above threshold.
193  //
194 
195  //
196  // The cells of the "inner cross" have here only 1 neighbour.
197  // [t]
198  // 0 o
199  // 1 i
200  // 2 o i C i o
201  // 3 i
202  // 4 o
203  //
204  // [p] 0 1 2 3 4
205 
206  // o i mTmpCluster[t][p]
207  if(chargeForCluster(mData[t-2]->at(p ).charge, mData[t-1]->at(p ).charge)) mTmpCluster[0][2] = mData[t-2]->at(p ); // t-X -> older
208  if(chargeForCluster(mData[t+2]->at(p ).charge, mData[t+1]->at(p ).charge)) mTmpCluster[4][2] = mData[t+2]->at(p ); // t+X -> newer
209  if(chargeForCluster(mData[t ]->at(p-2).charge, mData[t ]->at(p-1).charge)) mTmpCluster[2][0] = mData[t ]->at(p-2);
210  if(chargeForCluster(mData[t ]->at(p+2).charge, mData[t ]->at(p+1).charge)) mTmpCluster[2][4] = mData[t ]->at(p+2);
211 
212 
213  // The cells of the corners have 3 neighbours.
214  // o o o o
215  // o i i o
216  // C
217  // o i i o
218  // o o o o
219 
220  // bottom left
221  if(chargeForCluster(mData[t+1]->at(p-2).charge,mData[t+1]->at(p-1).charge)) mTmpCluster[3][0] = mData[t+1]->at(p-2);
222  if(chargeForCluster(mData[t+2]->at(p-2).charge,mData[t+1]->at(p-1).charge)) mTmpCluster[4][0] = mData[t+2]->at(p-2);
223  if(chargeForCluster(mData[t+2]->at(p-1).charge,mData[t+1]->at(p-1).charge)) mTmpCluster[4][1] = mData[t+2]->at(p-1);
224  // bottom right
225  if(chargeForCluster(mData[t+2]->at(p+1).charge,mData[t+1]->at(p+1).charge)) mTmpCluster[4][3] = mData[t+2]->at(p+1);
226  if(chargeForCluster(mData[t+2]->at(p+2).charge,mData[t+1]->at(p+1).charge)) mTmpCluster[4][4] = mData[t+2]->at(p+2);
227  if(chargeForCluster(mData[t+1]->at(p+2).charge,mData[t+1]->at(p+1).charge)) mTmpCluster[3][4] = mData[t+1]->at(p+2);
228  // top right
229  if(chargeForCluster(mData[t-1]->at(p+2).charge,mData[t-1]->at(p+1).charge)) mTmpCluster[1][4] = mData[t-1]->at(p+2);
230  if(chargeForCluster(mData[t-2]->at(p+2).charge,mData[t-1]->at(p+1).charge)) mTmpCluster[0][4] = mData[t-2]->at(p+2);
231  if(chargeForCluster(mData[t-2]->at(p+1).charge,mData[t-1]->at(p+1).charge)) mTmpCluster[0][3] = mData[t-2]->at(p+1);
232  // top left
233  if(chargeForCluster(mData[t-2]->at(p-1).charge,mData[t-1]->at(p-1).charge)) mTmpCluster[0][1] = mData[t-2]->at(p-1);
234  if(chargeForCluster(mData[t-2]->at(p-2).charge,mData[t-1]->at(p-1).charge)) mTmpCluster[0][0] = mData[t-2]->at(p-2);
235  if(chargeForCluster(mData[t-1]->at(p-2).charge,mData[t-1]->at(p-1).charge)) mTmpCluster[1][0] = mData[t-1]->at(p-2);
236 
237  //
238  // calculate cluster Properties
239  //
240 
241  qMax = mTmpCluster[2][2].charge;
242  qTot = 0;
243  meanP = 0;
244  meanT = 0;
245  sigmaP = 0;
246  sigmaT = 0;
247  minP = mClusterSizePads;
248  minT = mClusterSizeTime;
249  maxP = 0;
250  maxT = 0;
251  mClusterDigitIndices.emplace_back();
252  for (tt = 0; tt < mClusterSizeTime; ++tt) {
253  deltaT = tt - mClusterSizeTime/2;
254  for (pp = 0; pp < mClusterSizePads; ++pp) {
255  deltaP = pp - mClusterSizePads/2;
256 
257  charge = mTmpCluster[tt][pp].charge;
258  if (charge > 0 || mTmpCluster[tt][pp].event >= 0)
259  mClusterDigitIndices.back().emplace_back(std::make_pair(mTmpCluster[tt][pp].index, mTmpCluster[tt][pp].event));
260 
261  qTot += charge;
262 
263  meanP += charge * static_cast<float>(deltaP);
264  meanT += charge * static_cast<float>(deltaT);
265 
266  sigmaP += charge * static_cast<float>(deltaP)*static_cast<float>(deltaP);
267  sigmaT += charge * static_cast<float>(deltaT)*static_cast<float>(deltaT);
268 
269  if (charge > 0) {
270  minP = std::min(minP,pp); maxP = std::max(maxP,pp);
271  minT = std::min(minT,tt); maxT = std::max(maxT,tt);
272  }
273  }
274  }
275 
276  if (qTot > 0) {
277  meanP /= qTot;
278  meanT /= qTot;
279  sigmaP /= qTot;
280  sigmaT /= qTot;
281 
282  sigmaP = std::sqrt(sigmaP - (meanP*meanP));
283  sigmaT = std::sqrt(sigmaT - (meanT*meanT));
284 
285  meanP += p+mPadOffset;
286  meanT += mGlobalTimeOfLast-(mTimebins-1)+t;
287  }
288 
289  ++foundNclusters;
290  mClusterContainer.emplace_back(mCRU, mRow, qTot, qMax, meanP, sigmaP, meanT, sigmaT);
291 
292  if (mAssignChargeUnique) {
293  if (p < (pMin+4)) {
294  // If the cluster peak is in one of the 6 leftmost pads, the Cluster Finder
295  // on the left has to know about it to ignore the already used pads.
296  if (auto next = mNextCF.lock()) {
297  next->clusterAlreadyUsed(t,p+mPadOffset);//,mTmpCluster);
298  }
299  }
300 
301 
302  //
303  // subtract found cluster from storage
304  //
305  // TODO: really nexessary?? or just set to 0
306  for (tt=0; tt<5; ++tt) {
307  for (pp=0; pp<5; ++pp) {
308  //mData[t+(tt-2)][p+(pp-2)].charge -= mTmpCluster[tt][pp];
309  mData[t+(tt-2)]->at(p+(pp-2)).clear();
310  }
311  }
312  }
313  }
314  }
315 
316  if (foundNclusters > 0) return true;
317  return false;
318 }
