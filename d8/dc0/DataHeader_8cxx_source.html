237 void o2::header::hexDump (const char* desc, const void* voidaddr, size_t len, size_t max)
238 {
239  size_t i;
240  unsigned char buff[17]; // stores the ASCII data
241  memset(&buff[0],'\0',17);
242  const byte* addr = reinterpret_cast<const byte*>(voidaddr);
243 
244  // Output description if given.
245  if (desc != nullptr)
246  printf ("%s, ", desc);
247  printf("%zu bytes:", len);
248  if (max>0 && len>max) {
249  len = max; //limit the output if requested
250  printf(" output limited to %zu bytes\n", len);
251  } else {
252  printf("\n");
253  }
254 
255  // In case of null pointer addr
256  if (addr==nullptr) {printf(" nullptr, size: %zu\n", len); return;}
257 
258  // Process every byte in the data.
259  for (i = 0; i < len; i++) {
260  // Multiple of 16 means new line (with line offset).
261  if ((i % 16) == 0) {
262  // Just don't print ASCII for the zeroth line.
263  if (i != 0)
264  printf (" %s\n", buff);
265 
266  // Output the offset.
267  //printf (" %04x ", i);
268  printf (" %p ", &addr[i]);
269  }
270 
271  // Now the hex code for the specific character.
272  printf (" %02x", addr[i]);
273 
274  // And store a printable ASCII character for later.
275  if ((addr[i] < 0x20) || (addr[i] > 0x7e))
276  buff[i % 16] = '.';
277  else
278  buff[i % 16] = addr[i];
279  buff[(i % 16) + 1] = '\0';
280  fflush(stdout);
281  }
282 
283  // Pad out last line if not exactly 16 characters.
284  while ((i % 16) != 0) {
285  printf (" ");
286  fflush(stdout);
287  i++;
288  }
289 
290  // And print the final ASCII bit.
291  printf (" %s\n", buff);
292  fflush(stdout);
293 }
