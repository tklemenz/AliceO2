98 int SystemInterface::initSystem(unsigned long runNo)
99 {
101  int iResult = 0;
102 
103  string libraryPath = ALIHLTANALYSIS_INTERFACE_LIBRARY;
104 
105  void* libHandle = dlopen(libraryPath.c_str(), RTLD_NOW);
106  if (!libHandle) {
107  cerr << "error: can not load library " << libraryPath.c_str() << endl;
108 #ifdef __APPLE__
109  int returnvalue = -EFTYPE;
110 #else
111  int returnvalue = -ELIBACC;
112 #endif
113 
114  return returnvalue;
115  }
116 
117  AliHLTAnalysisFctGetInterfaceCall fctGetSystemCall =
118  (AliHLTAnalysisFctGetInterfaceCall)dlsym(libHandle, ALIHLTANALYSIS_FCT_GETINTERFACECALL);
119  if (!fctGetSystemCall) {
120  cerr << "error: can not find function '" << ALIHLTANALYSIS_FCT_GETINTERFACECALL << "' in " << libraryPath.c_str()
121  << endl;
122  return -ENOSYS;
123  }
124 
125  const char** arrayCalls = gInterfaceCallSignatures;
126  for (int i = 0; arrayCalls[i] != nullptr; i++) {
127  AliHLTExtFctInitSystem call = (AliHLTExtFctInitSystem) (*fctGetSystemCall)(arrayCalls[i]);
128  if (call == nullptr) {
129  cerr << "error: can not find function signature '" << arrayCalls[i] << "' in " << libraryPath.c_str() << endl;
130  } else {
131  cout << "function '" << arrayCalls[i] << "' loaded from " << libraryPath.c_str() << endl;
132  switch (i) {
133  case 0:
134  mpAliHLTExtFctInitSystem = (AliHLTExtFctInitSystem)call;
135  break;
136  case 1:
137  mpAliHLTExtFctDeinitSystem = (AliHLTExtFctDeinitSystem)call;
138  break;
139  case 2:
140  mpAliHLTExtFctLoadLibrary = (AliHLTExtFctLoadLibrary)call;
141  break;
142  case 3:
143  mpAliHLTExtFctUnloadLibrary = (AliHLTExtFctUnloadLibrary)call;
144  break;
145  case 4:
146  mpAliHLTExtFctCreateComponent = (AliHLTExtFctCreateComponent)call;
147  break;
148  case 5:
149  mpAliHLTExtFctDestroyComponent = (AliHLTExtFctDestroyComponent)call;
150  break;
151  case 6:
152  mpAliHLTExtFctProcessEvent = (AliHLTExtFctProcessEvent)call;
153  break;
154  case 7:
155  mpAliHLTExtFctGetOutputDataType = (AliHLTExtFctGetOutputDataType)call;
156  break;
157  case 8:
158  mpAliHLTExtFctGetOutputSize = (AliHLTExtFctGetOutputSize)call;
159  break;
160  default:
161  cerr << "error: number of function signatures does not match expected number of functions" << endl;
162  }
163  }
164  }
165 
166  if (mpAliHLTExtFctInitSystem) {
167  if ((iResult = (*mpAliHLTExtFctInitSystem)(ALIHLT_DATA_TYPES_VERSION, &mEnvironment, runNo, nullptr)) != 0) {
168  cerr << "error: AliHLTAnalysisInitSystem failed with error " << iResult << endl;
169  return -ENOSYS;
170  }
171  }
172 
173  return 0;
174 }
