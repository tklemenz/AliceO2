58 
59  const std::vector<o2::ITSMFT::Digit> *arr =
60  mgr->InitObjectAs<const std::vector<o2::ITSMFT::Digit> *>("ITSDigit");
61  if (!arr) {
62  LOG(ERROR)<<"ITS digits not registered in the FairRootManager. Exiting ..."<<FairLogger::endl;
63  return kERROR;
64  }
65  mReader.setDigitArray(arr);
66 
67  // Register output container
68  mgr->RegisterAny("ITSCluster", mClustersArray, kTRUE);

70  // Register MC Truth container
71  if (mClsLabels)

74  GeometryTGeo* geom = GeometryTGeo::Instance();

76  mGeometry = geom;
