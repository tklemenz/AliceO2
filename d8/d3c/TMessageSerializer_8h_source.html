84 inline void TMessageSerializer::serialize(FairTMessage& tm, const TObject* input, CacheStreamers streamers,
85  CompressionLevel compressionLevel)
86 {
87  if (streamers == CacheStreamers::yes) {
88  tm.EnableSchemaEvolution(true);
89  }
90 
91  if (compressionLevel >= 0) {
92  // if negative, skip to use ROOT default
93  tm.SetCompressionLevel(compressionLevel);
94  }
95 
96  tm.WriteObject(input);
97 
98  if (streamers == CacheStreamers::yes) {
99  updateStreamers(tm, sStreamers);
100  }
101 }

115 inline void TMessageSerializer::Serialize(FairMQMessage& msg, const TObject* input,
116  TMessageSerializer::CacheStreamers streamers,
117  TMessageSerializer::CompressionLevel compressionLevel)
118 {
119  std::unique_ptr<FairTMessage> tm = std::make_unique<FairTMessage>(kMESS_OBJECT);
120 
121  serialize(*tm, input, streamers, compressionLevel);
122 
123  msg.Rebuild(tm->Buffer(), tm->BufferSize(), FairTMessage::free, tm.get());
124  tm.release();
125 }
