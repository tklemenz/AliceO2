24 class ServiceRegistry
25 {
26 public:
27  // Register a service for the given interface T
28  // with actual implementation C, i.e. C is derived from T.
29  // Only one instance of type C can be registered per type T.
30  // The fact we use a bare pointer indicates that the ownership
31  // of the service still belongs to whatever created it, and is
32  // not passed to the registry. It's therefore responsibility of
33  // the creator of the service to properly dispose it.
34  template<class I, class C>
35  void registerService(C *service) {
36  // This only works for concrete implementations of the type T.
37  // We need type elision as we do not want to know all the services in
38  // advance
39  static_assert(std::is_base_of<I, C>::value == true,
40  "Registered service is not derived from declared interface");
41  mServices[typeid(I)] = reinterpret_cast<ServicePtr>(service);
42  }
43 
44  // Get a service for the given interface T. The returned reference
45  // exposed to the user.
46  // is actually of the last concrete type C, however this should not be
47  template<typename T>
48  T &get() {
49  auto service = mServices.find(typeid(T));
50  if (service == mServices.end()) {
51  throw std::runtime_error(std::string("Unable to find service of kind ") +
52  typeid(T).name() +
53  "did you register one?");
54  }
55  return *reinterpret_cast<T*>(service->second);
56  }
57 private:
58  using TypeInfoRef = std::reference_wrapper<const std::type_info>;
59  using ServicePtr = void *;
60  struct Hasher {
61  std::size_t operator()(TypeInfoRef code) const
62  {
63  return code.get().hash_code();
64  }
65  };
66  struct EqualTo {
67  bool operator()(TypeInfoRef lhs, TypeInfoRef rhs) const
68  {
69  return lhs.get() == rhs.get();
70  }
71  };
72  std::unordered_map<TypeInfoRef, ServicePtr, Hasher, EqualTo> mServices;
73 };
