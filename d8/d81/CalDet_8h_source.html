113  // bind row and pad to the maximum rows and pads in the requested region
114  const size_t nRows = mapper.getNumberOfRowsROC(roc);
115  const size_t mappedRow = row % nRows;
116  const size_t nPads = mapper.getNumberOfPadsInRowROC(roc, row);
117  const size_t mappedPad = pad % nPads;

120  // TODO: implement missing cases

139 template <class T>
140 inline const T CalDet<T>::getValue(const CRU cru, const size_t row, const size_t pad) const
141 {
142  // TODO: might need speedup and beautification
143  static const Mapper& mapper = Mapper::instance();
144  const auto& info = mapper.getPadRegionInfo(cru.region());
145 
146  // bind row and pad to the maximum rows and pads in the requested region
147  const size_t nRows = info.getNumberOfPadRows();
148  const size_t mappedRow = row % nRows;
149  const size_t nPads = info.getPadsInRowRegion(mappedRow);
150  const size_t mappedPad = pad % nPads;
151 
152  switch (mPadSubset) {
153  case PadSubset::ROC: {
154  const ROC roc = cru.roc();
155  const size_t irocOffset = (cru.rocType()==RocType::IROC)?0: mapper.getNumberOfRowsROC(0);
156  const size_t rowROC = mappedRow + info.getGlobalRowOffset() - irocOffset;
157  const size_t channel = mapper.getPadNumberInROC(PadROCPos(roc, rowROC, mappedPad));
158  //printf("roc %2d, row %2zu, pad %3zu, channel: %3zu\n", roc.getRoc(), rowROC, mappedPad, channel);
159  return mData[roc].getValue(channel);
160  break;
161  }
162  case PadSubset::Partition: {
163  break;
164  }
165  case PadSubset::Region: {
166  return mData[cru].getValue(mappedRow, mappedPad);
167  break;
168  }
169  }
170  return T{};
171 }

295 void CalDet<T>::initData()
296 {
297  const auto& mapper = Mapper::instance();
298 
299  // ---| Define number of sub pad regions |------------------------------------
300  size_t size = 0;
301 
302  switch (mPadSubset) {
303  case PadSubset::ROC: {
304  size = ROC::MaxROC;
305  break;
306  }
307  case PadSubset::Partition: {
308  size = Sector::MAXSECTOR * mapper.getNumberOfPartitions();
309  break;
310  }
311  case PadSubset::Region: {
312  size = Sector::MAXSECTOR * mapper.getNumberOfPadRegions();
313  break;
314  }
315  }
316 
317  for (size_t i=0; i<size; ++i) {
318  mData.push_back(CalType(mPadSubset, i));
319  }
320 }
