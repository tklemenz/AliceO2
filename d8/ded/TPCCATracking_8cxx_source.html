73 int TPCCATracking::convertClusters(TChain* inputClustersChain, const std::vector<o2::TPC::Cluster>* inputClustersArray, ClusterNativeAccessFullTPC& outputClusters, std::unique_ptr<ClusterNative[]>& clusterMemory) {
74  if (inputClustersChain && inputClustersArray) {
75  LOG(FATAL) << "Internal error, must not pass in both TChain and TClonesArray of clusters\n";
76  }
77  int numChunks;
78  if (inputClustersChain) {
79  inputClustersChain->SetBranchAddress("TPCClusterHW", &inputClustersArray);
80  numChunks = inputClustersChain->GetEntries();
81  } else {
82  numChunks = 1;
83  }
84 
85  Mapper& mapper = Mapper::instance();
86  for (int iter = 0;iter < 2;iter++)
87  {
88  for (int i = 0;i < Sector::MAXSECTOR;i++) {
89  for (int j = 0;j < Constants::MAXGLOBALPADROW;j++) {
90  outputClusters.nClusters[i][j] = 0;
91  }
92  }
93 
94  unsigned int nClusters = 0;
95  for (int iChunk = 0;iChunk < numChunks;iChunk++) {
96  if (inputClustersChain) {
97  inputClustersChain->GetEntry(iChunk);
98  }
99  for (int icluster = 0; icluster < inputClustersArray->size(); ++icluster) {
100  const auto& cluster = (*inputClustersArray)[icluster];
101  const CRU cru(cluster.getCRU());
102  const Sector sector = cru.sector();
103  const PadRegionInfo& region = mapper.getPadRegionInfo(cru.region());
104  const int rowInSector = cluster.getRow() + region.getGlobalRowOffset();
105 
106  if (iter == 1) {
107  ClusterNative& oCluster = outputClusters.clusters[sector][rowInSector][outputClusters.nClusters[sector][rowInSector]];
108  oCluster.setTimeFlags(cluster.getTimeMean(), 0);
109  oCluster.setPad(cluster.getPadMean());
110  oCluster.setSigmaTime(cluster.getTimeSigma());
111  oCluster.setSigmaPad(cluster.getPadSigma());
112  oCluster.qTot = cluster.getQmax();
113  oCluster.qMax = cluster.getQ();
114  }
115  outputClusters.nClusters[sector][rowInSector]++;
116  }
117  nClusters += inputClustersArray->size();
118  }
119  if (iter == 0)
120  {
121  clusterMemory.reset(new ClusterNative[nClusters]);
122  unsigned int pos = 0;
123  for (int i = 0;i < Sector::MAXSECTOR;i++) {
124  for (int j = 0;j < Constants::MAXGLOBALPADROW;j++) {
125  outputClusters.clusters[i][j] = &clusterMemory[pos];
126  pos += outputClusters.nClusters[i][j];
127  }
128  }
129  }
130  }
131  for (int i = 0;i < Sector::MAXSECTOR;i++) {
132  for (int j = 0;j < Constants::MAXGLOBALPADROW;j++) {
133  std::sort(outputClusters.clusters[i][j], outputClusters.clusters[i][j] + outputClusters.nClusters[i][j], ClusterNativeContainer::sortComparison);
134  }
135  }
136 
137  return(0);
138 }

140 int TPCCATracking::runTracking(const ClusterNativeAccessFullTPC& clusters, std::vector<TrackTPC>* outputTracks, MCLabelContainer* outputTracksMCTruth) {
141  const static ParameterDetector &detParam = ParameterDetector::defaultInstance();
142  const static ParameterGas &gasParam = ParameterGas::defaultInstance();
143  const static ParameterElectronics &elParam = ParameterElectronics::defaultInstance();
144 
145  int nClusters[Sector::MAXSECTOR] = {};
146  int nClustersConverted = 0;
147  int nClustersTotal = 0;
148 
149 
150  Mapper& mapper = Mapper::instance();
151  for (int i = 0; i< Sector::MAXSECTOR;i++) {
152  for (int j = 0;j < Constants::MAXGLOBALPADROW;j++) {
153  if (clusters.nClusters[i][j] > 0xFFFF) {
154  LOG(ERROR) << "Number of clusters in sector " << i << " row " << j << " exceeds 0xFFFF, which is currently a hard limit of the wrapper for the tracking!\n";
155  return(1);
156  }
157  nClusters[i] += clusters.nClusters[i][j];
158  }
159  nClustersTotal += nClusters[i];
160 
161  mClusterData[i].StartReading(i, nClusters[i]);
162 
163  for (int j = 0;j < Constants::MAXGLOBALPADROW;j++) {
164  Sector sector = i;
165  int regionNumber = 0;
166  while (j > mapper.getGlobalRowOffsetRegion(regionNumber) + mapper.getNumberOfRowsRegion(regionNumber)) regionNumber++;
167  CRU cru(sector, regionNumber);
168  const PadRegionInfo& region = mapper.getPadRegionInfo(cru.region());
169  for (int k = 0;k < clusters.nClusters[i][j];k++) {
170  const ClusterNative& cluster = clusters.clusters[i][j][k];
171  AliHLTTPCCAClusterData& cd = mClusterData[i];
172  AliHLTTPCCAClusterData::Data& hltCluster = cd.Clusters()[cd.NumberOfClusters()];
173 
174  const float padY = cluster.getPad();
175  const int padNumber = int(padY);
176  const GlobalPadNumber pad = mapper.globalPadNumber(PadPos(j, padNumber));
177  const PadCentre& padCentre = mapper.padCentre(pad);
178  const float localY = padCentre.Y() - (padY - padNumber - 0.5) * region.getPadWidth();
179  const float localYfactor = (cru.side() == Side::A) ? -1.f : 1.f;
180  float zPositionAbs = cluster.getTime()*elParam.getZBinWidth() * gasParam.getVdrift();
181  if (!mTrackingCAO2Interface->GetParamContinuous())
182  zPositionAbs = detParam.getTPClength() - zPositionAbs;
183  else
184  zPositionAbs = sContinuousTFReferenceLength * elParam.getZBinWidth() * gasParam.getVdrift() - zPositionAbs;
185 
186  // sanity checks
187  if (zPositionAbs < 0 || (!mTrackingCAO2Interface->GetParamContinuous() && zPositionAbs > detParam.getTPClength())) {
188  LOG(INFO) << "Removing cluster " << i << " time: " << cluster.getTime() << ", abs. z: " << zPositionAbs << "\n";
189  continue;
190  }
191 
192  hltCluster.fX = padCentre.X();
193  hltCluster.fY = localY * (localYfactor);
194  hltCluster.fZ = zPositionAbs * (-localYfactor);
195  hltCluster.fRow = j;
196  hltCluster.fAmp = cluster.qMax;
197  hltCluster.fId = (i << 24) | (j << 16) | (k);
198 
199  cd.SetNumberOfClusters(cd.NumberOfClusters() + 1);
200  nClustersConverted++;
201  }
202  }
203  }
204 
205  if (nClustersTotal != nClustersConverted) {
206  LOG(INFO) << "Passed " << nClustersConverted << " (out of " << nClustersTotal << ") clusters to CA tracker\n";
207  }
208 
209  const AliHLTTPCGMMergedTrack* tracks;
210  int nTracks;
211  const AliHLTTPCGMMergedTrackHit* trackClusters;
212  int retVal = mTrackingCAO2Interface->RunTracking(mClusterData, tracks, nTracks, trackClusters);
213  if (retVal == 0)
214  {
215  std::vector<std::pair<int, float>> trackSort(nTracks);
216  int tmp = 0, tmp2 = 0;
217  for (char side = 0;side < 2;side++)
218  {
219  for (int i = 0;i < nTracks;i++)
220  {
221  if (tracks[i].OK() && tracks[i].Side() == side) trackSort[tmp++] = {i, (side == 0 ? -1.f : 1.f) * tracks[i].GetParam().GetZOffset()};
222  }
223  std::sort(trackSort.data() + tmp2, trackSort.data() + tmp, [](const auto& a, const auto& b) {
224  return(a.second > b.second);
225  });
226  tmp2 = tmp;
227  if (side == 0) mNTracksASide = tmp;
228  }
229  nTracks = tmp;
230 
231  outputTracks->resize(nTracks);
232  for (int iTmp = 0; iTmp < nTracks; iTmp++) {
233  auto& oTrack = (*outputTracks)[iTmp];
234  const int i = trackSort[iTmp].first;
235  oTrack = TrackTPC(
236  tracks[i].GetParam().GetX(), tracks[i].GetAlpha(),
237  { tracks[i].GetParam().GetY(), tracks[i].GetParam().GetZ(), tracks[i].GetParam().GetSinPhi(),
238  tracks[i].GetParam().GetDzDs(), tracks[i].GetParam().GetQPt() },
239  { tracks[i].GetParam().GetCov(0), tracks[i].GetParam().GetCov(1), tracks[i].GetParam().GetCov(2),
240  tracks[i].GetParam().GetCov(3), tracks[i].GetParam().GetCov(4), tracks[i].GetParam().GetCov(5),
241  tracks[i].GetParam().GetCov(6), tracks[i].GetParam().GetCov(7), tracks[i].GetParam().GetCov(8),
242  tracks[i].GetParam().GetCov(9), tracks[i].GetParam().GetCov(10), tracks[i].GetParam().GetCov(11),
243  tracks[i].GetParam().GetCov(12), tracks[i].GetParam().GetCov(13), tracks[i].GetParam().GetCov(14) });
244 
245  if (!mTrackingCAO2Interface->GetParamContinuous())
246  oTrack.setTime0(0);
247  else
248  oTrack.setTime0(sContinuousTFReferenceLength - (tracks[i].Side() ? 1.f : -1.f) * tracks[i].GetParam().GetZOffset() / (elParam.getZBinWidth() * gasParam.getVdrift()));
249  oTrack.setLastClusterZ(trackClusters[tracks[i].FirstClusterRef()].fZ - tracks[i].GetParam().GetZOffset());
250  oTrack.resetClusterReferences(tracks[i].NClusters());
251  std::vector<std::pair<MCCompLabel, unsigned int>> labels;
252  for (int j = 0; j < tracks[i].NClusters(); j++) {
253  int clusterId = trackClusters[tracks[i].FirstClusterRef() + j].fId;
254  Sector sector = clusterId >> 24;
255  int globalRow = (clusterId >> 16) & 0xFF;
256  clusterId &= 0xFFFF;
257  const ClusterNative& cl = clusters.clusters[sector][globalRow][clusterId];
258  int regionNumber = 0;
259  while (globalRow > mapper.getGlobalRowOffsetRegion(regionNumber) + mapper.getNumberOfRowsRegion(regionNumber)) regionNumber++;
260  CRU cru(sector, regionNumber);
261  oTrack.addCluster(Cluster(cru, globalRow - mapper.getGlobalRowOffsetRegion(regionNumber), cl.qTot, cl.qMax, cl.getPad(), cl.getSigmaPad(), cl.getTime(), cl.getSigmaTime()));
262  oTrack.setClusterReference(j, sector, globalRow, clusterId);
263  if (outputTracksMCTruth) {
264  for (const auto& element : clusters.clustersMCTruth[sector][globalRow]->getLabels(clusterId)) {
265  bool found = false;
266  for (int l = 0;l < labels.size();l++) {
267  if (labels[l].first == element) {
268  labels[l].second++;
269  found = true;
270  break;
271  }
272  }
273  if (!found) labels.emplace_back(element, 1);
274  }
275  }
276  }
277  if (outputTracksMCTruth)
278  {
279  int bestLabelNum = 0, bestLabelCount = 0;
280  for (int j = 0;j < labels.size();j++) {
281  if (labels[j].second > bestLabelCount)
282  {
283  bestLabelNum = j;
284  bestLabelCount = labels[j].second;
285  }
286  }
287  MCCompLabel& bestLabel = labels[bestLabelNum].first;
288  if (bestLabelCount < sTrackMCMaxFake * tracks[i].NClusters()) bestLabel.set(-bestLabel.getTrackID(), bestLabel.getEventID(), bestLabel.getSourceID());
289  outputTracksMCTruth->addElement(iTmp, bestLabel);
290  }
291  int lastSector = trackClusters[tracks[i].FirstClusterRef() + tracks[i].NClusters() - 1].fId >> 24;
292  oTrack.setSide((Side) tracks[i].Side());
293  }
294  }
295  mTrackingCAO2Interface->Cleanup();
296  return (retVal);
297 }
