62 InitStatus ClustererTask::Init()
63 {
64  LOG(DEBUG) << "Enter Initializer of ClustererTask" << FairLogger::endl;
65 
66  FairRootManager *mgr = FairRootManager::Instance();
67  if( !mgr ) {
68 
69  LOG(ERROR) << "Could not instantiate FairRootManager. Exiting ..." << FairLogger::endl;
70  return kERROR;
71  }
72 
73  mDigitsArray = mgr->InitObjectAs<decltype(mDigitsArray)>("TPCDigit");
74  if( !mDigitsArray ) {
75  LOG(ERROR) << "TPC points not registered in the FairRootManager. Exiting ..." << FairLogger::endl;
76  return kERROR;
77  }
78 
79  mDigitMCTruthArray = mgr->InitObjectAs<decltype(mDigitMCTruthArray)>("TPCDigitMCTruth");
80  if( !mDigitMCTruthArray ) {
81  LOG(ERROR) << "TPC MC Truth not registered in the FairRootManager. Exiting ..." << FairLogger::endl;
82  return kERROR;
83  }
84 
85  if (mBoxClustererEnable) {
86 
87  // Create and register output container
88  mClustersArray = new std::vector<o2::TPC::Cluster>;
89  mgr->RegisterAny("TPCCluster", mClustersArray, kTRUE);
90 
91  // Register MC Truth output container
92  mClustersMCTruthArray = std::make_unique<MCLabelContainer>();
93  // a trick to register the unique pointer with FairRootManager
94  static auto tmp = mClustersMCTruthArray.get();
95  mgr->RegisterAny("TPCClusterMCTruth", tmp, kTRUE);
96 
97  // create clusterer and pass output pointer
98  mBoxClusterer = std::make_unique<BoxClusterer>(mClustersArray);
99  }
100 
101  if (mHwClustererEnable) {
102  // Register output container
103  mHwClustersArray = new std::vector<o2::TPC::Cluster>;
104  mgr->RegisterAny("TPCClusterHW", mHwClustersArray, kTRUE);
105 
106  // Register MC Truth output container

111 
112  // create clusterer and pass output pointer

124 void ClustererTask::Exec(Option_t *option)
125 {
126  LOG(DEBUG) << "Running clusterization on new event with " << mDigitsArray->size() << " digits" << FairLogger::endl;
127 
128  if (mBoxClustererEnable) {
129  mClustersArray->clear();
130  if(mClustersMCTruthArray) mClustersMCTruthArray->clear();
131  mBoxClusterer->Process(*mDigitsArray,mDigitMCTruthArray,mEventCount);
132  LOG(DEBUG) << "Box clusterer found " << mClustersArray->size() << " clusters" << FairLogger::endl << FairLogger::endl;
133  }
134 
135  if (mHwClustererEnable) {
136  if(mHwClustersArray) mHwClustersArray->clear();
137  if(mHwClustersMCTruthArray) mHwClustersMCTruthArray->clear();
138  mHwClusterer->Process(*mDigitsArray,mDigitMCTruthArray,mEventCount);
139  LOG(DEBUG) << "Hw clusterer found " << mHwClustersArray->size() << " clusters" << FairLogger::endl << FairLogger::endl;
140  }
141  ++mEventCount;
142 }
