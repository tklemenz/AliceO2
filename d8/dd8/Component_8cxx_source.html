93 int Component::init(int argc, char** argv)
94 {
96 
97  // the hidden options are not exposed to the outside
98  bpo::options_description od("component options");
99  od.add_options()
100  (OptionKeys[OptionKeyInstanceId],
101  bpo::value<string>()->required(),
102  "internal instance id");
103  // now add all the visible options
104  od.add(GetOptionsDescription());
105 
106  // HLT components are implemented in shared libraries, the library name
107  // and component id are used to factorize a component
108  // optionally, a list of configuration parameters can be specified as
109  // one single string which is translated in an array of string in the
110  // argc/argv format
111  string componentParameter;
112  string instanceId;
113 
114  // the configuration and calibration is fixed for every run and identified
115  // by the run no
116  int runNumber = 0;
117 
118  bpo::variables_map varmap;
119  bpo::store(bpo::parse_command_line(argc, argv, od), varmap);
120 
121  for (int option = 0; option < OptionKeyLast; ++option) {
122  if (varmap.count(OptionKeys[option]) == 0) continue;
123  switch (option) {
124  case OptionKeyLibrary: break;
125  case OptionKeyComponent: break;
126  case OptionKeyParameter:
127  componentParameter = varmap[OptionKeys[option]].as<string>();
128  break;
129  case OptionKeyRun:
130  stringstream(varmap[OptionKeys[option]].as<string>()) >> runNumber;
131  break;
132  case OptionKeyOCDB:
133  if (getenv("ALIHLT_HCDBDIR") != nullptr) {
134  LOG(WARN) << "overriding value of ALICEHLT_HCDBDIR by --ocdb command option";
135  }
136  setenv("ALIHLT_HCDBDIR", varmap[OptionKeys[option]].as<string>().c_str(), 1);
137  break;
138  case OptionKeyMsgsize: {
139  unsigned size = 0;
140  stringstream(varmap[OptionKeys[option]].as<string>()) >> size;
141  mOutputBuffer.resize(size);
142  } break;
143  case OptionKeyOutputMode: {
144  unsigned mode;
145  stringstream(varmap[OptionKeys[option]].as<string>()) >> mode;
146  mFormatHandler.setOutputMode(mode);
147  } break;
148  case OptionKeyInstanceId: break;
149  instanceId = varmap[OptionKeys[option]].as<string>();
150  break;
151  }
152  }
153 
154  // TODO: this can be a loop over all required options
155  // probably we won't come here anyhow as the parser will detect
156  // the absence of a required parameter, check and handle the parser
157  // exception
158  if (varmap.count(OptionKeys[OptionKeyLibrary]) == 0 ||
159  varmap.count(OptionKeys[OptionKeyComponent]) == 0 ||
160  runNumber < 0) {
161  LOG(ERROR) << "missing argument, required options: library, component,run";
162  return -EINVAL;
163  }
164 
165  // check the OCDB URI
166  // the HLT code relies on ALIHLT_HCDBDIR environment variable to be set
167  if (!getenv("ALIHLT_HCDBDIR")) {
168  LOG(ERROR) << "FATAL: OCDB URI is needed, use option --ocdb or environment variable ALIHLT_HCDBDIR";
169 // temporary fix to regain compilation on MacOS (which on some platforms does not define ENOKEY)
170 #ifndef ENOKEY
171 #define ENOKEY 126
172 #endif
173  return -ENOKEY;
174  }
175 
176  int iResult = 0;
177  // TODO: make the SystemInterface a singleton
178  unique_ptr<o2::alice_hlt::SystemInterface> iface(new SystemInterface);
179  if (iface.get() == nullptr || ((iResult = iface->initSystem(runNumber))) < 0) {
180  // LOG(ERROR) << "failed to set up SystemInterface " << iface.get() << " (" << iResult << ")";
181  return -ENOSYS;
182  }
183 
184  // basic initialization succeeded, make the instances persistent
185  mpSystem = iface.release();
186 
187  // load the component library
188  if ((iResult = mpSystem->loadLibrary(varmap[OptionKeys[OptionKeyLibrary]].as<string>().c_str())) != 0)
189  return iResult > 0 ? -iResult : iResult;
190 
191  // chop the parameter string in order to provide parameters in the argc/argv format
192  vector<const char*> parameters;
193  unsigned parameterLength = componentParameter.length();
194  unique_ptr<char> parameterBuffer(new char[parameterLength + 1]);
195  if (parameterLength > 0 && parameterBuffer.get() != nullptr) {
196  strcpy(parameterBuffer.get(), componentParameter.c_str());
197  char* iterator = parameterBuffer.get();
198  parameters.emplace_back(iterator);
199  for (; *iterator != 0; iterator++) {
200  if (*iterator != ' ') continue;
201  *iterator = 0; // separate strings
202  if (*(iterator + 1) != ' ' && *(iterator + 1) != 0) parameters.emplace_back(iterator + 1);
203  }
204  }
205 
206  // create component
207  string description;
208  description+=" chainid=" + instanceId;
209  if ((iResult=mpSystem->createComponent(varmap[OptionKeys[OptionKeyComponent]].as<string>().c_str(), nullptr, parameters.size(), &parameters[0], &mProcessor, description.c_str()))<0) {
210  // the ALICE HLT external interface uses the following error definition
211  // 0 success
212  // >0 error number
213  return iResult > 0 ? -iResult : iResult;
214  }
215 
216  mEventCount = 0;
217 
218  return iResult;
219 }

221 int Component::process(vector<MessageFormat::BufferDesc_t>& dataArray,
222  cballoc_signal_t* cbAllocate)
223 {
224  if (!mpSystem) return -ENOSYS;
225  int iResult = 0;
226 
227  unsigned outputBufferSize = 0;
228 
229  AliHLTComponentEventData evtData;
230  memset(&evtData, 0, sizeof(evtData));
231  evtData.fStructSize = sizeof(evtData);
232  if (mEventCount >= 0) {
233  // very simple approach to provide an event ID
234  // TODO: adjust to the relevant format if available
235  evtData.fEventID = mEventCount++;
236  }
237 
238  AliHLTComponentTriggerData trigData;
239  memset(&trigData, 0, sizeof(trigData));
240  trigData.fStructSize = sizeof(trigData);
241 
242  uint32_t outputBlockCnt = 0;
243  AliHLTComponentBlockData* pOutputBlocks = nullptr;
244  AliHLTComponentEventDoneData* pEventDoneData = nullptr;
245 
246  // prepare input structure for the ALICE HLT component
247  mFormatHandler.clear();
248  mFormatHandler.addMessages(dataArray);
249  vector<BlockDescriptor>& inputBlocks = mFormatHandler.getBlockDescriptors();
250  unsigned nofInputBlocks = inputBlocks.size();
251  if (dataArray.size() > 0 && nofInputBlocks == 0 && mFormatHandler.getEvtDataList().size() == 0) {
252  LOG(ERROR) << "warning: none of " << dataArray.size() << " input buffer(s) recognized as valid input";
253  }
254  dataArray.clear();
255 
256  if (mFormatHandler.getEvtDataList().size()>0) {
257  // copy the oldest event header
258  memcpy(&evtData, &mFormatHandler.getEvtDataList().front(), sizeof(AliHLTComponentEventData));
259  }
260 
261  // determine the total input size, needed later on for the calculation of the output buffer size
262  int totalInputSize = 0;
263  for (auto & ci : inputBlocks) {
264  totalInputSize += ci.fSize;
265  }
266 
267  // add event type data block
268  // this data block describes the type of the event, set it
269  // to 'data' by using specification gkAliEventTypeData
270  const AliHLTComponentDataType kDataTypeEvent = AliHLTComponentDataTypeInitializer("EVENTTYP", "PRIV");
271  inputBlocks.emplace_back(nullptr, 0, kDataTypeEvent, gkAliEventTypeData);
272 
273  // process
274  evtData.fBlockCnt = inputBlocks.size();
275  int nofTrials = 2;
276  do {
277  unsigned long constEventBase = 0;
278  unsigned long constBlockBase = 0;
279  double inputBlockMultiplier = 0.;
280  mpSystem->getOutputSize(mProcessor, &constEventBase, &constBlockBase, &inputBlockMultiplier);
281  outputBufferSize = constEventBase + nofInputBlocks * constBlockBase + totalInputSize * inputBlockMultiplier;
282  outputBufferSize+=sizeof(AliHLTComponentStatistics) + sizeof(AliHLTComponentTableEntry);
283  // take the full available buffer and increase if that
284  // is too little
285  mOutputBuffer.resize(mOutputBuffer.capacity());
286  if (mOutputBuffer.size() < outputBufferSize) {
287  mOutputBuffer.resize(outputBufferSize);
288  } else if (nofTrials < 2) {
289  // component did not update the output size
290  break;
291  }
292  outputBufferSize = mOutputBuffer.size();
293  outputBlockCnt = 0;
294  // TODO: check if that is working with the corresponding allocation method of the
295  // component environment
296  if (pOutputBlocks) delete[] pOutputBlocks;
297  pOutputBlocks = nullptr;
298  if (pEventDoneData) delete pEventDoneData;
299  pEventDoneData = nullptr;
300 
301  iResult = mpSystem->processEvent(mProcessor, &evtData, &inputBlocks[0], &trigData,
302  &mOutputBuffer[0], &outputBufferSize,
303  &outputBlockCnt, &pOutputBlocks,
304  &pEventDoneData);
305  if (outputBufferSize > 0) {
306  if (outputBufferSize > mOutputBuffer.size()) {
307  LOG(ERROR) << "FATAL: fatal error: component writing beyond buffer capacity";
308  return -EFAULT;
309  } else if (outputBufferSize < mOutputBuffer.size()) {
310  mOutputBuffer.resize(outputBufferSize);
311  }
312  } else {
313  mOutputBuffer.clear();
314  }
315 
316  } while (iResult == ENOSPC && --nofTrials > 0);
317 
318  // prepare output
319  { // keep this after removing condition to preserve formatting
320  uint8_t* pOutputBufferStart = &mOutputBuffer[0];
321  uint8_t* pOutputBufferEnd = pOutputBufferStart + mOutputBuffer.size();
322  // consistency check for data blocks
323  // 1) all specified data must be either inside the output buffer given
324  // to the component or in one of the input buffers
325  // 2) filter out special data blocks if they have been forwarded
326  //
327  // Some components set both pointer and offset in the block descriptor as data
328  // reference for blocks in the output buffer. Due to a misinterpretation in the
329  // early days of development that is only allowed if fPtr is set to the beginning
330  // of output buffer. The block is however fully described by offset relative to
331  // beginning of output buffer.
332  unsigned validBlocks = 0;
333  unsigned totalPayloadSize = 0;
334  AliHLTComponentBlockData* pOutputBlock = pOutputBlocks;
335  AliHLTComponentBlockData* pFiltered = pOutputBlocks;
336  for (unsigned blockIndex = 0; blockIndex < outputBlockCnt; blockIndex++, pOutputBlock++) {
337  // filter special data blocks
338  if (pOutputBlock->fDataType == kDataTypeEvent) continue;
339 
340  // block descriptors without any attached payload are propagated
341  bool bValid = pOutputBlock->fSize == 0;
342 
343  // calculate the data reference
344  uint8_t* pStart =
345  pOutputBlock->fPtr != nullptr ? reinterpret_cast<uint8_t*>(pOutputBlock->fPtr) : &mOutputBuffer[0];
346  pStart += pOutputBlock->fOffset;
347  uint8_t* pEnd = pStart + pOutputBlock->fSize;
348  pOutputBlock->fPtr = pStart;
349  pOutputBlock->fOffset = 0;
350 
351  // first search in the output buffer
352  bValid = bValid || (pStart >= pOutputBufferStart && pEnd <= pOutputBufferEnd);
353 
354  // possibly a forwarded data block, try the input buffers
355  if (!bValid) {
356  for (auto & ci : inputBlocks) {
357  uint8_t* pInputBufferStart = reinterpret_cast<uint8_t*>(ci.fPtr);
358  uint8_t* pInputBufferEnd = pInputBufferStart + ci.fSize;
359  if ((bValid = (pStart >= pInputBufferStart && pEnd <= pInputBufferEnd))) {
360  break;
361  }
362  }
363  }
364 
365  if (bValid) {
366  totalPayloadSize += pOutputBlock->fSize;
367  validBlocks++;
368  memcpy(pFiltered, pOutputBlock, sizeof(AliHLTComponentBlockData));
369  pFiltered++;
370  } else {
371  LOG(ERROR) << "Inconsistent data reference in output block " << blockIndex;
372  }
373  }
374  evtData.fBlockCnt=validBlocks;
375 
376  // create the messages
377  // TODO: for now there is an extra copy of the data, but it should be
378  // handled in place
379  vector<MessageFormat::BufferDesc_t> outputMessages =
380  mFormatHandler.createMessages(pOutputBlocks, validBlocks, totalPayloadSize, &evtData, cbAllocate);
381  dataArray.insert(dataArray.end(), outputMessages.begin(), outputMessages.end());
382  }
383 
384  // cleanup
385  // NOTE: don't cleanup mOutputBuffer as the data is going to be used outside the class
386  // until released.
387  inputBlocks.clear();
388  outputBlockCnt = 0;
389  if (pOutputBlocks) delete[] pOutputBlocks;
390  pOutputBlocks = nullptr;
391  if (pEventDoneData) delete pEventDoneData;
392  pEventDoneData = nullptr;
393 
394  return -iResult;
395 }
