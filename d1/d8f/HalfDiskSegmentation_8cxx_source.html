105 void HalfDiskSegmentation::createLadders(TXMLEngine* xml, XMLNodePointer_t node)
106 {
107  Int_t iladder;
108  Int_t nsensor;
109  Double_t pos[3];
110  Double_t ang[3]={0.,0.,0.};
111 
112  Geometry * mftGeom = Geometry::instance();
113 
114  TString nodeName = xml->GetNodeName(node);
115  if (!nodeName.CompareTo("ladder")) {
116  XMLAttrPointer_t attr = xml->GetFirstAttr(node);
117  while (attr != nullptr) {
118  TString attrName = xml->GetAttrName(attr);
119  TString attrVal = xml->GetAttrValue(attr);
120  if(!attrName.CompareTo("iladder")) {
121  iladder = attrVal.Atoi();
122  if (iladder >= getNLadders() || iladder < 0) {
123  LOG(FATAL) << "Wrong ladder number : " << iladder << FairLogger::endl;
124  }
125  } else
126  if(!attrName.CompareTo("nsensor")) {
127  nsensor = attrVal.Atoi();
128  } else
129  if(!attrName.CompareTo("xpos")) {
130  pos[0] = attrVal.Atof();
131  } else
132  if(!attrName.CompareTo("ypos")) {
133  pos[1] = attrVal.Atof();
134  } else
135  if(!attrName.CompareTo("zpos")) {
136  pos[2] = attrVal.Atof();
137  } else
138  if(!attrName.CompareTo("phi")) {
139  ang[0] = attrVal.Atof();
140  } else
141  if(!attrName.CompareTo("theta")) {
142  ang[1] = attrVal.Atof();
143  } else
144  if(!attrName.CompareTo("psi")) {
145  ang[2] = attrVal.Atof();
146  } else{
147  LOG(ERROR) << "Unknwon Attribute name " << xml->GetAttrName(attr) << FairLogger::endl;
148  }
149  attr = xml->GetNextAttr(attr);
150  }
151 
152  Int_t plane = -1;
153  Int_t ladderID=iladder;
154  if( iladder < getNLadders()/2) {
155  plane = 0;
156  } else {
157  plane = 1;
158  //ladderID -= getNLadders()/2;
159  }
160 
161  UInt_t ladderUniqueID = mftGeom->getObjectID(Geometry::LadderType,mftGeom->getHalfID(GetUniqueID()),mftGeom->getDiskID(GetUniqueID()),plane,ladderID);
162 
163  auto * ladder = new LadderSegmentation(ladderUniqueID);
164  ladder->setNSensors(nsensor);
165  ladder->setPosition(pos);
166  ladder->setRotationAngles(ang);
167 
170  // Find the position of the corner of the flex which is the ladder corrdinate system center.
171 
172  pos[0] = -Geometry::sSensorSideOffset;
173  pos[1] = -Geometry::sSensorTopOffset - SegmentationAlpide::SensorSizeRows;
174  pos[2] = -Geometry::sFlexThickness - Geometry::sChipThickness;
175  Double_t master[3];
176  ladder->getTransformation()->LocalToMaster(pos, master);
177  ladder->setPosition(master);
178 
179  ladder->createSensors();
180 
181  new ((*mLadders)[iladder]) LadderSegmentation(*ladder);
182  delete ladder;
183 
184  //getLadder(iladder)->Print();
185 
186  }
187 
188  // display all child nodes
189  XMLNodePointer_t child = xml->GetChild(node);
190  while (child!=nullptr) {
191  createLadders(xml, child);
192  child = xml->GetNext(child);
193  }
194 
195 }
