45 size_t
46 assignInputSpecId(void *data, std::vector<InputRoute> const &routes) {
47  for (size_t ri = 0, re = routes.size(); ri < re; ++ri) {
48  auto &route = routes[ri];
49  const DataHeader *h = o2::header::get<DataHeader>(data);
50  if (h == nullptr) {
51  return re;
52  }
53 
54  if (DataSpecUtils::match(route.matcher,
55  h->dataOrigin,
56  h->dataDescription,
57  h->subSpecification)) {
58  return ri;
59  }
60  }
61  return routes.size();
62 }

64 DataRelayer::RelayChoice
65 DataRelayer::relay(std::unique_ptr<FairMQMessage> &&header,
66  std::unique_ptr<FairMQMessage> &&payload) {
67  // STATE HOLDING VARIABLES
68  // This is the class level state of the relaying. If we start supporting
69  // multithreading this will have to be made thread safe before we can invoke
70  // relay concurrently.
71  const auto &inputs = mInputs;
72  std::vector<TimesliceId> &timeslices = mTimeslices;
73  auto &cache = mCache;
74  const auto &readonlyCache = mCache;
75 
76  // IMPLEMENTATION DETAILS
77  //
78  // This returns the identifier for the given input. We use a separate
79  // function because while it's trivial now, the actual matchmaking will
80  // become more complicated when we will start supporting ranges.
81  auto getInput = [&inputs,&header] () -> int {
82  return assignInputSpecId(header->GetData(), inputs);
83  };
84 
85  // This will check if the input is valid. We hide the details so that
86  // in principle the outer code will work regardless of the actual
87  // implementation.
88  auto isValidInput = [](int inputIdx) {
89  // If this is true, it means the message we got does
90  // not match any of the expected inputs.
91  return inputIdx != INVALID_INPUT;
92  };
93 
94  // This will check if the timeslice is valid. We hide the details so that in
95  // principle outer code will work regardless of the actual implementation.
96  auto isValidTimeslice = [](int64_t timesliceId) -> bool {
97  return timesliceId != INVALID_TIMESLICE;
98  };
99 
100  // The timeslice is embedded in the DataProcessingHeader header of the O2
101  // header stack. This is an extension to the DataHeader, because apparently
102  // we do have data which comes without a timestamp, although I am personally
103  // not sure what that would be.
104  auto getTimeslice = [&header,&timeslices]() -> int64_t {
105  const DataProcessingHeader *dph = o2::header::get<DataProcessingHeader>(header->GetData());
106  if (dph == nullptr) {
107  return -1;
108  }
109  size_t timesliceId = dph->startTime;
110  assert(timeslices.size());
111  return timesliceId;
112  };
113 
114  // Late arrival means that the incoming timeslice is actually older
115  // then the one hold in the current cache.
116  auto isInputFromObsolete = [&timeslices](int64_t timeslice) {
117  auto &current = timeslices[timeslice % timeslices.size()];
118  return current.value > timeslice;
119  };
120 
121  // A cache line is obsolete if the incoming one has a greater timestamp or
122  // if the slot contains an invalid timeslice.
123  auto isCacheEntryObsoleteFor = [&timeslices](int64_t timeslice){
124  auto &current = timeslices[timeslice % timeslices.size()];
125  return current.value == INVALID_TIMESLICE
126  || (current.value < timeslice);
127  };
128 
129  // We need to prune the cache from the old stuff, if any. Otherwise we
130  // simply store the payload in the cache and we mark relevant bit in the
131  // completion mask. Notice that late arrivals should simply be ignored,
132  // hence the first if.
133  auto pruneCacheSlotFor = [&cache,&inputs,&timeslices](int64_t timeslice) {
134  size_t slotIndex = timeslice % timeslices.size();
135  // Prune old stuff from the cache, hopefully deleting it...
136  // We set the current slot to the timeslice value, so that old stuff
137  // will be ignored.
138  assert(inputs.size() * slotIndex < cache.size());
139  timeslices[slotIndex].value = timeslice;
140  for (size_t ai = slotIndex*inputs.size(), ae = ai + inputs.size(); ai != ae ; ++ai) {
141  cache[ai].header.reset(nullptr);
142  cache[ai].payload.reset(nullptr);
143  }
144  };
145 
146  // We need to check if the slot for the current input is already taken for
147  // the current timeslice.
148  // This should never happen, however given this is dependent on the input
149  // we want to protect again malicious / bad upstream source.
150  auto hasCacheInputAlreadyFor = [&cache, &timeslices, &inputs](int64_t timeslice, int input) {
151  size_t slotIndex = timeslice % timeslices.size();
152  PartRef &currentPart = cache[inputs.size()*slotIndex + input];
153  return (currentPart.payload != nullptr) || (currentPart.header != nullptr);
154  };
155 
156  // Actually save the header / payload in the slot
157  auto saveInSlot = [&header, &payload, &cache, &timeslices, &inputs](int64_t timeslice, int input) {
158  size_t slotIndex = timeslice % timeslices.size();
159  PartRef &currentPart = cache[inputs.size()*slotIndex + input];
160  PartRef ref{std::move(header), std::move(payload)};
161  currentPart = std::move(ref);
162  timeslices[slotIndex] = {timeslice};
163  assert(header.get() == nullptr && payload.get() == nullptr);
164  };
165 
166  // OUTER LOOP
167  //
168  // This is the actual outer loop processing input as part of a given
169  // timeslice. All the other implementation details are hidden by the lambdas
170  auto input = getInput();
171 
172  if (isValidInput(input) == false) {
173  LOG(ERROR) << "A malformed message just arrived";
174  return WillNotRelay;
175  }
176 
177  auto timeslice = getTimeslice();
178  LOG(DEBUG) << "Received timeslice" << timeslice;
179  if (isValidTimeslice(timeslice) == false) {
180  LOG(ERROR) << "Could not determine the timeslice for input";
181  return WillNotRelay;
182  }
183 
184  if (isInputFromObsolete(timeslice)) {
185  LOG(ERROR) << "An entry for timeslice " << timeslice << " just arrived but too late to be processed";
186  return WillNotRelay;
187  }
188 
189  if (isCacheEntryObsoleteFor(timeslice)) {
190  pruneCacheSlotFor(timeslice);
191  }
192  if (hasCacheInputAlreadyFor(timeslice, input)) {
193  LOG(ERROR) << "Got a message with the same header and timeslice twice!!";
194  return WillNotRelay;
195  }
196  saveInSlot(timeslice, input);
197  return WillRelay;
198 }

200 std::vector<int>
201 DataRelayer::getReadyToProcess() {
202  // THE STATE
203  std::vector<int> completed;
204  const auto &cache = mCache;
205  const auto &inputs = mInputs;
206  //
207  // THE IMPLEMENTATION DETAILS
208  //
209  // We use this to bail out early from the check as soon as we find something
210  // which we know is not complete.
211  auto theLineWillBeIncomplete = [&cache, &inputs](int li, int ai) -> bool {
212  auto &input = cache[li*inputs.size() + ai];
213  if (input.header == nullptr || input.payload == nullptr) {
214  return true;
215  }
216  return false;
217  };
218 
219  // These two are trivial, but in principle the whole loop could be parallelised
220  // or vectorised so "completed" could be a thread local variable which needs
221  // merging at the end.
222  auto updateCompletionResults = [&completed](size_t li) {
223  completed.push_back(li);
224  };
225 
226  auto completionResults = [&completed]() -> std::vector<int> {
227  return completed;
228  };
229 
230  // THE OUTER LOOP
231  //
232  // To determine if a line is complete, we iterate on all the arguments
233  // and check if they are ready. We do it this way, because in the end
234  // the number of inputs is going to be small and having a more complex
235  // structure will probably result in a larger footprint in any case.
236  // Also notice that ai == inputsNumber only when we reach the end of the
237  // iteration, that means we have found all the required bits.
238  assert(!inputs.empty());
239  size_t cacheLines = cache.size() / inputs.size();
240  assert(cacheLines * inputs.size() == cache.size());
241 
242  for (size_t li = 0; li < cacheLines; ++li) {
243  size_t ai;
244  for (ai = 0; ai < inputs.size(); ++ai) {
245  if (theLineWillBeIncomplete(li, ai)) {
246  break;
247  }
248  }
249  if (ai == inputs.size()) {
250  updateCompletionResults(li);
251  }
252  }
253  return completionResults();
254 }

256 std::vector<std::unique_ptr<FairMQMessage>>
257 DataRelayer::getInputsForTimeslice(size_t timeslice) {
258  // State of the computation
259  std::vector<std::unique_ptr<FairMQMessage>> messages;
260  messages.reserve(mInputs.size()*2);
261  auto &cache = mCache;
262  auto &timeslices = mTimeslices;
263  const auto &inputs = mInputs;
264 
265  // Nothing to see here, this is just to make the outer loop more understandable.
266  auto jumpToCacheEntryAssociatedWith = [](size_t) {
267  return;
268  };
269 
270  // We move ownership so that the cache can be reused once the computation is
271  // finished. We bump by one the timeslice for the given cache entry, so that
272  // in case we get (for whatever reason) an old input, it will be
273  // automatically discarded by the relay method.
274  auto moveHeaderPayloadToOutput = [&messages, &cache, &timeslices, &inputs](size_t ti, size_t arg) {
275  messages.emplace_back(std::move(cache[ti*inputs.size() + arg].header));
276  messages.emplace_back(std::move(cache[ti*inputs.size() + arg].payload));
277  timeslices[ti % timeslices.size()].value += 1;
278  };
279 
280  // An invalid set of arguments is a set of arguments associated to an invalid
281  // timeslice, so I can simply do that. I keep the assertion there because in principle
282  // we should have dispatched the timeslice already!
283  // FIXME: what happens when we have enough timeslices to hit the invalid one?
284  auto invalidateCacheFor = [&inputs, &timeslices, &cache](size_t ti) {
285  for (size_t ai = ti*inputs.size(), ae = ai + inputs.size(); ai != ae; ++ai) {
286  assert(cache[ai].header.get() == nullptr);
287  assert(cache[ai].payload.get() == nullptr);
288  }
289  timeslices[ti % timeslices.size()] = INVALID_TIMESLICE_ID;
290  };
291 
292  // Outer loop here.
293  jumpToCacheEntryAssociatedWith(timeslice);
294  for (size_t ai = 0, ae = inputs.size(); ai != ae; ++ai) {
295  moveHeaderPayloadToOutput(timeslice, ai);
296  }
297  invalidateCacheFor(timeslice);
298 
299  return std::move(messages);
300 }
