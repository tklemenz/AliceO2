119 void HalfSegmentation::createHalfDisks(TXMLEngine* xml, XMLNodePointer_t node)
120 {
121  // this function display all accessible information about xml node and its children
122  Int_t idisk;
123  Int_t nladder;
124  Double_t pos[3]={0., 0., 0.};
125  Double_t ang[3]={0., 0., 0.};
126 
127  Geometry * mftGeom = Geometry::instance();
128 
129  TString nodeName = xml->GetNodeName(node);
130  if (!nodeName.CompareTo("disk")) {
131  XMLAttrPointer_t attr = xml->GetFirstAttr(node);
132  while (attr != nullptr) {
133  TString attrName = xml->GetAttrName(attr);
134  TString attrVal = xml->GetAttrValue(attr);
135  if(!attrName.CompareTo("idisk")) {
136  idisk = attrVal.Atoi();
137  if (idisk >= Constants::sNDisks || idisk < 0) {
138  LOG(FATAL) << "Wrong disk number : " << idisk << FairLogger::endl;
139  }
140  } else
141  if(!attrName.CompareTo("nladder")) {
142  nladder = attrVal.Atoi();
143  } else
144  if(!attrName.CompareTo("xpos")) {
145  pos[0] = attrVal.Atof();
146  } else
147  if(!attrName.CompareTo("ypos")) {
148  pos[1] = attrVal.Atof();
149  } else
150  if(!attrName.CompareTo("zpos")) {
151  pos[2] = attrVal.Atof();
152  } else
153  if(!attrName.CompareTo("phi")) {
154  ang[0] = attrVal.Atof();
155  } else
156  if(!attrName.CompareTo("theta")) {
157  ang[1] = attrVal.Atof();
158  } else
159  if(!attrName.CompareTo("psi")) {
160  ang[2] = attrVal.Atof();
161  } else{
162  LOG(ERROR) << "Unknwon Attribute name " << xml->GetAttrName(attr) << FairLogger::endl;
163  }
164  attr = xml->GetNextAttr(attr);
165  }
166 
167  UInt_t diskUniqueID = mftGeom->getObjectID(Geometry::HalfDiskType,mftGeom->getHalfID(GetUniqueID()),idisk);
168 
169  auto *halfDisk = new HalfDiskSegmentation(diskUniqueID);
170  halfDisk->setPosition(pos);
171  halfDisk->setRotationAngles(ang);
172  halfDisk->setNLadders(nladder);
173  halfDisk->createLadders(xml, node);
174  if(halfDisk->getNLaddersBuild() != halfDisk->getNLadders()) {
175  LOG(FATAL) << "Number of ladder build " << halfDisk->getNLaddersBuild() << " does not correspond to the number declared " << halfDisk->getNLadders() << " Check XML file" << FairLogger::endl;
176  }
177  new ((*mHalfDisks)[idisk]) HalfDiskSegmentation(*halfDisk);
178  delete halfDisk;
179 
180  }
181 
182  // display all child nodes
183  XMLNodePointer_t child = xml->GetChild(node);
184  while (child!=nullptr) {
185  createHalfDisks(xml, child);
186  child = xml->GetNext(child);
187  }
188 
189 }

194 void HalfSegmentation::findHalf(TXMLEngine* xml, XMLNodePointer_t node, XMLNodePointer_t &retnode){
195  // Find in the XML Geometry File the node corresponding to the Half-MFT being build
196  // Set Position and Orientation of the Half-MFT
197  Int_t isTop;
198  Int_t ndisk;
199  Double_t pos[3] = {0., 0., 0.};
200  Double_t ang[3] = {0., 0., 0.};
201 
202  TString nodeName = xml->GetNodeName(node);
203  if (!nodeName.CompareTo("half")) {
204  XMLAttrPointer_t attr = xml->GetFirstAttr(node);
205  while (attr!=nullptr) {
206  TString attrName = xml->GetAttrName(attr);
207  TString attrVal = xml->GetAttrValue(attr);
208  if(!attrName.CompareTo("top")){
209  isTop = attrVal.Atoi();
210  if (isTop>1 || isTop<0) {
211  LOG(FATAL) << "Wrong Half MFT number : " << isTop << FairLogger::endl;
212  }
213  } else
214  if(!attrName.CompareTo("ndisk")){
215  ndisk = attrVal.Atoi();
216  if (ndisk>5 || ndisk<0) {
217  LOG(ERROR) << "Wrong number of disk : " << ndisk << FairLogger::endl;
218  }
219 
220  } else
221  if(!attrName.CompareTo("xpos")){
222  pos[0] = attrVal.Atof();
223  } else
224  if(!attrName.CompareTo("ypos")){
225  pos[1] = attrVal.Atof();
226  } else
227  if(!attrName.CompareTo("zpos")){
228  pos[2] = attrVal.Atof();
229  } else
230  if(!attrName.CompareTo("phi")){
231  ang[0] = attrVal.Atof();
232  } else
233  if(!attrName.CompareTo("theta")){
234  ang[1] = attrVal.Atof();
235  } else
236  if(!attrName.CompareTo("psi")){
237  ang[2] = attrVal.Atof();
238  } else{
239  LOG(ERROR) << "Unknwon Attribute name " << xml->GetAttrName(attr) << FairLogger::endl;
240  }
241 
242  attr = xml->GetNextAttr(attr);
243  }
244 
245  Geometry * mftGeom = Geometry::instance();
246  if(isTop == mftGeom->getHalfID(GetUniqueID())) {
247  setPosition(pos);
248  setRotationAngles(ang);
249  retnode = node;
250  return;
251  }
252 
253  }
254 
255  // display all child nodes
256  XMLNodePointer_t child = xml->GetChild(node);
257  while (child!=nullptr) {
258  findHalf(xml, child, retnode);
259  child = xml->GetNext(child);
260  }
261 }
