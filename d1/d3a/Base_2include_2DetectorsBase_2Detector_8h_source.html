188  // generic implementation for the updateHitTrackIndices interface
189  // assumes Detectors have a GetHits(int) function that return some iterable
190  // hits which are o2::BaseHits
191  void updateHitTrackIndices(std::map<int, int> const& indexmapping) override
192  {
193  int probe = 0; // some Detectors have multiple hit vectors and we are probing
194  // them via a probe integer until we get a nullptr
195  while (auto hits = static_cast<Det*>(this)->Det::getHits(probe++)) {
196  for (auto& hit : *hits) {
197  auto iter = indexmapping.find(hit.GetTrackID());
198  hit.SetTrackID(iter->second);
199  }
200  }
201  }
202  ClassDefOverride(DetImpl, 0)
203 };
