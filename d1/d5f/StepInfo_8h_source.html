40 struct VolInfoContainer {
41  VolInfoContainer() = default;
42 
43  // keeps info about volumes (might exist somewhere else already)
44  // essentially a mapping from volumeID x copyNo to TGeoVolumes
45  std::vector<std::vector<TGeoVolume const*>*> volumes; // sparse container
46 
47  void insert(int id, int copyNo, TGeoVolume const* vol)
48  {
49  if (volumes.size() <= id) {
50  volumes.resize(id + 1, nullptr);
51  }
52  if (volumes[id] == nullptr) {
53  volumes[id] = new std::vector<TGeoVolume const*>;
54  }
55  if (volumes[id]->size() <= copyNo) {
56  volumes[id]->resize(copyNo + 1, nullptr);
57  }
58  (*volumes[id])[copyNo] = vol;
59  }
60 
61  TGeoVolume const* get(int id, int copy) const { return (*volumes[id])[copy]; }
62  ClassDefNV(VolInfoContainer, 1);
63 };

84  void insertPDG(int trackindex, int pdg)
85  {
86  constexpr int INVALIDPDG = 0;
87  if (trackindex >= tracktopdg.size()) {
88  tracktopdg.resize(trackindex + 1, INVALIDPDG);
89  }
90  auto prev = tracktopdg[trackindex];
91  if (prev != INVALIDPDG && prev != pdg) {
92  std::cerr << "Warning: Seeing more than one pdg for same trackID\n";
93  }
94  tracktopdg[trackindex] = pdg;
95  }
96 
97  void insertParent(int trackindex, int parent)
98  {
99  constexpr int PRIMARY = -1;
100  if (trackindex >= tracktoparent.size()) {
101  tracktoparent.resize(trackindex + 1, PRIMARY);
102  }
103  auto prev = tracktoparent[trackindex];
104  if (prev != PRIMARY && prev != parent) {
105  std::cerr << "Warning: Seeing more than one parent for same trackID\n";
106  }
107  tracktoparent[trackindex] = parent;
108  }
109 
110  private:
111  void insertValueAt(int index, std::string const& s, std::vector<std::string*>& container)
112  {
113  if (index >= container.size()) {
114  container.resize(index + 1, nullptr);
115  }
116  //#ifdef CHECKMODE
117  // check that if a value exists at some index it is the same that we want to write
118  if (container[index] != nullptr) {
119  auto previous = *(container[index]);
120  if (s.compare(previous) != 0) {
121  std::cerr << "trying to override " << previous << " with " << s << "\n";
122  }
123  }
124  //#endif
125  // can we use unique pointers??
126  container[index] = new std::string(s);
127  }
