57  // In case no extra argument is provided and the passed type is a POD,
58  // the most likely wanted behavior is to create a message with that POD,
59  // and so we do.
60  template <typename T>
61  typename std::enable_if<std::is_pod<T>::value == true, T &>::type
62  make(const OutputSpec &spec) {
63  DataChunk chunk = newChunk(spec, sizeof(T));
64  return *reinterpret_cast<T*>(chunk.data);
65  }
66 
67  // In case an extra argument is provided, we consider this an array /
68  // collection elements of that type
69  template <typename T>
70  typename std::enable_if<std::is_pod<T>::value == true, gsl::span<T>>::type
71  make(const OutputSpec &spec, size_t nElements) {
72  auto size = nElements*sizeof(T);
73  DataChunk chunk = newChunk(spec, size);
74  return gsl::span<T>(reinterpret_cast<T*>(chunk.data), nElements);

170  template <typename C>
171  typename std::enable_if<
172  is_specialization<C, std::vector>::value == true
173  && std::is_pointer<typename C::value_type>::value == false
174  && std::is_trivially_copyable<typename C::value_type>::value == true
175  >::type
176  snapshot(const OutputSpec &spec, C const &v) {
177  auto sizeInBytes = sizeof(typename C::value_type) * v.size();
178  FairMQMessagePtr payloadMessage(mDevice->NewMessage(sizeInBytes));
179 
180  typename C::value_type *tmp = const_cast<typename C::value_type*>(v.data());
181  memcpy(payloadMessage->GetData(), reinterpret_cast<void*>(tmp), sizeInBytes);
182 
183  addPartToContext(std::move(payloadMessage), spec, o2::header::gSerializationMethodNone);
184  }

190  template <typename C>
191  typename std::enable_if<
192  is_specialization<C, std::vector>::value == true
193  && std::is_pointer<typename C::value_type>::value == true
194  && std::is_trivially_copyable<typename std::remove_pointer<typename C::value_type>::type>::value == true
195  >::type
196  snapshot(const OutputSpec &spec, C const &v) {
197  using ElementType = typename std::remove_pointer<typename C::value_type>::type;
198  constexpr auto elementSizeInBytes = sizeof(ElementType);
199  auto sizeInBytes = elementSizeInBytes * v.size();
200  FairMQMessagePtr payloadMessage(mDevice->NewMessage(sizeInBytes));
201 
202  auto target = reinterpret_cast<unsigned char*>(payloadMessage->GetData());
203  for (auto const & pointer : v) {
204  memcpy(target, pointer, elementSizeInBytes);
205  target += elementSizeInBytes;
206  }
207 
208  addPartToContext(std::move(payloadMessage), spec, o2::header::gSerializationMethodNone);
209  }
