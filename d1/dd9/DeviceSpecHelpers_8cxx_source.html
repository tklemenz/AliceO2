81  // Edges are navigated in order for each device, so the device associaited to
82  // an edge is always the last one created.
83  auto deviceForEdge = [&actions, &workflow, &devices, &logicalEdges](size_t ei) {
84  auto &edge = logicalEdges[ei];
85  auto &action = actions[ei];
86 
87  if (action.requiresNewDevice == false) {
88  assert(devices.empty() == false);
89  return devices.size() - 1;

92  DeviceSpec device;
93  device.name = processor.name;
94  device.id = processor.name;
95  if (processor.maxInputTimeslices != 1) {
96  device.id = processor.name + "_t" + std::to_string(edge.producerTimeIndex);
97  }
98  device.algorithm = processor.algorithm;
99  device.options = processor.options;
100  device.rank = processor.rank;
101  device.nSlots = processor.nSlots;
102  device.inputTimesliceId = edge.timeIndex;
103  devices.push_back(device);
104  return devices.size() - 1;
105  };

138  // This creates a new channel for a given edge, if needed. Notice that we
139  // navigate edges in a per device fashion (creating those if they are not
140  // alredy there) and create a new channel only if it connects two new
141  // devices. Whether or not this is the case was previously computed
142  // in the action.requiresNewChannel field.
143  auto createChannelForDeviceEdge = [&devices,
144  &logicalEdges,
145  &nextPort,
146  &channelFromDeviceEdgeAndPort,
147  &connectionIdFromEdgeAndPort,
148  &outputsMatchers,
149  &deviceIndex,
150  &workflow](size_t di, size_t ei) {
151  auto &device = devices[di];
152  auto &edge = logicalEdges[ei];
153 
154  deviceIndex.emplace_back(DeviceId{edge.producer, edge.producerTimeIndex, di});
155 
156  OutputChannelSpec channel = channelFromDeviceEdgeAndPort(device, edge, nextPort);
157  const DeviceConnectionId &id = connectionIdFromEdgeAndPort(edge, nextPort);
158  nextPort++;
159 
160  device.outputChannels.push_back(channel);
161  return device.outputChannels.size() - 1;
162  };

164  // Notice how we need to behave in two different ways depending
165  // whether this is a real OutputRoute or if it's a forward from
166  // a previous consumer device.
167  // FIXME: where do I find the InputSpec for the forward?
168  auto appendOutputRouteToSourceDeviceChannel = [&outputsMatchers, &workflow, &devices, &logicalEdges]
169  (size_t ei, size_t di, size_t ci) {
170  assert(ei < logicalEdges.size());
171  assert(di < devices.size());
172  assert(ci < devices[di].outputChannels.size());
173  auto &edge = logicalEdges[ei];
174  auto &device = devices[di];
175  assert(edge.consumer < workflow.size());
176  auto &consumer = workflow[edge.consumer];
177  auto &channel = devices[di].outputChannels[ci];
178  assert(edge.outputGlobalIndex < outputsMatchers.size());
179 
180  if (edge.isForward == false) {
181  OutputRoute route;
182  route.matcher = outputsMatchers[edge.outputGlobalIndex];
183  route.timeslice = edge.timeIndex;
184  route.maxTimeslices = consumer.maxInputTimeslices;
185  route.channel = channel.name;
186  device.outputs.emplace_back(route);
187  } else {
188  ForwardRoute route;
189  route.matcher = workflow[edge.consumer].inputs[edge.consumerInputIndex];
190  route.channel = channel.name;
191  device.forwards.emplace_back(route);
192  }
193  };

199  auto lastChannelFor = [&devices](size_t di) {
200  assert(di < devices.size());
201  assert(devices[di].outputChannels.empty() == false);
202  return devices[di].outputChannels.size() - 1;
203  };

205  //
206  // OUTER LOOP
207  //
208  // We need to create all the channels going out of a device, and associate
209  // routes to them for this reason
210  // we iterate over all the edges (which are per-datatype exchanged) and
211  // whenever we need to connect to a new device we create the channel. `device`
212  // here refers to the source device. This loop will therefore not create the
213  // devices which acts as sink, which are done in the preocessInEdgeActions
214  // function.
215  for (auto edge : outEdgeIndex) {
216  auto device = deviceForEdge(edge);
217  size_t channel = -1;
218  if (isDifferentDestinationDeviceReferredBy(edge)) {
219  channel = createChannelForDeviceEdge(device, edge);
220  } else {
221  channel = lastChannelFor(device);
222  }
223  appendOutputRouteToSourceDeviceChannel(edge, device, channel);
224  }
225  sortDeviceIndex();

230  std::vector<DeviceSpec> &devices,
231  std::vector<DeviceId> &deviceIndex,
232  unsigned short &nextPort,

234  const std::vector<size_t> &inEdgeIndex,
235  const std::vector<DeviceConnectionEdge> &logicalEdges,
236  const std::vector<EdgeAction> &actions,
237  const WorkflowSpec &workflow,
238  std::vector<LogicalForwardInfo> const &availableForwardsInfo

242  auto findProducerForEdge = [&logicalEdges,&constDeviceIndex](size_t ei) {
243  auto &edge = logicalEdges[ei];
244 
245  DeviceId pid{edge.producer, edge.producerTimeIndex, 0};
246  auto deviceIt = std::lower_bound(constDeviceIndex.cbegin(),
247  constDeviceIndex.cend(), pid);
248  // By construction producer should always be there
249  assert(deviceIt != constDeviceIndex.end());
250  assert(deviceIt->processorIndex == pid.processorIndex
251  && deviceIt->timeslice == pid.timeslice);
252  return deviceIt->deviceIndex;
253  };
254 
255  // Notice that to start with, consumer exists only if they also are
256  // producers, so we need to create one if it does not exist. Given this is
257  // stateful, we keep an eye on what edge was last searched to make sure we
258  // are not screwing up.
259  //
260  // Notice this is not thread safe.
261  decltype(deviceIndex.begin()) lastConsumerSearch;
262  size_t lastConsumerSearchEdge;
263  auto hasConsumerForEdge = [&lastConsumerSearch, &lastConsumerSearchEdge, &deviceIndex, &logicalEdges] (size_t ei) -> int {
264  auto &edge = logicalEdges[ei];
265  DeviceId cid{edge.consumer, edge.timeIndex, 0};
266  lastConsumerSearchEdge = ei; // This will invalidate the cache
267  lastConsumerSearch = std::lower_bound(deviceIndex.begin(), deviceIndex.end(), cid);
268  return lastConsumerSearch != deviceIndex.end()
269  && cid.processorIndex == lastConsumerSearch->processorIndex
270  && cid.timeslice == lastConsumerSearch->timeslice;
271  };
272 
273  // The passed argument is there just to check. We do know that the last searched
274  // is the one we want.
275  auto getConsumerForEdge = [&lastConsumerSearch, &lastConsumerSearchEdge](size_t ei) {
276  assert(ei == lastConsumerSearchEdge);
277  return lastConsumerSearch->deviceIndex;
278  };

280  auto createNewDeviceForEdge = [&workflow, &logicalEdges, &devices, &deviceIndex] (size_t ei) {
281  auto &edge = logicalEdges[ei];
282  auto &processor = workflow[edge.consumer];
283  DeviceSpec device;
284  device.name = processor.name;
285  device.id = processor.name;
286  if (processor.maxInputTimeslices != 1) {
287  device.id += "_t" + std::to_string(edge.timeIndex);
288  }
289  device.algorithm = processor.algorithm;
290  device.options = processor.options;
291  device.rank = processor.rank;
292  device.nSlots = processor.nSlots;
293  device.inputTimesliceId = edge.timeIndex;
294  // FIXME: maybe I should use an std::map in the end
295  // but this is really not performance critical
296  auto id = DeviceId{edge.consumer,
297  edge.timeIndex,
298  devices.size()};
299  devices.push_back(device);
300  deviceIndex.push_back(id);
301  std::sort(deviceIndex.begin(), deviceIndex.end());
302  return devices.size() - 1;
303  };
304 
305  // We search for a preexisting outgoing connection associated to this edge.
306  // This is to retrieve the port of the source.
307  // This has to exists, because we already created all the outgoing connections
308  // so it's just a matter of looking it up.
309  auto findMatchingOutgoingPortForEdge = [&logicalEdges, &connections](size_t ei) {
310  auto const &edge = logicalEdges[ei];
311  DeviceConnectionId connectionId{
312  edge.producer,
313  edge.consumer,
314  edge.timeIndex,
315  edge.producerTimeIndex,
316  0
317  };
318 
319  auto it = std::lower_bound(connections.begin(), connections.end(), connectionId);
320 
321  assert(it != connections.end());
322  assert(it->producer == connectionId.producer);
323  assert(it->consumer == connectionId.consumer);
324  assert(it->timeIndex == connectionId.timeIndex);
325  assert(it->producerTimeIndex == connectionId.producerTimeIndex);
326  return it->port;
327  };

329  auto checkNoDuplicatesFor = [](std::vector<InputChannelSpec> const&channels,
330  const std::string &name) {
331  for (auto const &channel : channels) {
332  if (channel.name == name) {
333  return false;
334  }

338  auto appendInputChannelForConsumerDevice = [&devices,&connections,&checkNoDuplicatesFor]
339  (size_t pi, size_t ci, int16_t port) {
340  auto const &producerDevice = devices[pi];
341  auto &consumerDevice = devices[ci];
342  InputChannelSpec channel;
343  channel.name = "from_" + producerDevice.id + "_to_" + consumerDevice.id;
344  channel.method = ChannelMethod::Connect;
345  channel.type = ChannelType::Sub;
346  channel.port = port;
347  assert(checkNoDuplicatesFor(consumerDevice.inputChannels, channel.name));
348  consumerDevice.inputChannels.push_back(channel);
349  return consumerDevice.inputChannels.size() - 1;
350  };
351 
352  // I think this is trivial, since I think it should always be the last one,
353  // in case it's not actually the case, I should probably do an actual lookup
354  // here.
355  auto getChannelForEdge = [&devices](size_t pi, size_t ci) {
356  auto &consumerDevice = devices[ci];
357  return consumerDevice.inputChannels.size() - 1;
358  };
359 
360  // This is always called when adding a new channel, so we can simply refer
361  // to back. Notice also that this is the place where it makes sense to
362  // assign the forwarding, given that the forwarded stuff comes from some
363  // input.
364  auto appendInputRouteToDestDeviceChannel = [&devices,&logicalEdges,&workflow]
365  (size_t ei, size_t di, size_t ci) {
366  auto const &edge = logicalEdges[ei];
367  auto const &consumer = workflow[edge.consumer];
368  auto &consumerDevice = devices[di];
369  InputRoute route;
370  route.matcher = consumer.inputs[edge.consumerInputIndex];
371  route.sourceChannel = consumerDevice.inputChannels[ci].name;
372  consumerDevice.inputs.push_back(route);
373  };
374 
375  // Outer loop. A new device is needed for each
376  // of the sink data processors.
377  // New InputChannels need to refer to preexisting OutputChannels we create
378  // previously.
379  for (size_t edge : inEdgeIndex) {
380  auto &action = actions[edge];
381 
382  size_t consumerDevice;
383 
384  if (action.requiresNewDevice) {
385  if (hasConsumerForEdge(edge)) {
386  consumerDevice = getConsumerForEdge(edge);
387  } else {
388  consumerDevice = createNewDeviceForEdge(edge);
389  }
390  }
391  size_t producerDevice = findProducerForEdge(edge);
392 
393  size_t channel = -1;
394  if (action.requiresNewChannel) {
395  int16_t port = findMatchingOutgoingPortForEdge(edge);
396  channel = appendInputChannelForConsumerDevice(producerDevice, consumerDevice, port);
397  } else {
398  channel = getChannelForEdge(producerDevice, consumerDevice);
399  }
400  appendInputRouteToDestDeviceChannel(edge, consumerDevice, channel);
401  }

478 void
479 DeviceSpecHelpers::prepareArguments(int argc,
480  char **argv,
481  bool defaultQuiet,
482  bool defaultStopped,
483  const std::vector<DeviceSpec> &deviceSpecs,
484  std::vector<DeviceExecution> &deviceExecutions,
485  std::vector<DeviceControl> &deviceControls)
486 {
487  assert(deviceSpecs.size() == deviceExecutions.size());
488  assert(deviceControls.size() == deviceExecutions.size());
489  for (size_t si = 0; si < deviceSpecs.size(); ++si) {
490  auto &spec = deviceSpecs[si];
491  auto &control = deviceControls[si];
492  auto &execution = deviceExecutions[si];
493 
494  control.quiet = defaultQuiet;
495  control.stopped = defaultStopped;
496 
497  // We duplicate the list of options, filtering only those
498  // which are actually relevant for the given device. The additional
499  // four are to add
500  // * name of the executable
501  // * --framework-id <id> so that we can use the workflow
502  // executable also in other context where we do not fork, e.g. DDS.
503  // * final NULL required by execvp
504  //
505  // We do it here because we are still in the parent and we can therefore
506  // capture them to be displayed in the GUI or to populate the DDS configuration
507  // to dump
508 
509  // Set up options for the device running underneath
510  // FIXME: add some checksum in framework id. We could use this
511  // to avoid redeploys when only a portion of the workflow is changed.
512  // FIXME: this should probably be done in one go with char *, but I am lazy.
513  std::vector<std::string> tmpArgs = {
514  argv[0],
515  "--id",
516  spec.id.c_str(),
517  "--control",
518  "static",
519  "--log-color",
520  "0"
521  };
522 
523  // do the filtering of options, forward options belonging to this specific
524  // DeviceSpec, and some global options from getForwardedDeviceOptions
525  const char* name = spec.name.c_str();
526  bpo::options_description od;
527  prepareOptionsDescription(spec.options, od);
528  od.add(getForwardedDeviceOptions());
529  od.add_options()(name, bpo::value<std::string>());
530 
531  using FilterFunctionT = std::function<void (decltype(argc), decltype(argv),
532  decltype(od))>;
533 
534  FilterFunctionT filterArgsFct = [&] (int largc, char **largv,
535  const bpo::options_description &odesc) {
536  // spec contains options
537  bpo::command_line_parser parser{largc, largv};
538  parser.options(odesc).allow_unregistered();
539  bpo::parsed_options parsed_options = parser.run();
540 
541  bpo::variables_map varmap;
542  bpo::store(parsed_options, varmap);
543 
544  // options can be grouped per processor spec, the group is entered by
545  // the option created from the actual processor spec name
546  // if specified, the following string is interpreted as a sequence
547  // of arguments
548  if (varmap.count(name) > 0) {
549  // strangely enough, the first argument of the group argument string
550  // is marked as defaulted by the parser and is thus ignored. not fully
551  // understood but adding a dummy argument in front cures this
552  auto arguments = "--unused " + varmap[name].as<std::string>();
553  wordexp_t expansions;
554  wordexp(arguments.c_str(), &expansions, 0);
555  filterArgsFct(expansions.we_wordc, expansions.we_wordv, odesc);
556  wordfree(&expansions);
557  return;
558  }
559 
560  for (const auto varit : varmap) {
561  // find the option belonging to key, add if the option has been parsed
562  // and is not defaulted
563  const auto * description = odesc.find_nothrow(varit.first, false);
564  if (description && varmap.count(varit.first)) {
565  tmpArgs.emplace_back("--");
566  tmpArgs.back() += varit.first;
567  // check the semantics of the value
568  auto semantic = description->semantic();
569  const char* optarg = "";
570  if (semantic) {
571  // the value semantics allows different properties like
572  // multitoken, zero_token and composing
573  // currently only the simple case is supported
574  assert(semantic->min_tokens() <= 1);
575  assert(semantic->max_tokens() && semantic->min_tokens());
576  if (semantic->min_tokens() > 0 ) {
577  // add the token
578  tmpArgs.emplace_back(varit.second.as<std::string>());
579  optarg = tmpArgs.back().c_str();
580  }
581  }
582  control.options.insert(std::make_pair(varit.first,
583  optarg));
584  }
585  }
586  };
587 
588  filterArgsFct(argc, argv, od);
589 
590  // Add the channel configuration
591  for (auto &channel : spec.outputChannels) {
592  tmpArgs.emplace_back(std::string("--channel-config"));
593  tmpArgs.emplace_back(outputChannel2String(channel));
594  }
595  for (auto &channel : spec.inputChannels) {
596  tmpArgs.emplace_back(std::string("--channel-config"));
597  tmpArgs.emplace_back(inputChannel2String(channel));
598  }
599 
600  // We create the final option list, depending on the channels
601  // which are present in a device.
602  for (auto &arg : tmpArgs) {
603  execution.args.emplace_back(strdup(arg.c_str()));
604  }
605  // execvp wants a NULL terminated list.
606  execution.args.push_back(nullptr);
607 
608  //FIXME: this should probably be reflected in the GUI
609  std::ostringstream str;
610  for (size_t ai = 0; ai < execution.args.size() - 1; ai++) {
611  assert(execution.args[ai]);
612  str << " " << execution.args[ai];
613  }
614  LOG(DEBUG) << "The following options are being forwarded to "
615  << spec.id << ":" << str.str();
616  }
617 }
