199  code_type Encode(typename _BASE::value_type symbol, uint16_t& codeLength) const {
200  codeLength = 0;
201  auto nodeIndex = _BASE::alphabet_type::getIndex(symbol);
202  if (nodeIndex < mLeaveNodes.size()) {
203  // valid symbol/value
204  codeLength = mLeaveNodes[nodeIndex]->getBinaryCodeLength();
205  return mLeaveNodes[nodeIndex]->getBinaryCode();
206  } else {
207  std::string msg = "symbol "; msg += symbol;
208  msg += " not found in alphapet "; msg += _BASE::getName();
209  throw std::range_error(msg);
210  }

225  value_type Decode(code_type code, uint16_t& codeLength) const {
226  // TODO: need to check if there is a loaded tree, but don't
227  // want to check this every time when calling. Maybe its enough
228  // to let the dereferencing below throw an exception
229  codeLength = 0;
230  typename _BASE::value_type v = 0;
231  // dereference the iterator and get raw pointer from shared pointer
232  // TODO: work on shared pointers here as well
233  // the top node is the only element in the multiset after using the
234  // weighted sort algorithm to build the tree, all nodes are referenced
235  // from their parents in the tree.
236  const _NodeType* node = (*mTreeNodes.begin()).get();
237  uint16_t codeMSB = code.size() - 1;
238  while (node) {
239  // N.B.: nodes have either both child nodes or none of them
240  if (node->getLeftChild() == nullptr) {
241  // this is a leave node, retrieve value for corresponding index
242  // TODO: validity check for index, this can be done once after
243  // initializing the Huffman configuration, either after training
244  // or loading the configuration
245  return _BASE::alphabet_type::getSymbol(node->getIndex());
246  }
247  if (codeLength > codeMSB) {
248  // the size of the code type is shorter than the Huffman tree length
249  throw std::range_error("code type length insufficient for Huffman tree length");
250  break;
251  }
252  bool bit = false;
253  if (orderMSB) bit = code.test(codeMSB - codeLength);
254  else bit = code.test(codeLength);
255  ++codeLength;
256  if (bit) node = node->getLeftChild();
257  else node = node->getRightChild();
258  }
259  return v;
260  }

297  bool GenerateHuffmanTree() {
298  mLeaveNodes.clear();
299 
300  // probability model provides map of {symbol, weight}-pairs
301  _BASE& model = *this;
302  for ( auto i : model) {
303  // create nodes knowing about their index and the symbol weight
304  mLeaveNodes.push_back(std::make_shared<_NodeType>(i.second, _BASE::alphabet_type::getIndex(i.first)));
305  }
306 
307  // insert pointer to nodes into ordered structure to build tree
308  // since the type is a pointer, a specific 'less' functor needs to
309  // be provided to dereference before applying operator<
310  for ( auto &i : mLeaveNodes) {
311  mTreeNodes.insert(i);
312  }
313  while (mTreeNodes.size() > 1) {
314  // create new node combining the two with lowest probability
315  std::shared_ptr<_NodeType> combinedNode = std::make_shared<_NodeType>(*mTreeNodes.begin(), *++mTreeNodes.begin());
316  // remove those two nodes from the list
317  mTreeNodes.erase(mTreeNodes.begin());
318  mTreeNodes.erase(mTreeNodes.begin());
319  // insert the new node according to the less functor
320  mTreeNodes.insert(combinedNode);
321  }
322  //assign value, method works on pointer
323  // dereference iterator and shared_ptr to get the raw pointer
324  // TODO: change method to work on shared instead of raw pointers
325  assignCode((*mTreeNodes.begin()).get());
326  return true;
327  }

343  int assignCode(_NodeType* node) {
344  if (node == nullptr) return 0;
345  int codelen = node->getBinaryCodeLength();
346  int retcodelen = codelen;
347  if (node->getLeftChild()) {// bit '1' branch
348  code_type c = node->getBinaryCode();
349  if (orderMSB) {// note: this is a compile time switch
350  c <<= 1;
351  c.set(0);
352  } else {
353  c.set(codelen);
354  }
355  node->getLeftChild()->setBinaryCode(codelen+1, c);
356  int branchlen = assignCode(node->getLeftChild());
357  if (retcodelen < branchlen)
358  retcodelen = branchlen;
359  }
360  if (node->getRightChild()) {// bit '0' branch
361  code_type c = node->getBinaryCode();
362  if (orderMSB) {
363  c<<=1;
364  c.reset(0);
365  } else {
366  c.reset(codelen);
367  }
368  node->getRightChild()->setBinaryCode(codelen+1, c);
369  int branchlen = assignCode(node->getRightChild());
370  if (retcodelen < branchlen)
371  retcodelen = branchlen;
372  }
373  return retcodelen;
374  }

393  int read(std::istream& in) {
394  mLeaveNodes.clear();
395  mTreeNodes.clear();
396  int lineNo = -1;
397  std::string node, left, right, parameters;
398  std::set<int> nodeIndices;
399  struct treeNodeConfiguration {
400  treeNodeConfiguration(int _index, int _left, int _right) : index(_index), left(_left), right(_right) {}
401  int index; int left; int right;
402  bool operator<(const treeNodeConfiguration& other) const {return index < other.index;}
403  struct less {
404  bool operator()(const treeNodeConfiguration& a, const treeNodeConfiguration& b) {return a.index < b.index;}
405  };
406  };
407  std::set<treeNodeConfiguration> treeNodeConfigurations;
408  char firstChar = 0;
409  std::map<int, int> leaveNodeMap;
410  while ((in.get(firstChar)) && firstChar != '\n' && (in.putback(firstChar)) && (in >> node) && (in >> left) && (in >> right) && ++lineNo>=0) {
411  std::getline(in, parameters);
412  if (lineNo != std::stoi(node)) {
413  std::cerr << "format error: expected node no " << lineNo << ", but got " << node << " (" << left << " " << right << " " << parameters << ")" << std::endl;
414  std::cerr << "Note: Huffman table dump has to be terminated by blank line or eof" << std::endl;
415  break;
416  }
417  if (parameters.empty()) {
418  //std::cout << "tree node " << lineNo << " left=" << left << " right=" << right << std::endl;
419  int leftIndex = std::stoi(left);
420  int rightIndex = std::stoi(right);
421  auto it = nodeIndices.find(leftIndex);
422  if (it == nodeIndices.end()) {
423  std::cerr << "Format error: can not find left child node with index " << leftIndex << std::endl;
424  return -1;
425  }
426  nodeIndices.erase(it);
427  it = nodeIndices.find(rightIndex);
428  if (it == nodeIndices.end()) {
429  std::cerr << "Format error: can not find right child node with index " << rightIndex << std::endl;
430  return -1;
431  }
432  nodeIndices.erase(it);
433  treeNodeConfigurations.insert(treeNodeConfiguration(lineNo, leftIndex, rightIndex));
434  } else {
435  std::stringstream vs(left), ws(right);
436  typename _BASE::alphabet_type::value_type symbol; vs >> symbol;
437  typename _BASE::weight_type weight; ws >> weight;
438  int symbolIndex = _BASE::alphabet_type::getIndex(symbol);
439  // grow the vector as operator[] always expects index within range
440  if (mLeaveNodes.size() < symbolIndex + 1) mLeaveNodes.resize(symbolIndex + 1);
441  mLeaveNodes[symbolIndex] = std::make_shared<_NodeType>(weight, symbolIndex);
442  std::stringstream ps(parameters);
443  uint16_t codeLen = 0; ps >> codeLen;
444  code_type code = 0; ps >> code;
445  mLeaveNodes[symbolIndex]->setBinaryCode(codeLen, code);
446  leaveNodeMap[lineNo] = symbolIndex;
447  _BASE::addWeight(symbol, weight);
448  //std::cout << "leave node " << lineNo << " " << " value=" << value << " weight=" << weight << " " << codeLen << " " << code << std::endl;
449  }
450  nodeIndices.insert(lineNo);
451  }
452  std::map<int, std::shared_ptr<_NodeType>> treeNodes;
453  for (auto conf : treeNodeConfigurations) {
454  std::shared_ptr<_NodeType> left_local;
455  auto ln = leaveNodeMap.find(conf.left);
456  if ( ln != leaveNodeMap.end()) {
457  left_local = mLeaveNodes[ln->second];
458  leaveNodeMap.erase(ln);
459  } else {
460  auto tn = treeNodes.find(conf.left);
461  if (tn == treeNodes.end()) {
462  std::cerr << "Internal error: can not find left child node with index " << conf.left << std::endl;
463  return -1;
464  }
465  left_local = tn->second;
466  treeNodes.erase(tn);
467  }
468  std::shared_ptr<_NodeType> right_local;
469  auto rn = leaveNodeMap.find(conf.right);
470  if (rn != leaveNodeMap.end()) {
471  right_local = mLeaveNodes[rn->second];
472  leaveNodeMap.erase(rn);
473  } else {
474  auto tn = treeNodes.find(conf.right);
475  if (tn == treeNodes.end()) {
476  std::cerr << "Internal error: can not find right child node with index " << conf.right << std::endl;
477  return -1;
478  }
479  right_local = tn->second;
480  treeNodes.erase(tn);
481  }
482  // make combined node shared ptr and add to map
483  treeNodes[conf.index] = std::make_shared<_NodeType>(left_local, right_local);
484  }
485  if (leaveNodeMap.size() != 0 || treeNodes.size() != 1) {
486  std::cerr << "error: " << leaveNodeMap.size() << " unhandled leave node(s)"
487  << "; " << treeNodes.size() << " tree nodes(s), expected 1"
488  << std::endl;
489  }
490  mTreeNodes.insert(treeNodes.begin()->second);
491  return 0;
492  }
