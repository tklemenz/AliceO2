157 void Chip::MedianHitGlobal(const Hit *p1, const Hit *p2, Double_t &x, Double_t &y, Double_t &z) const
158 {
159  // Get hit positions in global coordinates
160  Double_t pos1Glob[3] = {p1->GetX(), p1->GetY(), p1->GetZ()},pos2Glob[3] = {p2->GetX(), p2->GetY(), p2->GetZ()};
161  Point3D<float> posMedianLocal;
162 
163  // Calculate mean positions
164  posMedianLocal.SetX(0.f);
165  if ((pos1Glob[1] * pos2Glob[1]) < 0.) {
166  posMedianLocal.SetY( (-pos1Glob[1] / (pos2Glob[1] - pos1Glob[1])) * (pos2Glob[0] - pos1Glob[0]) + pos1Glob[0] );
167  posMedianLocal.SetZ( (-pos1Glob[1] / (pos2Glob[1] - pos1Glob[1])) * (pos2Glob[2] - pos1Glob[2]) + pos1Glob[2] );
168  } else {
169  posMedianLocal.SetY( 0.5 * (pos1Glob[0] + pos2Glob[0]) );
170  posMedianLocal.SetZ( 0.5 * (pos1Glob[2] + pos2Glob[2]) );
171  }
172 
173  // Convert to global coordinates
174  auto posMedianGlobal = (*mMat)(posMedianLocal);
175  x = posMedianGlobal.X();
176  y = posMedianGlobal.Y();
177  z = posMedianGlobal.Z();
178 }

181 void Chip::MedianHitLocal(const Hit *p1, const Hit *p2, Double_t &x, Double_t &y, Double_t &z) const
182 {
183  // Convert hit positions into local positions inside the chip
184  auto pos1Loc = (*mMat)^(p1->GetPos());
185  auto pos2Loc = (*mMat)^(p2->GetPos());
186 
187  // Calculate mean positions
188  y = 0.;
189  if ((pos1Loc.Y() * pos2Loc.Y()) < 0.) {
190  x = (-pos1Loc.Y() / (pos2Loc.Y() - pos1Loc.Y())) * (pos2Loc.X() - pos1Loc.X()) + pos1Loc.X();
191  z = (-pos1Loc.Y() / (pos2Loc.Y() - pos1Loc.Y())) * (pos2Loc.Z() - pos1Loc.Z()) + pos1Loc.Z();
192  } else {
193  x = 0.5 * (pos1Loc.X() + pos2Loc.X());
194  z = 0.5 * (pos1Loc.Z() + pos2Loc.Z());
195  }
196 }

199 Digit* Chip::addDigit(UInt_t roframe, UShort_t row, UShort_t col, float charge, Label lbl, double timestamp)
200 {
201  auto key = Digit::getOrderingKey(roframe,row,col);
202  auto dig = findDigit(key);
203  if (dig) {
204  dig->addCharge(charge, lbl);
205  }
206  else {
207  auto digIter= mDigits.emplace(std::make_pair
208  (key,Digit(static_cast<UShort_t>(mChipIndex),roframe, row, col, charge, timestamp)));
209  auto pair = digIter.first;
210  dig = &(pair->second);
211  dig->setLabel(0, lbl);
212  }
213  return dig;
214 }

217 void Chip::fillOutputContainer(std::vector<Digit>* digits, UInt_t maxFrame)
218 {
219  // transfer digits with RO Frame < maxFrame to the output array
220  if (mDigits.empty()) return;
221  auto itBeg = mDigits.begin();
222  auto iter = itBeg;
223  ULong64_t maxKey = Digit::getOrderingKey(maxFrame+1,0,0);
224  for (; iter!=mDigits.end(); ++iter) {
225  if (iter->first > maxKey) break; // is the digit ROFrame from the key > the max requested frame
226  // apply thrshold
227  Digit &dig = iter->second;
228  //printf("Chip%d Fr:%d Q:%f R:%d C:%d\n",dig.getChipIndex(),dig.getROFrame(),dig.getCharge(), dig.getRow(),dig.getColumn());
229 
230  if (dig.getCharge()>mParams->getChargeThreshold() ) {
231  digits->emplace_back(dig);
232  }
233  }
234 
235  // if (iter!=mDigits.end()) iter--;
236  mDigits.erase(itBeg, iter);
