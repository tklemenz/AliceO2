44 //This is a prototype of a macro to test running the HLT O2 CA Tracking library on a root input file containg TClonesArray of clusters.

46 int runCATrackingClusterNative(TString inputFile, TString outputFile, TString options="") {
47  if (inputFile.EqualTo("") || outputFile.EqualTo("")) {printf("Filename missing\n");return(1);}
48  TPCCATracking tracker;
49 
50  if (tracker.initialize(options.Data())) {
51  printf("Error initializing tracker\n");
52  return(0);
53  }
54 
55  std::vector<ClusterNativeContainer> cont;
56  std::vector<MCLabelContainer> contMC;
57  bool doMC = true;
58 
59  TFile fin(inputFile);
60  for (int i = 0;i < Constants::MAXSECTOR;i++)
61  {
62  for (int j = 0;j < Constants::MAXGLOBALPADROW;j++)
63  {
64  TString contName = Form("clusters_sector_%d_row_%d", i, j);
65  TObject* tmp = fin.FindObjectAny(contName);
66  if (tmp == nullptr) {
67  printf("Error reading clusters %s\n", contName.Data());
68  } else {
69  cont.emplace_back(std::move(*reinterpret_cast<ClusterNativeContainer*>(tmp)));
70  tmp = fin.FindObjectAny(Form("clustersMCTruth_sector_%d_row_%d", i, j));
71 
72  if (tmp == nullptr) {
73  printf("Error, clustersMCTruth missing or clusters and clustersMCtruth out of sync! Disabling MC data\n");
74  doMC = false;
75  } else {
76  contMC.emplace_back(std::move(*reinterpret_cast<MCLabelContainer*>(tmp)));
77  }
78  }
79  }
80  }
81  fin.Close();
82 
83  std::unique_ptr<ClusterNativeAccessFullTPC> clusters = TPCClusterFormatHelper::accessNativeContainerArray(cont, doMC ? &contMC : nullptr);
84 
85  vector<TrackTPC> tracks;
86  MCLabelContainer tracksMC;
87 
88  TFile fout(outputFile, "recreate");
89  TTree tout("events","events");
90  tout.Branch("Tracks", &tracks);
91  tout.Branch("TracksMCTruth", &tracksMC);
92 
93  printf("Processing time frame\n");
94  if (tracker.runTracking(*clusters, &tracks, doMC ? &tracksMC : nullptr) == 0) {
95  printf("\tFound %d tracks\n", (int) tracks.size());
96  } else {
97  printf("\tError during tracking\n");
98  }
99 
100  float artificialVDrift = tracker.getPseudoVDrift();
101  float tfReferenceLength = tracker.getTFReferenceLength();
102  unsigned int nTracksASide = tracker.getNTracksASide();
103  for (unsigned int i = 0;i < tracks.size();i++)
104  {
105  bool isASide = i < nTracksASide;
106  if (isASide != (tracks[i].getSide() == Side::A)) printf("Incorrect sorting\n");
107  //Loop over clusters
108  for (int j = tracks[i].getNClusterReferences() - 1;j >= 0;j--)
109  {
110  //Get cluster references
111  uint8_t sector, row;
112  uint32_t clusterIndexInRow;
113  tracks[i].getClusterReference(j, sector, row, clusterIndexInRow);
114  const ClusterNative& cl = tracks[i].getCluster(j, *clusters, sector, row);
115  const ClusterNative& clLast = tracks[i].getCluster(0, *clusters);
116  float sideFactor = tracks[i].getSide() == Side::A ? -1.f : 1.f;
117 
118  printf("Track %d: Side %c Estimated timeVertex: %f, num clusters %d, innermost cluster: sector %d, row %d, ClusterTime %f, TrackParam X %f Z %f --> T %f, LastClusterZ %f --> T %f (T from cluster itself %f)\n",
119  i, tracks[i].getSide() == Side::A ? 'A' : 'C', tracks[i].getTime0(), tracks[i].getNClusterReferences(), (int) sector, (int) row, cl.getTime(), tracks[i].getX(),
120  tracks[i].getZ(), tracks[i].getTime0() - sideFactor * tracks[i].getZ() / artificialVDrift,
121  tracks[i].getLastClusterZ(), tracks[i].getTime0() - sideFactor * tracks[i].getLastClusterZ() / artificialVDrift, clLast.getTime());
122  break; //Reduce output in this example code
123  }
124  }
125 
126  tout.Fill();
127  fout.Write();
128  fout.Close();
129 
130  tracker.deinitialize();
131  return(0);
132 }
