234 /******************************************************************************
235  * @brief apply functor to the wrapped member object in the runtime container
236  * This meta function recurses through the list while incrementing the index
237  * and calls the functor at the required position
238  *
239  * @note internal meta function for the RuntimeContainers' apply function
240  */
241 template <
242  typename _ContainerT // container type
243  , typename _IndexT // data type of position index
244  , typename _Iterator // current iterator position
245  , typename _End // end iterator position
246  , _IndexT _Index // current index
247  , typename F // functor
248  >
249 struct rc_apply_at
250 {
251  static typename F::return_type apply( _ContainerT& c, _IndexT position, F& f )
252  {
253  if ( position == _Index ) {
254  // this is the queried position, make the type cast to the current
255  // stage of the runtime container and execute function for it.
256  // Terminate loop by forwarding _End as _Iterator and thus
257  // calling the specialization
258  typedef typename boost::mpl::deref< _Iterator >::type stagetype;
259  stagetype& stage = static_cast<stagetype&>(c);
260  return f(stage);
261  } else {
262  // go to next element
263  return rc_apply_at<
264  _ContainerT
265  , _IndexT
266  , typename boost::mpl::next< _Iterator >::type
267  , _End
268  , _Index + 1
269  , F
270  >::apply( c, position, f );
271  }
272  }
273 };
274 // specialization: end of recursive loop, kicks in if _Iterator matches
275 // _End.
276 // here we end up if the position parameter is out of bounds

466  /*
467  * Apply a functor to the runtime container at index
468  *
469  * For performance tests there is a template option to do an explicite loop
470  * unrolling for the first n (=10) elements. This is however only effective
471  * if the compiler optimization is switched of. This is in the end a nice
472  * demonstrator for the potential of compiler optimization. Unrolling is
473  * switched on with the compile time switch RC_UNROLL.
474  */
475  template<typename F
476 #ifdef RC_UNROLL
477  , bool unroll = true
478 #else
479  , bool unroll = false
480 #endif
481  >
482  typename F::return_type apply(int index, F f) {
483  if (unroll) {// this is a compile time switch
484  // do unrolling for the first n elements and forward to generic
485  // recursive function for the rest.
486  switch (index) {
487  case 0: return rc_dispatcher<mixin_type, F, boost::mpl::int_<0>, int>::apply(*this, 0, f);
488  case 1: return rc_dispatcher<mixin_type, F, boost::mpl::int_<1>, int>::apply(*this, 1, f);
489  case 2: return rc_dispatcher<mixin_type, F, boost::mpl::int_<2>, int>::apply(*this, 2, f);
490  case 3: return rc_dispatcher<mixin_type, F, boost::mpl::int_<3>, int>::apply(*this, 3, f);
491  case 4: return rc_dispatcher<mixin_type, F, boost::mpl::int_<4>, int>::apply(*this, 4, f);
492  case 5: return rc_dispatcher<mixin_type, F, boost::mpl::int_<5>, int>::apply(*this, 5, f);
493  case 6: return rc_dispatcher<mixin_type, F, boost::mpl::int_<6>, int>::apply(*this, 6, f);
494  case 7: return rc_dispatcher<mixin_type, F, boost::mpl::int_<7>, int>::apply(*this, 7, f);
495  case 8: return rc_dispatcher<mixin_type, F, boost::mpl::int_<8>, int>::apply(*this, 8, f);
496  case 9: return rc_dispatcher<mixin_type, F, boost::mpl::int_<9>, int>::apply(*this, 9, f);
497  }
498  }
499  return rc_dispatcher<mixin_type, F>::apply(*this, index, f);
500  }

503  T mMember;

534  typedef typename boost::mpl::lambda<
535  // mpl fold loops over all elements in the list of the first template
536  // parameter and provides this as placeholder _2; for every element the
537  // operation of the third template parameter is applied to the result of
538  // the previous stage which is provided as placeholder _1 to the operation
