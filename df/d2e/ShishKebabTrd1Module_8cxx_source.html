102 void ShishKebabTrd1Module::Init(Double_t A, Double_t B)
103 {
104  // Define parameter module from parameters A,B from previous.
105  Double_t yl = (sb / 2) * TMath::Sin(mTheta) + (sa / 2) * TMath::Cos(mTheta) + sr, y = yl;
106  Double_t xl = (yl - B) / A; // y=A*x+B
107 
108  // Double_t xp1 = (fga/2. + fgb/2.*fgtanBetta)/(TMath::Sin(fTheta) + fgtanBetta*TMath::Cos(fTheta));
109  // printf(" xp1 %9.3f \n ", xp1);
110  // xp1 == xp => both methods give the same results - 3-feb-05
111  Double_t alpha = TMath::Pi() / 2. + sangle / 2;
112  Double_t xt =
113  (sa + sa2) * TMath::Tan(mTheta) * TMath::Tan(alpha) / (4. * (1. - TMath::Tan(mTheta) * TMath::Tan(alpha)));
114  Double_t yt = xt / TMath::Tan(mTheta), xp = TMath::Sqrt(xt * xt + yt * yt);
115  Double_t x = xl + xp;
116  mOK.Set(x, y);
117  // printf(" yl %9.3f | xl %9.3f | xp %9.3f \n", yl, xl, xp);
118 
119  // have to define A and B;
120  Double_t yCprev = sr + sa * TMath::Cos(mTheta);
121  Double_t xCprev = (yCprev - B) / A;
122  Double_t xA = xCprev + sa * TMath::Sin(mTheta), yA = sr;
123 
124  mThetaA = mTheta - sangle / 2.;
125  mA = TMath::Tan(mThetaA); // !!
126  mB = yA - mA * xA;
127 
128  DefineAllStuff();
129 }

131 void ShishKebabTrd1Module::DefineAllStuff()
132 {
133  // Define some parameters
134  // DefineName(mTheta);
135  // Centers of cells - 2X2 case
136  Double_t kk1 = (sa + sa2) / (2. * 4.); // kk1=kk2
137 
138  Double_t xk1 = mOK.X() - kk1 * TMath::Sin(mTheta);
139  Double_t yk1 = mOK.Y() + kk1 * TMath::Cos(mTheta) - sr;
140  mOK1.Set(xk1, yk1);
141 
142  Double_t xk2 = mOK.X() + kk1 * TMath::Sin(mTheta);
143  Double_t yk2 = mOK.Y() - kk1 * TMath::Cos(mTheta) - sr;
144  mOK2.Set(xk2, yk2);
145 
146  // Centers of cells - 3X3 case; Nov 9,2006
147  mOK3X3[1].Set(mOK.X(), mOK.Y() - sr); // coincide with module center
148 
149  kk1 = ((sa + sa2) / 4. + sa / 6.) / 2.;
150 
151  xk1 = mOK.X() - kk1 * TMath::Sin(mTheta);
152  yk1 = mOK.Y() + kk1 * TMath::Cos(mTheta) - sr;
153  mOK3X3[0].Set(xk1, yk1);
154 
155  xk2 = mOK.X() + kk1 * TMath::Sin(mTheta);
156  yk2 = mOK.Y() - kk1 * TMath::Cos(mTheta) - sr;
157  mOK3X3[2].Set(xk2, yk2);
158 
159  // May 15, 2006; position of module(cells) center face
160  mOB.Set(mOK.X() - sb / 2. * TMath::Cos(mTheta), mOK.Y() - sb / 2. * TMath::Sin(mTheta) - sr);
161  mOB1.Set(mOB.X() - sa / 4. * TMath::Sin(mTheta), mOB.Y() + sa / 4. * TMath::Cos(mTheta));
162  mOB2.Set(mOB.X() + sa / 4. * TMath::Sin(mTheta), mOB.Y() - sa / 4. * TMath::Cos(mTheta));
163  // Jul 30, 2007 - for taking into account a position of shower maximum
164  mThetaOB1 = mTheta - sangle / 4.; // ??
165  mThetaOB2 = mTheta + sangle / 4.;
166 
167  // Position of right/top point of module
168  // Gives the posibility to estimate SM size in z direction
169  Double_t xBottom = (sr - mB) / mA;
170  Double_t yBottom = sr;
171  mORB.Set(xBottom, yBottom);
172 
173  Double_t l = sb / TMath::Cos(sangle / 2.); // length of lateral module side
174  Double_t xTop = xBottom + l * TMath::Cos(TMath::ATan(mA));
175  Double_t yTop = mA * xTop + mB;
176  mORT.Set(xTop, yTop);
177 }

179 void ShishKebabTrd1Module::DefineFirstModule(const Int_t key)
180 {
181  // Define first module
182  if (key == 0) {
183  // theta in radians ; first object theta=pi/2.
184  mTheta = TMath::PiOver2();
185  mOK.Set(sa2 / 2., sr + sb / 2.); // position the center of module vs o
186 
187  // parameters of right line : y = A*z + B in system where zero point is IP.
188  mThetaA = mTheta - sangle / 2.;
189  mA = TMath::Tan(mThetaA);
190  Double_t xA = sa / 2. + sa2 / 2.;
191  Double_t yA = sr;
192  mB = yA - mA * xA;
193  } else if (key == 1) {
194  // theta in radians ; first object theta = 90-0.75 = 89.25 degree
195  mTheta = 89.25 * TMath::DegToRad();
196  Double_t al1 = sangle / 2.;
197  Double_t x = 0.5 * (sa * TMath::Cos(al1) + sb * TMath::Sin(al1));
198  Double_t y = 0.5 * (sb + sa * TMath::Sin(al1)) * TMath::Cos(al1);
199  mOK.Set(x, sr + y);
200  // parameters of right line : y = A*z + B in system where zero point is IP.
201  mThetaA = mTheta - sangle / 2.;
202  mA = TMath::Tan(mThetaA);
203  Double_t xA = sa * TMath::Cos(al1);
204  Double_t yA = sr;
205  mB = yA - mA * xA;
206  } else {
207  LOG(ERROR) << "key=" << key << " : wrong case \n";
208  assert(0);
209  }
210 
211  DefineAllStuff();
212 }
