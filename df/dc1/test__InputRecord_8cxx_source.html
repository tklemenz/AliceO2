65  auto transport = FairMQTransportFactory::CreateTransportFactory("zeromq");

68  auto createMessage = [&transport, &inputs] (DataHeader &dh, int value) {
69  DataProcessingHeader dph{0,1};
70  Stack stack{dh, dph};
71  FairMQMessagePtr header = transport->CreateMessage(stack.size());
72  FairMQMessagePtr payload = transport->CreateMessage(sizeof(int));
73  memcpy(header->GetData(), stack.data(), stack.size());
74  memcpy(payload->GetData(), &value, sizeof(int));
75  inputs.emplace_back(std::move(header));
76  inputs.emplace_back(std::move(payload));
77  };
