93 bool Tracker::CellParams(int l, const ClsInfo_t& c1, const ClsInfo_t& c2, const ClsInfo_t& c3,
94  float &curv, array<float,3> &n) {
95  // Calculation of cell params and filtering using a DCA cut wrt beam line position.
96  // The hit are mapped on a paraboloid space: there a circle is described as plane.
97  // The parameter n of the cells is the normal vector to the plane describing the circle in the
98  // paraboloid.
99 
100  // Mapping the hits
101  const array<float,3> mHit0{c1.x, c1.y, c1.r * c1.r};
102  const array<float,3> mHit1{c2.x, c2.y, c2.r * c2.r};
103  const array<float,3> mHit2{c3.x, c3.y, c3.r * c3.r};
104  // Computing the deltas
105  const array<float,3> mD10{mHit1[0] - mHit0[0],mHit1[1] - mHit0[1],mHit1[2] - mHit0[2]};
106  const array<float,3> mD20{mHit2[0] - mHit0[0],mHit2[1] - mHit0[1],mHit2[2] - mHit0[2]};
107  // External product of the deltas -> n
108  n[0] = (mD10[1] * mD20[2]) - (mD10[2] * mD20[1]);
109  n[1] = (mD10[2] * mD20[0]) - (mD10[0] * mD20[2]);
110  n[2] = (mD10[0] * mD20[1]) - (mD10[1] * mD20[0]);
111  // Normalisation
112  float norm = sqrt((n[0] * n[0]) + (n[1] * n[1]) + (n[2] * n[2]));
113  if (norm < 1e-20f || fabs(n[2]) < 1e-20f)
114  return false;
115  else
116  norm = 1.f / norm;
117  n[0] *= norm;
118  n[1] *= norm;
119  n[2] *= norm;
120  // Center of the circle
121  const float c[2]{-0.5f * n[0] / n[2], -0.5f * n[1] / n[2]};
122  // Constant
123  const float k = - n[0] * mHit1[0] - n[1] * mHit1[1] - n[2] * mHit1[2];
124  // Radius of the circle
125  curv = sqrt((1.f - n[2] * n[2] - 4.f * k * n[2]) / (4.f * n[2] * n[2]));
126  // Distance of closest approach to the beam line
127  const float dca = fabs(curv - sqrt(c[0] * c[0] + c[1] * c[1]));
128  // Cut on the DCA
129  if (dca > mCDCAxy[l]) {
130  return false;
131  }
132 
133  curv = 1.f / curv;
134  return true;
135 }

137 void Tracker::CellsTreeTraversal(vector<Road> &roads,
138  const int &iD, const int &doubl) {
139 
140  // Each cells contains a list of neighbours. Each neighbour has presumably other neighbours.
141  // This chain of neighbours is, as a matter of fact, a tree and this function goes recursively
142  // through it. This function implements a depth first tree browsing.
143 
144  // End of the road
145  if (doubl < 0) return;
146 
147  // [1] add current cell to current cell
148  roads.back().AddElement(doubl,iD);
149  // We want the right number of elements in the roads also in the case of multiple neighbours
150  const int currentN = roads.back().N;
151 
152  // [2] loop on the neighbours of the current cell
153  for (size_t iN = 0; iN < mCells[doubl][iD].NumberOfNeighbours(); ++iN) {
154  const int currD = doubl - 1;
155  const int neigh = mCells[doubl][iD](iN);
156 
157  // [3] for each neighbour one road
158  if (iN > 0) {
159  roads.push_back(roads.back());
160  roads.back().N = currentN;
161  }
162  // [4] play this game again until the end of the road
163  CellsTreeTraversal(roads,neigh,currD);
164  }
165 
166  mCells[doubl][iD].SetLevel(0u); // Level = -1
167 }

245 void Tracker::FindTracksCA(int iteration) {
246  // Main pattern recognition routine. It has 4 steps (planning to split in different methods)
247  // 1. Tracklet finding (using vertex position)
248  // 2. Tracklet association, cells building
249  // 3. Handshake between neighbour cells
250  // 4. Candidates ("roads") finding and fitting
251 
252  // Road finding and fitting. The routine starts from cells with level 5 since they are the head
253  // of a full road (candidate with 7 points). Minimum level is 2, so candidates at the end have
254  // at least 4 points.
255  const int itLevelLimit[3] = {4, 4, 1};
256  for (int level = 5; level > itLevelLimit[iteration]; --level) {
257  vector<Road> roads;
258  // Road finding. For each cell at level $(level) a loop on their neighbours to start building
259  // the roads.
260  for (int iCL = 4; iCL >= level - 1; --iCL) {
261  for (size_t iCell = 0; iCell < mCells[iCL].size(); ++iCell) {
262  if (mCells[iCL][iCell].GetLevel() != level)
263  continue;
264  // [1] Add current cell to road
265  roads.emplace_back(iCL,iCell);
266  // [2] Loop on current cell neighbours
267  for(size_t iN = 0; iN < mCells[iCL][iCell].NumberOfNeighbours(); ++iN) {
268  const int currD = iCL - 1;
269  const int neigh = mCells[iCL][iCell](iN);
270  // [3] if more than one neighbour => more than one road, one road for each neighbour
271  if(iN > 0) {
272  roads.emplace_back(iCL,iCell);
273  }
274  // [4] Essentially the neighbour became the current cell and then go to [1]
275  CellsTreeTraversal(roads,neigh,currD);
276  }
277  mCells[iCL][iCell].SetLevel(0u); // Level = -1
278  }
279  }
280 
281  // Roads fitting
282  for (size_t iR = 0; iR < roads.size(); ++iR) {
283  if (roads[iR].N != level)
284  continue;
285  int indices[7] = {-1};
286  int first = -1,last = -1;
287  for(int i = 0; i < 5; ++i) {
288  if (roads[iR][i] < 0)
289  continue;
290 
291  if (first < 0) {
292  indices[i] = mCells[i][roads[iR][i]].x();
293  indices[i + 1] = mCells[i][roads[iR][i]].y();
294  first = i;
295  }
296  indices[i + 2] = mCells[i][roads[iR][i]].z();
297  last = i;
298  }
299  const int mid = (last + first) / 2;
300  const Cluster& cl0 = (*mLayer[first])[mCells[first][roads[iR][first]].x()];
301  const Cluster& cl1 = (*mLayer[mid + 1])[mCells[mid][roads[iR][mid]].y()];
302  const Cluster& cl2 = (*mLayer[last + 2])[mCells[last][roads[iR][last]].z()];
303  // Init track parameters
304  float cv = Curvature(cl0.x,cl0.y,cl1.x,cl1.y,cl2.x,cl2.y);
305  float tgl = TanLambda(cl0.x,cl0.y,cl2.x,cl2.y,cl0.z,cl2.z);
306 
307  ITSDetInfo_t det = (*mLayer[last + 2]).GetDetInfo(cl2.detid);
308  float x = det.xTF + cl2.x; // I'd like to avoit using AliITSUClusterPix...
309  float alp = det.phiTF;
310  std::array<float,5> par {cl2.y,cl2.z,0,tgl,cv};
311  std::array<float,15> cov {
312  5.f*5.f,
313  0.f, 5.f*5.f,
314  0.f, 0.f , 0.7f*0.7f,
315  0.f, 0.f, 0.f, 0.7f*0.7f,
316  0.f, 0.f, 0.f, 0.f, 10.f
317  };
318  Track tt{x,alp,par,cov,indices};
319  if (RefitAt(2.1, &tt))
320  mCandidates[level - 2].push_back(tt);
321  }
322  }
323 }

338 void Tracker::MakeCells(int iteration) {
339 
340  SetCuts(iteration);
341  if (iteration >= 1) {
342  for (int i = 0; i < 5; ++i)
343  vector<Cell>().swap(mCells[i]);
344  for (int i = 0; i < 6; ++i)
345  vector<Doublets>().swap(mDoublets[i]);
346  }
347 
348  // Trick to speed up the navigation of the doublets array. The lookup table is build like:
349  // dLUT[l][i] = n;
350  // where n is the index inside mDoublets[l+1] of the first doublets that uses the point
351  // mLayer[l+1][i]
352  vector<int> dLUT[5];
353  for (int iL = 0; iL < 6; ++iL) {
354  if (mLayer[iL]->GetNClusters() == 0) continue;
355  if (iL < 5)
356  dLUT[iL].resize((*mLayer[iL + 1]).GetNClusters(),-1);
357  if (dLUT[iL - 1].size() == 0u)
358  continue;
359  for (int iC = 0; iC < mLayer[iL]->GetNClusters(); ++iC) {
360  const ClsInfo_t& cls = mLayer[iL]->GetClusterInfo(iC);
361  if (mUsedClusters[iL][iC]) {
362  continue;
363  }
364  const float tanL = (cls.z - GetZ()) / cls.r;
365  const float extz = tanL * (mkR[iL + 1] - cls.r) + cls.z;
366  const int nClust = mLayer[iL + 1]->SelectClusters(extz - 2 * mCZ, extz + 2 * mCZ,
367  cls.phi - mCPhi, cls.phi + mCPhi);
368  bool first = true;
369 
370  for (int iC2 = 0; iC2 < nClust; ++iC2) {
371  const int iD2 = mLayer[iL + 1]->GetNextClusterInfoID();
372  const ClsInfo_t& cls2 = mLayer[iL + 1]->GetClusterInfo(iD2);
373  if (mUsedClusters[iL + 1][iC2]) {
374  continue;
375  }
376  const float dz = tanL * (cls2.r - cls.r) + cls.z - cls2.z;
377  if (fabs(dz) < mCDZ[iL] && CompareAngles(cls.phi, cls2.phi, mCPhi)) {
378  if (first && iL > 0) {
379  dLUT[iL - 1][iC] = mDoublets[iL].size();
380  first = false;
381  }
382  const float dTanL = (cls.z - cls2.z) / (cls.r - cls2.r);
383  const float phi = atan2(cls.y - cls2.y, cls.x - cls2.x);
384  mDoublets[iL].emplace_back(iC,iD2,dTanL,phi);
385  }
386  }
387  mLayer[iL + 1]->ResetFoundIterator();
388  }
389  }
390 
391  // Trick to speed up the navigation of the cells array. The lookup table is build like:
392  // tLUT[l][i] = n;
393  // where n is the index inside mCells[l+1] of the first cells that uses the doublet
394  // mDoublets[l+1][i]
395  vector<int> tLUT[4];
396  tLUT[0].resize(mDoublets[1].size(),-1);
397  tLUT[1].resize(mDoublets[2].size(),-1);
398  tLUT[2].resize(mDoublets[3].size(),-1);
399  tLUT[3].resize(mDoublets[4].size(),-1);
400 
401  for (int iD = 0; iD < 5; ++iD) {
402  if (mDoublets[iD + 1].size() == 0u || mDoublets[iD].size() == 0u) continue;
403 
404  for (size_t iD0 = 0; iD0 < mDoublets[iD].size(); ++iD0) {
405  const int idx = mDoublets[iD][iD0].y;
406  bool first = true;
407  if (dLUT[iD][idx] == -1) continue;
408  for (size_t iD1 = dLUT[iD][idx]; iD1 < mDoublets[iD + 1].size(); ++iD1) {
409  if (idx != mDoublets[iD + 1][iD1].x) break;
410  if (fabs(mDoublets[iD][iD0].tanL - mDoublets[iD + 1][iD1].tanL) < mCDTanL &&
411  fabs(mDoublets[iD][iD0].phi - mDoublets[iD + 1][iD1].phi) < mCDPhi) {
412  const float tan = 0.5f * (mDoublets[iD][iD0].tanL + mDoublets[iD + 1][iD1].tanL);
413  const float extz = -tan * (*mLayer[iD])[mDoublets[iD][iD0].x].r +
414  (*mLayer[iD])[mDoublets[iD][iD0].x].z;
415  if (fabs(extz - GetZ()) < mCDCAz[iD]) {
416  float curv = 0.f;
417  array<float,3> n {0.f};
418  if (CellParams(iD,(*mLayer[iD])[mDoublets[iD][iD0].x],(*mLayer[iD + 1])[mDoublets[iD][iD0].y],
419  (*mLayer[iD + 2])[mDoublets[iD + 1][iD1].y],curv,n)) {
420  if (first && iD > 0) {
421  tLUT[iD - 1][iD0] = mCells[iD].size();
422  first = false;
423  }
424  mCells[iD].emplace_back(mDoublets[iD][iD0].x,mDoublets[iD][iD0].y,
425  mDoublets[iD + 1][iD1].y,iD0,iD1,curv,n);
426  }
427  }
428  }
429  }
430  }
431  }

433  // Adjacent cells: cells that share 2 points. In the following code adjacent cells are combined.
434  // If they meet some requirements (~ same curvature, ~ same n) the innermost cell id is added
435  // to the list of neighbours of the outermost cell. When the cell is added to the neighbours of
436  // the outermost cell the "level" of the latter is set to the level of the innermost one + 1.
437  // ( only if $(level of the innermost) + 1 > $(level of the outermost) )
438  for (int iD = 0; iD < 4; ++iD) {
439  if (mCells[iD + 1].size() == 0u || tLUT[iD].size() == 0u) continue; // TODO: dealing with holes
440  for (size_t c0 = 0; c0 < mCells[iD].size(); ++c0) {
441  const int idx = mCells[iD][c0].d1();
442  if (tLUT[iD][idx] == -1) continue;
443  for (size_t c1 = tLUT[iD][idx]; c1 < mCells[iD + 1].size(); ++c1) {
444  if (idx != mCells[iD + 1][c1].d0()) break;
445  auto& n0 = mCells[iD][c0].GetN();
446  auto& n1 = mCells[iD + 1][c1].GetN();
447  const float dn2 = ((n0[0] - n1[0]) * (n0[0] - n1[0]) + (n0[1] - n1[1]) * (n0[1] - n1[1]) +
448  (n0[2] - n1[2]) * (n0[2] - n1[2]));
449  const float dp = fabs(mCells[iD][c0].GetCurvature() - mCells[iD + 1][c1].GetCurvature());
450  if (dn2 < mCDN[iD] && dp < mCDP[iD]) {
451  mCells[iD + 1][c1].Combine(mCells[iD][c0], c0);
452  }
453  }
454  }
455  }

493 bool Tracker::RefitAt(float xx, Track *track) {
494  // This function refits the track "t" at the position "x" using
495  // the clusters from "c"
496 
497  const int nLayers = 7;
498  int* index = track->Clusters();
499 
500  int from, to, step;
501  if (xx > track->getX()) {
502  from = 0;
503  to = nLayers;
504  step = +1;
505  } else {
506  from = nLayers - 1;
507  to = -1;
508  step = -1;
509  }
510 
511  for (int i = from; i != to; i += step) {
