111 void DigitizerTask::Exec(Option_t *option)
112 {
113  FairRootManager *mgr = FairRootManager::Instance();
114 
115  // time should be given in us
116  float eventTime = static_cast<float>(mgr->GetEventTime() * 0.001);
117  if (mEventTimes.size()) {
118  eventTime = mEventTimes[mCurrentEvent++];
119  LOG(DEBUG) << "Event time taken from bunch simulation";
120  }
121  const int eventTimeBin = Digitizer::getTimeBinFromTime(eventTime);
122 
123  LOG(DEBUG) << "Running digitization on new event at time " << eventTime << " us in time bin " << eventTimeBin << FairLogger::endl;
124  mDigitsArray->clear();
125  mMCTruthArray->clear();
126  if(mDigitDebugOutput) {
127  mDigitsDebugArray->clear();
128  }
129 
130  if (mHitSector == -1){
131  // treat all sectors
132  for (int s=0; s<Sector::MAXSECTOR; ++s){
133  LOG(DEBUG) << "Processing sector " << s << "\n";
134  mDigitContainer = mDigitizer->Process(*mSectorHitsArray[s], eventTime);
135  }
136  }
137  else {
138  // treat only chosen sector
139  mDigitContainer = mDigitizer->Process(*mSectorHitsArray[mHitSector], eventTime);
140  }
141  mDigitContainer->fillOutputContainer(mDigitsArray, *mMCTruthArray, mDigitsDebugArray, eventTimeBin, mIsContinuousReadout);
142 }

157 void DigitizerTask::initBunchTrainStructure(const size_t numberOfEvents)
158 {
159  LOG(DEBUG) << "Initialising bunch train structure for " << numberOfEvents << " evnets" << FairLogger::endl;
160  // Parameters for bunches
161  const double abortGap = 3e-6; //
162  const double collFreq = 50e3;
163  const double bSpacing = 50e-9; //bunch spacing
164  const int nTrainBunches = 48;
165  const int nTrains = 12;
166  const double revFreq = 1.11e4; //revolution frequency
167  const double collProb = collFreq/(nTrainBunches*nTrains*revFreq);
168  const double trainLength = bSpacing*(nTrainBunches-1);
169  const double totTrainLength = nTrains*trainLength;
170  const double trainSpacing = (1./revFreq - abortGap - totTrainLength)/(nTrains-1);
171 
172  // counters
173  double eventTime=0.; // all in seconds
174  size_t nGeneratedEvents = 0;
175  size_t bunchCounter = 0;
176  size_t trainCounter = 0;
177 
178  // reset vector
179  mEventTimes.clear();
180 
181  while (nGeneratedEvents < numberOfEvents+2){
182  // std::cout <<trainCounter << " " << bunchCounter << " "<< "eventTime " << eventTime << std::endl;
183 
184  int nCollsInCrossing = gRandom -> Poisson(collProb);
185  for(int iColl = 0; iColl<nCollsInCrossing; iColl++){
186  //printf("Generating event %3d (%.3g)\n",nGeneratedEvents,eventTime);
187  mEventTimes.emplace_back(static_cast<float>(eventTime*1e6)); // convert to us
188  nGeneratedEvents++;
189  }
190  bunchCounter++;
191 
192  if(bunchCounter>=nTrainBunches){
193 
194  trainCounter++;
195  if(trainCounter>=nTrains){
196  eventTime+=abortGap;
197  trainCounter=0;
198  }
199  else eventTime+=trainSpacing;
200 
201  bunchCounter=0;
202  }
203  else eventTime+= bSpacing;
204 
205  }
206 }
