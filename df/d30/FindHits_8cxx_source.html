118 void FindHits::Exec(Option_t* /*opt*/)
119 {
120 
121  //Info("Exec","Exec called",0,0);
122  LOG(INFO) << "FindHits::Exec >>>>>" << "";
123 
124  reset();
125 
126  o2::ITSMFT::Point *point;
127  TVector3 pos, dpos;
128  Int_t detID, trackID;
129  Double_t dx = SegmentationAlpide::PitchRow/TMath::Sqrt(12);
130  Double_t dy = SegmentationAlpide::PitchCol/TMath::Sqrt(12);
131  Double_t dz = 0.;
132 
133  // Loop over fPoints
134  Int_t nPoints = mPoints->GetEntriesFast();
135  for (Int_t iPoint = 0; iPoint < nPoints; iPoint++) {
136  point = static_cast<o2::ITSMFT::Point*>(mPoints->At(iPoint));
137  if (!point) continue;
138  detID = point->GetDetectorID();
139  trackID = point->GetTrackID();
140  // copy the coordinates from point to hit
141  pos.SetXYZ(point->GetStartX(),point->GetStartY(),point->GetStartZ());
142  dpos.SetXYZ(dx,dy,dz);
143  //new ((*fHits)[nHits]) Hit(detID, pos, dpos, iPoint);
144  new ((*mHits)[mNHits]) Hit(detID, pos, dpos, trackID);
145  mNHits++;
146  }
147 
148  LOG(INFO) << "Create " << mNHits << " hits out of "
149  << nPoints << " points." << "";
150 
151  mTNofEvents++;
152  mTNofHits += mNHits;
153 
154 }
