86 void Initializer::setupGeometry()
87 {
88  // read path to geometry files from config file
89  TEnv settings;
90  ConfigurationManager::getInstance().getConfig(settings);
91 
92  // get geometry from Geometry Manager and register in multiview
93  auto multiView = MultiView::getInstance();
94 
95  for(int iDet=0;iDet<NvisualisationGroups;++iDet){
96  EVisualisationGroup det = static_cast<EVisualisationGroup>(iDet);
97  string detName = gVisualisationGroupName[det];
98  if(settings.GetValue((detName+".draw").c_str(), false))
99  {
100  if( detName=="TPC" || detName=="MCH" || detName=="MTR"
101  || detName=="MID" || detName=="MFT" || detName=="AD0"
102  || detName=="FMD"){// don't load MUON+MFT and AD and standard TPC to R-Phi view
103 
104  multiView->drawGeometryForDetector(detName,true,false);
105  }
106  else if(detName=="RPH"){// special TPC geom from R-Phi view
107 
108  multiView->drawGeometryForDetector(detName,false,true,false);
109  }
110  else{// default
111  multiView->drawGeometryForDetector(detName);
112  }
113  }
114  }
115 }

117 void Initializer::setupCamera()
118 {
119  // move and rotate sub-views
120  TEnv settings;
121  ConfigurationManager::getInstance().getConfig(settings);
122 
123  // read settings from config file
124  const double angleHorizontal = settings.GetValue("camera.3D.rotation.horizontal",-0.4);
125  const double angleVertical = settings.GetValue("camera.3D.rotation.vertical",1.0);
126 
127  double zoom[MultiView::NumberOfViews];
128  zoom[MultiView::View3d] = settings.GetValue("camera.3D.zoom",1.0);
129  zoom[MultiView::ViewRphi] = settings.GetValue("camera.R-Phi.zoom",1.0);
130  zoom[MultiView::ViewZrho] = settings.GetValue("camera.Rho-Z.zoom",1.0);
131 
132  // get necessary elements of the multiview and set camera position
133  auto multiView = MultiView::getInstance();
134 
135  for(int viewIter=0;viewIter<MultiView::NumberOfViews;++viewIter){
136  TGLViewer *glView = multiView->getView(static_cast<MultiView::EViews>(viewIter))->GetGLViewer();
137  glView->CurrentCamera().Reset();
138 
139  if(viewIter==0){
140  glView->CurrentCamera().RotateRad(angleHorizontal, angleVertical);
141  }
142  glView->CurrentCamera().Dolly(zoom[viewIter], kFALSE, kTRUE);
143  }
144 }
