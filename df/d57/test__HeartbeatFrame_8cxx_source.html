76 template <typename T>
77 constexpr size_t sequenceInsert(byte* buffer, const T& element) noexcept {
78  size_t length = 0;
79  memcpy(buffer + length, &element.header, sizeof(typename T::HeaderType));
80  length += sizeof(typename T::HeaderType);
81  memcpy(buffer + length, element.data, element.getDataLength());
82  length += element.getDataLength();
83  memcpy(buffer + length, &element.trailer, sizeof(typename T::TrailerType));
84  length += sizeof(typename T::TrailerType);
