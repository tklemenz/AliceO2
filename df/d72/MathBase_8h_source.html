191

195  if ( entriesI>0 ) {
196  mean += xcenter*entriesI;
197  rms2 += xcenter*entriesI*xcenter;
198  sum += entriesI;
199  ++npoints;
200  }
201  }
202  if ( sum == 0 ) return data;
203  mean/=sum;
204 
205  data.mCOG = mean;
206  // exception in case of only one bin is filled

208  rms2 /= sum;

210  data.mStdDev = binWidth/std::sqrt(12.);
211  }
212  else {
213  data.mStdDev = std::sqrt(std::abs(rms2-mean*mean));
214  }
215
