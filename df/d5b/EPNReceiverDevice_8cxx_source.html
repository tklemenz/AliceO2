64 void EPNReceiverDevice::DiscardIncompleteTimeframes()
65 {
66  auto it = mTimeframeBuffer.begin();
67 
68  while (it != mTimeframeBuffer.end()) {
69  if (duration_cast<milliseconds>(steady_clock::now() - (it->second).start).count() > mBufferTimeoutInMs) {
70  LOG(WARN) << "Timeframe #" << it->first << " incomplete after " << mBufferTimeoutInMs << " milliseconds, discarding";
71  mDiscardedSet.insert(it->first);
72  mTimeframeBuffer.erase(it++);
73  LOG(WARN) << "Number of discarded timeframes: " << mDiscardedSet.size();
74  } else {
75  // LOG(INFO) << "Timeframe #" << it->first << " within timeout, buffering...";
76  ++it;
77  }
78  }

81 void EPNReceiverDevice::Run()
82 {
83  uint16_t id = 0; // holds the timeframe id of the currently arrived sub-timeframe.
84 
85  FairMQChannel& ackOutChannel = fChannels.at(mAckChannelName).at(0);
86 
87  // Simple multi timeframe index
88  using TimeframeId = int;
89  using FlpId = int;
90  std::multimap<TimeframeId, IndexElement> index;
91  std::multimap<TimeframeId, FlpId> flpIds;
92 
93  while (CheckCurrentState(RUNNING)) {
94  FairMQParts subtimeframeParts;
95  if (Receive(subtimeframeParts, mInChannelName, 0, 100) <= 0)
96  continue;
97 
98  assert(subtimeframeParts.Size() >= 2);
99 
100  const auto* dh = o2::header::get<header::DataHeader>(subtimeframeParts.At(0)->GetData());
101  assert(strncmp(dh->dataDescription.str, "SUBTIMEFRAMEMD", 16) == 0);
102  SubframeMetadata* sfm = reinterpret_cast<SubframeMetadata*>(subtimeframeParts.At(1)->GetData());
103  id = o2::DataFlow::timeframeIdFromTimestamp(sfm->startTime, sfm->duration);
104  auto flpId = sfm->flpIndex;
105 
106  if (mDiscardedSet.find(id) == mDiscardedSet.end())
107  {
108  if (mTimeframeBuffer.find(id) == mTimeframeBuffer.end())
109  {
110  // if this is the first part with this ID, save the receive time.
111  mTimeframeBuffer[id].start = steady_clock::now();
112  }
113  flpIds.insert(std::make_pair(id, flpId));
114  LOG(INFO) << "Timeframe ID " << id << " for startTime " << sfm->startTime << "\n";
115  // If the received ID has not previously been discarded, store
116  // the data part in the buffer For the moment we just concatenate
117  // the subtimeframes and add an index for their description at
118  // the end. Given every second part is a data header we skip
119  // every two parts to populate the index. Moreover we know that
120  // the SubframeMetadata is always in the second part, so we can
121  // extract the flpId from there.
122  for (size_t i = 0; i < subtimeframeParts.Size(); ++i)
123  {
124  if (i % 2 == 0)
125  {
126  const auto * adh = o2::header::get<header::DataHeader>(subtimeframeParts.At(i)->GetData());
127  auto ie = std::make_pair(*adh, index.count(id)*2);
128  index.insert(std::make_pair(id, ie));
129  }
130  mTimeframeBuffer[id].parts.AddPart(move(subtimeframeParts.At(i)));
131  }
132  }
133  else
134  {
135  // if received ID has been previously discarded.
136  LOG(WARN) << "Received part from an already discarded timeframe with id " << id;
137  }
138 
139  if (flpIds.count(id) == mNumFLPs) {
140  LOG(INFO) << "Timeframe " << id << " complete. Publishing.\n";
141  o2::header::DataHeader tih;
142  std::vector<IndexElement> flattenedIndex;
143 
144  tih.dataDescription = o2::header::DataDescription("TIMEFRAMEINDEX");
145  tih.dataOrigin = o2::header::DataOrigin("EPN");
146  tih.subSpecification = 0;
147  tih.payloadSize = index.count(id) * sizeof(flattenedIndex.front());
148  void *indexData = malloc(tih.payloadSize);
149  auto indexRange = index.equal_range(id);
150  for (auto ie = indexRange.first; ie != indexRange.second; ++ie)
151  {
152  flattenedIndex.push_back(ie->second);
153  }
154  memcpy(indexData, flattenedIndex.data(), tih.payloadSize);
155 
156  mTimeframeBuffer[id].parts.AddPart(NewSimpleMessage(tih));
157  mTimeframeBuffer[id].parts.AddPart(NewMessage(indexData, tih.payloadSize,
158  [](void* data, void* hint){ free(data); }, nullptr));
159  // LOG(INFO) << "Collected all parts for timeframe #" << id;
160  // when all parts are collected send then to the output channel
161  Send(mTimeframeBuffer[id].parts, mOutChannelName);
162  LOG(INFO) << "Index count for " << id << " " << index.count(id) << "\n";
163  index.erase(id);
164  LOG(INFO) << "Index count for " << id << " " << index.count(id) << "\n";
165  flpIds.erase(id);
166 
167  if (mTestMode > 0) {
168  // Send an acknowledgement back to the sampler to measure the round trip time
169  unique_ptr<FairMQMessage> ack(NewMessage(sizeof(uint16_t)));
170  memcpy(ack->GetData(), &id, sizeof(uint16_t));
171 
172  if (ackOutChannel.Send(ack, 0) <= 0) {
173  LOG(ERROR) << "Could not send acknowledgement without blocking";
174  }
175  }
176 
177  mTimeframeBuffer.erase(id);
178  }
179 
180  // LOG(WARN) << "Buffer size: " << fTimeframeBuffer.size();
181 
182  // Check if any incomplete timeframes in the buffer are older than
183  // timeout period, and discard them if they are
184  // QUESTION: is this really what we want to do?
185  DiscardIncompleteTimeframes();
186  }
187 }
