48 {
49  assert(mc);
50 
51  // init base time point
52  if (stepcounter == -1) {
53  starttime = std::chrono::high_resolution_clock::now();
54  }
55  stepcounter++;
56  stepid = stepcounter;
57 
58  auto stack = mc->GetStack();
59 
60  trackID = stack->GetCurrentTrackNumber();
61  lookupstructures.insertPDG(trackID, mc->TrackPid());
62 
63  auto id = mc->CurrentVolID(copyNo);
64  volId = id;
65 
66  auto curtrack = stack->GetCurrentTrack();
67  auto parentID = curtrack->IsPrimary() ? -1 : stack->GetCurrentParentTrackNumber();
68  lookupstructures.insertParent(trackID, parentID);
69 
70  // try to resolve the module via external map
71  // keep information in faster vector once looked up
72  auto volname = mc->CurrentVolName();
73  lookupstructures.insertVolName(volId, volname);
74 
75  if (volnametomodulemap && volnametomodulemap->size() > 0 && volId >= 0) {
76  if (lookupstructures.getModuleAt(volId) == nullptr) {
77  // lookup in map
78  auto iter = volnametomodulemap->find(volname);
79  if (iter != volnametomodulemap->end()) {
80  lookupstructures.insertModuleName(volId, iter->second);
81  }
82  }
83  }
84 
85  double xd, yd, zd;
86  mc->TrackPosition(xd, yd, zd);
87  x = xd;
88  y = yd;
89  z = zd;
90  E = curtrack->Energy();
91  auto now = std::chrono::high_resolution_clock::now();
92  // cputimestamp = std::chrono::duration_cast<std::chrono::nanoseconds>(now - starttime).count();
93  nsecondaries = mc->NSecondaries();
94 
95  if (nsecondaries > 0) {
96  secondaryprocesses = new int[nsecondaries];
97  // for the processes
98  for (int i = 0; i < nsecondaries; ++i) {
99  secondaryprocesses[i] = mc->ProdProcess(i);
100  }
101  }
102 
103  TArrayI procs;
104  mc->StepProcesses(procs);
105  nprocessesactive = procs.GetSize();
106 
107  // was track stopped due to energy limit ??
108  stopped = mc->IsTrackStop();
109 }
