345 {
346 
347  Int_t half, disk, ladder, sensor, ladderID;
348  getSensorID(index, half, disk, ladder, sensor);
349  ladderID = mLadderIndex2Id[disk][ladder];
350  //LOG(INFO) << "extractMatrixSensor index " << index << " half " << half << " disk " << disk << " ladder " << ladder << " ladderID " << ladderID << FairLogger::endl;
351 
352  TString path = Form("/cave_1/%s_0/", getMFTVolPattern());
353  path += Form("%s_%d_%d/%s_%d_%d_%d/%s_%d_%d_%d_%d/%s_%d_%d_%d_%d/%s_1",
354  getMFTHalfPattern(),half,half,
355  getMFTDiskPattern(),half,disk,disk,
356  getMFTLadderPattern(),half,disk,ladderID,ladderID,
357  getMFTChipPattern(),half,disk,ladderID,sensor,
358  getMFTSensorPattern());
359  //LOG(INFO) << "Volume path is " << path.Data() << FairLogger::endl;
360 
361  static TGeoHMatrix matTmp;
362  gGeoManager->PushPath();
363 
364  if (!gGeoManager->cd(path.Data())) {
365  gGeoManager->PopPath();
366  LOG(ERROR) << "Error in cd-ing to " << path.Data() << FairLogger::endl;
367  return nullptr;
368  } // end if !gGeoManager
369 
370  matTmp = *gGeoManager->GetCurrentMatrix(); // matrix may change after cd
371 
372  // Restore the modeler state.
373  gGeoManager->PopPath();
374 
375  // account for the difference between sensitive layer and physical sensor ticknesses
376  static TGeoTranslation tra(0.,0.5*(AlpideSegmentation::SensorThickness-AlpideSegmentation::SensLayerThickness),0.);
377 
378  matTmp *= tra;
379 
380  return &matTmp;

388  // populate matrix cache for requested transformations
389  //

407  if ( (mask & bit2Mask(TransformType::T2L)) && !getCacheT2L().isFilled() ) {
408  // matrices for Tracking to Local frame transformation

410  auto& cacheT2L = getCacheT2L();
411  cacheT2L.setSize(mSize);
412  for (int i=0;i<mSize;i++) {
413  TGeoHMatrix& hm = createT2LMatrix(i);
414  cacheT2L.setMatrix( Mat3D(hm) , i);
415  }
416  }
417 
418  if ( (mask & bit2Mask(TransformType::T2G)) && !getCacheT2G().isFilled() ) {
419  // matrices for Tracking to Global frame transformation
420  LOG(INFO) << "Loading MFT T2G matrices from TGeo" << FairLogger::endl;
421  auto& cacheT2G = getCacheT2G();
422  cacheT2G.setSize(mSize);
423  for (int i=0;i<mSize;i++) {

458 Bool_t GeometryTGeo::getSensorID(Int_t index, Int_t& half, Int_t& disk, Int_t& ladder, Int_t& sensor) const
459 {
460 
461  if (index < 0 || index >= mTotalNumberOfSensors) return kFALSE;
462 
463  half = index/(mTotalNumberOfSensors/mNumberOfHalves);
464  index = index%(mTotalNumberOfSensors/mNumberOfHalves);
465  disk = 0;
466  while (index > mLastSensorIndex[disk]) {
467  disk++;
468  }
469  index -= getFirstSensorIndex(disk);
470  Int_t nSensor = MinSensorsPerLadder;
471  Int_t nFirstSensorIndex = 0, nFirstSensorIndexSave = 0;
472  ladder = 0;
473  while (index > ((nFirstSensorIndex += nSensor*mNumberOfLadders[disk][nSensor])-1)) {
474  ladder += mNumberOfLadders[disk][nSensor];
475  nFirstSensorIndexSave = nFirstSensorIndex;
476  nSensor++;
477  }
478  index -= nFirstSensorIndexSave;
479  ladder += index/nSensor;
480  sensor = index%nSensor;
481 
482  return kTRUE;
