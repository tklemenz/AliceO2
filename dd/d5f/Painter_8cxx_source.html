30 void Painter::draw(const CalDet<T>& calDet)
31 {
32  using DetType = CalDet<T>;
33  using CalType = CalArray<T>;

35  static const Mapper& mapper = Mapper::instance();
36 
37  // ===| name and title |======================================================
38  const auto title = calDet.getName().c_str();
39  std::string name = calDet.getName();
40  std::replace(name.begin(), name.end(), ' ', '_');
41 
42  // ===| define histograms |===================================================
43  // TODO: auto scaling of ranges based on mean and variance?
44  // for the moment use roots auto scaling
45 
46  // set buffer size such that autoscaling uses the full range. This is about 2MB per histogram!
47  const int bufferSize = TH1::GetDefaultBufferSize();
48  TH1::SetDefaultBufferSize(Sector::MAXSECTOR*mapper.getPadsInSector());
49 
50  auto hAside1D = new TH1F(Form("h_Aside_1D_%s", name.c_str()), Form("%s (A-Side)", title),
51  300, 0, 0); //TODO: modify ranges
52 
53  auto hCside1D = new TH1F(Form("h_Cside_1D_%s", name.c_str()), Form("%s (C-Side)", title),
54  300, 0, 0); //TODO: modify ranges
55 
56  auto hAside2D = new TH2F(Form("h_Aside_2D_%s;x (cm);y (cm)", name.c_str()), Form("%s (A-Side)", title),
57  300, -300, 300, 300, -300, 300);
58 
59  auto hCside2D = new TH2F(Form("h_Cside_2D_%s;x (cm);y (cm)", name.c_str()), Form("%s (C-Side)", title),
60  300, -300, 300, 300, -300, 300);
61 
62 
63  for (ROC roc; !roc.looped(); ++roc) {
64 
65  auto hist2D = hAside2D;
66  auto hist1D = hAside1D;
67  if (roc.side() == Side::C) {
68  hist2D = hCside2D;
69  hist1D = hCside1D;
70  }
71 
72  const int nrows = mapper.getNumberOfRowsROC(roc);
73  for (int irow=0; irow<nrows; ++irow) {
74  const int npads = mapper.getNumberOfPadsInRowROC(roc, irow);
75  for (int ipad=0; ipad<npads; ++ipad) {
76  const auto val = calDet.getValue(roc, irow, ipad);
77  const GlobalPosition2D pos = mapper.getPadCentre(PadROCPos(roc, irow, ipad));
78  const int bin = hist2D->FindBin(pos.X(), pos.Y());
79  if (!hist2D->GetBinContent(bin)) hist2D->SetBinContent(bin, val);
80  hist1D->Fill(val);
81  }
82  }
83  }
84 
85  // ===| Draw histograms |=====================================================
86  auto c = new TCanvas(Form("c_%s", name.c_str()), title);
87  c->Divide(2,2);
88 
89  c->cd(1);
90  hAside2D->Draw("colz");
91 
92  c->cd(2);
93  hCside2D->Draw("colz");
94 
95  c->cd(3);
96  hAside1D->Draw();
97 
98  c->cd(4);
99  hCside1D->Draw();
100 
101  // reset the buffer size
102  TH1::SetDefaultBufferSize(bufferSize);
103 }

151 template <class T>
152 TH2* Painter::getHistogram2D(const CalArray<T>& calArray)
153 {
154  static const Mapper& mapper = Mapper::instance();
155 
156  const size_t position = calArray.getPadSubsetNumber();
157 
158  const PadSubset padSubset = calArray.getPadSubset();
159  // ===| maximum number of rows and pads |=====================================
160  const int nrows = mapper.getNumberOfPadRows(padSubset, position);
161  const int npads = mapper.getNumberOfPadsInRow(padSubset, position, nrows-1);
162 
163  // ===| create histogram |====================================================
164  const auto title = calArray.getName().c_str();
165  std::string name = calArray.getName();
166  std::replace(name.begin(), name.end(), ' ', '_');
167  auto hist = new TH2F(Form("h_%s", name.c_str()),
168  Form("%s;pad row;pad", title),
169  nrows, 0., nrows,
170  npads, -npads/2, npads/2);
171 
172  // ===| fill hist |===========================================================
173  for (int irow = 0; irow<nrows; ++irow) {
174  const int padsInRow = mapper.getNumberOfPadsInRow(padSubset, position, irow);
175  for (int ipad = 0; ipad<padsInRow; ++ipad) {
176  const GlobalPadNumber pad = mapper.getPadNumber(padSubset, position, irow, ipad);
177  const auto val = calArray.getValue(pad);
178  const int cpad = ipad - padsInRow/2;
179  hist->Fill(irow, cpad, val);
180  //printf("%d %d: %f\n", irow, cpad, (double)val);
181  }
182  }
183  return hist;
184 }
