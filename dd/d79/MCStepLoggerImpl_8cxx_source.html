72 // initializes a mapping from volumename to detector
73 // used for step resolution to detectors
74 void initVolumeMap()
75 {
76  auto volmap = new std::map<std::string, std::string>;
77  // open for reading or fail
78  std::ifstream ifs;
79  auto f = getVolMapFile();
80  std::cerr << "[MCLOGGER:] TRYING TO READ VOLUMEMAPS FROM " << f << "\n";
81  ifs.open(f);
82  if (ifs.is_open()) {
83  std::string line;
84  while (std::getline(ifs, line)) {
85  std::istringstream ss(line);
86  std::string token;
87  // split the line into key + value
88  int counter = 0;
89  std::string keyvalue[2] = { "NULL", "NULL" };
90  while (counter < 2 && std::getline(ss, token, ' ')) {
91  if (!token.empty()) {
92  keyvalue[counter] = token;
93  counter++;
94  }
95  }
96  // put into map
97  volmap->insert({ keyvalue[0], keyvalue[1] });
98  }
99  ifs.close();
100  StepInfo::volnametomodulemap = volmap;
101  } else {
102  std::cerr << "[MCLOGGER:] VOLUMEMAPSFILE NOT FILE\n";
103  StepInfo::volnametomodulemap = nullptr;
104  }
105 }

107 template <typename T>
108 void flushToTTree(const char* branchname, T* address)
109 {
110  TFile* f = new TFile(getLogFileName(), "UPDATE");
111  const char* treename = "StepLoggerTree";
112  auto tree = (TTree*)f->Get(treename);
113  if (!tree) {
114  // create tree
115  tree = new TTree(treename, "Tree container information from MC step logger");
116  }
117  auto branch = tree->GetBranch(branchname);
118  if (!branch) {
119  branch = tree->Branch(branchname, &address);
120  }
121  branch->SetAddress(&address);
122  branch->Fill();
123  tree->SetEntries(branch->GetEntries());
124  f->Write();
125  f->Close();
126  delete f;
127 }

148  void addStep(TVirtualMC* mc, const double* x, const double* b)
149  {
150  if (mTTreeIO) {
151  callcontainer.emplace_back(mc, x[0], x[1], x[2], b[0], b[1], b[2]);
152  return;
153  }
154  counter++;
155  int copyNo;
156  auto id = mc->CurrentVolID(copyNo);
157  if (volumetosteps.find(id) == volumetosteps.end()) {
158  volumetosteps.insert(std::pair<int, int>(id, 0));
159  } else {
160  volumetosteps[id]++;
161  }
162  if (idtovolname.find(id) == idtovolname.end()) {
163  idtovolname.insert(std::pair<int, std::string>(id, std::string(mc->CurrentVolName())));
164  }
165  }
166 
167  void clear()
168  {
169  counter = 0;
170  volumetosteps.clear();
171  idtovolname.clear();
172  if (mTTreeIO) {
173  callcontainer.clear();
174  }
175  }
176 
177  void flush()
178  {
179  if (mTTreeIO) {
180  flushToTTree("Calls", &callcontainer);
181  } else {
182  std::cerr << "[FIELDLOGGER]: did " << counter << " steps \n";
183  // summarize steps per volume
184  for (auto& p : volumetosteps) {
185  std::cerr << "[FIELDLOGGER]: VolName " << idtovolname[p.first] << " COUNT " << p.second;
186  std::cerr << "\n";
187  }
188  std::cerr << "[FIELDLOGGER]: ----- END OF EVENT ------\n";
189  }
190  clear();
191  }

220  void addStep(TVirtualMC* mc)
221  {
222  if (mTTreeIO) {
223  container.emplace_back(mc);
224  } else {
225  assert(mc);
226  stepcounter++;
227 
228  auto stack = mc->GetStack();
229  assert(stack);
230  trackset.insert(stack->GetCurrentTrackNumber());
231  pdgset.insert(mc->TrackPid());
232  int copyNo;
233  auto id = mc->CurrentVolID(copyNo);
234 
235  TArrayI procs;
236  mc->StepProcesses(procs);
237 
238  if (volumetosteps.find(id) == volumetosteps.end()) {
239  volumetosteps.insert(std::pair<int, int>(id, 0));
240  } else {
241  volumetosteps[id]++;
242  }
243  if (idtovolname.find(id) == idtovolname.end()) {
244  idtovolname.insert(std::pair<int, std::string>(id, std::string(mc->CurrentVolName())));
245  }
246 
247  // for the secondaries
248  if (volumetoNSecondaries.find(id) == volumetoNSecondaries.end()) {
249  volumetoNSecondaries.insert(std::pair<int, int>(id, mc->NSecondaries()));
250  } else {
251  volumetoNSecondaries[id] += mc->NSecondaries();
252  }
253 
254  // for the processes
255  for (int i = 0; i < mc->NSecondaries(); ++i) {
256  auto process = mc->ProdProcess(i);
257  auto p = std::pair<int, int>(id, process);
258  if (volumetoProcess.find(p) == volumetoProcess.end()) {
259  volumetoProcess.insert(std::pair<std::pair<int, int>, int>(p, 1));
260  } else {
261  volumetoProcess[p]++;
262  }
263  }
264  }
265  }

292  void flush()
293  {
294  if (!mTTreeIO) {
295  std::cerr << "[STEPLOGGER]: did " << stepcounter << " steps \n";
296  std::cerr << "[STEPLOGGER]: transported " << trackset.size() << " different tracks \n";
297  std::cerr << "[STEPLOGGER]: transported " << pdgset.size() << " different types \n";
298  // summarize steps per volume

322 // a helper template kernel describing generically the redispatching prodecure
323 template <typename Object /* the original object type */, typename MethodType /* member function type */,
324  typename... Args /* original arguments to function */>
325 void dispatchOriginalKernel(Object* obj, char const* libname, char const* origFunctionName, Args... args)
326 {
327  // Object, MethodType, and Args are of course related so we could do some static_assert checks or automatic deduction
328 
329  // static map to avoid having to lookup the right symbols in the shared lib at each call
330  // (We could do this outside of course)
331  static std::map<const char*, MethodType> functionNameToSymbolMap;
332  MethodType origMethod = nullptr;
333 
334  auto iter = functionNameToSymbolMap.find(origFunctionName);
335  if (iter == functionNameToSymbolMap.end()) {
336  auto libHandle = dlopen(libname, RTLD_NOW);
337  // try to make the library loading a bit more portable:
338  if (!libHandle) {
339  // try appending *.so
340  std::stringstream stream;
341  stream << libname << ".so";
342  libHandle = dlopen(stream.str().c_str(), RTLD_NOW);
343  }
344  if (!libHandle) {
345  // try appending *.dylib
346  std::stringstream stream;
347  stream << libname << ".dylib";
348  libHandle = dlopen(stream.str().c_str(), RTLD_NOW);
349  }
350  assert(libHandle);
351  void* symbolAddress = dlsym(libHandle, origFunctionName);
352  assert(symbolAddress);
353 // Purposely ignore compiler warning
354 #pragma GCC diagnostic push
355 #pragma GCC diagnostic ignored "-Wsizeof-pointer-memaccess"
356  // hack since C++ does not allow casting to C++ member function pointers
357  // thanks to gist.github.com/mooware/1174572
358  memcpy(&origMethod, &symbolAddress, sizeof(&symbolAddress));
359 #pragma GCC diagnostic pop
360  functionNameToSymbolMap[origFunctionName] = origMethod;
361  } else {
362  origMethod = iter->second;
363  }
364  // the final C++ member function call redispatch
365  (obj->*origMethod)(args...);
366 }
