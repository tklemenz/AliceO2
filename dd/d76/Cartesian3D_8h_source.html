139 class Transform3D : public ROOT::Math::Transform3D
140 {
141  //
142  // Class to perform geom.transformations (rotation and displacements only) in
143  // double precision over the cartesian points and vectors (float or double).
144  // Adds to the base ROOT::Math::Transform3D<double> class a convertor from
145  // TGeoMatrix.
146  // To be used instead of TGeoHMatrix for all transformations of hits,
147  // clusters etc.
148  //
149 
150  public:
151  Transform3D() = default;
152  Transform3D(const TGeoMatrix& m);
153  ~Transform3D() = default;
154 
155  // inherit assignment operators of the base class
156  using ROOT::Math::Transform3D::operator=;
157 
158  // to avoid conflict between the base Transform3D(const ForeignMatrix & m) and
159  // Transform3D(const TGeoMatrix &m) constructors we cannot inherit base c-tors,
160  // therefore we redefine them here
161  Transform3D(const ROOT::Math::Rotation3D& r, const Vector& v) : ROOT::Math::Transform3D(r, v) {}
162  Transform3D(const ROOT::Math::Rotation3D& r, const ROOT::Math::Translation3D& t) : ROOT::Math::Transform3D(r, t) {}
163  template <class IT>
164  Transform3D(IT begin, IT end) : ROOT::Math::Transform3D(begin, end) {}
165 
166  // conversion operator to TGeoHMatrix
167  operator TGeoHMatrix&() const {
168  static TGeoHMatrix tmp;
169  double rot[9],tra[3];
170  GetComponents(rot[0],rot[1],rot[2],tra[0],rot[3],rot[4],rot[5],tra[1],rot[6],rot[7],rot[8],tra[2]);
171  tmp.SetRotation(rot);
172  tmp.SetTranslation(tra);
173  return tmp;
174  }
175 
176  void set(const TGeoMatrix& m); // set parameters from TGeoMatrix
177 
178  using ROOT::Math::Transform3D::operator();
179  // the local->master transformation for points and vectors can be
180  // done in operator form (inherited from base Transform3D) as
181  // Point3D pnt;
182  // Transform3D trans;
183  // auto pntTr0 = trans(pnt); // 1st version
184  // auto pntTr1 = trans*pnt; // 2nd version
185  //
186  // For the inverse transform we define our own operator^
187 
188  template <typename T>
189  Point3D<T> operator^(const Point3D<T>& p) const
190  { // master->local
191  return ApplyInverse(p);
192  }
193
