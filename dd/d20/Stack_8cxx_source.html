128 void Stack::PushTrack(Int_t toBeDone, Int_t parentId, Int_t pdgCode, Double_t px, Double_t py, Double_t pz, Double_t e,
129  Double_t vx, Double_t vy, Double_t vz, Double_t time, Double_t polx, Double_t poly, Double_t polz,
130  TMCProcess proc, Int_t &ntr, Double_t weight, Int_t is, Int_t secondparentID)

132  // Create new TParticle and add it to the TParticle array
133  Int_t trackId = mNumberOfEntriesInParticles;
134  Int_t nPoints = 0;
135  Int_t daughter1Id = -1;
136  Int_t daughter2Id = -1;
137 
138  // LOG(INFO) << "Pushing " << trackId << " with parent " << parentId << FairLogger::endl;
139 
140  TParticle p(pdgCode, trackId, parentId, nPoints, daughter1Id, daughter2Id, px, py, pz, e, vx, vy, vz, time);
141  p.SetPolarisation(polx, poly, polz);
142  p.SetWeight(weight);
143  p.SetUniqueID(proc); // using the unique ID to transfer process ID
144  mNumberOfEntriesInParticles++;
145 
146  // currently I only know of G4 who pushes particles like this (but never pops)
147  // so we have to register the particles here
148  if (mIsG4Like && parentId >= 0) {
149  //p.SetStatusCode(mParticles.size());
150  mParticles.emplace_back(p);
151  mTransportedIDs.emplace_back(p.GetStatusCode());
152  mCurrentParticle = p;
153  }
154 
155  // Increment counter
156  if (parentId < 0) {
157  mNumberOfPrimaryParticles++;
158  mPrimaryParticles.push_back(p);
159  }
160 
161  // Set argument variable
162  ntr = trackId;
163 
164  // Push particle on the stack if toBeDone is set
165  if (toBeDone == 1) {
166  mStack.push(p);
167  }

204 TParticle* Stack::PopNextTrack(Int_t& iTrack)
205 {
206  // This functions is mainly used by Geant3?
207 
208  // If end of stack: Return empty pointer
209  if (mStack.empty()) {
210  notifyFinishPrimary();
211  iTrack = -1;
212  return nullptr;
213  }
214 
215  // If not, get next particle from stack
216  mCurrentParticle = mStack.top();
217  mStack.pop();
218 
219  if (mCurrentParticle.GetMother(0) < 0) {
220  // particle is primary -> indicates that previous particle finished
221  if(mParticles.size() > 0){
222  notifyFinishPrimary();
223  }
224  mIndexOfPrimaries.emplace_back(mParticles.size());
225  }
226  mParticles.emplace_back(mCurrentParticle);
227  mTransportedIDs.emplace_back(mCurrentParticle.GetStatusCode());
228 
229  mIndexOfCurrentTrack = mCurrentParticle.GetStatusCode();
230  iTrack = mIndexOfCurrentTrack;
231 
232  // LOG(INFO) << "transporting ID " << mIndexOfCurrentTrack << "\n";
233  return &mCurrentParticle;

263 void Stack::finishPrimary() {
264  // Here transport of a primary and all its secondaries is finished
265  // we can do some cleanup of the memory structures
266  LOG(DEBUG) << "STACK: Cleaning up" << FairLogger::endl;
267  auto selected = selectTracks();
268  // loop over current particle buffer
269  int index=0;
270  for(const auto& particle : mParticles) {
271  if (particle.getStore()) {
272  // map the global track index to the new persistent index
273  // FIXME: only fill the map for non-trivial mappings in which mTransportedIDs[index]!=mTracks->size();
274  mIndexMap[mTransportedIDs[index]] = mTracks->size();
275  mTracks->emplace_back(particle);
276  }
277  index++;
278  mTracksDone++;
279  }
280  // we can now clear the particles buffer!
281  mParticles.clear();
282  mTransportedIDs.clear();
283  mIndexOfPrimaries.clear();
284 }

286 void Stack::UpdateTrackIndex(TRefArray *detList)
287 {
288  // we can avoid any updating in case no tracks have been filtered out
289  // check this like this
290  if (mIndexMap.size() == 0) {
291  LOG(INFO) << "No TrackIndex update necessary\n";
292  return;
293  }
294 
295  // we are getting the detectorlist from FairRoot as TRefArray
296  // at each call, but this list never changes so we cache it here
297  // as the right type to avoid repeated dynamic casts
298  if (mActiveDetectors.size() == 0) {
299  if (detList == nullptr) {
300  LOG(FATAL) << "No detList passed to Stack" << FairLogger::endl;
301  }
302  auto iter = detList->MakeIterator();
303  while (auto det = iter->Next()) {
304  auto o2det = dynamic_cast<o2::Base::Detector*>(det);
305  if (o2det) {
306  mActiveDetectors.emplace_back(o2det);
307  } else {
308  LOG(INFO) << "Found nonconforming detector" << FairLogger::endl;
309  }
310  }
311  }
312 
313  if (mLogger) {
314  mLogger->Debug(MESSAGE_ORIGIN, "Stack: Updating track indices...");
315  } else {
316  cout << "Stack: Updating track indices..." << endl;
317  }
318  Int_t nColl = 0;
319 
320  // First update mother ID in MCTracks
321  for (Int_t i = 0; i < mNumberOfEntriesInTracks; i++) {
322  auto& track = (*mTracks)[i];
323  Int_t iMotherOld = track.getMotherTrackId();
324  if (iMotherOld == -1) {
325  // no need to lookup this case
326  continue;
327  }
328  auto iter = mIndexMap.find(iMotherOld);
329  if (iter == mIndexMap.end()) {
330  if (mLogger) {
331  mLogger->Fatal(MESSAGE_ORIGIN, "Stack: Track index %i not found index map! ", iMotherOld);
332  }
333  Fatal("Stack::UpdateTrackIndex", "Track index not found in map");
334  }
335  track.SetMotherTrackId(iter->second);
336  }
337 
338  for(auto det : mActiveDetectors) {
339  // update the track indices by delegating to specialized detector functions
340  det->updateHitTrackIndices(mIndexMap);
341  } // List of active detectors
342 
343  if (mLogger) {
344  mLogger->Debug(MESSAGE_ORIGIN, "...stack and %i collections updated.", nColl);
345  } else {
346  cout << "...stack and " << nColl << " collections updated." << endl;
347  }
348 }

350 void Stack::Reset()
351 {
352  mIndex = 0;
353  mIndexOfCurrentTrack = -1;
354  mNumberOfPrimaryParticles = mNumberOfEntriesInParticles = mNumberOfEntriesInTracks = 0;
355  while (!mStack.empty()) {
356  mStack.pop();
357  }
358  mParticles.clear();
359  mTracks->clear();
360  if (mPrimariesDone != mPrimaryParticles.size()) {
361  LOG(FATAL) << "Inconsistency in primary particles treated vs expected (This points "
362  << "to a flaw in the stack logic)" << FairLogger::endl;
363  }
364  mPrimariesDone = 0;
365  mPrimaryParticles.clear();

413 bool Stack::selectTracks()
414 {
415  bool tracksdiscarded = false;
416  // Check particles in the fParticle array
417  int prim = -1; // counter how many primaries seen (mainly to constrain search in motherindex remapping)
418  for(auto& thisPart : mParticles) {
419  Bool_t store = kTRUE;
420 
421  // Get track parameters
422  Int_t iMother = thisPart.getMotherTrackId();
423  if (iMother < 0) {
424  prim++;
425  // for primaries we are done quickly
426  store = kTRUE;
427  }
428  else {
429  // for other particles we (potentially need to correct the mother indices
430  auto rangestart = mTransportedIDs.begin() + mIndexOfPrimaries[prim];
431  auto rangeend = (prim < (mIndexOfPrimaries.size()-1)) ? mTransportedIDs.begin() + mIndexOfPrimaries[prim+1] : mTransportedIDs.end();
432  // auto rangeend = mTransportedIDs.end();
433 
434  auto iter = std::find_if(rangestart, rangeend, [iMother](int x){return x == iMother;});
435  if (iter!=rangeend) {
436  // complexity should be constant
437  auto newmother = std::distance(mTransportedIDs.begin(), iter);
438  // LOG(INFO) << "Fixing mother from " << iMother << " to " << newmother << FairLogger::endl;
439  thisPart.SetMotherTrackId(newmother);
440  }
441 
442  // no secondaries; also done
443  if (!mStoreSecondaries) {
444  store = kFALSE;
445  tracksdiscarded = true;
446  }
447  else {
448  // Calculate number of hits created by this track
449  // Note: we only distinguish between no hit and more than 0 hits
450  int nHits = thisPart.hasHits();
451 
452  // Check for cuts (store primaries in any case)
453  if (nHits < mMinHits) {
454  store = kFALSE;
455  tracksdiscarded = true;
456  }
457  // only if we have non-trival energy cut
458  if (mEnergyCut > 0.) {
459  Double_t energy = thisPart.GetEnergy();
460  Double_t mass = thisPart.GetMass();
461  Double_t eKin = energy - mass;
462 
463  if (eKin < mEnergyCut) {
464  store = kFALSE;
465  tracksdiscarded = true;
466  }
467  }
468  }
469  }
470  // LOG(INFO) << "storing " << store << FairLogger::endl;
471  thisPart.setStore(store);
472  }
473 
474  // If flag is set, flag recursively mothers of selected tracks
475  if (mStoreMothers) {
476  for (auto& particle : mParticles) {
477  if (particle.getStore()) {
478  Int_t iMother = particle.getMotherTrackId();
479  while (iMother >= 0) {
480  auto& mother = mParticles[iMother];
481  mother.setStore(true);
482  iMother = mother.getMotherTrackId();
483  }
484  }
485  }
486  }
487 
488  return !tracksdiscarded;
489 }
