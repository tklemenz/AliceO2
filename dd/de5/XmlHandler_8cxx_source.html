79 void XmlHandler::OnStartElement(const char *name, const TList *attributes)
80 {
81  // when a new XML element is found, it is processed here
82 
83  // set the current system if necessary
84  TString strName(name);
85  LOG(DEBUG) << "name = " << strName.Data() << FairLogger::endl;
86  Int_t startRun = -1;
87  Int_t endRun = -1;
88  TXMLAttr *attr;
89  TIter next(attributes);
90  while ((attr = (TXMLAttr *) next())) {
91  TString attrName = attr->GetName();
92  LOG(DEBUG) << "Name = " << attrName.Data() << FairLogger::endl;
93  if (attrName == "StartIdRunRange") {
94  startRun = (Int_t) (((TString) (attr->GetValue())).Atoi());
95  LOG(DEBUG) << "startRun = " << startRun << FairLogger::endl;
96  }
97  if (attrName == "EndIdRunRange") {
98  endRun = (Int_t) (((TString) (attr->GetValue())).Atoi());
99  LOG(DEBUG) << "endRun = " << endRun << FairLogger::endl;
100  }
101  if (attrName == "OCDBFolder") {
102  if (mRun >= startRun && mRun <= endRun && startRun != -1 && endRun != -1) {
103  mOCDBFolder = (TString) (attr->GetValue());
104  LOG(DEBUG) << "OCDBFolder = " << mOCDBFolder.Data() << FairLogger::endl;
105  mStartIdRunRange = startRun;
106  mEndIdRunRange = endRun;
107  }
108  }
109  }
110  return;
111 }
