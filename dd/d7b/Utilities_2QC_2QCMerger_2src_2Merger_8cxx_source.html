34 TObject* Merger::mergeObject(TObject* object)
35 {
36  auto foundEntry = mTitlesToDataObjectsMap.find(object->GetTitle());
37 
38  if (foundEntry != mTitlesToDataObjectsMap.end() &&
39  foundEntry->second->GetSize() >= NUMBER_OF_QC_OBJECTS_FOR_COMPLETE_DATA - 1) {
40  TObject* output = mergeObjectWithGivenCollection(object);
41  eraseCollection(object->GetTitle());
42  return output;
43  } else {
44  addReceivedObjectToMapByName(object);
45  return nullptr;
46  }

81 TObject* Merger::mergeObjectWithGivenCollection(TObject* mergedObject)
82 {
83  TCollection* mergeList = mTitlesToDataObjectsMap.find(mergedObject->GetTitle())->second;
84 
85  TObject* result = nullptr;
86  TH1F* histogram1F = nullptr;
87  TH2F* histogram2F = nullptr;
88  TH3F* histogram3F = nullptr;
89  THnF* histogramNF = nullptr;
90  TTree* tree = nullptr;
91  const char* className = mergedObject->ClassName();
92 
93  auto measureTime = chrono::high_resolution_clock::now();
94 
95  if (strcmp(className, "TH1F") == 0) {
96  histogram1F = reinterpret_cast<TH1F*>(mergedObject);
97  histogram1F->Merge(mergeList);
98  result = histogram1F;
99  } else if (strcmp(className, "TH2F") == 0) {
100  histogram2F = reinterpret_cast<TH2F*>(mergedObject);
101  histogram2F->Merge(mergeList);
102  result = histogram2F;
103  } else if (strcmp(className, "TH3F") == 0) {
104  histogram3F = reinterpret_cast<TH3F*>(mergedObject);
105  histogram3F->Merge(mergeList);
106  result = histogram3F;
107  } else if (strcmp(className, "THnT<float>") == 0) {
108  histogramNF = reinterpret_cast<THnF*>(mergedObject);
109  histogramNF->Merge(mergeList);
110  result = histogramNF;
111  } else if (strcmp(className, "TTree") == 0) {
112  tree = reinterpret_cast<TTree*>(mergedObject);
113  tree->Merge(mergeList);
114  result = tree;
115  } else {
116  LOG(ERROR) << "Object with type " << className << " is not one of mergable type.";
117  }
118 
119  mMergeTime = chrono::duration_cast<std::chrono::microseconds>(chrono::high_resolution_clock::now() - measureTime);
120 
121  return mergedObject;
122 }
