62 bool ProducerDevice::outputLimitReached()
63 {
64  bool output;
65  int now = getCurrentSecond();
66 
67  if (now == lastCheckedSecond && sentObjectsInCurrentSecond > OUTPUT_LIMIT_PER_SECOND - 1) {
68  output = true;
69  } else {
70  output = false;
71  }
72 
73  if (now != lastCheckedSecond) {
74  sentObjectsInCurrentSecond = 0;
75  }
76  lastCheckedSecond = now;
77 
78  return output;
79 }

100 void ProducerDevice::sendDataToMerger(unique_ptr<FairMQMessage> request)
101 {
102  if (fChannels.at("data-out").at(0).SendAsync(request) == -2) {
103  mLastBufferOverloadTime = clock();
104  mBufferOverloaded = true;
105  LOG(DEBUG) << "Buffer of data-out channel is full. Waiting for free buffer...";
106 
107  while (fChannels.at("data-out").at(0).SendAsync(request) == -2) {
108  this_thread::sleep_for(chrono::milliseconds(10));
109  }
110 
111  mBufferOverloaded = false;
112  LOG(DEBUG) << "Buffer was released after " << double(clock() - mLastBufferOverloadTime) / CLOCKS_PER_SEC
113  << " seconds.";
114  }
