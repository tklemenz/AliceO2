40 TEveGeoShape* GeometryManager::getGeometryForDetector(string detectorName)
41 {
42  TEnv settings;
43  ConfigurationManager::getInstance().getConfig(settings);
44 
45  // read geometry path from config file
46  string geomPath = settings.GetValue("simple.geom.path","");
47 
48  // TODO:
49  // we need a way to set O2 installation path here
50  //
51  const string o2basePathSpecifier = "${ALICE_ROOT}";
52  const string o2basePath = "";//= gSystem->Getenv("ALICE_ROOT");
53  const size_t o2pos = geomPath.find(o2basePathSpecifier);
54 
55  if(o2pos != string::npos){
56  geomPath.replace(o2pos,o2pos+o2basePathSpecifier.size(),o2basePath);
57  }
58 
59  // load ROOT file with geometry
60  TFile *f = TFile::Open(Form("%s/simple_geom_%s.root",geomPath.c_str(),detectorName.c_str()));
61  if(!f){
62  cout<<"GeometryManager::GetSimpleGeom -- no file with geometry found for: "<<detectorName<<"!"<<endl;
63  return nullptr;
64  }
65 
66  TEveGeoShapeExtract *geomShapreExtract = static_cast<TEveGeoShapeExtract*>(f->Get(detectorName.c_str()));
67  TEveGeoShape *geomShape = TEveGeoShape::ImportShapeExtract(geomShapreExtract);
68  f->Close();
69 
70  // tricks for different R-Phi geom of TPC:
71  if(detectorName=="RPH"){ // use all other parameters of regular TPC geom
72  detectorName = "TPC";
73  }
74 
75  // prepare geometry to be drawn including all children
76  drawDeep(geomShape,
77  settings.GetValue(Form("%s.color",detectorName.c_str()),-1),
78  settings.GetValue(Form("%s.trans",detectorName.c_str()),-1),
79  settings.GetValue(Form("%s.line.color",detectorName.c_str()),-1));
80 
81  gEve->GetDefaultGLViewer()->UpdateScene();
82 
83  return geomShape;
84 }
