119 void SpaceFrame::CreateGeometry()
120 {
121  LOG(DEBUG) << "Create CalFrame Geometry" << std::endl;
122 
124  TGeoVolume* top = gGeoManager->GetVolume("cave");
125  TGeoMedium* steel = gGeoManager->GetMedium("EMC_S steel$");
126  TGeoMedium* air = gGeoManager->GetMedium("EMC_Air$");
127 
129  TGeoVolume* calFrameMO = gGeoManager->MakeTubs("CalFrame", air, mBeginRadius - 2.1, mEndRadius, mTotalHalfWidth * 3,
130  mBeginPhi - 3, mEndPhi + 3); // Mother Volume
131 
132  calFrameMO->SetVisibility(kFALSE);
133 
134  // Half Frame Mother Volume
135  TGeoVolume* calHalfFrameMO = gGeoManager->MakeTubs("HalfFrame", air, mBeginRadius - 2, mEndRadius, mTotalHalfWidth,
136  mBeginPhi - 2.9, mEndPhi + 2.9);
137 
138  calHalfFrameMO->SetVisibility(kFALSE);
139 
140  TGeoVolume* endBeams =
141  gGeoManager->MakeBox("End Beams", steel, mEndBeamRadThick, mCrossTopHeight, mTotalHalfWidth); // End Beams
142 
143  TGeoVolume* skin = gGeoManager->MakeTubs("skin", steel, mRibHeight + 0.15, mEndRadius, mTotalHalfWidth, mBeginPhi,
144  mEndPhi); // back frame
145 
146  TGeoVolume* flangeVolume = gGeoManager->MakeTubs("supportBottom", steel, mBeginRadius, mFlangeHeight, mFlangeWidth,
147  mBeginPhi, mEndPhi); // FlangeVolume Beams
148 
149  TGeoVolume* ribVolume =
150  gGeoManager->MakeTubs("RibVolume", steel, mFlangeHeight, mRibHeight, mRibWidth, mBeginPhi, mEndPhi);
151 
152  TGeoVolume* subSetCross = gGeoManager->MakeTubs(
153  "subSetCross", air, mBeginRadius - 1, mBeginRadius + 2 * mCrossBottomRadThick + 2 * mCrossTopRadThick + 0.15,
154  mCrossBottomWidth, mBeginPhi, mEndPhi); // Cross Beam Containers
155  subSetCross->SetVisibility(kFALSE);
156  /* // Obsolete for now
157  TGeoVolume *subSetCrossTop =
158  gGeoManager->MakeTubs("SubSetCrossTop", air, mBeginRadius+2*mCrossBottomRadThick-1,
159  mBeginRadius+2*mCrossBottomRadThick+ 2*mCrossTopRadThick+1, mCrossTopWidth, mBeginPhi, mEndPhi); // Cross
160  subSetCrossTop->SetVisibility(kFALSE);
161  */
162  TGeoVolume* crossBottomBeams = gGeoManager->MakeBox("crossBottom", steel, mCrossBottomRadThick, mCrossBottomHeight,
163  mCrossBottomWidth); // Cross Beams
164 
165  TGeoVolume* crossTopBeams =
166  gGeoManager->MakeBox("crossTop", steel, mCrossTopRadThick, mCrossTopHeight, mCrossTopWidth); // Cross Beams
167 
168  TGeoTranslation* trTEST = new TGeoTranslation();
169  TGeoRotation* rotTEST = new TGeoRotation();
170 
171  Double_t conv = TMath::Pi() / 180.;
172  Double_t radAngle = 0;
173  Double_t endBeamParam = .4;
174  // cout<<"\nmCrossBottomStartRadius: "<<mCrossBottomStartRadius<<"\n";
175 
176  for (Int_t i = 0; i < mNumCross; i++) {
177  Double_t loopPhi = mBeginPhi + 1.8;
178 
179  // Cross Bottom Beams
180 
181  radAngle = (loopPhi + i * mCrossBeamArcLength) * conv;
182 
183  rotTEST->SetAngles(mBeginPhi + i * mCrossBeamArcLength, 0,
184  0); // SetTranslation(Double_t dx, Double_t dy, Double_t dz);
185  trTEST->SetTranslation(cos(radAngle) * mCrossBottomStartRadius, sin(radAngle) * mCrossBottomStartRadius, 0);
186 
187  TGeoCombiTrans* combo = new TGeoCombiTrans(*trTEST, *rotTEST); // TGeoTranslation &tr, const TGeoRotation &rot);
188  combo->RegisterYourself();
189  crossBottomBeams->SetVisibility(1);
190  subSetCross->AddNode(crossBottomBeams, i + 1, combo);
191  if (i != 0 && i != mNumCross - 1) {
192  // Cross Bottom Beams
193  rotTEST->SetAngles(mBeginPhi + i * mCrossBeamArcLength, 0,
194  0); // SetTranslation(Double_t dx, Double_t dy, Double_t dz);
195  trTEST->SetTranslation(cos(radAngle) * mCrossTopStart, sin(radAngle) * mCrossTopStart, 0);
196  crossTopBeams->SetVisibility(1);
197  subSetCross->AddNode(crossTopBeams, i + 1, new TGeoCombiTrans(*trTEST, *rotTEST));
198  }
199 
200  else if (i == 0) {
201  rotTEST->SetAngles(mBeginPhi + i * mCrossBeamArcLength, 0,
202  0); // SetTranslation(Double_t dx, Double_t dy, Double_t dz);
203  trTEST->SetTranslation(cos((77 - endBeamParam) * conv) * (mEndBeamBeginRadius),
204  sin((77 - endBeamParam) * conv) * (mEndBeamBeginRadius), 0);
205  endBeams->SetVisibility(1);
206  calHalfFrameMO->AddNode(endBeams, 1, new TGeoCombiTrans(*trTEST, *rotTEST));
207  } else {
208  rotTEST->SetAngles(193.03, 0, 0); // SetTranslation(Double_t dx, Double_t dy, Double_t dz);
209  trTEST->SetTranslation(cos((193.03 + endBeamParam) * conv) * (mEndBeamBeginRadius) /*more duct tape*/,
210  sin((193.03 + endBeamParam) * conv) * (mEndBeamBeginRadius), 0);
211  endBeams->SetVisibility(1);
212  calHalfFrameMO->AddNode(endBeams, 2, new TGeoCombiTrans(*trTEST, *rotTEST));
213  }
214  }
215 
216  // Beam Containers
217 
218  // Translations
219 
220  TGeoTranslation* origin1 = new TGeoTranslation(0, 0, 0); // Equivalent to gGeoIdentity
221  TGeoTranslation* origin2 = new TGeoTranslation(0, 0, 2 * (mCrossBottomWidth + mFlangeWidth));
222  TGeoTranslation* origin3 = new TGeoTranslation(0, 0, -2 * (mCrossBottomWidth + mFlangeWidth));
223 
224  // FlangeVolume translations

228  TGeoTranslation* str4 = new TGeoTranslation(0, 0, 3 * (mCrossBottomWidth + mFlangeWidth));
229 
230  // Half Frame Translations

232  TGeoTranslation* halfTrans2 = new TGeoTranslation(0, 0, -mHalfFrameTrans);
233 
234  // Beams Volume
235  calHalfFrameMO->AddNode(flangeVolume, 1, str1);
236  calHalfFrameMO->AddNode(flangeVolume, 2, str2);
237  calHalfFrameMO->AddNode(flangeVolume, 3, str3);
238  calHalfFrameMO->AddNode(flangeVolume, 4, str4);

242  calHalfFrameMO->AddNode(ribVolume, 3, str3);
243  calHalfFrameMO->AddNode(ribVolume, 4, str4);

246  calHalfFrameMO->AddNode(subSetCross, 1, origin1);
247  calHalfFrameMO->AddNode(subSetCross, 2, origin2);
248  calHalfFrameMO->AddNode(subSetCross, 3, origin3);

250  calHalfFrameMO->AddNode(subSetCrossTop, 1, origin1);

252  calHalfFrameMO->AddNode(subSetCrossTop, 3, origin3);
253  */
254

260  top->AddNode(calFrameMO, 1, gGeoIdentity);
261  LOG(DEBUG) << "**********************************\nmEndRadius:\t" << mEndRadius << std::endl;
