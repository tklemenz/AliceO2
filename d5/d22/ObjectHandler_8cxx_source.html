26 void ObjectHandler::GetObject(const std::string& path, std::string& object)
27 {
28  TFile* file = new TFile(path.c_str());
29 
30  // If file was not found or empty
31  if (file->IsZombie()) {
32  LOG(ERROR) << "The object was not found at " << path;
33  }
34 
35  // Get the AliCDBEntry from the root file
36  // we cast it directly to TObject (to avoid a link dependency on AliRoot here)
37  TObject* entry = file->Get("AliCDBEntry");
38 
39  // Create an outcoming buffer
40  TBufferFile* buffer = new TBufferFile(TBuffer::kWrite);
41 
42  // Stream and serialize the AliCDBEntry object to the buffer
43  buffer->WriteObject((const TObject*)entry);
44 
45  // Obtain a pointer to the buffer
46  char* pointer = buffer->Buffer();
47 
48  // Store the object to the referenced string
49  object.assign(pointer, buffer->Length());
50 
51  // LOG(INFO) << "Object length: " << object.size();
52 
53  // Release the open file
54  delete file;
55 
56  delete buffer;
57 
58  delete entry;
59 }
