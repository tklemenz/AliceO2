25 // FIXME: Should enforce the fact that DataRefs are read only...
26 struct DataRefUtils {
27  // SFINAE makes this available only for the case we are using
28  // a POD, this is to distinguish it from the alternative below,
29  // which works for TObject (which are serialised).
30  template <typename T>
31  static typename std::enable_if<std::is_pod<T>::value == true, gsl::span<T>>::type
32  as(DataRef const &ref) {
33  using DataHeader = o2::header::DataHeader;
34  auto header = o2::header::get<const DataHeader>(ref.header);
35  if (header->payloadSerializationMethod != o2::header::gSerializationMethodNone) {
36  throw std::runtime_error("Attempt to extract a POD from a wrong message kind");
37  }
38  if ((header->payloadSize % sizeof(T)) != 0) {
39  throw std::runtime_error("Cannot extract POD from message as size do not match");
40  }
41  //FIXME: provide a const collection
42  return gsl::span<T>(reinterpret_cast<T *>(const_cast<char *>(ref.payload)), header->payloadSize/sizeof(T));
43  }
44 
45  // See above. SFINAE allows us to use this to extract a TObject with
46  // a somewhat uniform API.
47  template <typename T>
48  static typename std::enable_if<std::is_base_of<TObject, T>::value == true, std::unique_ptr<T>>::type
49  as(DataRef const &ref) {
50  using DataHeader = o2::header::DataHeader;
51  auto header = o2::header::get<const DataHeader>(ref.header);
52  if (header->payloadSerializationMethod != o2::header::gSerializationMethodROOT) {
53  throw std::runtime_error("Attempt to extract a TMessage from non-ROOT serialised message");
54  }
55 
56  o2::framework::FairTMessage ftm(const_cast<char *>(ref.payload), header->payloadSize);
57  auto cobj = ftm.ReadObject(ftm.GetClass());
58  std::unique_ptr<T> result;
59  result.reset(dynamic_cast<T*>(cobj));
60  if (result.get() == nullptr) {
61  std::ostringstream ss;
62  ss << "Attempting to extract a " << T::Class()->GetName()
63  << " but a " << cobj->ClassName()
64  << " is actually stored which cannot be casted to the requested one.";
65  throw std::runtime_error(ss.str());
66  }
67  return std::move(result);
68  }
69 };
70 
71 }
