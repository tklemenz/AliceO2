118  template <typename ValueType, typename WriterT>
119  int writeRaw(ValueType value, uint16_t bitlength, WriterT writer) {
120  auto bitsToWrite = bitlength;
121  if (bitlength > 8*sizeof(ValueType)) {
122  // TODO: error policy
123  throw std::runtime_error("bit length exceeds width of the data type");
124  }
125  while (bitsToWrite > 0) {
126  if (mFilledBits == TargetBitWidth) {
127  mFilledBits=0;
128  writer(mCurrent);
129  mCurrent = 0;
130  }
131  // write at max what is left to be written
132  auto writeNow = bitsToWrite;
133  // write one element of the target buffer at a time
134  if (writeNow > TargetBitWidth) writeNow = TargetBitWidth;
135  // write the remaining space in the current element
136  auto capacity = TargetBitWidth - mFilledBits;
137  if (writeNow > capacity) writeNow = capacity;
138  auto mask = (((ValueType)1 << writeNow) - 1) << (bitsToWrite - writeNow);
139  auto activebits = (value&mask) >> (bitsToWrite - writeNow);
140  mCurrent |= activebits << (capacity-writeNow);
141  mFilledBits += writeNow;
142  bitsToWrite -= writeNow;
143  assert(mFilledBits <= TargetBitWidth);
144  }
145  return bitlength - bitsToWrite;
146  }
