79 static double radii2Turbo(double rMin, double rMid, double rMax, double sensW)
80 {
81  // compute turbo angle from radii and sensor width
82  return TMath::ASin((rMax * rMax - rMin * rMin) / (2 * rMid * sensW)) * TMath::RadToDeg();
83 }

85 static void configITS(Detector *its) {
86  // build ITS upgrade detector
87  const double kSiThickIB = 50e-4;
88  const double kSiThickOB = 50e-4;
89  //
90  const int kNLr = 7;
91  const int kNLrInner = 3;
92  const int kBuildLevel = 0;
93  const int kSensTypeID = 0; // dummy id for Alpide sensor
94 
95  enum { kRmn, kRmd, kRmx, kNModPerStave, kPhi0, kNStave, kNPar };
96  // Radii are from last TDR (ALICE-TDR-017.pdf Tab. 1.1, rMid is mean value)
97  const double tdr5dat[kNLr][kNPar] = {
98  {2.24, 2.34, 2.67, 9., 16.42, 12}, // for each inner layer: rMin,rMid,rMax,NChip/Stave, phi0, nStaves
99  {3.01, 3.15, 3.46, 9., 12.18, 16},
100  {3.78, 3.93, 4.21, 9., 9.55, 20},
101  {-1, 19.6 , -1, 4., 0. , 24}, // for others: -, rMid, -, NMod/HStave, phi0, nStaves // 24 was 49
102  {-1, 24.55, -1, 4., 0. , 30}, // 30 was 61
103  {-1, 34.39, -1, 7., 0. , 42}, // 42 was 88
104  {-1, 39.34, -1, 7., 0. , 48} // 48 was 100
105  };
106  const int nChipsPerModule = 7; // For OB: how many chips in a row
107  const double zChipGap = 0.01; // For OB: gap in Z between chips
108  const double zModuleGap = 0.01;// For OB: gap in Z between modules
109 
110  double dzLr, rLr, phi0, turbo;
111  int nStaveLr, nModPerStaveLr;
112 
113  its->setStaveModelIB(o2::ITS::Detector::kIBModel4);
114  its->setStaveModelOB(o2::ITS::Detector::kOBModel2);
115 
116  const int kNWrapVol = 3;
117  const double wrpRMin[kNWrapVol] = { 2.1, 15.0, 32.0};
118  const double wrpRMax[kNWrapVol] = {14.0, 30.0, 46.0};
119  const double wrpZSpan[kNWrapVol] = {70., 95., 200.};
120 
121  for (int iw = 0; iw < kNWrapVol; iw++) {
122  its->defineWrapperVolume(iw, wrpRMin[iw], wrpRMax[iw], wrpZSpan[iw]);
123  }
124 
125  for (int idLr = 0; idLr < kNLr; idLr++) {
126  rLr = tdr5dat[idLr][kRmd];
127  phi0 = tdr5dat[idLr][kPhi0];
128 
129  nStaveLr = TMath::Nint(tdr5dat[idLr][kNStave]);
130  nModPerStaveLr = TMath::Nint(tdr5dat[idLr][kNModPerStave]);
131  int nChipsPerStaveLr = nModPerStaveLr;
132  if (idLr >= kNLrInner) {
133  its->defineLayer(idLr, phi0, rLr, nStaveLr, nModPerStaveLr,
134  kSiThickOB, Segmentation::SensorThickness, kSensTypeID, kBuildLevel);
135  } else {
136  turbo = -radii2Turbo(tdr5dat[idLr][kRmn], rLr, tdr5dat[idLr][kRmx], Segmentation::SensorSizeRows);
137  its->defineLayerTurbo(idLr, phi0, rLr, nStaveLr,
138  nChipsPerStaveLr, Segmentation::SensorSizeRows, turbo, kSiThickIB,
139  Segmentation::SensorThickness, kSensTypeID, kBuildLevel);
140  }
141  }

315 Bool_t Detector::ProcessHits(FairVolume *vol)
316 {
317  auto vmc = TVirtualMC::GetMC();
318  // This method is called from the MC stepping
319  if (!(vmc->TrackCharge())) {
320  return kFALSE;
321  }
322 
323  Int_t lay=0, volID = vol->getMCid();
324 
325  // FIXME: Determine the layer number. Is this information available directly from the FairVolume?
326  bool notSens = false;
327  while ((lay < sNumberLayers) && (notSens=(volID != mLayerID[lay]))) {
328  ++lay;
329  }
330  if (notSens) return kFALSE; //RS: can this happen? This method must be called for sensors only?
331 
332  // FIXME: Is it needed to keep a track reference when the outer ITS volume is encountered?
333  // if(vmc->IsTrackExiting()) {
334  // AddTrackReference(gAlice->GetMCApp()->GetCurrentTrackNumber(), AliTrackReference::kITS);
335  // } // if Outer ITS mother Volume
336  bool startHit=false, stopHit=false;
337  unsigned char status = 0;
338  if (vmc->IsTrackEntering()) { status |= Hit::kTrackEntering; }
339  if (vmc->IsTrackInside()) { status |= Hit::kTrackInside; }
340  if (vmc->IsTrackExiting()) { status |= Hit::kTrackExiting; }
341  if (vmc->IsTrackOut()) { status |= Hit::kTrackOut; }
342  if (vmc->IsTrackStop()) { status |= Hit::kTrackStopped; }
343  if (vmc->IsTrackAlive()) { status |= Hit::kTrackAlive; }
344 
345  // track is entering or created in the volume
346  if ( (status & Hit::kTrackEntering) || (status & Hit::kTrackInside && !mTrackData.mHitStarted) ) {
347  startHit = true;
348  }
349  else if ( (status & (Hit::kTrackExiting|Hit::kTrackOut|Hit::kTrackStopped)) ) {
350  stopHit = true;
351  }
352 
353  // increment energy loss at all steps except entrance
354  if (!startHit) mTrackData.mEnergyLoss += vmc->Edep();
355  if (!(startHit|stopHit)) return kFALSE; // do noting
356 
357  if (startHit) {
358  mTrackData.mEnergyLoss = 0.;
359  vmc->TrackMomentum(mTrackData.mMomentumStart);
360  vmc->TrackPosition(mTrackData.mPositionStart);
361  mTrackData.mTrkStatusStart = status;
362  mTrackData.mHitStarted = true;
363  }
364  if (stopHit) {
365  TLorentzVector positionStop;
366  vmc->TrackPosition(positionStop);
367  // Retrieve the indices with the volume path
368  int stave(0), halfstave(0), chipinmodule(0), module;
369  vmc->CurrentVolOffID(1, chipinmodule);
370  vmc->CurrentVolOffID(2, module);
371  vmc->CurrentVolOffID(3, halfstave);
372  vmc->CurrentVolOffID(4, stave);
373  int chipindex = mGeometryTGeo->getChipIndex(lay, stave, halfstave, module, chipinmodule);
374 
375  Hit *p = addHit(vmc->GetStack()->GetCurrentTrackNumber(), chipindex,
376  mTrackData.mPositionStart.Vect(),positionStop.Vect(),mTrackData.mMomentumStart.Vect(),
377  mTrackData.mMomentumStart.E(),positionStop.T(),mTrackData.mEnergyLoss,
378  mTrackData.mTrkStatusStart,status);
379  //p->SetTotalEnergy(vmc->Etot());
380 
381  // RS: not sure this is needed
382  // Increment number of Detector det points in TParticle
383  o2::Data::Stack *stack = (o2::Data::Stack *) TVirtualMC::GetMC()->GetStack();
384  stack->addHit(GetDetId());
385  }
386 
387  return kTRUE;
388 }

778 void Detector::constructDetectorGeometry()
779 {
780  // Create the geometry and insert it in the mother volume ITSV
781  TGeoManager *geoManager = gGeoManager;
782 
783  TGeoVolume *vALIC = geoManager->GetVolume("cave");
784 
785  if (!vALIC) {
786  LOG(FATAL) << "Could not find the top volume" << FairLogger::endl;
787  }
788 
789  new TGeoVolumeAssembly(GeometryTGeo::getITSVolPattern());
790  TGeoVolume *vITSV = geoManager->GetVolume(GeometryTGeo::getITSVolPattern());
791  vALIC->AddNode(vITSV, 2, nullptr); // Copy number is 2 to cheat AliGeoManager::CheckSymNamesLUT
792 
793  const Int_t kLength = 100;
794  Char_t vstrng[kLength] = "xxxRS"; //?
795  vITSV->SetTitle(vstrng);
796 
797  // Check that we have all needed parameters
798  for (Int_t j = 0; j < sNumberLayers; j++) {
799  if (mLayerRadii[j] <= 0) {
800  LOG(FATAL) << "Wrong layer radius for layer " << j << "(" << mLayerRadii[j] << ")"
801  << FairLogger::endl;
802  }
803  if (mStavePerLayer[j] <= 0) {
804  LOG(FATAL) << "Wrong number of staves for layer " << j << "(" << mStavePerLayer[j] << ")"
805  << FairLogger::endl;
806  }
807  if (mUnitPerStave[j] <= 0) {
808  LOG(FATAL) << "Wrong number of chips for layer " << j << "(" << mUnitPerStave[j] << ")"
809  << FairLogger::endl;
810  }
811  if (mChipThickness[j] < 0) {
812  LOG(FATAL) << "Wrong chip thickness for layer " << j << "(" << mChipThickness[j] << ")"
813  << FairLogger::endl;
814  }
815  if (mTurboLayer[j] && mStaveWidth[j] <= 0) {
816  LOG(FATAL) << "Wrong stave width for layer " << j << "(" << mStaveWidth[j] << ")"
817  << FairLogger::endl;
818  }
819  if (mDetectorThickness[j] < 0) {
820  LOG(FATAL) << "Wrong Sensor thickness for layer " << j << "(" << mDetectorThickness[j] << ")"
821  << FairLogger::endl;
822  }
823 
824  if (j > 0) {
825  if (mLayerRadii[j] <= mLayerRadii[j - 1]) {
826  LOG(FATAL) << "Layer " << j << " radius (" << mLayerRadii[j] << ") is smaller than layer "
827  << j - 1 << " radius (" << mLayerRadii[j - 1] << ")" << FairLogger::endl;
828  }
829  }
830 
831  if (mChipThickness[j] == 0) {
832  LOG(INFO) << "Chip thickness for layer " << j << " not set, using default"
833  << FairLogger::endl;
834  }
835  if (mDetectorThickness[j] == 0) {
836  LOG(INFO) << "Sensor thickness for layer " << j << " not set, using default"
837  << FairLogger::endl;
838  }
839  }
840 
841  // Create the wrapper volumes
842  TGeoVolume **wrapVols = nullptr;
843 
844  if (sNumberOfWrapperVolumes) {
845  wrapVols = new TGeoVolume *[sNumberOfWrapperVolumes];
846  for (int id = 0; id < sNumberOfWrapperVolumes; id++) {
847  wrapVols[id] = createWrapperVolume(id);
848  vITSV->AddNode(wrapVols[id], 1, nullptr);
849  }
850  }
851 
852  // Now create the actual geometry
853  for (Int_t j = 0; j < sNumberLayers; j++) {
854  TGeoVolume *dest = vITSV;
855  mWrapperLayerId[j] = -1;
856 
857  if (mTurboLayer[j]) {
858  mGeometry[j] = new V3Layer(j, kTRUE, kFALSE);
859  mGeometry[j]->setStaveWidth(mStaveWidth[j]);
860  mGeometry[j]->setStaveTilt(mStaveTilt[j]);
861  } else {
862  mGeometry[j] = new V3Layer(j, kFALSE);
863  }
864 
865  mGeometry[j]->setPhi0(mLayerPhi0[j]);
866  mGeometry[j]->setRadius(mLayerRadii[j]);
867  mGeometry[j]->setNumberOfStaves(mStavePerLayer[j]);
868  mGeometry[j]->setNumberOfUnits(mUnitPerStave[j]);
869  mGeometry[j]->setChipType(mChipTypeID[j]);
870  mGeometry[j]->setBuildLevel(mBuildLevel[j]);
871 
872  if (j < sNumberInnerLayers) {
873  mGeometry[j]->setStaveModel(mStaveModelInnerBarrel);
874  } else {
875  mGeometry[j]->setStaveModel(mStaveModelOuterBarrel);
876  }
877 
878  LOG(DEBUG1) << "mBuildLevel: " << mBuildLevel[j] << FairLogger::endl;
879 
880  if (mChipThickness[j] != 0) {
881  mGeometry[j]->setChipThick(mChipThickness[j]);
882  }
883  if (mDetectorThickness[j] != 0) {
884  mGeometry[j]->setSensorThick(mDetectorThickness[j]);
885  }
886 
887  for (int iw = 0; iw < sNumberOfWrapperVolumes; iw++) {
888  if (mLayerRadii[j] > mWrapperMinRadius[iw] && mLayerRadii[j] < mWrapperMaxRadius[iw]) {
889  LOG(INFO) << "Will embed layer " << j << " in wrapper volume " << iw << FairLogger::endl;
890 
891  dest = wrapVols[iw];
892  mWrapperLayerId[j] = iw;
893  break;
894  }
895  }
896  mGeometry[j]->createLayer(dest);
897  }
898  createServiceBarrel(kTRUE, wrapVols[0]);
899  createServiceBarrel(kFALSE, wrapVols[2]);
900 
901  delete[] wrapVols; // delete pointer only, not the volumes
902 }
