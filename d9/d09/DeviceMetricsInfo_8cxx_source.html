24 // Parses a metric in the form

26 // METRIC:<type>:<name>:<timestamp>:<value>
27 bool parseMetric(const std::string &s, std::smatch &match) {
28  const static std::regex metricsRE(".*METRIC:(int|float|string):([a-zA-Z0-9/_-]+):([0-9]+):(.*)");
29  return std::regex_match(s, match, metricsRE);
30 }

32 // and fills the appropriatate DeviceInfo plot accordingly.
33 //
34 // @matches is the regexp_matches from the metric identifying regex
35 // @info is the DeviceInfo associated to the device posting the metric
36 bool processMetric(const std::smatch &match,
37  DeviceMetricsInfo &info) {
38  auto type = match[1];
39  auto name = match[2];
40  char *ep = nullptr;
41  auto timestamp = strtol(match[3].str().c_str(), &ep, 10);
42  if (ep == nullptr || *ep != '\0') {
43  return false;
44  }
45  auto stringValue = match[4];
46 
47  size_t metricIndex = -1;
48 
49  auto metricType = MetricType::Unknown;
50  if (type.str() == "int") {
51  metricType = MetricType::Int;
52  } else if (type.str() == "float") {
53  metricType = MetricType::Float;
54  }
55 
56  // Find the metric based on the label. Create it if not found.
57  using IndexElement = std::pair<std::string, size_t>;
58  auto cmpFn = [](const IndexElement &a, const IndexElement &b) -> bool {
59  return std::tie(a.first, a.second) < std::tie(b.first, b.second);
60  };
61  IndexElement metricLabelIdx = std::make_pair(name.str(), 0);
62  auto mi = std::lower_bound(info.metricLabelsIdx.begin(),
63  info.metricLabelsIdx.end(),
64  metricLabelIdx,
65  cmpFn);
66 
67  // We could not find the metric, lets insert a new one.
68  if (mi == info.metricLabelsIdx.end()
69  || mi->first != metricLabelIdx.first) {
70  // Create a new metric
71  MetricInfo metricInfo;
72  metricInfo.pos = 0;
73  metricInfo.type = metricType;
74  // Add a new empty buffer for it of the correct kind
75  switch(metricType) {
76  case MetricType::Int:
77  metricInfo.storeIdx = info.intMetrics.size();
78  info.intMetrics.emplace_back(std::array<int, 1024>{});
79  break;
80  case MetricType::Float:
81  metricInfo.storeIdx = info.floatMetrics.size();
82  info.floatMetrics.emplace_back(std::array<float, 1024>{});
83  break;
84  default:
85  return false;
86  };
87  // Add the timestamp buffer for it
88  info.timestamps.emplace_back(std::array<size_t, 1024>{});
89 
90  // Add the index by name in the correct position
91  // this will require moving the tail of the index,
92  // but inserting should happen only once for each metric,
93  // so who cares.
94  metricLabelIdx.second = info.metrics.size();
95  info.metricLabelsIdx.insert(mi, metricLabelIdx);
96  // Add the the actual Metric info to the store
97  metricIndex = info.metrics.size();
98  info.metrics.push_back(metricInfo);
99  } else {
100  metricIndex = mi->second;
101  }
102  assert(metricIndex != -1);
103  // We are now guaranteed our metric is present at metricIndex.
104  MetricInfo &metricInfo = info.metrics[metricIndex];
105 
106  int intValue = 0;
107  float floatValue = 0;
108  auto mod = info.timestamps[metricIndex].size();
109 
110  switch(metricInfo.type) {
111  case MetricType::Int:
112  intValue = strtol(stringValue.str().c_str(), &ep, 10);
113  if (!ep || *ep != '\0') {
114  return false;
115  }
116  info.intMetrics[metricInfo.storeIdx][metricInfo.pos] = intValue;
117  break;
118  case MetricType::Float:
119  floatValue = strtof(stringValue.str().c_str(), &ep);
120  if (!ep || *ep != '\0') {
121  return false;
122  }
123  info.floatMetrics[metricInfo.storeIdx][metricInfo.pos] = floatValue;
124  break;
125  default:
126  return false;
127  break;
128  };
129 
130  // Save the timestamp for the current metric we do it here
131  // so that we do not update timestamps for broken metrics
132  info.timestamps[metricIndex][metricInfo.pos] = timestamp;
133  // Update the position where to write the next metric
134  metricInfo.pos = (metricInfo.pos + 1) % mod;
135  return true;
136 }
