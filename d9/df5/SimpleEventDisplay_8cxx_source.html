72 Int_t SimpleEventDisplay::updateROC(const Int_t roc,
73  const Int_t row,
74  const Int_t pad,
75  const Int_t timeBin,
76  const Float_t signal)
77 {
78  //
79  // Signal filling methode on the fly pedestal and time offset correction if necessary.
80  // no extra analysis necessary. Assumes knowledge of the signal shape!
81  // assumes that it is looped over consecutive time bins of one pad
82  //
83  //printf("update called: %d, %d, %d, %d, %.3f\n", roc, row, pad, timeBin, signal);
84  if (row<0) return 0;
85  if (pad<0) return 0;
86  if (timeBin<0) return 0;
87  if ( (timeBin>mLastTimeBin) || (timeBin<mFirstTimeBin) ) return 0;
88  if (mSectorLoop && roc%36!=mSelectedSector%36) return 0;
89 
90  if ( row<0 || pad<0 ){
91  printf("Wrong Pad or Row number, skipping!");
92  return 0;
93  }
94 
95  float corrSignal = signal;
96 
97  // ===| get pedestal |========================================================
98  if (mPedestals) {
99  corrSignal -= mPedestals->getValue(ROC(roc), row, pad);
100  }
101 
102  Int_t iChannel = mTPCmapper.globalPadNumber(PadPos(row, pad));
103 
104  //init first pad and roc in this event
105  if ( mCurrentChannel == -1 ) {
106  mCurrentChannel = iChannel;
107  mCurrentROC = roc;
108  mCurrentRow = row;
109  mCurrentPad = pad;
110  }
111 
112  //process last pad if we change to a new one
113  if ( iChannel != mCurrentChannel ){
114  //ProcessPad();
115  mLastSector=mCurrentROC;
116  mCurrentChannel = iChannel;
117  mCurrentROC = roc;
118  mCurrentRow = row;
119  mCurrentPad = pad;
120  mMaxPadSignal=0;
121  }
122 // if (signal>0) printf("%02d:%03d:%03d:%05d: %.3f\n",mCurrentROC,mCurrentRow,mCurrentPad,mCurrentChannel,signal);
123  //fill signals for current pad
124  if (mCurrentROC%36==mSelectedSector%36){
125  const Int_t nbins = mLastTimeBin - mFirstTimeBin;
126  const Int_t offset = (nbins+2)*(iChannel+1)+timeBin+1;
127 
128  if ((UInt_t)roc<mTPCmapper.getNumberOfIROCs()){
129  mHSigIROC->GetArray()[offset]=corrSignal;
130  }else{
131  mHSigOROC->GetArray()[offset]=corrSignal;
132  }
133  }
134 
135  CalROC& calROC = mPadMax.getCalArray(mCurrentROC);
136  //auto val = calROC.getValue(mCurrentRow, mCurrentPad);
137  auto val = calROC.getValue(row, pad);
138 
139  if ( corrSignal > val ){
140  //printf("sec: %2d, row: %2d, pad: %3d, val: %.2f, sig: (%.2f) %.2f\n", mCurrentROC, mCurrentRow, mCurrentPad, val, corrSignal, signal);
141  //calROC.setValue(mCurrentRow,mCurrentPad,signal);
142  calROC.setValue(row, pad, corrSignal);
143  mMaxPadSignal = corrSignal;
144  mMaxTimeBin = timeBin;
145  }
146  //printf("update done\n");
147  return 0;
148 }
