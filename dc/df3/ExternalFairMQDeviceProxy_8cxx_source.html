28 void broadcastMessage(FairMQDevice &device, o2::header::Stack &&headerStack, FairMQMessagePtr &&payloadMessage, int index) {
29  for (auto &channelInfo : device.fChannels) {
30  auto channel = channelInfo.first;
31  assert(channelInfo.second.size() == 1);
32  // FIXME: I need to make sure the input channel is not used... For the moment
33  // I rely on the convention dpl channels start with "from_".
34  if (strncmp(channel.data(), "from_", 5) != 0) {
35  continue;
36  }
37 
38  // FIXME: this assumes there is only one output from here... This should
39  // really do the matchmaking between inputs and output channels.
40  FairMQMessagePtr headerMessage = device.NewMessageFor(channel, index,
41  headerStack.buffer.get(),
42  headerStack.bufferSize,
43  &o2::header::Stack::freefn,
44  headerStack.buffer.get());
45  headerStack.buffer.release();
46  FairMQParts out;
47  out.AddPart(std::move(headerMessage));
48  out.AddPart(std::move(payloadMessage));
49  device.Send(out, channel, index);
50  }
51 }

54 InjectorFunction o2DataModelAdaptor(OutputSpec const &spec, uint64_t startTime, uint64_t step) {
55  auto timesliceId = std::make_shared<size_t>(startTime);
56  return [timesliceId, step](FairMQDevice &device, FairMQParts &parts, int index) {
57  for (size_t i = 0; i < parts.Size()/2; ++i) {
58  auto dh = o2::header::get<DataHeader>(parts.At(i*2)->GetData());
59 
60  DataProcessingHeader dph{*timesliceId, 0};
61  o2::header::Stack headerStack{*dh, dph};
62  broadcastMessage(device, std::move(headerStack), std::move(parts.At(i*2+1)), index);
63  auto oldTimesliceId = *timesliceId;
64  *timesliceId += 1;
65  }
66  };
67 }

69 InjectorFunction incrementalConverter(OutputSpec const &spec, uint64_t startTime, uint64_t step) {
70  auto timesliceId = std::make_shared<size_t>(startTime);
71 
72  return [timesliceId, spec, step](FairMQDevice &device, FairMQParts &parts, int index) {
73  // We iterate on all the parts and we send them two by two,
74  // adding the appropriate O2 header.
75  for (size_t i = 0; i < parts.Size(); ++i) {
76  DataHeader dh;
77  dh.dataOrigin = spec.origin;
78  dh.dataDescription = spec.description;
79  dh.subSpecification = spec.subSpec;
80  dh.payloadSize = parts.At(i)->GetSize();
81 
82  DataProcessingHeader dph{*timesliceId, 0};
83  *timesliceId += step;
84  //we have to move the incoming data
85  o2::header::Stack headerStack{dh, dph};
86 
87  broadcastMessage(device, std::move(headerStack), std::move(parts.At(i)), index);
88  }
89  };
90 }

92 DataProcessorSpec specifyExternalFairMQDeviceProxy(char const *name,
93  std::vector<OutputSpec> const &outputs,
94  char const *channelConfig,
95  std::function<void(FairMQDevice &device,
96  FairMQParts& inputs,
97  int index)> converter) {
98  DataProcessorSpec spec;
99  spec.name = strdup(name);
100  spec.inputs = {};
101  spec.outputs = outputs;
102  // The Init method will register a new "Out of band" channel and
103  // attach an OnData to it which is responsible for converting incoming
104  // messages into DPL messages.
105  spec.algorithm = AlgorithmSpec{[converter, channel = spec.name](InitContext &ctx) {
106  auto device = ctx.services().get<RawDeviceService>().device();
107  assert(device);
108 
109  // Converter should pump messages
110  auto handler = [device,converter](FairMQParts &inputs, int idx) {
111  converter(*device, inputs, idx);
112  return true;
113  };
114 
115  device->OnData(channel, handler);
116  return [](ProcessingContext &) {};
117  }
118  };
119  const char *d = strdup((std::string("name=") + name + "," + channelConfig).c_str());
120  spec.options = {
121  ConfigParamSpec{"channel-config", VariantType::String, d, {"Out-of-band channel config"}}
122  };
123  return spec;
124 }
