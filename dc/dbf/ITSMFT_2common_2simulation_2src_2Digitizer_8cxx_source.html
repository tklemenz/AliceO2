52 void Digitizer::process(const std::vector<Hit>* hits, std::vector<Digit>* digits)
53 {
54  // digitize single event
55 
56  const Int_t numOfChips = mGeometry->getNumberOfChips();
57 
58  // estimate the smalles RO Frame this event may have
59  double hTime0 = mEventTime - mParams.getTimeOffset();
60  if (hTime0 > UINT_MAX) {
61  LOG(WARNING) << "min Hit RO Frame undefined: time: " << hTime0 << " is in far future: "
62  << " EventTime: " << mEventTime << " TimeOffset: "
63  << mParams.getTimeOffset() << FairLogger::endl;
64  return;
65  }
66 
67  if (hTime0<0) hTime0 = 0.;
68  UInt_t minNewROFrame = static_cast<UInt_t>(hTime0/mParams.getROFrameLenght());
69 
70  LOG(INFO) << "Digitizing ITS event at time " << mEventTime
71  << " (TOffset= " << mParams.getTimeOffset() << " ROFrame= " << minNewROFrame << ")"
72  << " cont.mode: " << isContinuous() << " current Min/Max RO Frames "
73  << mROFrameMin << "/" << mROFrameMax << FairLogger::endl ;
74 
75  if (mParams.isContinuous() && minNewROFrame>mROFrameMin) {
76  // if there are already digits cached for previous RO Frames AND the new event
77  // cannot contribute to these digits, move them to the output container
78  if (mROFrameMax<minNewROFrame) mROFrameMax = minNewROFrame-1;
79  for (auto rof=mROFrameMin; rof<minNewROFrame; rof++) {
80  fillOutputContainer(digits, rof);
81  }
82  // fillOutputContainer(digits, minNewROFrame-1);
83  }
84 
85  // accumulate hits for every chip
86  for(auto& hit : *hits) {
87  // RS: ATTENTION: this is just a trick until we clarify how the hits from different source are
88  // provided and identified. At the moment we just create a combined identifier from eventID
89  // and sourceID and store it TEMPORARILY in the cached Point's TObject UniqueID
90  const_cast<Hit&>(hit).SetSrcEvID(mCurrSrcID,mCurrEvID);
91  mSimulations[hit.GetDetectorID()].InsertHit(&hit);
92  }
93 
94  // Convert hits to digits
95  for (auto &simulation : mSimulations) {
96  simulation.Hits2Digits(mEventTime, mROFrameMin, mROFrameMax);
97  simulation.ClearHits();
98  }
99 
100  // in the triggered mode store digits after every MC event
101  if (!mParams.isContinuous()) {
102  fillOutputContainer(digits, mROFrameMax);
103  }
104 }

108 void Digitizer::setEventTime(double t)
109 {
110  // assign event time, it should be in a strictly increasing order
111  // convert to ns
112  t *= mCoeffToNanoSecond;
113 
114  if (t<mEventTime && mParams.isContinuous()) {
115  LOG(FATAL) << "New event time (" << t << ") is < previous event time (" << mEventTime << ")" << FairLogger::endl;
116  }
117  mEventTime = t;
118  // to limit the range of RO Frame IDs we subtract the meaningless offset
119  if (mParams.isContinuous()) { // in continuous mode we set the offset only in the very beginning
120  if (!mParams.isTimeOffsetSet()) { // offset is initially at -inf
121  mParams.setTimeOffset(mEventTime + mParams.getROFrameLenght()*(gRandom->Rndm()-0.5));
122  }
123  }
124  else { // in the triggered mode we start from 0 ROFrame in every event
125  mParams.setTimeOffset( mEventTime + mParams.getROFrameLenght()*(gRandom->Rndm()-0.5));
126  mROFrameMin = 0; // so we reset the frame counters
127  mROFrameMax = 0;
128  }
129 }

132 void Digitizer::fillOutputContainer(std::vector<Digit>* digits, UInt_t maxFrame)
133 {
134  // fill output with digits ready to be stored, generating the noise beforehand
135  if (maxFrame>mROFrameMax) maxFrame = mROFrameMax;
136 
137  LOG(INFO) << "Filling ITS digits output for RO frames " << mROFrameMin << ":" << maxFrame << FairLogger::endl ;
138 
139  for (auto &simulation : mSimulations) {
140  // add the random noise to all ROFrame being stored
141  simulation.addNoise(mROFrameMin,maxFrame);
142  }
143 
144  // we have to write chips in RO increasing order, therefore have to loop over the frames here
145  for (auto rof=mROFrameMin;rof<=maxFrame;rof++) {
146  for (auto &simulation : mSimulations) {
147  simulation.fillOutputContainer(digits,rof);
148  }
149  }
150  mROFrameMin = maxFrame+1;
151 }
