149 inline void Cluster::setClusterUsage(Int_t n)
150 {
151  // set cluster usage counter
152  mNxNzN &= ~(kMaskClUse << kOffsClUse);
153  mNxNzN |= (n & kMaskClUse) << kOffsClUse;
154  if (n < 2) resetBit(kShared);
