85  Int_t nChips = mSegmentation->getNSensors();

91  Double_t shiftY = 2*Geometry::sSensorTopOffset+SegmentationAlpide::SensorSizeRows-Geometry::sFlexHeight/2; // strange
92  TGeoVolumeAssembly * flexVol = mFlex->makeFlex(mSegmentation->getNSensors(), flexLength);
93  mLadderVolume->AddNode(flexVol, 1, new TGeoTranslation(flexLength/2+Geometry::sSensorSideOffset/2, shiftY,
94  Geometry::sFlexThickness/2-Geometry::sRohacell));
95 
96  // Create the CMOS Sensors
97  createSensors();

106 void Ladder::createSensors()
107 {
108 
109  Geometry * mftGeom = Geometry::instance();
110 
111  // Create Shapes
112 
113  // sensor = sensitive volume
114  TString namePrefixS = "MFTSensor";
115 
116  // chip = sensor + readout
117  TString namePrefixC = Form("MFT_C_%d_%d_%d",
118  mftGeom->getHalfID(mSegmentation->GetUniqueID()),
119  mftGeom->getDiskID(mSegmentation->GetUniqueID()),
120  mftGeom->getLadderID(mSegmentation->GetUniqueID()));
121 
122  // the MFT glue
123  TString namePrefixG = Form("MFT_G_%d_%d_%d",
124  mftGeom->getHalfID(mSegmentation->GetUniqueID()),
125  mftGeom->getDiskID(mSegmentation->GetUniqueID()),
126  mftGeom->getLadderID(mSegmentation->GetUniqueID()));
127 
128  TGeoMedium *kMedGlue = gGeoManager->GetMedium("MFT_SE4445$");
129 
130  TGeoVolume * glue = gGeoManager->MakeBox(namePrefixG.Data(), kMedGlue,
131  (SegmentationAlpide::SensorSizeCols-Geometry::sGlueEdge)/2.,
132  (SegmentationAlpide::SensorSizeRows-Geometry::sGlueEdge)/2.,
133  Geometry::sGlueThickness/2.);
134  glue->SetVisibility(kTRUE);
135  glue->SetLineColor(kRed-10);
136  glue->SetLineWidth(1);
137  glue->SetFillColor(glue->GetLineColor());
138  glue->SetFillStyle(4000); // 0% transparent
139 
140  // common with ITS
141  TGeoVolume* chipVol = AlpideChip::createChip(Geometry::sChipThickness/2.,
142  Geometry::sSensorThickness/2.,namePrefixC,namePrefixS,kFALSE);
143 
144  //chipVol->Print();
145 
146  for (int ichip = 0; ichip < mSegmentation->getNSensors(); ichip++) {
147 
148  ChipSegmentation* chipSeg = mSegmentation->getSensor(ichip);
149  TGeoCombiTrans* chipPos = chipSeg->getTransformation();
150  TGeoCombiTrans* chipPosGlue = chipSeg->getTransformation();
151 
152  // Position of the center on the chip in the chip coordinate system
153  Double_t pos[3] = {SegmentationAlpide::SensorSizeCols/2., SegmentationAlpide::SensorSizeRows/2.,
154  Geometry::sChipThickness/2. - Geometry::sGlueThickness - Geometry::sRohacell};
155 
156  Double_t posglue[3] = {SegmentationAlpide::SensorSizeCols/2., SegmentationAlpide::SensorSizeRows/2.,
157  Geometry::sGlueThickness/2-Geometry::sChipThickness-Geometry::sRohacell};
158 
159  Double_t master[3];
160  Double_t masterglue[3];
161  chipPos->LocalToMaster(pos, master);
162  chipPosGlue->LocalToMaster(posglue, masterglue);
163 
164  TGeoBBox* shape = (TGeoBBox*)mLadderVolume->GetShape();
165  master[0] -= shape->GetDX();
166  master[1] -= shape->GetDY();
167  master[2] -= shape->GetDZ();
168 
169  masterglue[0] -= shape->GetDX();
170  masterglue[1] -= shape->GetDY();
171  masterglue[2] -= shape->GetDZ();
172 
173  LOG(DEBUG1) << "CreateSensors " << Form("adding chip %s_%d ",namePrefixS.Data(),ichip) << FairLogger::endl;
174  //chipPos->Print();
175 
176  TGeoTranslation* trans = new TGeoTranslation(master[0],master[1],master[2]);
177  TGeoHMatrix* final = new TGeoHMatrix((*trans)*(Geometry::sTransMFT2ITS).Inverse());
178  mLadderVolume->AddNode(chipVol, ichip, final);
179  mLadderVolume->AddNode(glue, ichip, new TGeoTranslation(masterglue[0],masterglue[1],masterglue[2]));
180 
181  }
182 
183 }
