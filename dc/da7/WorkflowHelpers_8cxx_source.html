23 std::vector<size_t>
24 WorkflowHelpers::topologicalSort(size_t nodeCount,
25  const size_t *edgeIn,
26  const size_t *edgeOut,
27  size_t stride,
28  size_t edgesCount) {
29  using NodeIndex = size_t;
30  using EdgeIndex = size_t;
31  // Create the index which will be returned.
32  std::vector<NodeIndex> index(nodeCount);
33  for (NodeIndex wi = 0; wi < nodeCount; ++wi) {
34  index[wi] = wi;
35  }
36  // Temporary vector holding vertices to be processed
37  std::vector<EdgeIndex> remainingEdgesIndex(edgesCount);
38  for (EdgeIndex ei = 0; ei < edgesCount; ++ei) {
39  remainingEdgesIndex[ei] = ei;
40  }
41 
42  // Create a vector where at each position we have true
43  // if the vector has dependencies, false otherwise
44  std::vector<bool> nodeDeps(nodeCount, false);
45  for (EdgeIndex ei = 0; ei < edgesCount; ++ei) {
46  nodeDeps[*(edgeOut+ei*stride)] = true;
47  }
48 
49  std::list<NodeIndex> L;
50  std::vector<NodeIndex> S;
51  std::set<NodeIndex> nextVertex;
52  std::vector<EdgeIndex> nextEdges;
53 
54  for (size_t ii = 0, ie = index.size(); ii < ie; ++ii) {
55  if (nodeDeps[ii] == false) {
56  L.push_back(ii);
57  }
58  }
59  while (!L.empty()) {
60  auto node = L.front();
61  S.push_back(node);
62  L.pop_front();
63  nextVertex.clear();
64  nextEdges.clear();
65 
66  for (EdgeIndex ei = 0, ee = remainingEdgesIndex.size(); ei < ee; ++ei) {
67  if (*(edgeIn+ei*stride) == node) {
68  nextVertex.insert(*(edgeOut+ei*stride));
69  } else {
70  nextEdges.push_back(remainingEdgesIndex[ei]);
71  }
72  }
73  remainingEdgesIndex.swap(nextEdges);
74 
75  std::set<NodeIndex> hasPredecessors;
76  for (auto &ei : remainingEdgesIndex) {
77  for (auto &m : nextVertex) {
78  if (m == *(edgeOut+ei*stride)) {
79  hasPredecessors.insert(m);
80  }
81  }
82  }
83  std::vector<NodeIndex> withPredecessor;
84  std::set_difference(nextVertex.begin(), nextVertex.end(),
85  hasPredecessors.begin(), hasPredecessors.end(),
86  std::back_inserter(withPredecessor));
87  std::copy(withPredecessor.begin(), withPredecessor.end(), std::back_inserter(L));
88  }
89  return S;
90 }

92 void
93 WorkflowHelpers::constructGraph(const WorkflowSpec &workflow,
94  std::vector<DeviceConnectionEdge> &logicalEdges,
95  std::vector<OutputSpec> &outputs,
96  std::vector<LogicalForwardInfo> &forwardedInputsInfo) {
97  assert(!workflow.empty());
98  // This is the state. Oif is the iterator I use for the searches.
99  std::list<LogicalOutputInfo> availableOutputsInfo;
100  auto const &constOutputs = outputs; // const version of the outputs
101  decltype(availableOutputsInfo.begin()) oif;
102  // Forwards is a local cache to avoid adding forwards before time.
103  std::vector<LogicalOutputInfo> forwards;
104 
105  // Notice that availableOutputsInfo MUST be updated first, since it relies on
106  // the size of outputs to be the one before the update.
107  auto enumerateAvailableOutputs = [&workflow,&outputs,&availableOutputsInfo]() {
108  for (size_t wi = 0; wi < workflow.size(); ++wi) {
109  auto &producer = workflow[wi];
110 
111  for (size_t oi = 0; oi < producer.outputs.size(); ++oi) {
112  auto &out = producer.outputs[oi];
113  auto uniqueOutputId = outputs.size();
114  availableOutputsInfo.emplace_back(LogicalOutputInfo{wi, uniqueOutputId, false});
115  outputs.push_back(out);
116  }
117  }
118  };
119 
120  // Notice that if the output is actually a forward, we need to store that
121  // information so that when we add it at device level we know which output
122  // channel we need to connect it too.
123  auto hasMatchingOutputFor = [&workflow, &constOutputs,
124  &availableOutputsInfo, &oif,
125  &forwardedInputsInfo](size_t ci, size_t ii) {
126  assert(ci < workflow.size());
127  assert(ii < workflow[ci].inputs.size());
128  auto &input = workflow[ci].inputs[ii];
129  auto matcher = [&input, &constOutputs](const LogicalOutputInfo &outputInfo) -> bool {
130  auto &output = constOutputs[outputInfo.outputGlobalIndex];
131  return matchDataSpec2Channel(input, outputSpec2LogicalChannel(output));
132  };
133  oif = std::find_if(availableOutputsInfo.begin(),
134  availableOutputsInfo.end(),
135  matcher);
136  if (oif != availableOutputsInfo.end() && oif->forward) {
137  LogicalForwardInfo forward;
138  forward.consumer = ci;
139  forward.inputLocalIndex = ii;
140  forward.outputGlobalIndex = oif->outputGlobalIndex;
141  forwardedInputsInfo.emplace_back(LogicalForwardInfo{ci, ii, oif->outputGlobalIndex});
142  }
143  return oif != availableOutputsInfo.end();
144  };
145 
146  // We have consumed the input, therefore we remove it from the list.
147  // We will insert the forwarded inputs only at the end of the iteration.
148  auto findNextOutputFor = [&availableOutputsInfo,&constOutputs,&oif,&workflow](
149  size_t ci, size_t &ii) {
150  auto &input = workflow[ci].inputs[ii];
151  auto matcher = [&input, &constOutputs](const LogicalOutputInfo &outputInfo) -> bool {
152  auto &output = constOutputs[outputInfo.outputGlobalIndex];
153  return matchDataSpec2Channel(input, outputSpec2LogicalChannel(output));
154  };
155  oif = availableOutputsInfo.erase(oif);
156  oif = std::find_if(oif, availableOutputsInfo.end(), matcher);
157  return oif;
158  };
159 
160  auto numberOfInputsFor = [&workflow](size_t ci) {
161  auto &consumer = workflow[ci];
162  return consumer.inputs.size();
163  };
164 
165  auto maxInputTimeslicesFor = [&workflow](size_t pi) {
166  auto &processor = workflow[pi];
167  return processor.maxInputTimeslices;
168  };
169 
170  // Trivial, but they make reading easier..
171  auto getOutputAssociatedProducer = [&oif]() {
172  return oif->specIndex;
173  };
174 
175  // Trivial, but they make reading easier..
176  auto getAssociateOutput = [&oif]() {
177  return oif->outputGlobalIndex;
178  };
179 
180  auto isForward = [&oif]() {
181  return oif->forward;
182  };
183 
184  // Trivial but makes reasing easier in the outer loop.
185  auto createEdge = [&logicalEdges](size_t producer,
186  size_t consumer,
187  size_t tpi,
188  size_t ptpi,
189  size_t uniqueOutputId,
190  size_t matchingInputInConsumer,
191  bool doForward) {
192  logicalEdges.emplace_back(
193  DeviceConnectionEdge{producer,
194  consumer,
195  tpi,
196  ptpi,
197  uniqueOutputId,
198  matchingInputInConsumer,
199  doForward});
200  };
201 
202  auto errorDueToMissingOutputFor = [&workflow](size_t ci, size_t ii) {
203  auto input = workflow[ci].inputs[ii];
204  std::ostringstream str;
205  str << "No matching output found for " << input.origin.str << " "
206  << input.description.str << " "
207  << input.subSpec << "\n";
208  throw std::runtime_error(str.str());
209  };
210 
211 
212  // Whenever we have a set of forwards, we need to append it
213  // the the global list of outputs, so that they can be matched
214  // and we need to add a ForwardRoute for the current consumer
215  // because it is the one who will actually do the forwarding.
216  auto appendForwardsToPossibleOutputs = [&availableOutputsInfo, &forwards]() {
217  for(auto &forward : forwards) {
218  availableOutputsInfo.push_back(forward);
219  }
220  };
221 
222  // Given we create a forward every time we match and input and an
223  // output, having no forwards means we did not find any matching.
224  auto noMatchingOutputFound = [&forwards]() {
225  return forwards.empty();
226  };
227 
228  // Forwards is basically a cache to record
229  auto newEdgeBetweenDevices = [&forwards]() {
230  forwards.clear();
231  };
232 
233  auto forwardOutputFrom = [&forwards](size_t consumer, size_t uniqueOutputId) {
234  forwards.push_back(LogicalOutputInfo{consumer, uniqueOutputId, true});
235  };
236 
237  // This is the outer loop
238  //
239  // Here we iterate over dataprocessor items in workflow and we consider them
240  // as consumer, since we are interested in their inputs.
241  // Notice also we need to search for all the matching inputs, since
242  // we could have more than one source that matches (e.g. in the
243  // case of a time merger).
244  // Once consumed, an output is not actually used anymore, however
245  // we append it as a forward.
246  // Finally, If a device has n-way pipelining, we need to create one node per
247  // parallel pipeline and add an edge for each.
248  enumerateAvailableOutputs();
249 
250  for (size_t consumer = 0; consumer < workflow.size(); ++consumer) {
251  for (size_t input = 0; input < numberOfInputsFor(consumer); ++input) {
252  newEdgeBetweenDevices();
253 
254  while (hasMatchingOutputFor(consumer, input)) {
255  auto producer = getOutputAssociatedProducer();
256  auto uniqueOutputId = getAssociateOutput();
257  for (size_t tpi = 0; tpi < maxInputTimeslicesFor(consumer); ++tpi) {
258  for (size_t ptpi = 0; ptpi < maxInputTimeslicesFor(producer); ++ptpi) {
259  createEdge(producer, consumer, tpi, ptpi, uniqueOutputId, input, isForward());
260  }
261  forwardOutputFrom(consumer, uniqueOutputId);
262  }
263  findNextOutputFor(consumer, input);
264  }
265  if (noMatchingOutputFound()) {
266  errorDueToMissingOutputFor(consumer, input);
267  }
268  appendForwardsToPossibleOutputs();
269  }
270  }

273 std::vector<EdgeAction>
274 WorkflowHelpers::computeOutEdgeActions(
275  const std::vector<DeviceConnectionEdge> &edges,
276  const std::vector<size_t> &index
277  ) {
278  DeviceConnectionEdge last{ULONG_MAX,ULONG_MAX,ULONG_MAX,ULONG_MAX,ULONG_MAX,ULONG_MAX};
279 
280  assert(edges.size() == index.size());
281  std::vector<EdgeAction> actions(edges.size(), EdgeAction{false, false});
282  for (size_t i : index) {
283  auto &edge = edges[i];
284  auto &action = actions[i];
285  action.requiresNewDevice = last.producer != edge.producer
286  || last.producerTimeIndex != edge.producerTimeIndex;
287  action.requiresNewChannel = last.consumer != edge.consumer
288  || last.producer != edge.producer
289  || last.timeIndex != edge.timeIndex
290  || last.producerTimeIndex != edge.producerTimeIndex;
291  last = edge;
292  }
293  return actions;
294 }

322 void
323 WorkflowHelpers::sortEdges(std::vector<size_t> &inEdgeIndex,
324  std::vector<size_t> &outEdgeIndex,
325  const std::vector<DeviceConnectionEdge> &edges) {
326  inEdgeIndex.resize(edges.size());
327  outEdgeIndex.resize(edges.size());
328  std::iota(inEdgeIndex.begin(), inEdgeIndex.end(), 0);
329  std::iota(outEdgeIndex.begin(), outEdgeIndex.end(), 0);
330 
331  // Two indexes, one to bind the outputs, the other
332  // one to connect the inputs. The
333  auto outSorter = [&edges](size_t i, size_t j) {
334  auto &a = edges[i];
335  auto &b = edges[j];
336  return std::tie(a.producer, a.producerTimeIndex, a.timeIndex, a.consumer)
337  < std::tie(b.producer, b.producerTimeIndex, b.timeIndex, b.consumer);
338  };
339  auto inSorter = [&edges](size_t i, size_t j) {
340  auto &a = edges[i];
341  auto &b = edges[j];
342  return std::tie(a.consumer, a.timeIndex, a.producer, a.producerTimeIndex)
343  < std::tie(b.consumer, b.timeIndex, b.producer, b.producerTimeIndex);
344  };
345 
346  std::sort(inEdgeIndex.begin(), inEdgeIndex.end(), inSorter);
347  std::sort(outEdgeIndex.begin(), outEdgeIndex.end(), outSorter);
348 }

350 void
351 WorkflowHelpers::verifyWorkflow(const o2::framework::WorkflowSpec &workflow) {
352  if (workflow.empty()){
353  throw std::runtime_error("Empty workflow!");
354  }
355  std::set<std::string> validNames;
356  std::vector<OutputSpec> availableOutputs;
357  std::vector<InputSpec> requiredInputs;
358 
359  // An index many to one index to go from a given input to the
360  // associated spec
361  std::map<size_t, size_t> inputToSpec;
362  // A one to one index to go from a given output to the Spec emitting it
363  std::map<size_t, size_t> outputToSpec;
364 
365  std::ostringstream ss;
366 
367  for (auto &spec : workflow)
368  {
369  if (spec.name.empty())
370  throw std::runtime_error("Invalid DataProcessorSpec name");
371  if (validNames.find(spec.name) != validNames.end())
372  throw std::runtime_error("Name " + spec.name + " is used twice.");
373  for (auto &option : spec.options) {
374  if (option.defaultValue.type() != VariantType::Empty &&
375  option.type != option.defaultValue.type()) {
376  ss << "Mismatch between declared option type and default value type"
377  << " for " << option.name << " in DataProcessorSpec of "
378  << spec.name;
379  throw std::runtime_error(ss.str());
380  }
381  }
382  for (size_t ii = 0; ii < spec.inputs.size(); ++ii) {
383  InputSpec const &input = spec.inputs[ii];
384  if (input.binding.empty()
385  || input.description == o2::header::DataDescription("")
386  || input.origin == o2::header::DataOrigin("")) {
387  ss << "In spec " << spec.name << " input specification "
388  << ii << " requires binding, description and origin"
389  " to be fully specified";
390  throw std::runtime_error(ss.str());
391  }
392  }
393  }
394 }
