104 Bool_t Detector::ProcessHits(FairVolume* v)
105 {
106  // TODO Implement handling of parents and primary particle
107  auto* mcapp = TVirtualMC::GetMC();
108  Double_t eloss = mcapp->Edep();
109  if (eloss < DBL_EPSILON)
110  return false; // only process hits which actually deposit some energy in the EMCAL
111  Geometry* geom = GetGeometry();
112  // Obtain detector ID
113  // This is not equal to the volume ID of the fair volume
114  // EMCAL geometry implementation in VMC works with a copy of the volume and placing it n-times into the mother volume
115  // via translation / rotation, so the copy index is the index of a tower / module / supermodule node within a mother
116  // volume Additional care needs to be taken for the supermodule index: The copy is connected to a certain supermodule
117  // type, which differs for various parts of the detector
118  Int_t copyEta, copyPhi, copyMod, copySmod;
119  mcapp->CurrentVolID(copyEta); // Tower in module - x-direction
120  mcapp->CurrentVolOffID(1, copyPhi); // Tower in module - y-direction
121  mcapp->CurrentVolOffID(3, copyMod); // Module in supermodule
122  mcapp->CurrentVolOffID(
123  4, copySmod); // Supermodule in EMCAL - attention, with respect to a given supermodule type (offsets needed)
124  std::string smtype(mcapp->CurrentVolOffName(4));
125  int offset(0);
126  if (smtype == "SM3rd") {
127  offset = 10;
128  } else if (smtype == "DCSM") {
129  offset = 12;
130  } else if (smtype == "DCEXT") {
131  offset = 18;
132  }
133  LOG(DEBUG3) << "Supermodule copy " << copySmod << ", module copy " << copyMod << ", y-dir " << copyPhi << ", x-dir "
134  << copyEta << ", supermodule ID " << copySmod + offset - 1 << std::endl;
135  LOG(DEBUG3) << "path " << mcapp->CurrentVolPath() << std::endl;
136  LOG(DEBUG3) << "Name of the supermodule type " << mcapp->CurrentVolOffName(4) << ", Module name "
137  << mcapp->CurrentVolOffName(3) << std::endl;
138 
139  Int_t partID = mcapp->GetStack()->GetCurrentTrackNumber(),
140  parent = mcapp->GetStack()->GetCurrentTrack()->GetMother(0),
141  detID = geom->GetAbsCellId(offset + copySmod - 1, copyMod - 1, copyPhi - 1, copyEta - 1);
142 
143  Double_t lightyield(eloss);
144  if (mcapp->TrackCharge())
145  lightyield = CalculateLightYield(eloss, mcapp->TrackStep(), mcapp->TrackCharge());
146  lightyield /= geom->GetSampling();
147 
148  if (partID != mCurrentTrackID || detID != mCurrentCellID || !mCurrentHit) {
149  // Condition for new hit:
150  // - Processing different track
151  // - Inside different cell
152  // - First track of the event
153  // std::cout << "New track / cell started\n";
154  Double_t posX, posY, posZ, momX, momY, momZ, energy;
155  mcapp->TrackPosition(posX, posY, posZ);
156  mcapp->TrackMomentum(momX, momY, momZ, energy);
157  Double_t estart = mcapp->Etot(), time = mcapp->TrackTime() * 1e9; // time in ns
158 
160  mCurrentHit =
161  AddHit(partID, parent, 0, estart, detID, Point3D<float>(float(posX), float(posY), float(posZ)),
162  Vector3D<float>(float(momX), float(momY), float(momZ)), time, lightyield);
163  static_cast<o2::Data::Stack*>(mcapp->GetStack())->addHit(GetDetId());
164  mCurrentTrackID = partID;
165  mCurrentCellID = detID;
166 
167  } else {
168  // std::cout << "Adding energy to the current hit\n";
169  mCurrentHit->SetEnergyLoss(mCurrentHit->GetEnergyLoss() + lightyield);
170  }
171 
172  return true;
173 }

185 Double_t Detector::CalculateLightYield(Double_t energydeposit, Double_t tracklength, Int_t charge) const
186 {
187  if (charge == 0)
188  return energydeposit; // full energy deposit for neutral particles (photons)
189  // Apply Birk's law (copied from G3BIRK)
190 
191  Float_t birkC1Mod = 0;
192  if (mBirkC0 == 1) { // Apply correction for higher charge states
193  if (std::abs(charge) >= 2)
194  birkC1Mod = mBirkC1 * 7.2 / 12.6;
195  else
196  birkC1Mod = mBirkC1;
197  }
198 
199  Float_t dedxcm = 0.;
200  if (tracklength > 0)
201  dedxcm = 1000. * energydeposit / tracklength;
202  else
203  dedxcm = 0;
204 
205  return energydeposit / (1. + birkC1Mod * dedxcm + mBirkC2 * dedxcm * dedxcm);
206 }

233 {

244  // Create the EMCAL Mother Volume (a polygone) within which to place the Detector and named XEN1

255  // Look to AliEMCALWsuCosmicRaySetUp.
256  TVirtualMC::GetMC()->Gspos(geom->GetNameOfEMCALEnvelope(), 1, "WSUC", 0.0, 0.0, +265., rotMatrixID, "ONLY");
257  } else {
258  Double_t envelopA[10];
259  envelopA[0] = geom->GetArm1PhiMin(); // minimum phi angle
260  envelopA[1] = geom->GetArm1PhiMax() - geom->GetArm1PhiMin(); // angular range in phi
261  envelopA[2] = envelopA[1] / geom->GetPhiSuperModule(); // Section of that
262  envelopA[3] = 2; // 2: z coordinates
263  envelopA[4] = -geom->GetEnvelop(2) / 2.; // zmin - includes padding

269  envelopA[9] = envelopA[6];
270  // radii are the same.
271  TVirtualMC::GetMC()->Gsvolu(geom->GetNameOfEMCALEnvelope(), "PGON", getMediumID(ID_AIR), envelopA,
272  10); // Polygone filled with air

274  LOG(DEBUG2) << "ConstructGeometry: " << geom->GetNameOfEMCALEnvelope() << " = " << envelopA[5] << ", "
275  << envelopA[6] << FairLogger::endl;
276  LOG(DEBUG2) << "ConstructGeometry: XU0 = " << envelopA[5] << ", " << envelopA[6] << FairLogger::endl;

278  // Position the EMCAL Mother Volume (XEN1) in Alice (ALIC)
279  TVirtualMC::GetMC()->Gspos(geom->GetNameOfEMCALEnvelope(), 1, "cave", 0.0, 0.0, 0.0, rotMatrixID, "ONLY");

283 void Detector::CreateShiskebabGeometry()
284 {
285  // TRD1
286  using boost::algorithm::contains;
287  Geometry* g = GetGeometry();
288  std::string gn = g->GetName();
289  std::transform(gn.begin(), gn.end(), gn.begin(), ::toupper);
290 
291  Double_t trd1Angle = g->GetTrd1Angle() * TMath::DegToRad(), tanTrd1 = TMath::Tan(trd1Angle / 2.);
292  // see AliModule::fFIdTmedArr
293  // fIdTmedArr = fIdtmed->GetArray() - 1599 ; // see AliEMCAL::::CreateMaterials()
294  // Int_t kIdAIR=1599, kIdPB = 1600, kIdSC = 1601, kIdSTEEL = 1603;
295  // idAL = 1602;
296  Double_t par[10], xpos = 0., ypos = 0., zpos = 0.;
297 
298  LOG(DEBUG2) << "Name of mother volume: " << g->GetNameOfEMCALEnvelope() << FairLogger::endl;
299  CreateSupermoduleGeometry(g->GetNameOfEMCALEnvelope());
300 
301  auto SMTypeList = g->GetEMCSystem();
302  auto tmpType = NOT_EXISTENT;
303  for (auto i : boost::irange(0, g->GetNumberOfSuperModules())) {
304  if (SMTypeList[i] == tmpType)
305  continue;
306  else
307  tmpType = SMTypeList[i];
308 
309  if (tmpType == EMCAL_STANDARD)
310  CreateEmcalModuleGeometry("SMOD", "EMOD"); // 18-may-05
311  else if (tmpType == EMCAL_HALF)
312  CreateEmcalModuleGeometry("SM10", "EMOD"); // Nov 1,2006 1/2 SM
313  else if (tmpType == EMCAL_THIRD)
314  CreateEmcalModuleGeometry("SM3rd", "EMOD"); // Feb 1,2012 1/3 SM
315  else if (tmpType == DCAL_STANDARD)
316  CreateEmcalModuleGeometry("DCSM", "EMOD"); // Mar 13, 2012, 6 or 10 DCSM
317  else if (tmpType == DCAL_EXT)
318  CreateEmcalModuleGeometry("DCEXT", "EMOD"); // Mar 13, 2012, DCAL extension SM
319  else
320  LOG(ERROR) << "Unkown SM Type!!\n";
321  }
322 
323  // Sensitive SC (2x2 tiles)
324  Double_t parSCM0[5] = { 0, 0, 0, 0 }, *dummy = nullptr, parTRAP[11];
325  if (!contains(gn, "V1")) {
326  Double_t wallThickness = g->GetPhiModuleSize() / g->GetNPHIdiv() - g->GetPhiTileSize();
327  for (Int_t i = 0; i < 3; i++)
328  parSCM0[i] = mParEMOD[i] - wallThickness;
329  parSCM0[3] = mParEMOD[3];
330  TVirtualMC::GetMC()->Gsvolu("SCM0", "TRD1", getMediumID(ID_AIR), parSCM0, 4);
331  TVirtualMC::GetMC()->Gspos("SCM0", 1, "EMOD", 0., 0., 0., 0, "ONLY");
332  } else {
333  Double_t wTh = g->GetLateralSteelStrip();
334  parSCM0[0] = mParEMOD[0] - wTh + tanTrd1 * g->GetTrd1AlFrontThick();
335  parSCM0[1] = mParEMOD[1] - wTh;
336  parSCM0[2] = mParEMOD[2] - wTh;
337  parSCM0[3] = mParEMOD[3] - g->GetTrd1AlFrontThick() / 2.;
338  TVirtualMC::GetMC()->Gsvolu("SCM0", "TRD1", getMediumID(ID_AIR), parSCM0, 4);
339  Double_t zshift = g->GetTrd1AlFrontThick() / 2.;
340  TVirtualMC::GetMC()->Gspos("SCM0", 1, "EMOD", 0., 0., zshift, 0, "ONLY");
341  //
342  CreateAlFrontPlate("EMOD", "ALFP");
343  }
344 
345  if (g->GetNPHIdiv() == 2 && g->GetNETAdiv() == 2) {
346  // Division to tile size - 1-oct-04
347  LOG(DEBUG2) << " Divide SCM0 on y-axis " << g->GetNETAdiv() << FairLogger::endl;
348  TVirtualMC::GetMC()->Gsdvn("SCMY", "SCM0", g->GetNETAdiv(), 2); // y-axis
349 
350  // Trapesoid 2x2
351  parTRAP[0] = parSCM0[3]; // dz
352  parTRAP[1] = TMath::ATan2((parSCM0[1] - parSCM0[0]) / 2., 2. * parSCM0[3]) * 180. / TMath::Pi(); // theta
353  parTRAP[2] = 0.; // phi
354 
355  // bottom
356  parTRAP[3] = parSCM0[2] / 2.; // H1
357  parTRAP[4] = parSCM0[0] / 2.; // BL1
358  parTRAP[5] = parTRAP[4]; // TL1
359  parTRAP[6] = 0.0; // ALP1
360 
361  // top
362  parTRAP[7] = parSCM0[2] / 2.; // H2
363  parTRAP[8] = parSCM0[1] / 2.; // BL2
364  parTRAP[9] = parTRAP[8]; // TL2
365  parTRAP[10] = 0.0; // ALP2
366 
367  LOG(DEBUG2) << " ** TRAP ** \n";
368  for (Int_t i = 0; i < 11; i++)
369  LOG(DEBUG3) << " par[" << std::setw(2) << std::setprecision(2) << i << "] " << std::setw(9)
370  << std::setprecision(4) << parTRAP[i] << FairLogger::endl;
371 
372  TVirtualMC::GetMC()->Gsvolu("SCMX", "TRAP", getMediumID(ID_SC), parTRAP, 11);
373  xpos = +(parSCM0[1] + parSCM0[0]) / 4.;
374  TVirtualMC::GetMC()->Gspos("SCMX", 1, "SCMY", xpos, 0.0, 0.0, 0, "ONLY");
375 
376  // Using rotation because SCMX should be the same due to Pb tiles
377  xpos = -xpos;
378  Int_t rotMatrixID(-1);
379  Matrix(rotMatrixID, 90.0, 180., 90.0, 270.0, 0.0, 0.0);
380  TVirtualMC::GetMC()->Gspos("SCMX", 2, "SCMY", xpos, 0.0, 0.0, rotMatrixID, "ONLY");
381 
382  // put LED to the SCM0
383  const ShishKebabTrd1Module& mod = g->GetShishKebabTrd1Modules()[0];
384  Double_t tanBetta = mod.GetTanBetta();
385 
386  Int_t nr = 0;
387  ypos = 0.0;
388  Double_t xCenterSCMX = (parTRAP[4] + parTRAP[8]) / 2.;
389  if (!contains(gn, "V1")) {
390  par[1] = parSCM0[2] / 2; // y
391  par[2] = g->GetECPbRadThick() / 2.; // z
392  TVirtualMC::GetMC()->Gsvolu("PBTI", "BOX", getMediumID(ID_PB), dummy, 0);
393 
394  zpos = -mSampleWidth * g->GetNECLayers() / 2. + g->GetECPbRadThick() / 2.;
395  LOG(DEBUG2) << " Pb tiles \n";
396 
397  for (Int_t iz = 0; iz < g->GetNECLayers(); iz++) {
398  par[0] = (parSCM0[0] + tanBetta * mSampleWidth * iz) / 2.;
399  xpos = par[0] - xCenterSCMX;
400  TVirtualMC::GetMC()->Gsposp("PBTI", ++nr, "SCMX", xpos, ypos, zpos, 0, "ONLY", par, 3);
401  LOG(DEBUG3) << iz + 1 << " xpos " << xpos << " zpos " << zpos << " par[0] " << par[0] << FairLogger::endl;
402  zpos += mSampleWidth;
403  }
404 
405  LOG(DEBUG2) << " Number of Pb tiles in SCMX " << nr << FairLogger::endl;
406  } else {
407  // Oct 26, 2010
408  // First sheet of paper
409  par[1] = parSCM0[2] / 2.; // y
410  par[2] = g->GetTrd1BondPaperThick() / 2.; // z
411  par[0] = parSCM0[0] / 2.; // x
412  TVirtualMC::GetMC()->Gsvolu("PAP1", "BOX", getMediumID(ID_PAPER), par, 3);
413 
414  xpos = par[0] - xCenterSCMX;
415  zpos = -parSCM0[3] + g->GetTrd1BondPaperThick() / 2.;
416  TVirtualMC::GetMC()->Gspos("PAP1", 1, "SCMX", xpos, ypos, zpos, 0, "ONLY");
417 
418  for (auto iz : boost::irange(0, g->GetNECLayers() - 1)) {
419  nr = iz + 1;
420  Double_t dz = g->GetECScintThick() + g->GetTrd1BondPaperThick() + mSampleWidth * iz;
421 
422  // PB + 2 paper sheets
423  par[2] = g->GetECPbRadThick() / 2. + g->GetTrd1BondPaperThick(); // z
424  par[0] = (parSCM0[0] + tanBetta * dz) / 2.;
425  TString pa(Form("PA%2.2i", nr));
426  TVirtualMC::GetMC()->Gsvolu(pa.Data(), "BOX", getMediumID(ID_PAPER), par, 3);
427 
428  xpos = par[0] - xCenterSCMX;
429  zpos = -parSCM0[3] + dz + par[2];
430  TVirtualMC::GetMC()->Gspos(pa.Data(), 1, "SCMX", xpos, ypos, zpos, 0, "ONLY");
431 
432  // Pb
433  TString pb(Form("PB%2.2i", nr));
434  par[2] = g->GetECPbRadThick() / 2.; // z
435  TVirtualMC::GetMC()->Gsvolu(pb.Data(), "BOX", getMediumID(ID_PB), par, 3);
436  TVirtualMC::GetMC()->Gspos(pb.Data(), 1, pa.Data(), 0.0, 0.0, 0.0, 0, "ONLY");
437  }
438  }
439  }
