30 void BackendRiak::Compress(const std::string& uncompressed_string, std::string& compressed_string)
31 {
32  // z_stream is zlib's control structure
33  z_stream zs;
34  memset(&zs, 0, sizeof(zs));
35 
36  if (deflateInit(&zs, Z_DEFAULT_COMPRESSION) != Z_OK) {
37  LOG(ERROR) << "deflateInit failed while compressing";
38  }
39 
40  zs.next_in = (Bytef*)uncompressed_string.data();
41  zs.avail_in = uncompressed_string.size();
42 
43  int ret;
44  char outbuffer[32768];
45  std::string outstring;
46 
47  // Get the compressed bytes in blocks of 32768 bytes using repeated calls to deflate
48  do {
49  zs.next_out = reinterpret_cast<Bytef*>(outbuffer);
50  zs.avail_out = sizeof(outbuffer);
51 
52  ret = deflate(&zs, Z_FINISH);
53 
54  if (outstring.size() < zs.total_out) {
55  // append the block to the output string
56  outstring.append(outbuffer, zs.total_out - outstring.size());
57  }
58  } while (ret == Z_OK);
59 
60  deflateEnd(&zs);
61 
62  if (ret != Z_STREAM_END) {
63  LOG(ERROR) << "Exception during zlib compression: (" << ret << ") " << zs.msg;
64  }
65 
66  compressed_string.assign(outstring);
67 }

69 void BackendRiak::Decompress(std::string& uncompressed_string, const std::string& compressed_string)
70 {
71  // z_stream is zlib's control structure
72  z_stream zs;
73  memset(&zs, 0, sizeof(zs));
74 
75  if (inflateInit(&zs) != Z_OK) {
76  LOG(ERROR) << "deflateInit failed while decompressing";
77  }
78 
79  zs.next_in = (Bytef*)compressed_string.data();
80  zs.avail_in = compressed_string.size();
81 
82  int ret;
83  char outbuffer[32768];
84  std::string outstring;
85 
86  // Get the decompressed bytes in blocks of 32768 bytes using repeated calls to inflate
87  do {
88  zs.next_out = reinterpret_cast<Bytef*>(outbuffer);
89  zs.avail_out = sizeof(outbuffer);
90 
91  ret = inflate(&zs, 0);
92 
93  if (outstring.size() < zs.total_out) {
94  outstring.append(outbuffer, zs.total_out - outstring.size());
95  }
96  } while (ret == Z_OK);
97 
98  inflateEnd(&zs);
99 
100  if (ret != Z_STREAM_END) {
101  LOG(ERROR) << "Exception during zlib compression: (" << ret << ") " << zs.msg;
102  }
103 
104  uncompressed_string.assign(outstring);
105 }
