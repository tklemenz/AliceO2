23 TrackPar::TrackPar(const array<float, 3>& xyz, const array<float, 3>& pxpypz, int charge, bool sectorAlpha)
24  : mX{ 0.f }, mAlpha{ 0.f }, mP{ 0.f }
25 {
26  // construct track param from kinematics
27 
28  // Alpha of the frame is defined as:
29  // sectorAlpha == false : -> angle of pt direction
30  // sectorAlpha == true : -> angle of the sector from X,Y coordinate for r>1
31  // angle of pt direction for r==0
32  //
33  //
34  constexpr float kSafe = 1e-5f;
35  float radPos2 = xyz[0] * xyz[0] + xyz[1] * xyz[1];
36  float alp = 0;
37  if (sectorAlpha || radPos2 < 1) {
38  alp = atan2f(pxpypz[1], pxpypz[0]);
39  } else {
40  alp = atan2f(xyz[1], xyz[0]);
41  }
42  if (sectorAlpha) {
43  alp = Utils::Angle2Alpha(alp);
44  }
45  //
46  float sn, cs;
47  Utils::sincosf(alp, sn, cs);
48  // protection: avoid alpha being too close to 0 or +-pi/2
49  if (fabs(sn) < 2 * kSafe) {
50  if (alp > 0) {
51  alp += alp < kPIHalf ? 2 * kSafe : -2 * kSafe;
52  } else {
53  alp += alp > -kPIHalf ? -2 * kSafe : 2 * kSafe;
54  }
55  Utils::sincosf(alp, sn, cs);
56  } else if (fabs(cs) < 2 * kSafe) {
57  if (alp > 0) {
58  alp += alp > kPIHalf ? 2 * kSafe : -2 * kSafe;
59  } else {
60  alp += alp > -kPIHalf ? 2 * kSafe : -2 * kSafe;
61  }
62  Utils::sincosf(alp, sn, cs);
63  }
64  // get the vertex of origin and the momentum
65  array<float, 3> ver{ xyz[0], xyz[1], xyz[2] };
66  array<float, 3> mom{ pxpypz[0], pxpypz[1], pxpypz[2] };
67  //
68  // Rotate to the local coordinate system
69  Utils::RotateZ(ver, -alp);
70  Utils::RotateZ(mom, -alp);
71  //
72  float ptI = 1.f / sqrt(mom[0] * mom[0] + mom[1] * mom[1]);
73  mX = ver[0];
74  mAlpha = alp;
75  mP[kY] = ver[1];
76  mP[kZ] = ver[2];
77  mP[kSnp] = mom[1] * ptI;
78  mP[kTgl] = mom[2] * ptI;
79  mP[kQ2Pt] = ptI * charge;
80  //
81  if (fabs(1 - getSnp()) < kSafe) {
82  mP[kSnp] = 1. - kSafe; // Protection
83  } else if (fabs(-1 - getSnp()) < kSafe) {
84  mP[kSnp] = -1. + kSafe; // Protection
85  }
86  //
87 }

90 bool TrackPar::getPxPyPzGlo(array<float, 3>& pxyz) const
91 {
92  // track momentum
93  if (fabs(getQ2Pt()) < kAlmost0 || fabs(getSnp()) > kAlmost1) {
94  return false;
95  }
96  float cs, sn, pt = fabs(1.f / getQ2Pt());
97  float r = sqrtf((1.f - getSnp()) * (1.f + getSnp()));
98  Utils::sincosf(getAlpha(), sn, cs);
99  pxyz[0] = pt * (r * cs - getSnp() * sn);
100  pxyz[1] = pt * (getSnp() * cs + r * sn);
101  pxyz[2] = pt * getTgl();
102  return true;
103 }

106 bool TrackPar::getPosDirGlo(array<float, 9>& posdirp) const
107 {
108  // fill vector with lab x,y,z,px/p,py/p,pz/p,p,sinAlpha,cosAlpha
109  float ptI = fabs(getQ2Pt());
110  float snp = getSnp();
111  if (ptI < kAlmost0 || fabs(snp) > kAlmost1) {
112  return false;
113  }
114  float &sn = posdirp[7], &cs = posdirp[8];
115  float csp = sqrtf((1.f - snp) * (1.f + snp));
116  float cstht = sqrtf(1.f + getTgl() * getTgl());
117  float csthti = 1.f / cstht;
118  Utils::sincosf(getAlpha(), sn, cs);
119  posdirp[0] = getX() * cs - getY() * sn;
120  posdirp[1] = getX() * sn + getY() * cs;
121  posdirp[2] = getZ();
122  posdirp[3] = (csp * cs - snp * sn) * csthti; // px/p
123  posdirp[4] = (snp * cs + csp * sn) * csthti; // py/p
124  posdirp[5] = getTgl() * csthti; // pz/p
125  posdirp[6] = cstht / ptI; // p
126  return true;
127 }

130 bool TrackPar::rotateParam(float alpha)
131 {
132  // rotate to alpha frame
133  if (fabs(getSnp()) > kAlmost1) {
134  // FairLogger::GetLogger()->Error(MESSAGE_ORIGIN,
135  printf("Precondition is not satisfied: |sin(phi)|>1 ! %f\n", getSnp());
136  return false;
137  }
138  //
139  Utils::BringToPMPi(alpha);
140  //
141  float ca = 0, sa = 0;
142  Utils::sincosf(alpha - getAlpha(), sa, ca);
143  float snp = getSnp(), csp = sqrtf((1.f - snp) * (1.f + snp)); // Improve precision
144  // RS: check if rotation does no invalidate track model (cos(local_phi)>=0, i.e. particle
145  // direction in local frame is along the X axis
146  if ((csp * ca + snp * sa) < 0) {
147  // FairLogger::GetLogger()->Warning(MESSAGE_ORIGIN,
148  printf("Rotation failed: local cos(phi) would become %.2f\n", csp * ca + snp * sa);
149  return false;
150  }
151  //
152  float tmp = snp * ca - csp * sa;
153  if (fabs(tmp) > kAlmost1) {
154  // FairLogger::GetLogger()->Warning(MESSAGE_ORIGIN,
155  printf("Rotation failed: new snp %.2f\n", tmp);
156  return false;
157  }
158  float xold = getX(), yold = getY();
159  mAlpha = alpha;
160  mX = xold * ca + yold * sa;
161  mP[kY] = -xold * sa + yold * ca;
162  mP[kSnp] = tmp;
163  return true;
164 }

167 bool TrackPar::propagateParamTo(float xk, const array<float, 3>& b)
168 {
169  //----------------------------------------------------------------
170  // Extrapolate this track params (w/o cov matrix) to the plane X=xk in the field b[].
171  //
172  // X [cm] is in the "tracking coordinate system" of this track.
173  // b[]={Bx,By,Bz} [kG] is in the Global coordidate system.
174  //----------------------------------------------------------------
175 
176  float dx = xk - getX();
177  if (fabs(dx) < kAlmost0) {
178  return true;
179  }
180  // Do not propagate tracks outside the ALICE detector
181  if (fabs(dx) > 1e5 || fabs(getY()) > 1e5 || fabs(getZ()) > 1e5) {
182  printf("Anomalous track, target X:%f\n", xk);
183  // Print();
184  return false;
185  }
186  float crv = (fabs(b[2]) < kAlmost0) ? 0.f : getCurvature(b[2]);
187  float x2r = crv * dx;
188  float f1 = getSnp(), f2 = f1 + x2r;
189  if (fabs(f1) > kAlmost1 || fabs(f2) > kAlmost1) {
190  return false;
191  }
192  if (fabs(getQ2Pt()) < kAlmost0) {
193  return false;
194  }
195  float r1 = sqrtf((1.f - f1) * (1.f + f1));
196  if (fabs(r1) < kAlmost0) {
197  return false;
198  }
199  float r2 = sqrtf((1.f - f2) * (1.f + f2));
200  if (fabs(r2) < kAlmost0) {
201  return false;
202  }
203  float dy2dx = (f1 + f2) / (r1 + r2);
204  float step = (fabs(x2r) < 0.05f) ? dx * fabs(r2 + f2 * dy2dx) // chord
205  : 2.f * asinf(0.5f * dx * sqrtf(1.f + dy2dx * dy2dx) * crv) / crv; // arc
206  step *= sqrtf(1.f + getTgl() * getTgl());
207  //
208  // get the track x,y,z,px/p,py/p,pz/p,p,sinAlpha,cosAlpha in the Global System
209  array<float, 9> vecLab{ 0.f };
210  if (!getPosDirGlo(vecLab)) {
211  return false;
212  }
213 
214  // rotate to the system where Bx=By=0.
215  float bxy2 = b[0] * b[0] + b[1] * b[1];
216  float bt = sqrtf(bxy2);
217  float cosphi = 1.f, sinphi = 0.f;
218  if (bt > kAlmost0) {
219  cosphi = b[0] / bt;
220  sinphi = b[1] / bt;
221  }
222  float bb = sqrtf(bxy2 + b[2] * b[2]);
223  float costet = 1., sintet = 0.;
224  if (bb > kAlmost0) {
225  costet = b[2] / bb;
226  sintet = bt / bb;
227  }
228  array<float, 7> vect{ costet * cosphi * vecLab[0] + costet * sinphi * vecLab[1] - sintet * vecLab[2],
229  -sinphi * vecLab[0] + cosphi * vecLab[1],
230  sintet * cosphi * vecLab[0] + sintet * sinphi * vecLab[1] + costet * vecLab[2],
231  costet * cosphi * vecLab[3] + costet * sinphi * vecLab[4] - sintet * vecLab[5],
232  -sinphi * vecLab[3] + cosphi * vecLab[4],
233  sintet * cosphi * vecLab[3] + sintet * sinphi * vecLab[4] + costet * vecLab[5],
234  vecLab[6] };
235 
236  // Do the helix step
237  float sgn = getSign();
238  g3helx3(sgn * bb, step, vect);
239 
240  // rotate back to the Global System
241  vecLab[0] = cosphi * costet * vect[0] - sinphi * vect[1] + cosphi * sintet * vect[2];
242  vecLab[1] = sinphi * costet * vect[0] + cosphi * vect[1] + sinphi * sintet * vect[2];
243  vecLab[2] = -sintet * vect[0] + costet * vect[2];
244 
245  vecLab[3] = cosphi * costet * vect[3] - sinphi * vect[4] + cosphi * sintet * vect[5];
246  vecLab[4] = sinphi * costet * vect[3] + cosphi * vect[4] + sinphi * sintet * vect[5];
247  vecLab[5] = -sintet * vect[3] + costet * vect[5];
248 
249  // rotate back to the Tracking System
250  float sinalp = -vecLab[7], cosalp = vecLab[8];
251  float t = cosalp * vecLab[0] - sinalp * vecLab[1];
252  vecLab[1] = sinalp * vecLab[0] + cosalp * vecLab[1];
253  vecLab[0] = t;
254  t = cosalp * vecLab[3] - sinalp * vecLab[4];
255  vecLab[4] = sinalp * vecLab[3] + cosalp * vecLab[4];
256  vecLab[3] = t;
257 
258  // Do the final correcting step to the target plane (linear approximation)
259  float x = vecLab[0], y = vecLab[1], z = vecLab[2];
260  if (fabs(dx) > kAlmost0) {
261  if (fabs(vecLab[3]) < kAlmost0) {
262  return false;
263  }
264  dx = xk - vecLab[0];
265  x += dx;
266  y += vecLab[4] / vecLab[3] * dx;
267  z += vecLab[5] / vecLab[3] * dx;
268  }
269 
270  // Calculate the track parameters
271  t = 1.f / sqrtf(vecLab[3] * vecLab[3] + vecLab[4] * vecLab[4]);
272  mX = x;
273  mP[kY] = y;
274  mP[kZ] = z;
275  mP[kSnp] = vecLab[4] * t;
276  mP[kTgl] = vecLab[5] * t;
277  mP[kQ2Pt] = sgn * t / vecLab[6];
278 
279  return true;
280 }

283 bool TrackPar::propagateParamTo(float xk, float b)
284 {
285  //----------------------------------------------------------------
286  // propagate this track to the plane X=xk (cm) in the field "b" (kG)
287  // Only parameters are propagated, not the matrix. To be used for small
288  // distances only (<mm, i.e. misalignment)
289  //----------------------------------------------------------------
290  float dx = xk - getX();
291  if (fabs(dx) < kAlmost0) {
292  return true;
293  }
294  float crv = (fabs(b) < kAlmost0) ? 0.f : getCurvature(b);
295  float x2r = crv * dx;
296  float f1 = getSnp(), f2 = f1 + x2r;
297  if ((fabs(f1) > kAlmost1) || (fabs(f2) > kAlmost1) || (fabs(getQ2Pt()) < kAlmost0)) {
298  return false;
299  }
300  float r1 = sqrtf((1.f - f1) * (1.f + f1));
301  if (fabs(r1) < kAlmost0) {
302  return false;
303  }
304  float r2 = sqrtf((1.f - f2) * (1.f + f2));
305  if (fabs(r2) < kAlmost0) {
306  return false;
307  }
308  mX = xk;
309  double dy2dx = (f1 + f2) / (r1 + r2);
310  mP[kY] += dx * dy2dx;
311  mP[kSnp] += x2r;
312  if (fabs(x2r) < 0.05f) {
313  mP[kZ] += dx * (r2 + f2 * dy2dx) * getTgl();
314  } else {
315  // for small dx/R the linear apporximation of the arc by the segment is OK,
316  // but at large dx/R the error is very large and leads to incorrect Z propagation
317  // angle traversed delta = 2*asin(dist_start_end / R / 2), hence the arc is: R*deltaPhi
318  // The dist_start_end is obtained from sqrt(dx^2+dy^2) = x/(r1+r2)*sqrt(2+f1*f2+r1*r2)
319  // double chord = dx*TMath::Sqrt(1+dy2dx*dy2dx); // distance from old position to new one
320  // double rot = 2*TMath::ASin(0.5*chord*crv); // angular difference seen from the circle center
321  // track1 += rot/crv*track3;
322  //
323  float rot = asinf(r1 * f2 - r2 * f1); // more economic version from Yura.
324  if (f1 * f1 + f2 * f2 > 1.f && f1 * f2 < 0.f) { // special cases of large rotations or large abs angles
325  if (f2 > 0.f) {
326  rot = kPI - rot; //
327  } else {
328  rot = -kPI - rot;
329  }
330  }
331  mP[kZ] += getTgl() / crv * rot;
332  }
333  return true;
334 }

515  c40 += b40;

529 //______________________________________________________________
530 bool TrackParCov::rotate(float alpha)
531 {
532  // rotate to alpha frame
533  if (fabs(getSnp()) > kAlmost1) {
534  // FairLogger::GetLogger()->Error(MESSAGE_ORIGIN,
535  printf("Precondition is not satisfied: |sin(phi)|>1 ! %f\n", getSnp());
536  return false;
537  }
538  //
539  Utils::BringToPMPi(alpha);
540  //
541  float ca = 0, sa = 0;
542  Utils::sincosf(alpha - getAlpha(), sa, ca);
543  float snp = getSnp(), csp = sqrtf((1.f - snp) * (1.f + snp)); // Improve precision
544  // RS: check if rotation does no invalidate track model (cos(local_phi)>=0, i.e. particle
545  // direction in local frame is along the X axis
546  if ((csp * ca + snp * sa) < 0) {
547  // FairLogger::GetLogger()->Warning(MESSAGE_ORIGIN,
548  printf("Rotation failed: local cos(phi) would become %.2f\n", csp * ca + snp * sa);
549  return false;
550  }
551  //
552 
553  float updSnp = snp * ca - csp * sa;
554  if (fabs(updSnp) > kAlmost1) {
555  // FairLogger::GetLogger()->Warning(MESSAGE_ORIGIN,
556  printf("Rotation failed: new snp %.2f\n", updSnp);
557  return false;
558  }
559  float xold = getX(), yold = getY();
560  setAlpha(alpha);
561  setX(xold * ca + yold * sa);
562  setY(-xold * sa + yold * ca);
563  setSnp(updSnp);
564 
565  if (fabs(csp) < kAlmost0) {
566  printf("Too small cosine value %f\n", csp);
567  csp = kAlmost0;
568  }
569 
570  float rr = (ca + snp / csp * sa);
571 
572  mC[kSigY2] *= (ca * ca);
573  mC[kSigZY] *= ca;
574  mC[kSigSnpY] *= ca * rr;
575  mC[kSigSnpZ] *= rr;
576  mC[kSigSnp2] *= rr * rr;
577  mC[kSigTglY] *= ca;
578  mC[kSigTglSnp] *= rr;
579  mC[kSigQ2PtY] *= ca;
580  mC[kSigQ2PtSnp] *= rr;
581 
582  checkCovariance();
583  return true;
584 }

586 //______________________________________________________________
587 TrackParCov::TrackParCov(const array<float, 3>& xyz, const array<float, 3>& pxpypz,
588  const array<float, kLabCovMatSize>& cv, int charge, bool sectorAlpha)
589 {
590  // construct track param and covariance from kinematics and lab errors
591 
592  // Alpha of the frame is defined as:
593  // sectorAlpha == false : -> angle of pt direction
594  // sectorAlpha == true : -> angle of the sector from X,Y coordinate for r>1
595  // angle of pt direction for r==0
596  //
597  //
598  constexpr float kSafe = 1e-5f;
599  float radPos2 = xyz[0] * xyz[0] + xyz[1] * xyz[1];
600  float alp = 0;
601  if (sectorAlpha || radPos2 < 1) {
602  alp = atan2f(pxpypz[1], pxpypz[0]);
603  } else {
604  alp = atan2f(xyz[1], xyz[0]);
605  }
606  if (sectorAlpha) {
607  alp = Utils::Angle2Alpha(alp);
608  }
609  //
610  float sn, cs;
611  Utils::sincosf(alp, sn, cs);
612  // protection: avoid alpha being too close to 0 or +-pi/2
613  if (fabs(sn) < 2.f * kSafe) {
614  if (alp > 0) {
615  alp += alp < kPIHalf ? 2.f * kSafe : -2.f * kSafe;
616  } else {
617  alp += alp > -kPIHalf ? -2.f * kSafe : 2.f * kSafe;
618  }
619  Utils::sincosf(alp, sn, cs);
620  } else if (fabs(cs) < 2.f * kSafe) {
621  if (alp > 0) {
622  alp += alp > kPIHalf ? 2.f * kSafe : -2.f * kSafe;
623  } else {
624  alp += alp > -kPIHalf ? 2.f * kSafe : -2.f * kSafe;
625  }
626  Utils::sincosf(alp, sn, cs);
627  }
628  // get the vertex of origin and the momentum
629  array<float, 3> ver{ xyz[0], xyz[1], xyz[2] };
630  array<float, 3> mom{ pxpypz[0], pxpypz[1], pxpypz[2] };
631  //
632  // Rotate to the local coordinate system
633  Utils::RotateZ(ver, -alp);
634  Utils::RotateZ(mom, -alp);
635  //
636  float pt = sqrt(mom[0] * mom[0] + mom[1] * mom[1]);
637  float ptI = 1.f / pt;
638  setX(ver[0]);
639  setAlpha(alp);
640  setY(ver[1]);
641  setZ(ver[2]);
642  setSnp(mom[1] * ptI); // cos(phi)
643  setTgl(mom[2] * ptI); // tg(lambda)
644  setQ2Pt(ptI * charge);
645  //
646  if (fabs(1.f - getSnp()) < kSafe) {
647  setSnp(1.f - kSafe); // Protection
648  } else if (fabs(-1.f - getSnp()) < kSafe) {
649  setSnp(-1.f + kSafe); // Protection
650  }
651  //
652  // Covariance matrix (formulas to be simplified)
653  float r = mom[0] * ptI; // cos(phi)
654  float cv34 = sqrtf(cv[3] * cv[3] + cv[4] * cv[4]);
655  //
656  int special = 0;
657  float sgcheck = r * sn + getSnp() * cs;
658  if (fabs(sgcheck) > 1 - kSafe) { // special case: lab phi is +-pi/2
659  special = 1;
660  sgcheck = sgcheck < 0 ? -1.f : 1.f;
661  } else if (fabs(sgcheck) < kSafe) {
662  sgcheck = cs < 0 ? -1.0f : 1.0f;
663  special = 2; // special case: lab phi is 0
664  }
665  //
666  mC[kSigY2] = cv[0] + cv[2];
667  mC[kSigZY] = (-cv[3] * sn) < 0 ? -cv34 : cv34;
668  mC[kSigZ2] = cv[5];
669  //
670  float ptI2 = ptI * ptI;
671  float tgl2 = getTgl() * getTgl();
672  if (special == 1) {
673  mC[kSigSnpY] = cv[6] * ptI;
674  mC[kSigSnpZ] = -sgcheck * cv[8] * r * ptI;
675  mC[kSigSnp2] = fabs(cv[9] * r * r * ptI2);
676  mC[kSigTglY] = (cv[10] * getTgl() - sgcheck * cv[15]) * ptI / r;
677  mC[kSigTglZ] = (cv[17] - sgcheck * cv[12] * getTgl()) * ptI;
678  mC[kSigTglSnp] = (-sgcheck * cv[18] + cv[13] * getTgl()) * r * ptI2;
679  mC[kSigTgl2] = fabs(cv[20] - 2 * sgcheck * cv[19] * mC[4] + cv[14] * tgl2) * ptI2;
680  mC[kSigQ2PtY] = cv[10] * ptI2 / r * charge;
681  mC[kSigQ2PtZ] = -sgcheck * cv[12] * ptI2 * charge;
682  mC[kSigQ2PtSnp] = cv[13] * r * ptI * ptI2 * charge;
683  mC[kSigQ2PtTgl] = (-sgcheck * cv[19] + cv[14] * getTgl()) * r * ptI2 * ptI;
684  mC[kSigQ2Pt2] = fabs(cv[14] * ptI2 * ptI2);
685  } else if (special == 2) {
686  mC[kSigSnpY] = -cv[10] * ptI * cs / sn;
687  mC[kSigSnpZ] = cv[12] * cs * ptI;
688  mC[kSigSnp2] = fabs(cv[14] * cs * cs * ptI2);
689  mC[kSigTglY] = (sgcheck * cv[6] * getTgl() - cv[15]) * ptI / sn;
690  mC[kSigTglZ] = (cv[17] - sgcheck * cv[8] * getTgl()) * ptI;
691  mC[kSigTglSnp] = (cv[19] - sgcheck * cv[13] * getTgl()) * cs * ptI2;
692  mC[kSigTgl2] = fabs(cv[20] - 2 * sgcheck * cv[18] * getTgl() + cv[9] * tgl2) * ptI2;
693  mC[kSigQ2PtY] = sgcheck * cv[6] * ptI2 / sn * charge;
694  mC[kSigQ2PtZ] = -sgcheck * cv[8] * ptI2 * charge;
695  mC[kSigQ2PtSnp] = -sgcheck * cv[13] * cs * ptI * ptI2 * charge;
696  mC[kSigQ2PtTgl] = (-sgcheck * cv[18] + cv[9] * getTgl()) * ptI2 * ptI * charge;
697  mC[kSigQ2Pt2] = fabs(cv[9] * ptI2 * ptI2);
698  } else {
699  double m00 = -sn; // m10=cs;
700  double m23 = -pt * (sn + getSnp() * cs / r), m43 = -pt * pt * (r * cs - getSnp() * sn);
701  double m24 = pt * (cs - getSnp() * sn / r), m44 = -pt * pt * (r * sn + getSnp() * cs);
702  double m35 = pt, m45 = -pt * pt * getTgl();
703  //
704  m43 *= charge;
705  m44 *= charge;
706  m45 *= charge;
707  //
708  double a1 = cv[13] - cv[9] * (m23 * m44 + m43 * m24) / m23 / m43;
709  double a2 = m23 * m24 - m23 * (m23 * m44 + m43 * m24) / m43;
710  double a3 = m43 * m44 - m43 * (m23 * m44 + m43 * m24) / m23;
711  double a4 = cv[14] + 2. * cv[9];
712  double a5 = m24 * m24 - 2. * m24 * m44 * m23 / m43;
713  double a6 = m44 * m44 - 2. * m24 * m44 * m43 / m23;
714  //
715  mC[kSigSnpY] = (cv[10] * m43 - cv[6] * m44) / (m24 * m43 - m23 * m44) / m00;
716  mC[kSigQ2PtY] = (cv[6] / m00 - mC[kSigSnpY] * m23) / m43;
717  mC[kSigTglY] = (cv[15] / m00 - mC[kSigQ2PtY] * m45) / m35;
718  mC[kSigSnpZ] = (cv[12] * m43 - cv[8] * m44) / (m24 * m43 - m23 * m44);
719  mC[kSigQ2PtZ] = (cv[8] - mC[kSigSnpZ] * m23) / m43;
720  mC[kSigTglZ] = cv[17] / m35 - mC[kSigQ2PtZ] * m45 / m35;
721  mC[kSigSnp2] = fabs((a4 * a3 - a6 * a1) / (a5 * a3 - a6 * a2));
722  mC[kSigQ2Pt2] = fabs((a1 - a2 * mC[kSigSnp2]) / a3);
723  mC[kSigQ2PtSnp] = (cv[9] - mC[kSigSnp2] * m23 * m23 - mC[kSigQ2Pt2] * m43 * m43) / m23 / m43;
724  double b1 = cv[18] - mC[kSigQ2PtSnp] * m23 * m45 - mC[kSigQ2Pt2] * m43 * m45;
725  double b2 = m23 * m35;
726  double b3 = m43 * m35;
727  double b4 = cv[19] - mC[kSigQ2PtSnp] * m24 * m45 - mC[kSigQ2Pt2] * m44 * m45;
728  double b5 = m24 * m35;
729  double b6 = m44 * m35;
730  mC[kSigTglSnp] = (b4 - b6 * b1 / b3) / (b5 - b6 * b2 / b3);
731  mC[kSigQ2PtTgl] = b1 / b3 - b2 * mC[kSigTglSnp] / b3;
732  mC[kSigTgl2] = fabs((cv[20] - mC[kSigQ2Pt2] * (m45 * m45) - mC[kSigQ2PtTgl] * 2. * m35 * m45) / (m35 * m35));
733  }
734  checkCovariance();
735 }

737 //____________________________________________________________
738 bool TrackParCov::propagateTo(float xk, const array<float, 3>& b)
739 {
740  //----------------------------------------------------------------
741  // Extrapolate this track to the plane X=xk in the field b[].
742  //
743  // X [cm] is in the "tracking coordinate system" of this track.
744  // b[]={Bx,By,Bz} [kG] is in the Global coordidate system.
745  //----------------------------------------------------------------
746 
747  float dx = xk - getX();
748  if (fabs(dx) < kAlmost0) {
749  return true;
750  }
751  // Do not propagate tracks outside the ALICE detector
752  if (fabs(dx) > 1e5 || fabs(getY()) > 1e5 || fabs(getZ()) > 1e5) {
753  printf("Anomalous track, target X:%f\n", xk);
754  // Print();
755  return false;
756  }
757  float crv = (fabs(b[2]) < kAlmost0) ? 0.f : getCurvature(b[2]);
758  float x2r = crv * dx;
759  float f1 = getSnp(), f2 = f1 + x2r;
760  if ((fabs(f1) > kAlmost1) || (fabs(f2) > kAlmost1) || (fabs(getQ2Pt()) < kAlmost0)) {
761  return false;
762  }
763  float r1 = sqrtf((1.f - f1) * (1.f + f1));
764  if (fabs(r1) < kAlmost0) {
765  return false;
766  }
767  float r2 = sqrtf((1.f - f2) * (1.f + f2));
768  if (fabs(r2) < kAlmost0) {
769  return false;
770  }
771 
772  float dy2dx = (f1 + f2) / (r1 + r2);
773  float step = (fabs(x2r) < 0.05f) ? dx * fabs(r2 + f2 * dy2dx) // chord
774  : 2.f * asinf(0.5f * dx * sqrtf(1.f + dy2dx * dy2dx) * crv) / crv; // arc
775  step *= sqrtf(1.f + getTgl() * getTgl());
776  //
777  // get the track x,y,z,px/p,py/p,pz/p,p,sinAlpha,cosAlpha in the Global System
778  array<float, 9> vecLab{ 0.f };
779  if (!getPosDirGlo(vecLab)) {
780  return false;
781  }
782  //
783  // matrix transformed with Bz component only
784  float &c00 = mC[kSigY2], &c10 = mC[kSigZY], &c11 = mC[kSigZ2], &c20 = mC[kSigSnpY], &c21 = mC[kSigSnpZ],
785  &c22 = mC[kSigSnp2], &c30 = mC[kSigTglY], &c31 = mC[kSigTglZ], &c32 = mC[kSigTglSnp], &c33 = mC[kSigTgl2],
786  &c40 = mC[kSigQ2PtY], &c41 = mC[kSigQ2PtZ], &c42 = mC[kSigQ2PtSnp], &c43 = mC[kSigQ2PtTgl],
787  &c44 = mC[kSigQ2Pt2];
788  // evaluate matrix in double prec.
789  double rinv = 1. / r1;
790  double r3inv = rinv * rinv * rinv;
791  double f24 = dx * b[2] * kB2C; // x2r/track[kQ2Pt];
792  double f02 = dx * r3inv;
793  double f04 = 0.5 * f24 * f02;
794  double f12 = f02 * getTgl() * f1;
795  double f14 = 0.5 * f24 * f12; // 0.5*f24*f02*getTgl()*f1;
796  double f13 = dx * rinv;
797 
798  // b = C*ft
799  double b00 = f02 * c20 + f04 * c40, b01 = f12 * c20 + f14 * c40 + f13 * c30;
800  double b02 = f24 * c40;
801  double b10 = f02 * c21 + f04 * c41, b11 = f12 * c21 + f14 * c41 + f13 * c31;
802  double b12 = f24 * c41;
803  double b20 = f02 * c22 + f04 * c42, b21 = f12 * c22 + f14 * c42 + f13 * c32;
804  double b22 = f24 * c42;
805  double b40 = f02 * c42 + f04 * c44, b41 = f12 * c42 + f14 * c44 + f13 * c43;
806  double b42 = f24 * c44;
807  double b30 = f02 * c32 + f04 * c43, b31 = f12 * c32 + f14 * c43 + f13 * c33;
808  double b32 = f24 * c43;
809 
810  // a = f*b = f*C*ft
811  double a00 = f02 * b20 + f04 * b40, a01 = f02 * b21 + f04 * b41, a02 = f02 * b22 + f04 * b42;
812  double a11 = f12 * b21 + f14 * b41 + f13 * b31, a12 = f12 * b22 + f14 * b42 + f13 * b32;
813  double a22 = f24 * b42;
814 
815  // F*C*Ft = C + (b + bt + a)
816  c00 += b00 + b00 + a00;
817  c10 += b10 + b01 + a01;
818  c20 += b20 + b02 + a02;
819  c30 += b30;
820  c40 += b40;
821  c11 += b11 + b11 + a11;
822  c21 += b21 + b12 + a12;
823  c31 += b31;
824  c41 += b41;
825  c22 += b22 + b22 + a22;
826  c32 += b32;
827  c42 += b42;
828 
829  checkCovariance();
830 
831  // Rotate to the system where Bx=By=0.
832  float bxy2 = b[0] * b[0] + b[1] * b[1];
833  float bt = sqrtf(bxy2);
834  float cosphi = 1.f, sinphi = 0.f;
835  if (bt > kAlmost0) {
836  cosphi = b[0] / bt;
837  sinphi = b[1] / bt;
838  }
839  float bb = sqrtf(bxy2 + b[2] * b[2]);
840  float costet = 1., sintet = 0.;
841  if (bb > kAlmost0) {
842  costet = b[2] / bb;
843  sintet = bt / bb;
844  }
845  array<float, 7> vect{ costet * cosphi * vecLab[0] + costet * sinphi * vecLab[1] - sintet * vecLab[2],
846  -sinphi * vecLab[0] + cosphi * vecLab[1],
847  sintet * cosphi * vecLab[0] + sintet * sinphi * vecLab[1] + costet * vecLab[2],
848  costet * cosphi * vecLab[3] + costet * sinphi * vecLab[4] - sintet * vecLab[5],
849  -sinphi * vecLab[3] + cosphi * vecLab[4],
850  sintet * cosphi * vecLab[3] + sintet * sinphi * vecLab[4] + costet * vecLab[5],
851  vecLab[6] };
852 
853  // Do the helix step
854  float sgn = getSign();
855  g3helx3(sgn * bb, step, vect);
856 
857  // Rotate back to the Global System
858  vecLab[0] = cosphi * costet * vect[0] - sinphi * vect[1] + cosphi * sintet * vect[2];
859  vecLab[1] = sinphi * costet * vect[0] + cosphi * vect[1] + sinphi * sintet * vect[2];
860  vecLab[2] = -sintet * vect[0] + costet * vect[2];
861 
862  vecLab[3] = cosphi * costet * vect[3] - sinphi * vect[4] + cosphi * sintet * vect[5];
863  vecLab[4] = sinphi * costet * vect[3] + cosphi * vect[4] + sinphi * sintet * vect[5];
864  vecLab[5] = -sintet * vect[3] + costet * vect[5];
865 
866  // Rotate back to the Tracking System
867  float sinalp = -vecLab[7], cosalp = vecLab[8];
868  float t = cosalp * vecLab[0] - sinalp * vecLab[1];
869  vecLab[1] = sinalp * vecLab[0] + cosalp * vecLab[1];
870  vecLab[0] = t;
871  t = cosalp * vecLab[3] - sinalp * vecLab[4];
872  vecLab[4] = sinalp * vecLab[3] + cosalp * vecLab[4];
873  vecLab[3] = t;
874 
875  // Do the final correcting step to the target plane (linear approximation)
876  float x = vecLab[0], y = vecLab[1], z = vecLab[2];
877  if (fabs(dx) > kAlmost0) {
878  if (fabs(vecLab[3]) < kAlmost0) {
879  return false;
880  }
881  dx = xk - vecLab[0];
882  x += dx;
883  y += vecLab[4] / vecLab[3] * dx;
884  z += vecLab[5] / vecLab[3] * dx;
885  }
886 
887  // Calculate the track parameters
888  t = 1.f / sqrtf(vecLab[3] * vecLab[3] + vecLab[4] * vecLab[4]);
889  setX(x);
890  setY(y);
891  setZ(z);
892  setSnp(vecLab[4] * t);
893  setTgl(vecLab[5] * t);
894  setQ2Pt(sgn * t / vecLab[6]);
895 
896  return true;
897 }

899 //______________________________________________
900 void TrackParCov::checkCovariance()
901 {
902  // This function forces the diagonal elements of the covariance matrix to be positive.
903  // In case the diagonal element is bigger than the maximal allowed value, it is set to
904  // the limit and the off-diagonal elements that correspond to it are set to zero.
905 
906  mC[kSigY2] = fabs(mC[kSigY2]);
907  if (mC[kSigY2] > kCY2max) {
908  float scl = sqrtf(kCY2max / mC[kSigY2]);
909  mC[kSigY2] = kCY2max;
910  mC[kSigZY] *= scl;
911  mC[kSigSnpY] *= scl;
912  mC[kSigTglY] *= scl;
913  mC[kSigQ2PtY] *= scl;
914  }
915  mC[kSigZ2] = fabs(mC[kSigZ2]);
916  if (mC[kSigZ2] > kCZ2max) {
917  float scl = sqrtf(kCZ2max / mC[kSigZ2]);
918  mC[kSigZ2] = kCZ2max;
919  mC[kSigZY] *= scl;
920  mC[kSigSnpZ] *= scl;
921  mC[kSigTglZ] *= scl;
922  mC[kSigQ2PtZ] *= scl;
923  }
924  mC[kSigSnp2] = fabs(mC[kSigSnp2]);
925  if (mC[kSigSnp2] > kCSnp2max) {
926  float scl = sqrtf(kCSnp2max / mC[kSigSnp2]);
927  mC[kSigSnp2] = kCSnp2max;
928  mC[kSigSnpY] *= scl;
929  mC[kSigSnpZ] *= scl;

1212 bool TrackParCov::correctForMaterial(float x2x0, float xrho, float mass, bool anglecorr, float dedx)
1213 {
1214  //------------------------------------------------------------------
1215  // This function corrects the track parameters for the crossed material.
1216  // "x2x0" - X/X0, the thickness in units of the radiation length.
1217  // "xrho" - is the product length*density (g/cm^2).
1218  // It should be passed as negative when propagating tracks
1219  // from the intreaction point to the outside of the central barrel.
1220  // "mass" - the mass of this particle (GeV/c^2). Negative mass means charge=2 particle
1221  // "dedx" - mean enery loss (GeV/(g/cm^2), if <=kCalcdEdxAuto : calculate on the fly
1222  // "anglecorr" - switch for the angular correction
1223  //------------------------------------------------------------------
1224  constexpr float kMSConst2 = 0.0136f * 0.0136f;
1225  constexpr float kMaxELossFrac = 0.3f; // max allowed fractional eloss
1226  constexpr float kMinP = 0.01f; // kill below this momentum
1227 
1228  float& fC22 = mC[kSigSnp2];
1229  float& fC33 = mC[kSigTgl2];
1230  float& fC43 = mC[kSigQ2PtTgl];
1231  float& fC44 = mC[kSigQ2Pt2];
1232  //
1233  float csp2 = (1.f - getSnp()) * (1.f + getSnp()); // cos(phi)^2
1234  float cst2I = (1.f + getTgl() * getTgl()); // 1/cos(lambda)^2
1235  // Apply angle correction, if requested
1236  if (anglecorr) {
1237  float angle = sqrtf(cst2I / (csp2));
1238  x2x0 *= angle;
1239  xrho *= angle;
1240  }
1241 
1242  float p = getP();
1243  if (mass < 0) {
1244  p += p; // q=2 particle
1245  }
1246  float p2 = p * p, mass2 = mass * mass;
1247  float e2 = p2 + mass2;
1248  float beta2 = p2 / e2;
1249 
1250  // Calculating the multiple scattering corrections******************
1251  float cC22(0.f), cC33(0.f), cC43(0.f), cC44(0.f);
1252  if (x2x0 != 0.f) {
1253  float theta2 = kMSConst2 / (beta2 * p2) * fabs(x2x0);
1254  if (mass < 0) {
1255  theta2 *= 4.f; // q=2 particle
1256  }
1257  if (theta2 > kPI * kPI) {
1258  return false;
1259  }
1260  float fp34 = getTgl() * getQ2Pt();
1261  float t2c2I = theta2 * cst2I;
1262  cC22 = t2c2I * csp2;
1263  cC33 = t2c2I * cst2I;
1264  cC43 = t2c2I * fp34;
1265  cC44 = theta2 * fp34 * fp34;
1266  // optimes this
1267  // cC22 = theta2*((1.-getSnp())*(1.+getSnp()))*(1. + getTgl()*getTgl());
1268  // cC33 = theta2*(1. + getTgl()*getTgl())*(1. + getTgl()*getTgl());
1269  // cC43 = theta2*getTgl()*getQ2Pt()*(1. + getTgl()*getTgl());
1270  // cC44 = theta2*getTgl()*getQ2Pt()*getTgl()*getQ2Pt();
1271  }
1272 
1273  // Calculating the energy loss corrections************************
1274  float cP4 = 1.f;
1275  if ((xrho != 0.f) && (beta2 < 1.f)) {
1276  if (dedx < kCalcdEdxAuto + kAlmost1) { // request to calculate dedx on the fly
1277  dedx = BetheBlochSolid(p / fabs(mass));
1278  if (mass < 0) {
1279  dedx *= 4.f; // z=2 particle
1280  }
1281  }
1282 
1283  float dE = dedx * xrho;
1284  float e = sqrtf(e2);
1285  if (fabs(dE) > kMaxELossFrac * e) {
1286  return false; // 30% energy loss is too much!
1287  }
1288  float eupd = e + dE;
1289  float pupd2 = eupd * eupd - mass2;
1290  if (pupd2 < kMinP * kMinP) {
1291  return false;
1292  }
1293  cP4 = p / sqrtf(pupd2);
1294  //
1295  // Approximate energy loss fluctuation (M.Ivanov)
1296  constexpr float knst = 0.07f; // To be tuned.
1297  float sigmadE = knst * sqrtf(fabs(dE)) * e / p2 * getQ2Pt();
1298  cC44 += sigmadE * sigmadE;
1299  }
1300 
1301  // Applying the corrections*****************************
1302  fC22 += cC22;
1303  fC33 += cC33;
1304  fC43 += cC43;
1305  fC44 += cC44;
1306  setQ2Pt( getQ2Pt()*cP4 );
1307 
1308  checkCovariance();
1309 
1310  return true;
1311 }
