32 bool Detector::ProcessHits(FairVolume* v)
33 {
34  // very rudimentatary hit creation
35  // TODO: needs upgrade to the level of AliROOT
36 
37  // TODO: reference to vmc --> put this as member of detector
38  static auto vmc = TVirtualMC::GetMC();
39 
40  // If not charged track or already stopped or disappeared, just return.
41  if ((!vmc->TrackCharge()) || vmc->IsTrackDisappeared()) {
42  return false;
43  }
44 
45  // just record position and basic quantities for the moment
46  // TODO: needs to be interpreted properly
47  double x, y, z;
48  vmc->TrackPosition(x, y, z);
49 
50  float enDep = vmc->Edep();
51  float time = vmc->TrackTime() * 1.0e09;
52  auto stack = (o2::Data::Stack *) TVirtualMC::GetMC()->GetStack();
53  auto trackID = stack->GetCurrentTrackNumber();
54  auto sensID = v->getMCid();
55  addHit((float)x, (float)y, (float)z, time, enDep, trackID, sensID);
56  stack->addHit(GetDetId());
57 
58  return true;
59 }
