35 Bool_t Cluster::testPixel(UShort_t row, UShort_t col) const
36 {
37  // test if pixel at relative row,col is fired
38  int nbits = row * getPatternColSpan() + col;
39  if (nbits >= kMaxPatternBits)
40  return kFALSE;
41  int bytn = nbits >> 3; // 1/8
42  int bitn = 7 - (nbits % 8);
43  return (mPattern[bytn] & (0x1 << bitn)) != 0;
44  //
45 }
