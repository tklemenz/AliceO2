90  template<typename T>
91  return_type operator()(T& stage) {
92  // the addWeight function belongs to the probability model as base
93  // of the specific model; funcions of the base can be accessed by
94  // static casting. This avoids an extra level of function calls.
95  return static_cast<typename T::wrapped_type::base_type&>(*stage).addWeight(value, weight);
96  }

110  bool addWeight(ValueType v, WeightType w, bool switchToNextModel = true) {
111  bool result = mContainer.apply(mPosition, addWeightFctr<ValueType, WeightType>(v, w));
112  if (switchToNextModel && ++mPosition >= getNumberOfModels()) mPosition = 0;
113  return result;

297  class readFctr {
298  public:

300  ~readFctr() {}
301 
302  typedef bool return_type;
303 
304  template<typename T>
305  return_type operator()(boost::type<T>) {
306  T& stage = static_cast<T&>(mContainer);
307  std::string level, name, remaining;
308  mIn >> level; mIn >> name;
309  if (!mIn) return false;
310  if (std::stoi(level) != T::level::value ||
311  name.compare((*stage).getName())) {
312  std::cerr << "Format error: expecting level '" << T::level::value << "' and name '" << (*stage).getName() << "', got: " << level << " " << name << std::endl;
313  }
314  std::cout << "reading configuration for model " << name << std::endl;
315  std::getline(mIn, remaining); // flush the current line
316  (*stage).read(mIn);
317  return true;
318  }
319 
320  private:
321  std::istream& mIn;
322  container_type& mContainer;
323  };
