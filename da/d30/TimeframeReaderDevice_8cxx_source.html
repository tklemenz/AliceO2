40  parts.AddPart(this->NewMessage(buffer,
41  size,
42  [](void* data, void* hint) { delete[] (char*)data; },
43  nullptr));
44  };
45  auto sendFn = [this](FairMQParts &parts) {this->Send(parts, this->mOutChannelName);};
46 
47  // FIXME: For the moment we support a single file. This should really be a glob. We
48  // should also have a strategy for watching directories.
49  std::vector<std::string> files;
50  files.push_back(mInFileName);
51  for (auto &&fn : files) {
52  mFile.open(fn, std::ofstream::in | std::ofstream::binary);
53  try {
54  streamTimeframe(mFile,
55  addPartFn,
56  sendFn);
57  } catch(std::runtime_error &e) {
58  LOG(ERROR) << e.what() << "\n";
59  }
60  mSeen.push_back(fn);
61  }
62 
63  return false;
