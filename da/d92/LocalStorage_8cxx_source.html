52 Bool_t LocalStorage::filenameToId(const char *filename, IdRunRange &runRange, Int_t &version, Int_t &subVersion)
53 {
54  // build ConditionId from filename numbers
55 
56  Ssiz_t mSize;
57 
58  // valid filename: Run#firstRun_#lastRun_v#version_s#subVersion.root
59  TRegexp keyPattern("^Run[0-9]+_[0-9]+_v[0-9]+_s[0-9]+.root$");
60  keyPattern.Index(filename, &mSize);
61  if (!mSize) {
62  LOG(DEBUG) << "Bad filename <" << filename << ">." << FairLogger::endl;
63  return kFALSE;
64  }
65 
66  TString idString(filename);
67  idString.Resize(idString.Length() - sizeof(".root") + 1);
68 
69  TObjArray *strArray = (TObjArray *) idString.Tokenize("_");
70 
71  TString firstRunString(((TObjString *) strArray->At(0))->GetString());
72  runRange.setFirstRun(atoi(firstRunString.Data() + 3));
73  runRange.setLastRun(atoi(((TObjString *) strArray->At(1))->GetString()));
74 
75  TString verString(((TObjString *) strArray->At(2))->GetString());
76  version = atoi(verString.Data() + 1);
77 
78  TString subVerString(((TObjString *) strArray->At(3))->GetString());
79  subVersion = atoi(subVerString.Data() + 1);
80 
81  delete strArray;
82 
83  return kTRUE;
84 }

114 Bool_t LocalStorage::prepareId(ConditionId &id)
115 {
116  // prepare id (version, subVersion) of the object that will be stored (called by putCondition)
117 
118  TString dirName = Form("%s/%s", mBaseDirectory.Data(), id.getPathString().Data());
119 
120  // go to the path; if directory does not exist, create it
121  void *dirPtr = gSystem->OpenDirectory(dirName);
122  if (!dirPtr) {
123  gSystem->mkdir(dirName, kTRUE);
124  dirPtr = gSystem->OpenDirectory(dirName);
125 
126  if (!dirPtr) {
127  LOG(ERROR) << R"(Can't create directory ")" << dirName.Data() << R"("!)" << FairLogger::endl;
128  return kFALSE;
129  }
130  }
131 
132  const char *filename;
133  IdRunRange aIdRunRange; // the runRange got from filename
134  IdRunRange lastIdRunRange(-1, -1); // highest runRange found
135  Int_t aVersion, aSubVersion; // the version subVersion got from filename
136  Int_t lastVersion = 0, lastSubVersion = -1; // highest version and subVersion found
137 
138  if (!id.hasVersion()) { // version not specified: look for highest version & subVersion
139 
140  while ((filename = gSystem->GetDirEntry(dirPtr))) { // loop on the files
141 
142  TString aString(filename);
143  if (aString == "." || aString == "..") {
144  continue;
145  }
146 
147  if (!filenameToId(filename, aIdRunRange, aVersion, aSubVersion)) {
148  LOG(DEBUG) << "Bad filename <" << filename << ">! I'll skip it." << FairLogger::endl;
149  continue;
150  }
151 
152  if (!aIdRunRange.isOverlappingWith(id.getIdRunRange())) {
153  continue;
154  }
155  if (aVersion < lastVersion) {
156  continue;
157  }
158  if (aVersion > lastVersion) {
159  lastSubVersion = -1;
160  }
161  if (aSubVersion < lastSubVersion) {
162  continue;
163  }
164  lastVersion = aVersion;
165  lastSubVersion = aSubVersion;
166  lastIdRunRange = aIdRunRange;
167  }
168 
169  id.setVersion(lastVersion);
170  id.setSubVersion(lastSubVersion + 1);
171 
172  } else { // version specified, look for highest subVersion only
173 
174  while ((filename = gSystem->GetDirEntry(dirPtr))) { // loop on the files
175 
176  TString aString(filename);
177  if (aString == "." || aString == "..") {
178  continue;
179  }
180 
181  if (!filenameToId(filename, aIdRunRange, aVersion, aSubVersion)) {
182  LOG(DEBUG) << "Bad filename <" << filename << ">!I'll skip it." << FairLogger::endl;
183  continue;
184  }
185 
186  if (aIdRunRange.isOverlappingWith(id.getIdRunRange()) && aVersion == id.getVersion() &&
187  aSubVersion > lastSubVersion) {
188  lastSubVersion = aSubVersion;
189  lastIdRunRange = aIdRunRange;
190  }
191  }
192 
193  id.setSubVersion(lastSubVersion + 1);
194  }
195 
196  gSystem->FreeDirectory(dirPtr);
197 
198  TString lastStorage = id.getLastStorage();
199  if (lastStorage.Contains(TString("grid"), TString::kIgnoreCase) && id.getSubVersion() > 0) {
200  LOG(ERROR) << "GridStorage to LocalStorage Storage error! local object with version v" << id.getVersion() << "_s"
201  << id.getSubVersion() - 1 << " found:" << FairLogger::endl;
202  LOG(ERROR) << "This object has been already transferred from GridStorage (check v" << id.getVersion() << "_s0)!"
203  << FairLogger::endl;
204  return kFALSE;
205  }
206 
207  if (lastStorage.Contains(TString("new"), TString::kIgnoreCase) && id.getSubVersion() > 0) {
208  LOG(DEBUG) << "A NEW object is being stored with version v" << id.getVersion() << "_s" << id.getSubVersion()
209  << FairLogger::endl;
210  LOG(DEBUG) << "and it will hide previously stored object with v" << id.getVersion() << "_s"
211  << id.getSubVersion() - 1 << "!" << FairLogger::endl;
212  }
213 
214  if (!lastIdRunRange.isAnyRange() && !(lastIdRunRange.isEqual(&id.getIdRunRange())))
215  LOG(WARNING) << "Run range modified w.r.t. previous version (Run" << lastIdRunRange.getFirstRun() << "_"
216  << lastIdRunRange.getLastRun() << "_v" << id.getVersion() << "_s" << id.getSubVersion() - 1
217  << FairLogger::endl;
218 
219  return kTRUE;
220 }

387 Condition *LocalStorage::getCondition(const ConditionId &queryId)
388 {
389  // get Condition from the storage (the CDB file matching the query is
390  // selected by getConditionId and the contained id is passed here)
391 
392  ConditionId *dataId = getConditionId(queryId);
393 
394  TString errMessage(TString::Format("No valid CDB object found! request was: %s", queryId.ToString().Data()));
395  if (!dataId || !dataId->isSpecified()) {
396  LOG(ERROR) << "No file found matching this id!" << FairLogger::endl;
397  throw std::runtime_error(errMessage.Data());
398  return nullptr;
399  }
400 
401  TString filename;
402  if (!idToFilename(*dataId, filename)) {
403  LOG(ERROR) << "Bad data ID encountered!" << FairLogger::endl;
404  delete dataId;
405  throw std::runtime_error(errMessage.Data());
406  return nullptr;
407  }
408 
409  TFile file(filename, "READ"); // open file
410  if (!file.IsOpen()) {
411  LOG(ERROR) << "Can't open file <" << filename.Data() << ">!" << FairLogger::endl;
412  delete dataId;
413  throw std::runtime_error(errMessage.Data());
414  return nullptr;
415  }
416 
417  // get the only Condition object from the file
418  // the object in the file is an Condition entry named " Condition"
419 
420  Condition *anCondition = dynamic_cast<Condition *>(file.Get(" Condition"));
421  if (!anCondition) {
422  LOG(ERROR) << "Bad storage data: No Condition in file!" << FairLogger::endl;
423  file.Close();
424  delete dataId;
425  throw std::runtime_error(errMessage.Data());
426  return nullptr;
427  }
428 
429  ConditionId &entryId = anCondition->getId();
430 
431  // The object's ConditionId are not reset during storage
432  // If object's ConditionId runRange or version do not match with filename,
433  // it means that someone renamed file by hand. In this case a warning msg is issued.
434 
435  anCondition->setLastStorage("local");
436 
437  if (!entryId.isEqual(dataId)) {
438  LOG(WARNING) << "Mismatch between file name and object's ConditionId!" << FairLogger::endl;
439  LOG(WARNING) << "File name: " << dataId->ToString().Data() << FairLogger::endl;
440  LOG(WARNING) << "Object's ConditionId: " << entryId.ToString().Data() << FairLogger::endl;
441  }
442 
443  // Check whether entry contains a TTree. In case load the tree in memory!
444  loadTreeFromFile(anCondition);
445 
446  // close file, return retrieved entry
447  file.Close();
448  delete dataId;
449 
450  return anCondition;
451 }

521 void LocalStorage::getEntriesForLevel1(const char *level0, const char *level1, const ConditionId &queryId,
522  TList *result)
523 {
524  // multiple request ( Storage::GetAllObjects)
525 
526  TString level1Dir = Form("%s/%s/%s", mBaseDirectory.Data(), level0, level1);
527 
528  void *level1DirPtr = gSystem->OpenDirectory(level1Dir);
529  if (!level1DirPtr) {
530  LOG(DEBUG) << "Can't open level1 directory <" << level1Dir.Data() << ">!" << FairLogger::endl;
531  return;
532  }
533 
534  const char *level2;
535  Long_t flag = 0;
536  while ((level2 = gSystem->GetDirEntry(level1DirPtr))) {
537 
538  TString level2Str(level2);
539  // skip directories starting with a dot (".svn" and similar in old svn working copies)
540  if (level2Str.BeginsWith('.')) {
541  continue;
542  }
543 
544  TString fullPath = Form("%s/%s", level1Dir.Data(), level2);
545 
546  Int_t res = gSystem->GetPathInfo(fullPath.Data(), nullptr, (Long64_t *) nullptr, &flag, nullptr);
547 
548  if (res) {
549  LOG(DEBUG) << "Error reading entry " << level2Str.Data() << " !" << FairLogger::endl;
550  continue;
551  }
552  if (!(flag & 2)) {
553  continue;
554  } // skip if not a directory
555 
556  if (queryId.getPath().doesLevel2Contain(level2)) {
557 
558  IdPath entryPath(level0, level1, level2);
559 
560  ConditionId entryId(entryPath, queryId.getIdRunRange(), queryId.getVersion(), queryId.getSubVersion());
561 
562  // check filenames to see if any includes queryId.getIdRunRange()
563  void *level2DirPtr = gSystem->OpenDirectory(fullPath);
564  if (!level2DirPtr) {
565  LOG(DEBUG) << "Can't open level2 directory <" << fullPath.Data() << ">!" << FairLogger::endl;
566  return;
567  }
568  const char *level3;
569  Long_t file_flag = 0;
570  while ((level3 = gSystem->GetDirEntry(level2DirPtr))) {
571  TString fileName(level3);
572  TString fullFileName = Form("%s/%s", fullPath.Data(), level3);
573 
574  Int_t file_res = gSystem->GetPathInfo(fullFileName.Data(), nullptr, (Long64_t *) nullptr, &file_flag, nullptr);
575 
576  if (file_res) {
577  LOG(DEBUG) << "Error reading entry " << level2Str.Data() << " !" << FairLogger::endl;
578  continue;
579  }
580  if (file_flag) {
581  continue;
582  } // it is not a regular file!
583 
584  // skip if result already contains an entry for this path
585  Bool_t alreadyLoaded = kFALSE;
586  Int_t nEntries = result->GetEntries();
587  for (int i = 0; i < nEntries; i++) {
588  Condition *lCondition = (Condition *) result->At(i);
589  ConditionId lId = lCondition->getId();
590  TString lIdPath = lId.getPathString();
591  if (lIdPath.EqualTo(entryPath.getPathString())) {
592  alreadyLoaded = kTRUE;
593  break;
594  }
595  }
596  if (alreadyLoaded)
597  continue;
598 
599  // skip filenames not matching the regex below
600  TRegexp re("^Run[0-9]+_[0-9]+_");
601  if (!fileName.Contains(re))
602  continue;
603  // Extract first- and last-run and version and subversion.
604  // This allows to avoid quering for a calibration path if we did not find a filename with
605  // run-range including the one specified in the query and
606  // with version, subversion matching the query
607  TString fn = fileName(3, fileName.Length() - 3);
608  TString firstRunStr = fn(0, fn.First('_'));
609  fn.Remove(0, firstRunStr.Length() + 1);
610  TString lastRunStr = fn(0, fn.First('_'));
611  fn.Remove(0, lastRunStr.Length() + 1);
612  TString versionStr = fn(1, fn.First('_') - 1);
613  fn.Remove(0, versionStr.Length() + 2);
614  TString subvStr = fn(1, fn.First('.') - 1);
615  Int_t firstRun = firstRunStr.Atoi();
616  Int_t lastRun = lastRunStr.Atoi();
617  IdRunRange rr(firstRun, lastRun);
618  Int_t version = versionStr.Atoi();
619  Int_t subVersion = subvStr.Atoi();
620 
621  Condition *anCondition = nullptr;
622  Bool_t versionOK = kTRUE, subVersionOK = kTRUE;
623  if (queryId.hasVersion() && version != queryId.getVersion())
624  versionOK = kFALSE;
625  if (queryId.hasSubVersion() && subVersion != queryId.getSubVersion())
626  subVersionOK = kFALSE;
627  if (rr.isSupersetOf(queryId.getIdRunRange()) && versionOK && subVersionOK) {
628  anCondition = getCondition(entryId);
629  result->Add(anCondition);
630  }
631  }
632  }
633  }
634 
635  gSystem->FreeDirectory(level1DirPtr);
636 }

827 void LocalStorage::queryValidFiles()
828 {
829  // Query the CDB for files valid for Storage::mRun.
830  // Fills list mValidFileIds with ConditionId objects extracted from CDB files
831  // present in the local storage.
832  // If mVersion was not set, mValidFileIds is filled with highest versions.
833  // In the CVMFS case, the mValidFileIds is filled from the file containing
834  // the filepaths corresponding to the highest versions for the give OCDB tag
835  // by launching the script which extracts the last versions for the given run.
836  //
837 
838  if (mVersion != -1)
839  LOG(WARNING) << "Version parameter is not used by local storage query!" << FairLogger::endl;
840  if (mConditionMetaDataFilter) {
841  LOG(WARNING) << "CDB meta data parameters are not used by local storage query!" << FairLogger::endl;
842  delete mConditionMetaDataFilter;
843  mConditionMetaDataFilter = nullptr;
844  }
845 
846  // Check if in CVMFS case
847  TString cvmfsOcdbTag(gSystem->Getenv("OCDB_PATH"));
848  if (!cvmfsOcdbTag.IsNull()) {
849  queryValidCVMFSFiles(cvmfsOcdbTag);
850  return;
851  }
852 
853  void *storageDirPtr = gSystem->OpenDirectory(mBaseDirectory);
854 
855  const char *level0;
856  while ((level0 = gSystem->GetDirEntry(storageDirPtr))) {
857 
858  TString level0Str(level0);
859  if (level0Str.BeginsWith(".")) {
860  continue;
861  }
862 
863  if (mPathFilter.doesLevel0Contain(level0)) {
864  TString level0Dir = Form("%s/%s", mBaseDirectory.Data(), level0);
865  void *level0DirPtr = gSystem->OpenDirectory(level0Dir);
866  const char *level1;
867  while ((level1 = gSystem->GetDirEntry(level0DirPtr))) {
868 
869  TString level1Str(level1);
870  if (level1Str.BeginsWith(".")) {
871  continue;
872  }
873 
874  if (mPathFilter.doesLevel1Contain(level1)) {
875  TString level1Dir = Form("%s/%s/%s", mBaseDirectory.Data(), level0, level1);
876 
877  void *level1DirPtr = gSystem->OpenDirectory(level1Dir);
878  const char *level2;
879  while ((level2 = gSystem->GetDirEntry(level1DirPtr))) {
880 
881  TString level2Str(level2);
882  if (level2Str.BeginsWith(".")) {
883  continue;
884  }
885 
886  if (mPathFilter.doesLevel2Contain(level2)) {
887  TString dirName = Form("%s/%s/%s/%s", mBaseDirectory.Data(), level0, level1, level2);
888 
889  void *dirPtr = gSystem->OpenDirectory(dirName);
890 
891  const char *filename;
892 
893  IdRunRange aIdRunRange; // the runRange got from filename
894  IdRunRange hvIdRunRange; // the runRange of the highest version valid file
895  Int_t aVersion, aSubVersion; // the version and subVersion got from filename
896  Int_t highestV = -1, highestSubV = -1; // the highest version and subVersion for this calibration type
897 
898  while ((filename = gSystem->GetDirEntry(dirPtr))) { // loop on files
899 
900  TString aString(filename);
901  if (aString.BeginsWith(".")) {
902  continue;
903  }
904 
905  if (!filenameToId(filename, aIdRunRange, aVersion, aSubVersion)) {
906  continue;
907  }
908 
909  IdRunRange runrg(mRun, mRun);
910  if (!aIdRunRange.isSupersetOf(runrg)) {
911  continue;
912  }
913 
914  // check to keep the highest version/subversion (in case of more than one)
915  if (aVersion > highestV) {
916  highestV = aVersion;
917  highestSubV = aSubVersion;
918  hvIdRunRange = aIdRunRange;
919  } else if (aVersion == highestV) {
920  if (aSubVersion > highestSubV) {
921  highestSubV = aSubVersion;
922  hvIdRunRange = aIdRunRange;
923  }
924  }
925  }
926  if (highestV >= 0) {
927  IdPath validPath(level0, level1, level2);
928  ConditionId *validId = new ConditionId(validPath, hvIdRunRange, highestV, highestSubV);
929  mValidFileIds.AddLast(validId);
930  }
931 
932  gSystem->FreeDirectory(dirPtr);
933  }
934  }
935  gSystem->FreeDirectory(level1DirPtr);
936  }
937  }
938  gSystem->FreeDirectory(level0DirPtr);
939  }
940  }
941  gSystem->FreeDirectory(storageDirPtr);
942 }

944 void LocalStorage::queryValidCVMFSFiles(TString &cvmfsOcdbTag)
945 {
946  // Called in the CVMFS case to fill the mValidFileIds from the file containing
947  // the filepaths corresponding to the highest versions for the given OCDB tag
948  // by launching the script which extracts the last versions for the given run.
949  //
950 
951  TString command = cvmfsOcdbTag;
952  LOG(DEBUG) << R"(Getting valid files from CVMFS-OCDB tag ")" << cvmfsOcdbTag.Data() << R"(")" << FairLogger::endl;
953  // CVMFS-OCDB tag. This is the file $OCDB_PATH/catalogue/20??.list.gz
954  // containing all CDB file paths (for the given AR tag)
955  cvmfsOcdbTag.Strip(TString::kTrailing, '/');
956  cvmfsOcdbTag.Append("/");
957  gSystem->ExpandPathName(cvmfsOcdbTag);
958  if (gSystem->AccessPathName(cvmfsOcdbTag))
959  LOG(FATAL) << "cvmfs OCDB set to an invalid path: " << cvmfsOcdbTag.Data() << FairLogger::endl;
960 
961  // The file containing the list of valid files for the current run has to be generated
962  // by running the (shell+awk) script on the CVMFS OCDB tag file.
963 
964  // the script in cvmfs to extract CDB filepaths for the given run has the following fullpath
965  // w.r.t. $OCDB_PATH: bin/OCDBperRun.sh
966  command = command.Strip(TString::kTrailing, '/');
967  command.Append("/bin/getOCDBFilesPerRun.sh ");
968  command += cvmfsOcdbTag;
969  // from URI define the last two levels of the path of the cvmfs ocdb tag (e.g. data/2012.list.gz)
970  TString uri(getUri());
971  uri.Remove(TString::kTrailing, '/');
972  TObjArray *osArr = uri.Tokenize('/');
973  TObjString *mcdata_os = dynamic_cast<TObjString *>(osArr->At(osArr->GetEntries() - 3));
974  TObjString *yeartype_os = nullptr;
975  TString mcdata = mcdata_os->GetString();
976  if (mcdata == TString("data")) {
977  yeartype_os = dynamic_cast<TObjString *>(osArr->At(osArr->GetEntries() - 2));
978  } else {
979  mcdata_os = dynamic_cast<TObjString *>(osArr->At(osArr->GetEntries() - 2));
980  yeartype_os = dynamic_cast<TObjString *>(osArr->At(osArr->GetEntries() - 1));
981  }
982  mcdata = mcdata_os->GetString();
983  TString yeartype = yeartype_os->GetString();
984  command += mcdata;
985  command += '/';
986  command += yeartype;
987  command += ".list.gz cvmfs ";
988  command += TString::Itoa(mRun, 10);
989  command += ' ';
990  command += TString::Itoa(mRun, 10);
991  command += " -y > ";
992  TString runValidFile(gSystem->WorkingDirectory());
993  runValidFile += '/';
994  runValidFile += mcdata;
995  runValidFile += '_';
996  runValidFile += yeartype;
997  runValidFile += '_';
998  runValidFile += TString::Itoa(mRun, 10);
999  command += runValidFile;
1000  LOG(DEBUG) << R"(Running command: ")" << command.Data() << R"(")" << FairLogger::endl;
1001  Int_t result = gSystem->Exec(command.Data());
1002  if (result != 0) {
1003  LOG(ERROR) << R"(Was not able to execute ")" << command.Data() << R"(")" << FairLogger::endl;
1004  }
1005 
1006  // We expect the file with valid paths for this run to be generated in the current directory
1007  // and to be named as the CVMFS OCDB tag, without .gz, with '_runnumber' appended
1008  // Fill mValidFileIds from file
1009  std::ifstream file(runValidFile.Data());
1010  if (!file.is_open()) {
1011  LOG(FATAL) << R"(Error opening file ")" << runValidFile.Data() << R"("!)" << FairLogger::endl;
1012  }
1013  TString filepath;
1014  while (filepath.ReadLine(file)) {
1015  // skip line in case it is not a root file path
1016  if (!filepath.EndsWith(".root")) {
1017  continue;
1018  }
1019  // extract three-level path and basename
1020  TObjArray *tokens = filepath.Tokenize('/');
1021  if (tokens->GetEntries() < 5) {
1022  LOG(ERROR) << R"(")" << filepath.Data() << R"(" is not a valid cvmfs path for an OCDB object)" << FairLogger::endl;
1023  continue;
1024  }
1025  TObjString *baseNameOstr = (TObjString *) tokens->At(tokens->GetEntries() - 1);
1026  TString baseName(baseNameOstr->String());
1027  TObjString *l0oStr = (TObjString *) tokens->At(tokens->GetEntries() - 4);
1028  TObjString *l1oStr = (TObjString *) tokens->At(tokens->GetEntries() - 3);
1029  TObjString *l2oStr = (TObjString *) tokens->At(tokens->GetEntries() - 2);
1030  TString l0(l0oStr->String());
1031  TString l1(l1oStr->String());
1032  TString l2(l2oStr->String());
1033  TString threeLevels = l0 + '/' + l1 + '/' + l2;
1034 
1035  IdPath validPath(threeLevels);
1036  // use basename and three-level path to create ConditionId
1037  IdRunRange aIdRunRange; // the runRange got from filename
1038  Int_t aVersion, aSubVersion; // the version and subVersion got from filename
1039  if (!filenameToId(baseName, aIdRunRange, aVersion, aSubVersion))
1040  LOG(ERROR) << R"(Could not create a valid CDB id from path: ")" << filepath.Data() << R"(")" << FairLogger::endl;
1041 
1042  IdRunRange runrg(mRun, mRun);
1043  if (!aIdRunRange.isSupersetOf(runrg)) {
1044  continue;
1045  } // should never happen (would mean awk script wrong output)
1046  // aIdRunRange contains requested run!
1047  ConditionId *validId = new ConditionId(validPath, aIdRunRange, aVersion, aSubVersion);
1048  mValidFileIds.AddLast(validId);
1049  }
1050 
1051  file.close();
1052  return;
1053 }
