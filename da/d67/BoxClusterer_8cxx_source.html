220 void BoxClusterer::FindLocalMaxima(const Int_t iCRU)
221 {
226 
227  R__ASSERT(mAllBins);
228 
229  Int_t nLocalMaxima = 0;
230  // loop over rows
231  for (Int_t iRow = 0; iRow < mRowsMax; iRow++) {
232 
233  Float_t* allBins = mAllBins[iRow];
234  Int_t* sigBins = mAllSigBins[iRow];
235  const Int_t nSigBins = mAllNSigBins[iRow];
236 
237  // loop over all signals
238  for (Int_t iSig = 0; iSig < nSigBins; iSig++) {
239 
240  Int_t bin = sigBins[iSig];
241  // Array of charged centered at the current signal
242  Float_t *qArray = &allBins[bin];
243  Float_t qMax = qArray[0];
244 
245  // First check that the charge is bigger than the threshold
246  if ( qMax < mMinQMax )
247  continue;
248 
249  // Require at least one neighboring time bin with signal
250  if ( qArray[-1] + qArray[1] <= 0 ) continue;
251  // Require at least one neighboring pad with signal
252  const Int_t maxTimeBin = mTimeBinsMax+4; // Used to step between neighboring
253  if ( mRequireNeighbouringPad
254  && (qArray[-maxTimeBin]+qArray[maxTimeBin]<=0) ) continue;
255  //
256  // Check that this is a local maximum
257  // Note that the checking is done so that if 2 charges has the same
258  // qMax then only 1 cluster is generated
259  // (that is why there is BOTH > and >=)
260  //
261  if (qArray[-maxTimeBin] >= qMax) continue;
262  if (qArray[+maxTimeBin] > qMax) continue;
263  if (qArray[-1 ] >= qMax) continue;
264  if (qArray[+1 ] > qMax) continue;
265  if (qArray[-maxTimeBin-1] >= qMax) continue;
266  if (qArray[+maxTimeBin+1] > qMax) continue;
267  if (qArray[+maxTimeBin-1] >= qMax) continue;
268  if (qArray[-maxTimeBin+1] > qMax) continue;
269 
270 // Short_t tb; Short_t pa;
271 // GetPadAndTimeBin(bin,pa,tb);
272 // if ((iCRU == 179 && iRow == 2 && pa == 104 && tb == 171) || (iCRU == 256 && iRow == 10 && pa == 27 && tb == 181) ) {
273 // std::cout << qArray[+maxTimeBin-1] << " " << qArray[+maxTimeBin] << " " << qArray[+maxTimeBin+1] << std::endl;
274 // std::cout << qArray[-1] << " " << qArray[0] << " " << qArray[+1] << std::endl;
275 // std::cout << qArray[-maxTimeBin-1] << " " << qArray[-maxTimeBin] << " " << qArray[-maxTimeBin+1] << std::endl;
276 // }
277  // We accept the local maximum as a cluster and calculates its
278  // parameters
279  ++nLocalMaxima;
280 
281  //
282  // Calculate the total charge as the sum over the region:
283  //
284  // o o o o o
285  // o i i i o
286  // o i C i o
287  // o i i i o
288  // o o o o o
289  //
290  // with qmax at the center C.
291  //
292  // The inner charge (i) we always add, but we only add the outer
293  // charge (o) if the neighboring inner bin (i) has a signal.
294  //
295  Short_t minP = 0, maxP = 0, minT = 0, maxT = 0;
296  Double_t meanP = 0; // mean pad position
297  Double_t meanT = 0; // mean time position
298  Double_t sigmaP = 0; // sigma pad position
299  Double_t sigmaT = 0; // sigma time position
300  Float_t qTot = qMax; // total charge
301  for(Short_t dTime = -1; dTime<=1; dTime++) { // delta time
302  for(Short_t dPad = -1; dPad<=1; dPad++) { // delta pad
303 
304  if( dPad==0 && dTime==0 ) // central pad
305  continue;
306 
307  Float_t charge = GetQ(qArray, dPad, dTime, minT, maxT, minP, maxP);
308  UpdateCluster(charge, dPad, dTime, qTot,
309  meanP, sigmaP, meanT, sigmaT);
310 
311  if( !mRequirePositiveCharge || charge>0 ) {
312  // see if the next neighbor is also above threshold
313 
314  if(dPad*dTime==0) { // we are above/below or to the sides
315  // (dPad, dTime) = (+-1, 0), or (0, +-1)
316  // so we only have 1 neighbor
317  charge = GetQ(qArray, 2*dPad, 2*dTime, minT, maxT, minP, maxP);
318  UpdateCluster(charge, 2*dPad, 2*dTime, qTot,
319  meanP, sigmaP, meanT, sigmaT);
320  } else { // we are in a diagonal corner so we have 3 neighbors
321  // (dPad, dTime) = (+-1, +-1), or (+-1, -+1)
322  charge = GetQ(qArray, dPad, 2*dTime, minT, maxT, minP, maxP);
323  UpdateCluster(charge, dPad, 2*dTime, qTot,
324  meanP, sigmaP, meanT, sigmaT);
325  charge = GetQ(qArray, 2*dPad, dTime, minT, maxT, minP, maxP);
326  UpdateCluster(charge, 2*dPad, dTime, qTot,
327  meanP, sigmaP, meanT, sigmaT);
328  charge = GetQ(qArray, 2*dPad, 2*dTime, minT, maxT, minP, maxP);
329  UpdateCluster(charge, 2*dPad, 2*dTime, qTot,
330  meanP, sigmaP, meanT, sigmaT);
331  }
332  }
333  }
334  }
335 
336  // calculate cluster parameters
337  if(qTot > 0) {
338  meanP /= qTot;
339  meanT /= qTot;
340  sigmaP /= qTot;
341  sigmaT /= qTot;
342  sigmaP = TMath::Sqrt(sigmaP - meanP*meanP);
343  sigmaT = TMath::Sqrt(sigmaT - meanT*meanT);
344  Short_t pad, timebin;
345  GetPadAndTimeBin(bin, pad, timebin);
346  meanP += pad;
347  meanT += timebin;
348  Short_t nPad = maxP-minP+1;
349  Short_t nTimeBins = maxT-minT+1;
350  Short_t size = 10*nPad+nTimeBins;
351  Cluster* cluster = ClusterContainer::AddCluster<Cluster>(mClusterArray, iCRU, iRow, qTot, qMax, meanP, meanT,sigmaP, sigmaT);
352

356 // cluster->Print(std::cout);

407 Int_t BoxClusterer::Update(const Int_t iCRU,
408  const Int_t iRow,
409  const Int_t iPad,
410  const Int_t iTimeBin,
411  Float_t signal)
412 {
414 
415  // -ne could have a list of active chambers
416  // if (!fActiveChambers[iSector]) return 0;
417 
418  // Stop processing if input is out of range
419  R__ASSERT(iRow>=0 && iRow<mRowsMax);
420  R__ASSERT(iPad>=0 && iPad<(mPadsMax+4));
421  R__ASSERT(iTimeBin>=0 && iTimeBin<(mTimeBinsMax+4));
422 
423  if (mRequirePositiveCharge && (signal <= 0)){
424  return 0; // signal was not accepted
425  }
426 
427  // ===| get pedestal |========================================================
428  if (mPedestals) {
429  signal -= mPedestals->getValue(CRU(iCRU), iRow, iPad);
430  }
431 
432  // Fill signal in array. Add 2 to pad and time to make sure that the 2D
433  // array even for (0, 0) has a valid 5x5 matrix.
434  Int_t bin = (iPad+2)*(mTimeBinsMax+4) + (iTimeBin+2);
435 
436  mAllBins[iRow][bin] = signal;
437  mAllSigBins[iRow][mAllNSigBins[iRow]] = bin;
438  mAllNSigBins[iRow]++;
439 
440  return 1; // signal was accepted
441 }
