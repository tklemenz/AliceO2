92 void ConditionsMQServer::Run()
93 {
94  std::unique_ptr<FairMQPoller> poller(
95  fTransportFactory->CreatePoller(fChannels, { "data-put", "data-get", "broker-get" }));
96 
97  while (CheckCurrentState(RUNNING)) {
98 
99  poller->Poll(-1);
100 
101  if (poller->CheckInput("data-get", 0)) {
102  std::unique_ptr<FairMQMessage> input(fTransportFactory->CreateMessage());
103 
104  if (Receive(input, "data-get") > 0) {
105  std::string serialString(static_cast<char*>(input->GetData()), input->GetSize());
106 
107  //LOG(DEBUG) << "Received a GET client message: " << serialString;
108 
109  std::string dataSource;
110  ParseDataSource(dataSource, serialString);
111 
112  if (dataSource == "OCDB") {
113  // Retrieve the key from the serialized message
114  std::string key;
115  Deserialize(serialString, key);
116 
117  getFromOCDB(key);
118  } else if (dataSource == "Riak") {
119  // No need to de-serialize, just forward message to the broker
120  fChannels.at("broker-get").at(0).Send(input);
121  }
122  }
123  }
124 
125  if (poller->CheckInput("data-put", 0)) {
126  std::unique_ptr<FairMQMessage> input(fTransportFactory->CreateMessage());
127 
128  if (Receive(input, "data-put") > 0) {
129  std::string serialString(static_cast<char*>(input->GetData()), input->GetSize());
130 
131  // LOG(DEBUG) << "Received a PUT client message: " << serialString;
132  LOG(DEBUG) << "Message size: " << input->GetSize();
133 
134  std::string dataSource;
135  ParseDataSource(dataSource, serialString);
136 
137  if (dataSource == "OCDB") {
138  LOG(ERROR) << "The GET operation is not supported for the OCDB data source yet";
139  } else if (dataSource == "Riak") {
140  fChannels.at("broker-put").at(0).Send(input);
141  }
142  }
143  }
144 
145  if (poller->CheckInput("broker-get", 0)) {
146  std::unique_ptr<FairMQMessage> input(fTransportFactory->CreateMessage());
147 
148  if (Receive(input, "broker-get") > 0) {
149  LOG(DEBUG) << "Received object from broker with a size of: " << input->GetSize();
150 
151  fChannels.at("data-get").at(0).Send(input);
152  }
153  }
154  }
155 }

157 // Query OCDB for the condition
158 void ConditionsMQServer::getFromOCDB(std::string key)
159 {
160  // Change key from i.e. "/DET/Calib/Histo/Run2008_2008_v1_s0" to (DET/Calib/Histo, 2008)
161  // FIXME: This will have to be changed in the future by adapting IdPath and getObject accordingly
162  std::size_t pos = key.rfind("/");
163  std::string identifier = key.substr(0, pos);
164  key.erase(0, pos + 4);
165  std::size_t pos2 = key.find("_");
166  int runId = atoi(key.substr(0, pos2).c_str());
167 
168  Condition* aCondition = nullptr;
169 
170  mCdbManager->setRun(runId);
171  aCondition = mCdbManager->getCondition(IdPath(identifier), runId);
172 
173  if (aCondition) {
174  LOG(DEBUG) << "Sending following parameter to the client:";
175  aCondition->printConditionMetaData();
176  TMessage* tmsg = new TMessage(kMESS_OBJECT);
177  tmsg->WriteObject(aCondition);
178 
179  std::unique_ptr<FairMQMessage> message(
180  fTransportFactory->CreateMessage(tmsg->Buffer(), tmsg->BufferSize(), free_tmessage, tmsg));
181 
182  fChannels.at("data-get").at(0).Send(message);
183  } else {
184  LOG(ERROR) << R"(Could not get a condition for ")" << key << R"(" and run )" << runId << "!";
185  }
186 }
